133class MyGeneratorAction(G4VUserPrimaryGeneratorAction):
134 "My Generator Action"
135
136 def GeneratePrimaries(self, anEvent):
137 global debug, nevTot
138 t_0 = time.time()
139 npart = 0
140 while npart == 0:
141 myPythia.GenerateEvent()
142 nevTot += 1
143 myTimer["pythia"] += time.time() - t_0
144 # pythia interaction happens at 0,0,0
145 # x = rnr.Uniform(-3.,3.)
146 # y = rnr.Uniform(-3.,3.)
147 # leave this smearing for later
148 pos = G4ThreeVector(0 * cm, 0 * cm, -50 * m)
149 vertex = G4PrimaryVertex(pos, 0.0)
150 # create new primaries and set them to the vertex
151 particles = myPythia.GetListOfParticles()
152 for p in particles:
153 if p.GetStatusCode() != 1:
154 continue
155 pid = p.GetPdgCode()
156 if tauOnly and abs(pid) != 16:
157 continue
158 if pid in notWanted:
159 continue
160 G4particle = G4PrimaryParticle(pid)
161 v = TLorentzVector()
162 p.Momentum(v)
163 if v.E() * GeV < ecut:
164 continue
165 G4particle.Set4Momentum(v.Px() * GeV, v.Py() * GeV, v.Pz() * GeV, v.E() * GeV)
166 vertex.SetPrimary(G4particle)
167 # store mother ID
168 mkey = p.GetMother(0) + 1
169 mother = myPythia.GetParticle(mkey)
170 curPid = p.GetPdgCode() + 10000 # make it positive
171 moPid = mother.GetPdgCode() + 10000
172 w = curPid + moPid * 100000
173 G4particle.SetWeight(w)
174 npart += 1
175 if tauOnly and debug:
176 myPythia.EventListing()
177 anEvent.AddPrimaryVertex(vertex)
178 myTimer["geant4_conv"] += time.time() - t_0
