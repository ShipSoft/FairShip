22 def create_fibre_to_simp_map(self):
23 """
24 Build mappings from optical fibres to SiPM channels for U and V planes.
25
26 Retrieves the U and V SiPM maps from the SciFi module and constructs
27 dictionaries mapping each fibre index to its associated SiPM channels,
28 including weight and x-position information.
29
30 For full details, see SND/MTC/MTCDetector.cxx in the SND framework.
31
32 Side Effects
33 ------------
34 Sets attributes 'fibre_to_simp_map_U' and 'fibre_to_simp_map_V'.
35 """
36 FU, FV = self.scifi.GetSiPMmapU(), self.scifi.GetSiPMmapV()
37 self.fibre_to_simp_map_U, self.fibre_to_simp_map_V = {}, {}
38 for x1, x2 in zip(FU, FV):
39 self.fibre_to_simp_map_U[x1.first] = {}
40 for z in x1.second:
41 self.fibre_to_simp_map_U[x1.first][z.first] = {
42 "weight": z.second[0],
43 "xpos": z.second[1],
44 }
45 self.fibre_to_simp_map_V[x2.first] = {}
46 for z in x2.second:
47 self.fibre_to_simp_map_V[x2.first][z.first] = {
48 "weight": z.second[0],
49 "xpos": z.second[1],
50 }

118 def draw_channel(self, channel):
119 """
120 Draw a single channel mapping showing fibre positions and the SiPM sensor.
121
122 Parameters
123 ----------
124 channel : int
125 Global channel identifier encoding plane type, SiPM unit, and channel index.
126
127 Side Effects
128 ------------
129 Saves a PDF file named 'scifi_mapping_channel_1_{channel}.pdf' with the plot.
130 """
131 AF = ROOT.TVector3()
132 BF = ROOT.TVector3()
133 plane_type = int(channel / 1e5) % 10
134 locChannel = channel % 1000000
135 fibreVol = sGeo.FindVolumeFast("FiberVol")
136 R = fibreVol.GetShape().GetDX()
137 DX = 0.025
138 DZ = 0.135
139 xmin = 999.0
140 xmax = -999.0
141 ymin = 999.0
142 ymax = -999.0
143 fibre_positions = []
144 fibresSiPM = self.fibre_to_simp_map_U if plane_type == 0 else self.fibre_to_simp_map_V
145 for fibre in fibresSiPM[locChannel]:
146 globfiberID = (
147 fibre + 100000000 + 1000000 + 0 * 100000
148 if plane_type == 0
149 else fibre + 100000000 + 1000000 + 1 * 100000
150 )
151 self.scifi.GetPosition(globfiberID, AF, BF)
152 loc = self.scifi.GetLocalPos(globfiberID, AF)
153 fibre_positions.append((loc[0], loc[2]))
154 if xmin > loc[0]:
155 xmin = loc[0]
156 if ymin > loc[2]:
157 ymin = loc[2]
158 if xmax < loc[0]:
159 xmax = loc[0]
160 if ymax < loc[2]:
161 ymax = loc[2]
162 D = ymax - ymin + 3 * R
163 x0 = (xmax + xmin) / 2.0
164 fig, ax = plt.subplots(figsize=(8, 8))
165 ax.set_xlim(x0 - D / 2, x0 + D / 2)
166 ax.set_ylim(ymin - 1.5 * R, ymax + 1.5 * R)
167 for i, (x, y) in enumerate(fibre_positions):
168 ellipse = patches.Ellipse(
169 (x, y), width=2 * R, height=2 * R, color="orange", alpha=0.6
170 )
171 ax.add_patch(ellipse)
172 self.scifi.GetSiPMPosition(locChannel, AF, BF)
173 loc = self.scifi.GetLocalPos(globfiberID, AF)
174 rect = patches.Rectangle(
175 (loc[0] - DX, loc[2] - DZ),
176 2 * DX,
177 2 * DZ,
178 linewidth=1,
179 edgecolor="blue",
180 facecolor="blue",
181 alpha=0.3,
182 hatch="//",
183 )
184 ax.add_patch(rect)
185 ax.set_xlabel("X [cm]")
186 ax.set_ylabel("Z [cm]")
187 ax.set_title(f"SiPM Mapping for Channel {channel}")
188 ax.grid(True)
189 ax.set_aspect("equal")
190 plt.savefig(f"scifi_mapping_channel_1_{channel}.pdf")

193 self,
194 output_file="scifi_mapping_all_channels.pdf",
195 figsize=(16, 16),
196 dpi=300,
197 cmap_name="tab20",

200 """
201 Draw overlay plot of multiple channel mappings on a single figure.
202
203 Parameters
204 ----------
205 output_file : str, optional
206 Filename for the saved PDF plot. Default is 'scifi_mapping_all_channels.pdf'.
207 figsize : tuple of float, optional
208 Figure size in inches. Default is (16, 16).
209 dpi : int, optional
210 Resolution of the figure in dots per inch. Default is 300.
211 cmap_name : str, optional
212 Matplotlib colormap name for differentiating channels. Default is 'tab20'.
213 alpha_fibre : float, optional
214 Transparency for fibre ellipses. Default is 0.4.
215
216 Side Effects
217 ------------
218 Saves an overlay PDF plot with the specified filename.
219 """
220 # Prepare fig & ax
221 fig, ax = plt.subplots(figsize=figsize, dpi=dpi)
222
223 # Select your channels. Hardcoded range for demonstration
224 channelsU = sorted(self.fibre_to_simp_map_U.keys())[:10]
225 channelsV = sorted(self.fibre_to_simp_map_V.keys())[::-1][86:96]
226 channels = channelsU + channelsV
227 n_chan = len(channels)
228
229 # Build a colormap for rectangles
230 cmap = plt.get_cmap(cmap_name)
231 colors = [cmap(i / (n_chan - 1)) for i in range(n_chan)]
232
233 # Geometry constants
234 fibreVol = sGeo.FindVolumeFast("FiberVol")
235 R = fibreVol.GetShape().GetDX()
236 DX, DZ = 0.025, 0.135 / 2
237
238 AF = ROOT.TVector3()
239 BF = ROOT.TVector3()
240
241 # Loop through all channels
242 for idx, chan in enumerate(channels):
243 col = colors[idx]
244
245 # decode local channel
246 locChan = chan % 1_000_000
247
248 # collect fibre positions
249 xs, ys = [], []
250 for fibreID in (
251 self.fibre_to_simp_map_U if chan in channelsU else self.fibre_to_simp_map_V
252 )[locChan]:
253 globfiberID = (
254 fibreID + 100000000 + 1000000 + 0 * 100000
255 if chan in channelsU
256 else fibreID + 100000000 + 1000000 + 1 * 100000
257 )
258 self.scifi.GetPosition(globfiberID, AF, BF)
259 loc = self.scifi.GetLocalPos(fibreID, AF)
260 xs.append(loc[0])
261 ys.append(loc[2])
262
263 # draw fibres (still orange)
264 for x, y in zip(xs, ys):
265 ell = patches.Ellipse(
266 (x, y), 2 * R, 2 * R, color="orange", alpha=alpha_fibre
267 )
268 ax.add_patch(ell)
269
270 # draw SiPM rect in its unique shade
271 self.scifi.GetSiPMPosition(chan, BF, AF)
272 loc_siPM = self.scifi.GetLocalPos(fibreID, AF)
273 rx, ry = loc_siPM[0], loc_siPM[2]
274 rect = patches.Rectangle(
275 (rx - DX, ry - DZ),
276 2 * DX,
277 2 * DZ,
278 linewidth=1,
279 edgecolor="black",
280 facecolor=col,

282 hatch="//",
283 )
284 ax.add_patch(rect)
285
286 # compute a fontsize that fits the rect width
287 # 1) data→display coords:
288 p0 = ax.transData.transform((rx - DX, ry))
289 p1 = ax.transData.transform((rx + DX, ry))
290 disp_width = abs(p1[0] - p0[0])
291 # 2) convert px→points (1pt = 1/72in; fig.dpi px/inch):
292 pts = disp_width * 72.0 / fig.dpi
293 fontsize = max(4, min(12, pts * 0.8))
294
295 # label above the rect
296 text = f"SiPM: {(chan // 1000) % 10} \n ch: {chan % 1000}"
297 ax.text(
298 rx - DX / 2,
299 ry + DZ + R * 0.1,
300 text,
301 ha="left",
302 va="bottom",
303 fontsize=fontsize,
304 color="black",
305 clip_on=True,
306 )
307

312 ax.tick_params(axis="both", which="major", labelsize=18)
313 ax.tick_params(axis="both", which="minor", labelsize=18)
314 ax.set_xlabel("X [cm]", fontsize=20)
315 ax.set_ylabel("Z [cm]", fontsize=20)
316 ax.set_title("SiPM-Channel Mappings Overlaid", fontsize=24)
317 ax.grid(True)
318
319 plt.tight_layout()
320 plt.savefig(output_file)
321 plt.close(fig)
322 print(f"Saved overlay plot of {n_chan} channels to {output_file}")
