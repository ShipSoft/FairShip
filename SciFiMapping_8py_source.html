32 def create_fibre_to_simp_map(self):
33 """Build mappings from optical fibres to SiPM channels for U and V planes.
34
35 Retrieves the U and V SiPM maps from the SciFi module and constructs
36 dictionaries mapping each fibre index to its associated SiPM channels,
37 including weight and x-position information.
38
39 For full details, see SND/MTC/MTCDetector.cxx in the SND framework.
40
41 Side Effects
42 ------------
43 Sets attributes 'fibre_to_simp_map_U' and 'fibre_to_simp_map_V'.
44 """
45 FU, FV = self.scifi.GetSiPMmapU(), self.scifi.GetSiPMmapV()
46 self.fibre_to_simp_map_U, self.fibre_to_simp_map_V = {}, {}
47 for x1, x2 in zip(FU, FV):
48 self.fibre_to_simp_map_U[x1.first] = {}
49 for z in x1.second:
50 self.fibre_to_simp_map_U[x1.first][z.first] = {
51 "weight": z.second[0],
52 "xpos": z.second[1],
53 }
54 self.fibre_to_simp_map_V[x2.first] = {}
55 for z in x2.second:
56 self.fibre_to_simp_map_V[x2.first][z.first] = {
57 "weight": z.second[0],
58 "xpos": z.second[1],
59 }

90 def create_sipm_to_position_map(self):
91 """Create a mapping from SiPM channels to their positions in the SciFi detector.
92
93 This method retrieves the SiPM positions for both U and V planes and constructs
94 dictionaries mapping each SiPM channel index to its corresponding position.
95
96 Side Effects
97 ------------
98 Sets attributes 'sipm_pos_U' and 'sipm_pos_V'.
99 """

102 self.sipm_pos_U, self.sipm_pos_V = {}, {}

105 self.sipm_pos_U[pair.first] = pair.second

145 def draw_channel(self, sGeo, channel):
146 """Draw a single channel mapping showing fibre positions and the SiPM sensor.
147
148 Parameters
149 ----------
150 sGeo: ROOT.TGeoManager
151 The geometry manager for the detector setup.
152 channel : int
153 Global channel identifier encoding plane type, SiPM unit, and channel index.
154
155 Side Effects
156 ------------
157 Saves a PDF file named 'scifi_mapping_channel_1_{channel}.pdf' with the plot.
158
159 """
160 AF = ROOT.TVector3()
161 BF = ROOT.TVector3()
162 plane_type = int(channel / 1e5) % 10
163 locChannel = channel % 1000000
164 fibreVol = sGeo.FindVolumeFast("FiberVol")
165 R = fibreVol.GetShape().GetDX()
166 DX = 0.025
167 DZ = 0.135
168 xmin = 999.0
169 xmax = -999.0
170 ymin = 999.0
171 ymax = -999.0
172 fibre_positions = []
173 fibresSiPM = (
174 self.fibre_to_simp_map_U if plane_type == 0 else self.fibre_to_simp_map_V
175 )
176 for fibre in fibresSiPM[locChannel]:
177 globfiberID = (
178 fibre + 100000000 + 1000000 + 0 * 100000
179 if plane_type == 0
180 else fibre + 100000000 + 1000000 + 1 * 100000
181 )
182 self.scifi.GetPosition(globfiberID, AF, BF)
183 loc = self.scifi.GetLocalPos(globfiberID, BF)
184 print(f"Position for fibre {globfiberID} A: {BF.X()}, {BF.Y()}, {BF.Z()}")
185 fibre_positions.append((loc[0], loc[2]))
186 if xmin > loc[0]:
187 xmin = loc[0]
188 if ymin > loc[2]:
189 ymin = loc[2]
190 if xmax < loc[0]:
191 xmax = loc[0]
192 if ymax < loc[2]:
193 ymax = loc[2]
194 D = ymax - ymin + 3 * R
195 x0 = (xmax + xmin) / 2.0
196 fig, ax = plt.subplots(figsize=(8, 8))
197 ax.set_xlim(x0 - D / 2, x0 + D / 2)
198 ax.set_ylim(ymin - 1.5 * R, ymax + 1.5 * R)
199 for i, (x, y) in enumerate(fibre_positions):
200 ellipse = patches.Ellipse(
201 (x, y), width=2 * R, height=2 * R, color="orange", alpha=0.6
202 )
203 ax.add_patch(ellipse)
204 self.scifi.GetSiPMPosition(locChannel, AF, BF)
205 loc = self.scifi.GetLocalPos(globfiberID, BF)
206 print(f"SiPM position for channel {channel}: {loc[0]}, {loc[1]}, {loc[2]}")
207 rect = patches.Rectangle(
208 (loc[0] - DX, loc[2] - DZ),
209 2 * DX,
210 2 * DZ,
211 linewidth=1,
212 edgecolor="blue",
213 facecolor="blue",
214 alpha=0.3,
215 hatch="//",
216 )
217 ax.add_patch(rect)
218 ax.set_xlabel("X [cm]")
219 ax.set_ylabel("Z [cm]")
220 ax.set_title(f"SiPM Mapping for Channel {channel}")
221 ax.grid(True)
222 ax.set_aspect("equal")
223 plt.savefig(f"scifi_mapping_channel_1_{channel}.pdf")

284 def get_surrounding_keys(d: dict, target_key, N: int):
285 """Return neighbors of `target_key` in value-sorted order.
286
287 Given a dict `d`, find `target_key` in the ordering of keys sorted by
288 their values, and return up to N//2 keys that come before it and N//2 keys
289 that come after it (excluding the target_key itself).
290
291 Parameters
292 ----------
293 d : dict
294 Mapping from keys to comparable values.
295 target_key : hashable
296 The key around which to collect neighbors.
297 N : int
298 Total number of neighbors to return (N//2 before, N//2 after).
299
300 Returns
301 -------
302 list
303 List of up to N keys: first the keys before, then the keys after.
304
305 """
306 # 1) sort keys by their associated values
307 sorted_keys = sorted(d.keys(), key=lambda k: d[k])
308 # 2) locate the target
309 try:
310 idx = sorted_keys.index(target_key)
311 except ValueError:
312 raise KeyError(f"Target key {target_key!r} not found in dictionary")
313
314 half = N // 2
315 # 3) slice out the neighbors
316 start = max(0, idx - half)
317 end_before = idx
318 start_after = idx + 1
319 end_after = idx + 1 + half
320
321 before = sorted_keys[start:end_before]
322 after = sorted_keys[start_after:end_after]
323
324 return before + [target_key] + after
325
326 channelsU = get_surrounding_keys(
327 self.sipm_pos_U, center_channel_U, number_of_channels
328 )
329 channelsV = get_surrounding_keys(
330 self.sipm_pos_V, center_channel_V, number_of_channels
331 )
332
333 channels = channelsU + channelsV
334 n_chan = len(channels)
335
336 AF = ROOT.TVector3()
337 BF = ROOT.TVector3()
338
339 # Loop through all channels
340 for chan in channels:
341 isU = chan in channelsU
342 # decode local channel
343 locChan = chan % 1_000_000
344
345 # collect fibre positions
346 xs, ys = [], []
347 for fibreID in (
348 self.fibre_to_simp_map_U
349 if chan in channelsU
350 else self.fibre_to_simp_map_V
351 )[locChan]:
352 globfiberID = (
353 fibreID + 100000000 + 1000000 + 0 * 100000
354 if chan in channelsU
355 else fibreID + 100000000 + 1000000 + 1 * 100000
356 )
357 self.scifi.GetPosition(globfiberID, AF, BF)
358 loc = self.scifi.GetLocalPos(fibreID, BF)
359 xs.append(loc[0])
360 ys.append(loc[2])
361
362 # draw fibres (still orange)
363 for x, y in zip(xs, ys):
364 ell = patches.Ellipse(
365 (x, y),
366 2 * R,
367 2 * R,
368 fill=False,
369 edgecolor="black",
370 linewidth=1,
371 alpha=alpha_fibre,
372 )
373 ax.add_patch(ell)
374
375 # draw SiPM rect in its unique shade
376 self.scifi.GetSiPMPosition(chan, BF, AF)
377 loc_siPM = self.scifi.GetLocalPos(fibreID, BF)
378 rx, ry = loc_siPM[0], loc_siPM[2]
379 rect = patches.Rectangle(
380 (rx - DX, ry - DZ),
381 2 * DX,
382 2 * DZ,
383 linewidth=1,
384 edgecolor="black",
385 facecolor="blue"
386 if isU
387 else "red", # override with solid colors for clarity,
388 alpha=alpha_fibre * 0.8,
389 hatch="//" if isU else "\\\\",
390 )
391 ax.add_patch(rect)
392
393 if labeling:
394 # compute a fontsize that fits the rect width
395 # 1) datadisplay coords:
396 p0 = ax.transData.transform((rx - DX, ry))
397 p1 = ax.transData.transform((rx + DX, ry))
398 disp_width = abs(p1[0] - p0[0])
399 # 2) convert pxpoints (1pt = 1/72in; fig.dpi px/inch):
400 pts = disp_width * 72.0 / fig.dpi
401 fontsize = max(2, min(12, pts * 0.8))
402
403 # label above the rect
404 text = f"SiPM: {(chan // 1000) % 10} \n ch: {chan % 1000}"
405 ax.text(
406 rx - DX / 2,
407 ry + DZ + R * 0.1,
408 text,
409 ha="left",
410 va="bottom",
411 fontsize=fontsize,
412 color="black",
413 clip_on=True,
414 )
415
416 # finalize plot
417 ax.relim()
418 ax.autoscale_view()
419 ax.set_aspect("equal")
420 ax.tick_params(axis="both", which="major", labelsize=18)
421 ax.tick_params(axis="both", which="minor", labelsize=18)
422 ax.set_xlabel("X [cm]", fontsize=20)
423 ax.set_ylabel("Z [cm]", fontsize=20)
424 ax.set_title("SiPM-Channel Mappings Overlaid", fontsize=24)
425 ax.grid(True)
426
427 plt.tight_layout()
428 plt.savefig(output_file)
429 plt.close(fig)
430 print(f"Saved overlay plot of {n_chan} channels to {output_file}")

432 def draw_channel_XY(
433 self,
434 number_of_channels=20,
435 real_event=False,
436 x_coords=None,

438 figsize=(16, 16),

441 """Plot SciFi channel ribbons and SiPM boxes in XY view.
442
443 Simplified XY view: one ribbon per channel (no per-fiber loops).
444 Ribbons span Y from -25 to +25 (length=50 cm), thickness equal
445 to channel height = 2*DZ, tilted +5 for U-plane channels, -5 for
446 V-plane. SiPM boxes sit on top at y=2525+2*DZ.
447
448 Parameters
449 ----------
450 number_of_channels : int
451 Total ribbons (split around center) per plane.
452 real_event : bool
453 If True, highlight ribbons corresponding to given x_coords.
454 x_coords : list of float
455 If real_event is True, list of two x-coordinates [x_U, x_V] to highlight.
456 output_file : str
457 Path to save the PDF.
458 figsize : tuple of float
459 Figure size in inches.
460 dpi : int
461 Figure resolution in dots per inch.
462 Side Effects
463 ------------
464 Saves a PDF file with the specified filename containing the plot.
465
466 """
467 # 2) Geometry constants
468 DZ = 0.135 / 2 # SiPM half-height [cm]
469 DX = 0.025 # SiPM half-width [cm]
470 L = 50.0 # ribbon length (full Y-span) [cm]
471 angle_U = -5.0 # tilt for U
472 angle_V = +5.0 # tilt for V
473

485 channels_x_V = next(
486 c for c, x in self.sipm_pos_V.items() if abs(x - x_coords[1]) < DX
487 )
488 channel_x_V = [
489 c for c, x in self.sipm_pos_V.items() if abs(x - x_coords[1]) < DX
490 ][0]
491 # 1) Figure setup
492 fig, ax = plt.subplots(figsize=figsize, dpi=dpi)
493
494 # get center channels + neighbors
495 centerU = next(c for c, x in self.sipm_pos_U.items() if abs(x) < DX)
496 centerV = next(c for c, x in self.sipm_pos_V.items() if abs(x) < DX)

498 def get_surrounding(d, target, N):
499 keys = sorted(d.keys(), key=lambda k: d[k])
500 i = keys.index(target)
501 half = N // 2
502 start = max(0, i - half)
503 end = min(len(keys), i + half + 1)

573

576 2 * DX,
577 DZ,
578 facecolor=col, # same fill color
579 edgecolor="black",
580 alpha=0.75, # a bit more opaque
581 hatch="//" if isU else "\\\\",
582 )
583 ax.add_patch(box)
584 else:
585 ribbon = patches.Polygon(
586 coords,
587 closed=True,
588 facecolor=col, # fill in your color
589 edgecolor="black",
590 alpha=alpha_fibre, # semi-transparent

615 else:

617 ax.set_ylim(-25, 26)
618 ax.set_xlim(min_x, max_x)
619 # ax.set_ylim(22, 26)
620 ax.set_xlabel("X [cm]")
621 ax.set_ylabel("Y [cm]")
622 ax.set_title("SciFi channel ribbons & SiPM boxes (XY view)")

627 facecolor="blue",
628 edgecolor="black",
629 hatch="//",
630 label="U plane",

646 print(f"Saved channel ribbons + SiPM boxes to {output_file}")
