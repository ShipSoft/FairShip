28 def create_fibre_to_simp_map(self):
29 """Build mappings from optical fibres to SiPM channels for U and V planes.
30
31 Retrieves the U and V SiPM maps from the SciFi module and constructs
32 dictionaries mapping each fibre index to its associated SiPM channels,
33 including weight and x-position information.
34
35 For full details, see SND/MTC/MTCDetector.cxx in the SND framework.
36
37 Side Effects
38 ------------
39 Sets attributes 'fibre_to_simp_map_U' and 'fibre_to_simp_map_V'.
40 """
41 FU, FV = self.scifi.GetSiPMmapU(), self.scifi.GetSiPMmapV()
42 self.fibre_to_simp_map_U, self.fibre_to_simp_map_V = {}, {}
43 for x1, x2 in zip(FU, FV):
44 self.fibre_to_simp_map_U[x1.first] = {}
45 for z in x1.second:
46 self.fibre_to_simp_map_U[x1.first][z.first] = {
47 "weight": z.second[0],
48 "xpos": z.second[1],
49 }
50 self.fibre_to_simp_map_V[x2.first] = {}
51 for z in x2.second:
52 self.fibre_to_simp_map_V[x2.first][z.first] = {
53 "weight": z.second[0],
54 "xpos": z.second[1],
55 }

280 def get_surrounding_keys(d: dict, target_key, N: int):
281 """Return neighbors of `target_key` in value-sorted order.
282
283 Given a dict `d`, find `target_key` in the ordering of keys sorted by
284 their values, and return up to N//2 keys that come before it and N//2 keys
285 that come after it (excluding the target_key itself).
286
287 Parameters
288 ----------
289 d : dict
290 Mapping from keys to comparable values.
291 target_key : hashable
292 The key around which to collect neighbors.
293 N : int
294 Total number of neighbors to return (N//2 before, N//2 after).
295
296 Returns
297 -------
298 list
299 List of up to N keys: first the keys before, then the keys after.
300
301 """
302 # 1) sort keys by their associated values
303 sorted_keys = sorted(d.keys(), key=lambda k: d[k])
304 # 2) locate the target
305 try:
306 idx = sorted_keys.index(target_key)
307 except ValueError:
308 raise KeyError(f"Target key {target_key!r} not found in dictionary")
309
310 half = N // 2
311 # 3) slice out the neighbors
312 start = max(0, idx - half)
313 end_before = idx
314 start_after = idx + 1
315 end_after = idx + 1 + half
316
317 before = sorted_keys[start:end_before]
318 after = sorted_keys[start_after:end_after]
319
320 return before + [target_key] + after
321
