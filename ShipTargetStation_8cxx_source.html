41Int_t ShipTargetStation::InitMedium(const char* name) {
42 static FairGeoLoader* geoLoad = FairGeoLoader::Instance();
43 static FairGeoInterface* geoFace = geoLoad->getGeoInterface();
44 static FairGeoMedia* media = geoFace->getMedia();
45 static FairGeoBuilder* geoBuild = geoLoad->getGeoBuilder();
46
47 FairGeoMedium* ShipMedium = media->getMedium(name);
48
49 if (!ShipMedium) {
50 Fatal("InitMedium", "Material %s not defined in media file.", name);
51 return -1111;
52 }
53 TGeoMedium* medium = gGeoManager->GetMedium(name);
54 if (medium != nullptr) return ShipMedium->getMediumIndex();
55 return geoBuild->createMedium(ShipMedium);
56}

58void ShipTargetStation::ConstructGeometry() {
59 TGeoVolume* top = gGeoManager->GetTopVolume();
60
61 InitMedium("tungsten");
62 TGeoMedium* tungsten = gGeoManager->GetMedium("tungsten");
63 InitMedium("tantalum");
64 TGeoMedium* tantalum = gGeoManager->GetMedium("tantalum");
65 InitMedium("molybdenum");
66 TGeoMedium* mo = gGeoManager->GetMedium("molybdenum");
67 InitMedium("iron");
68 TGeoMedium* iron = gGeoManager->GetMedium("iron");
69 InitMedium("steel316L");
70 TGeoMedium* steel316L = gGeoManager->GetMedium("steel316L");
71 InitMedium("copper");
72 TGeoMedium* copper = gGeoManager->GetMedium("copper");
73
74 InitMedium("Inconel718");
75 TGeoMedium* inc718 = gGeoManager->GetMedium("Inconel718");
76
77 InitMedium("vacuums");
78 TGeoMedium* vacuum = gGeoManager->GetMedium("vacuums");
79
80 double He_T = 273.15 + fHeT; // in K, use average 90 degrees C.
81 double He_P = 1.6e6 * 6.241509e3; // 16 bar in MeV/mm3
82
83 std::ostringstream lHename;
84 lHename << "PressurisedHe" << fHeT;
85 InitMedium(lHename.str().c_str());
86 TGeoMedium* pressurised_He = gGeoManager->GetMedium(lHename.str().c_str());
87
88 // CAMM- dirty fix to have pressure and temperature correct for Geant4.
89 // Should fix this properly in future...
90 TGeoMaterial* fixedCooler = pressurised_He->GetMaterial();
91 fixedCooler->SetTemperature(He_T);
92 fixedCooler->SetPressure(He_P);
93 TGeoMedium* cooler = pressurised_He;
94
95 LOG(info) << "-- Target cooler: " << cooler->GetName()
96 << " T=" << cooler->GetMaterial()->GetTemperature()
97 << " K, P=" << cooler->GetMaterial()->GetPressure()
98 << " MeV/mm3, Density=" << cooler->GetMaterial()->GetDensity();
99
100 TGeoVolume* tTarget = new TGeoVolumeAssembly("TargetArea");
101
102 // Target vessel - inner dimensions for diameter, shifts and length
103 double vessel_thickness = 8 * mm;
104 double vessel_diameter = fDiameter + 150 * mm; // Inner diameter of vessel
105 double vessel_shift =
106 62 *
107 mm; // Shift in z at y=0 (round shape not implemented for vessel lids)
108 double vessel_length = fTargetLength + 2 * vessel_shift;
109
110 TGeoVolume* vessel;
111 vessel = gGeoManager->MakeTube("TargetVessel", inc718, vessel_diameter / 2.,
112 vessel_diameter / 2. + vessel_thickness,
113 vessel_length / 2.);
114 vessel->SetLineColor(28);
115 tTarget->AddNode(
116 vessel, 1,
117 new TGeoTranslation(0, 0, -1. * vessel_shift + vessel_length / 2.));
118 // Front face
119 vessel = gGeoManager->MakeTube("TargetVesselFront", inc718, 0,
120 vessel_diameter / 2. + vessel_thickness,
121 vessel_thickness / 2.);
122 vessel->SetLineColor(28);
123 tTarget->AddNode(
124 vessel, 1,
125 new TGeoTranslation(0, 0, -1. * vessel_shift - vessel_thickness / 2.));
126 // Back face
127 vessel = gGeoManager->MakeTube("TargetVesselBack", inc718, 0,
128 vessel_diameter / 2. + vessel_thickness,
129 vessel_thickness / 2.);
130 vessel->SetLineColor(28);
131 tTarget->AddNode(
132 vessel, 1,
133 new TGeoTranslation(
134 0, 0, -1. * vessel_shift + vessel_length + vessel_thickness / 2.));
135 // He inside
136 vessel = gGeoManager->MakeTube("HeVolume", cooler, 0, vessel_diameter / 2.,
137 vessel_length / 2.);
138 vessel->SetLineColor(7);
139
140 // Steel enclosure around target inside He volume
141 // Inner radius must be larger than target radius (fDiameter/2) to avoid
142 // overlaps
143 double enclosure_clearance =
144 0.1 * mm; // Clearance between target and enclosure
145 double enclosure_inner_radius = fDiameter / 2. + enclosure_clearance;
146 double enclosure_thickness = 66.9 * mm; // Updated to match BDF model
147 double enclosure_outer_radius = enclosure_inner_radius + enclosure_thickness;
148 double enclosure_cutout_width_x = 160 * mm; // Width in x direction
149 double enclosure_cutout_width_y = 280 * mm; // Width in y direction
150 double enclosure_length = fTargetLength;
151
152 [[maybe_unused]] auto enclosure_outer_tube =
153 new TGeoTube("enclosure_outer_tube", enclosure_inner_radius,
154 enclosure_outer_radius, enclosure_length / 2.);
155 [[maybe_unused]] auto enclosure_cutout_box =
156 new TGeoBBox("enclosure_cutout_box", enclosure_cutout_width_x / 2.,
157 enclosure_cutout_width_y / 2., enclosure_length / 2.);
158 auto enclosure_shape = new TGeoCompositeShape(
159 "enclosure_shape", "enclosure_outer_tube - enclosure_cutout_box");
160 auto enclosure =
161 new TGeoVolume("target_enclosure", enclosure_shape, steel316L);
162 enclosure->SetLineColor(46); // Reddish color for steel
163 vessel->AddNode(
164 enclosure, 1,
165 new TGeoTranslation(
166 0, 0, -vessel_length / 2. + vessel_shift + enclosure_length / 2.));
167
168 // now place target inside He volume (and inside steel enclosure)
169 // Using nested volumes: tantalum cladding contains target core
170 double cladding_width = 1.5 * mm;
171
172 Double_t zPos = 0.;
173 unsigned slots = fnS;
174 if (slots > 0) slots = slots - 1;
175
176 TGeoVolume* claddedTarget;
177 TGeoVolume* targetCore;
178 // Double_t zPos = fTargetZ - fTargetLength/2.;
179 for (unsigned i = 0; i < fnS; i++) { // loop on layers
180 TString nmi_cladded = "CladdedTarget_";
181 nmi_cladded += i + 1;
182 TString nmi_core = "TargetCore_";
183 nmi_core += i + 1;
184
185 TGeoMedium* material = nullptr;
186 if (fM.at(i) == "molybdenum") {
187 material = mo;
188 } else if (fM.at(i) == "tungsten") {
189 material = tungsten;
190 }
191
192 // Create outer cladded volume (tantalum, full dimensions)
193 claddedTarget = gGeoManager->MakeTube(nmi_cladded, tantalum, 0.,
194 fDiameter / 2., fL.at(i) / 2.);
195 claddedTarget->SetLineColor(8); // Green for tantalum
196
197 // Create inner target core (W or Mo, reduced dimensions)
198 // Positioned at centre (z=0) of cladded volume - tantalum fills the gaps
199 // automatically
200 targetCore = gGeoManager->MakeTube(nmi_core, material, 0.,
201 fDiameter / 2. - cladding_width,
202 (fL.at(i) - 2 * cladding_width) / 2.);
203 if (fM.at(i) == "molybdenum") {
204 targetCore->SetLineColor(28);
205 } else {
206 targetCore->SetLineColor(38);
207 }; // silver/blue
208
209 // Nest core inside cladding (at centre, z=0 in cladded volume's coordinate
210 // system)
211 claddedTarget->AddNode(targetCore, 1, new TGeoTranslation(0, 0, 0));
212
213 // Place complete cladded target in He volume
214 vessel->AddNode(
215 claddedTarget, 1,
216 new TGeoTranslation(
217 0, 0, -vessel_length / 2. + vessel_shift + zPos + fL.at(i) / 2.));
218
219 if (i < slots) {
220 // slits will already be filled with He, no need to define volume
221 zPos += fL.at(i) + fG.at(i);
222 } else {
223 zPos += fL.at(i);
224 }
225 } // loop on layers
226
227 // now add the He+target to the target area.
228 tTarget->AddNode(
229 vessel, 1,
230 new TGeoTranslation(0, 0, -1. * vessel_shift + vessel_length / 2.));
231
232 // Proximity shielding
233
234 double start_of_target = fTargetZ - fTargetLength / 2.;
235 fTargetLength =
236 1586.4 * mm; // Make shielding independent of actual target length
237 double shielding_width = 1600 * mm;
238 double shielding_length = 3000 * mm;
239 double proximity_shielding_height = 1126 * mm;
240 double proximity_shielding_thickness = 250 * mm;
241 double proximity_shielding_thickness_front = 550 * mm;
242 double proximity_shielding_hole_diameter = 200 * mm;
243 double proximity_shielding_hole_height = 735 * mm;
244 double proximity_shielding_distance_after_target = 96.1 * mm;
245 double top_shielding_height = 600 * mm;
246 double bottom_shielding_height = 545 * mm;
247 double shielding_position = start_of_target + fTargetLength +
248 proximity_shielding_distance_after_target +
249 proximity_shielding_thickness -
250 shielding_length / 2;
251 double target_box_shift = 14.45 * cm;
252 [[maybe_unused]] auto proximity_shielding_envelope =
253 new TGeoBBox("proximity_shielding_envelope", shielding_width / 2,
254 proximity_shielding_height / 2, shielding_length / 2);
255 [[maybe_unused]] auto proximity_shielding_inner = new TGeoBBox(
256 "proximity_shielding_inner",
257 shielding_width / 2 - proximity_shielding_thickness,
258 proximity_shielding_height / 2,
259 (shielding_length -
260 (proximity_shielding_thickness_front + proximity_shielding_thickness)) /
261 2);
262 [[maybe_unused]] auto proximity_shielding_hole = new TGeoTube(
263 "proximity_shielding_hole", 0, proximity_shielding_hole_diameter / 2,
264 proximity_shielding_thickness_front / 2);
265 auto proximity_shielding_inner_shift = new TGeoTranslation(
266 "proximity_shielding_inner_shift", 0, 0,
267 (proximity_shielding_thickness_front - proximity_shielding_thickness) /
268 2);
269 proximity_shielding_inner_shift->RegisterYourself();
270 auto proximity_shielding_hole_shift = new TGeoTranslation(
271 "proximity_shielding_hole_shift", 0,
272 proximity_shielding_hole_height - proximity_shielding_height / 2,
273 -shielding_length / 2 + proximity_shielding_thickness_front / 2);
274 proximity_shielding_hole_shift->RegisterYourself();
275 auto proximity_shielding_shape = new TGeoCompositeShape(
276 "proximity_shielding_shape",
277 "proximity_shielding_envelope"
278 "- proximity_shielding_inner:proximity_shielding_inner_shift"
279 "- proximity_shielding_hole:proximity_shielding_hole_shift");
280 auto proximity_shielding =
281 new TGeoVolume("proximity_shielding", proximity_shielding_shape, copper);
282 auto proximity_shielding_centre = new TGeoTranslation(
283 0.,
284 -(proximity_shielding_hole_height - proximity_shielding_height / 2) +
285 target_box_shift,
286 0);
287
288 auto target_vacuum_box =
289 gGeoManager->MakeBox("target_vacuum_box", vacuum, shielding_width / 2,
290 (proximity_shielding_height + top_shielding_height +
291 bottom_shielding_height) /
292 2,
293 shielding_length / 2);
294 auto vacuum_vessel_centre = new TGeoTranslation(
295 0.,
296 // -(proximity_shielding_hole_height - proximity_shielding_height / 2)
297 // + (bottom_shielding_height - top_shielding_height) / 2,
298 -target_box_shift, shielding_position);
299 top->AddNode(target_vacuum_box, 1, vacuum_vessel_centre);
300
301 target_vacuum_box->AddNode(proximity_shielding, 1,
302 proximity_shielding_centre);
303
304 // Top/bottom copper shielding
305 auto top_shielding =
306 gGeoManager->MakeBox("top_shielding", copper, shielding_width / 2,
307 top_shielding_height / 2, shielding_length / 2);
308 target_vacuum_box->AddNode(
309 top_shielding, 1,
310 new TGeoTranslation(
311 0.,
312 top_shielding_height / 2 + proximity_shielding_height -
313 proximity_shielding_hole_height + target_box_shift,
314 0));
315 auto bottom_shielding =
316 gGeoManager->MakeBox("bottom_shielding", copper, shielding_width / 2,
317 bottom_shielding_height / 2, shielding_length / 2);
318 target_vacuum_box->AddNode(
319 bottom_shielding, 1,
320 new TGeoTranslation(0.,
321 -bottom_shielding_height / 2 -
322 proximity_shielding_hole_height +
323 target_box_shift,
324 0));
325 double pedestal_length = 2170 * mm;
326 double pedestal_width = 1070 * mm;
327 double pedestal_height = 150 * mm;
328 auto shielding_pedestal =
329 gGeoManager->MakeBox("shielding_pedestal", iron, pedestal_width / 2,
330 pedestal_height / 2, pedestal_length / 2);
331 target_vacuum_box->AddNode(
332 shielding_pedestal, 1,
333 new TGeoTranslation(
334 0.,
335 pedestal_height / 2 - proximity_shielding_hole_height +
336 target_box_shift,
337 -shielding_length / 2 + 565 * mm + pedestal_length / 2));
338
339 target_vacuum_box->AddNode(
340 tTarget, 1,
341 new TGeoTranslation(0, target_box_shift,
342 start_of_target - shielding_position));
343}
