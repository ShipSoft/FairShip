105Int_t TargetTracker::InitMedium(const char* name)
106{
107 static FairGeoLoader* geoLoad = FairGeoLoader::Instance();
108 static FairGeoInterface* geoFace = geoLoad->getGeoInterface();
109 static FairGeoMedia* media = geoFace->getMedia();
110 static FairGeoBuilder* geoBuild = geoLoad->getGeoBuilder();
111
112 FairGeoMedium* ShipMedium = media->getMedium(name);
113
114 if (!ShipMedium) {
115 Fatal("InitMedium", "Material %s not defined in media file.", name);
116 return -1111;
117 }
118 TGeoMedium* medium = gGeoManager->GetMedium(name);
119 if (medium != NULL)
120 return ShipMedium->getMediumIndex();
121 return geoBuild->createMedium(ShipMedium);
122}

172void TargetTracker::ConstructGeometry()
173{
174
175 InitMedium("vacuum");
176 TGeoMedium* vacuum = gGeoManager->GetMedium("vacuum");
177
178 InitMedium("CarbonComposite");
179 TGeoMedium* CarbonComposite = gGeoManager->GetMedium("CarbonComposite");
180
181 InitMedium("SciFiMat");
182 TGeoMedium* SciFiMat = gGeoManager->GetMedium("SciFiMat");
183
184 InitMedium("Airex");
185 TGeoMedium* Airex = gGeoManager->GetMedium("Airex");
186
187 // Target Tracker
188 TGeoVolume* volTarget = gGeoManager->GetVolume("volTarget");
189
190 TGeoBBox* TT_box = new TGeoBBox("TT_box", TTrackerX / 2, TTrackerY / 2, TTrackerZ / 2);
191 TGeoVolume* TT_volume = new TGeoVolume("TT", TT_box, vacuum);
192 TT_volume->SetLineColor(kBlue - 1);
193 // TT_volume->SetTransparency(1);
194 TT_volume->SetVisibility(1);
195 TT_volume->SetVisDaughters(1);
196
197 // Support Carbon Composite
198 TGeoBBox* TT_support_box = new TGeoBBox("TT_support_box", TTrackerX / 2, TTrackerY / 2, support_z / 2);
199 TGeoVolume* TT_support_volume = new TGeoVolume("TT_support", TT_support_box, CarbonComposite);
200 TT_support_volume->SetLineColor(kGray - 2);
201 TT_support_volume->SetVisibility(1);
202
203 // Honeycomb Airex (or Nomex)
204 TGeoBBox* TT_honeycomb_box = new TGeoBBox("TT_honeycomb_box", TTrackerX / 2, TTrackerY / 2, honeycomb_z / 2);
205 TGeoVolume* TT_honeycomb_volume = new TGeoVolume("TT_honeycomb", TT_honeycomb_box, Airex);
206 TT_honeycomb_volume->SetLineColor(kYellow);
207 TT_honeycomb_volume->SetVisibility(1);
208
209 // SciFi planes
210 TGeoBBox* TT_scifi_plane_hor_box =
211 new TGeoBBox("TT_scifi_plane_hor_box", TTrackerX / 2, TTrackerY / 2, scifimat_z / 2);
212 TGeoVolume* TT_scifi_plane_hor_volume = new TGeoVolume("TT_scifi_plane_hor", TT_scifi_plane_hor_box, SciFiMat);
213 TT_scifi_plane_hor_volume->SetVisibility(1);
214
215 TGeoBBox* TT_scifi_plane_vert_box =
216 new TGeoBBox("TT_scifi_plane_vert_box", TTrackerX / 2, TTrackerY / 2, scifimat_z / 2);
217 TGeoVolume* TT_scifi_plane_vert_volume = new TGeoVolume("TT_scifi_plane_vert", TT_scifi_plane_vert_box, SciFiMat);
218 TT_scifi_plane_vert_volume->SetVisibility(1);
219
220 // SciFi mats for X and Y
221 TGeoBBox* TT_scifimat_hor_box =
222 new TGeoBBox("TT_scifimat_hor_box", scifimat_hor / 2, scifimat_width / 2, scifimat_z / 2);
223 TGeoVolume* TT_scifimat_hor_volume = new TGeoVolume("TT_scifimat_hor", TT_scifimat_hor_box, SciFiMat);
224 TT_scifimat_hor_volume->SetLineColor(kCyan - 9);
225
226 TGeoBBox* TT_scifimat_vert_box =
227 new TGeoBBox("TT_scifimat_vert_box", scifimat_width / 2, scifimat_vert / 2, scifimat_z / 2);
228 TGeoVolume* TT_scifimat_vert_volume = new TGeoVolume("TT_scifimat_vert", TT_scifimat_vert_box, SciFiMat);
229 TT_scifimat_vert_volume->SetLineColor(kGreen - 7);
230
231 // Add SciFi mat as sensitive unit
232 AddSensitiveVolume(TT_scifimat_hor_volume);
233 AddSensitiveVolume(TT_scifimat_vert_volume);
234
235 // Creating physical volumes and multiply
236 for (int i = 0; i < n_hor_planes; i++) {
237 TT_scifi_plane_hor_volume->AddNode(
238 TT_scifimat_hor_volume, i + 1, new TGeoTranslation(0, (-(n_hor_planes - 1) / 2.0 + i) * scifimat_width, 0));
239 }
240 for (int i = 0; i < n_vert_planes; i++) {
241 TT_scifi_plane_vert_volume->AddNode(
242 TT_scifimat_vert_volume,
243 100 + i + 1,
244 new TGeoTranslation((-(n_vert_planes - 1) / 2.0 + i) * scifimat_width, 0, 0));
245 }
246
247 TT_volume->AddNode(TT_support_volume, 0, new TGeoTranslation(0, 0, -TTrackerZ / 2 + support_z / 2));
248 TT_volume->AddNode(
249 TT_scifi_plane_hor_volume, 0, new TGeoTranslation(0, 0, -TTrackerZ / 2 + support_z + scifimat_z / 2));
250 TT_volume->AddNode(TT_scifi_plane_vert_volume,
251 0,
252 new TGeoTranslation(0, 0, -TTrackerZ / 2 + support_z + scifimat_z + scifimat_z / 2));
253 TT_volume->AddNode(TT_honeycomb_volume,
254 0,
255 new TGeoTranslation(0, 0, -TTrackerZ / 2 + support_z + 2 * scifimat_z + honeycomb_z / 2));
256 TT_volume->AddNode(
257 TT_support_volume,
258 1,
259 new TGeoTranslation(0, 0, -TTrackerZ / 2 + support_z + 2 * scifimat_z + honeycomb_z + support_z / 2));
260
261 Double_t first_tt_position = -ZDimension / 2 + TTrackerZ / 2;
262
263 // fNTT - number of TT walls
264 for (int l = 0; l < fNTT; ++l) {
265 volTarget->AddNode(
266 TT_volume, 1000 * (l + 1), new TGeoTranslation(0, 0, first_tt_position + l * (TTrackerZ + CellWidth)));
267 }
268}

270Bool_t TargetTracker::ProcessHits(FairVolume* vol)
271{
273 // Set parameters at entrance of volume. Reset ELoss.
274 if (gMC->IsTrackEntering()) {
275 fELoss = 0.;
276 fTime = gMC->TrackTime() * 1.0e09;
277 fLength = gMC->TrackLength();
278 gMC->TrackPosition(fPos);
279 gMC->TrackMomentum(fMom);
280 }
281 // Sum energy loss for all steps in the active volume
282 fELoss += gMC->Edep();
283
284 // Create muonPoint at exit of active volume
285 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
286 if (fELoss == 0.) {
287 return kFALSE;
288 }
289 TParticle* p = gMC->GetStack()->GetCurrentTrack();
290 Int_t pdgCode = p->GetPdgCode();
291 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
292
293 gMC->CurrentVolID(fVolumeID);
294 Int_t detID = fVolumeID;
295 Int_t TTstationID;
296 gMC->CurrentVolOffID(2, TTstationID);
297 fVolumeID = TTstationID + detID;
298
299 TLorentzVector Pos;
300 gMC->TrackPosition(Pos);
301 Double_t xmean = (fPos.X() + Pos.X()) / 2.;
302 Double_t ymean = (fPos.Y() + Pos.Y()) / 2.;
303 Double_t zmean = (fPos.Z() + Pos.Z()) / 2.;
304
305 AddHit(fTrackID,
306 fVolumeID,
307 TVector3(xmean, ymean, zmean),
308 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
309 fTime,
310 fLength,
311 fELoss,
312 pdgCode);
313
314 // Increment number of muon det points in TParticle
315 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
316 stack->AddPoint(ktauTT);
317 }
318
319 return kTRUE;
320}

322void TargetTracker::DecodeTTID(Int_t detID, Int_t& NTT, int& nplane, Bool_t& ishor)
323{
324 NTT = detID / 1000;
325 int idir = (detID - NTT * 1000) / 100;
326
327 if (idir == 1)
328 ishor = kFALSE;
329 else if (idir == 0)
330 ishor = kTRUE;
331
332 nplane = (detID - NTT * 1000 - idir * 100);
333}
