26def muonUpdateWeight(sTree, diMuboost, xSecboost, noCharm=True):
27 nMu = 0
28 for v in sTree.vetoPoint:
29 mu = v.GetTrackID()
30 t = sTree.MCTrack[mu]
31 if abs(t.GetPdgCode()) != 13:
32 continue
33 moID = abs(sTree.MCTrack[t.GetMotherId()].GetPdgCode())
34 if moID in charmExtern and noCharm:
35 continue # take heavy flavours from separate production
36 nMu += 1
37 pName = t.GetProcName().Data()
38 t.SetWeight(weight)
39 if not pName.find("Hadronic inelastic") < 0:
40 t.MultiplyWeight(1.0 / diMuboost)
41 elif not pName.find("Lepton pair") < 0:
42 t.MultiplyWeight(1.0 / xSecboost)
43 elif not pName.find("Positron annihilation") < 0:
44 t.MultiplyWeight(1.0 / xSecboost)
45 elif not pName.find("Primary particle") < 0 or not pName.find("Decay") < 0:
46 if moID in muSources.values():
47 t.MultiplyWeight(1.0 / diMuboost)
48 return nMu

51def PoT(f):
52 nTot = 0
53 # POT = 1000000000 with ecut=10.0 diMu100.0 X100.0
54 diMuboost = 0.0
55 xSecboost = 0.0
56 ncycles = 0
57 for k in f.GetListOfKeys():
58 if k.GetName() == "FileHeader":
59 txt = k.GetTitle()
60 tmp = txt.split("=")[1]
61 tmp2 = tmp.split("with")[0]
62 if tmp2.find("E") < 0:
63 nTot += int(tmp2)
64 else:
65 nTot += float(tmp2)
66 ncycles += 1
67 i = txt.find("diMu")
68 if i < 0:
69 diMuboost += 1.0
70 else:
71 diMuboost += float(txt[i + 4 :].split(" ")[0])
72 i = txt.find("X")
73 if i < 0:
74 xSecboost += 1.0
75 else:
76 xSecboost += float(txt[i + 1 :].split(" ")[0])
77 diMuboost = diMuboost / float(ncycles)
78 xSecboost = xSecboost / float(ncycles)
79 print(
80 "POT = ",
81 nTot,
82 " number of events:",
83 f.Get("cbmsim").GetEntries(),
84 " diMuboost=",
85 diMuboost,
86 " XsecBoost=",
87 xSecboost,
88 )
89 return nTot, diMuboost, xSecboost

102def processFile(fin, noCharm=True):
103 f = ROOT.TFile.Open(os.environ["EOSSHIP"] + path + fin)
104 nPot, diMuboost, xSecboost = PoT(f)
105 sTree = f.Get("cbmsim")
106 outFile = fin.replace(".root", "_mu.root")
107 fmu = ROOT.TFile(outFile, "recreate")
108 newTree = sTree.CloneTree(0)
109 for n in range(sTree.GetEntries()):
110 sTree.GetEntry(n)
111 nMu = muonUpdateWeight(sTree, diMuboost, xSecboost, noCharm)
112 if nMu > 0:
113 newTree.Fill()
114 ff = f.FileHeader.Clone("Extracted Muon Background File")
115 txt = ff.GetTitle()
116 tmp = txt.split("=")[1]
117 newTxt = txt.replace(tmp.split("with")[0].replace(" ", ""), str(nPot))
118 ff.SetTitle(newTxt)
119 fmu.cd()
120 ff.Write("FileHeader", ROOT.TObject.kSingleKey)
121 newTree.AutoSave()
122 fmu.Close()
123 return 0
