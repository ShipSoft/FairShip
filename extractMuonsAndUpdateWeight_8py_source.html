23def muonUpdateWeight(sTree,diMuboost,xSecboost,noCharm=True):
24 nMu=0
25 for v in sTree.vetoPoint:
26 mu = v.GetTrackID()
27 t = sTree.MCTrack[mu]
28 if abs(t.GetPdgCode())!=13: continue
29 moID = abs(sTree.MCTrack[t.GetMotherId()].GetPdgCode())
30 if moID in charmExtern and noCharm: continue # take heavy flavours from separate production
31 nMu+=1
32 pName = t.GetProcName().Data()
33 t.SetWeight(weight)
34 if not pName.find('Hadronic inelastic')<0:
35 t.MultiplyWeight(1./diMuboost)
36 elif not pName.find('Lepton pair')<0:
37 t.MultiplyWeight(1./xSecboost)
38 elif not pName.find('Positron annihilation')<0:
39 t.MultiplyWeight(1./xSecboost)
40 elif not pName.find('Primary particle')<0 or not pName.find('Decay')<0:
41 if moID in muSources.values():
42 t.MultiplyWeight(1./diMuboost)
43 return nMu

45def PoT(f):
46 nTot = 0
47 # POT = 1000000000 with ecut=10.0 diMu100.0 X100.0
48 diMuboost = 0.
49 xSecboost = 0.
50 ncycles = 0
51 for k in f.GetListOfKeys():
52 if k.GetName() == 'FileHeader':
53 txt = k.GetTitle()
54 tmp = txt.split('=')[1]
55 tmp2 = tmp.split('with')[0]
56 if tmp2.find('E')<0: nTot += int(tmp2)
57 else: nTot += float(tmp2)
58 ncycles+=1
59 i = txt.find('diMu')
60 if i < 0:
61 diMuboost+=1.
62 else:
63 diMuboost+=float(txt[i+4:].split(' ')[0])
64 i = txt.find('X')
65 if i < 0:
66 xSecboost+=1.
67 else:
68 xSecboost+=float(txt[i+1:].split(' ')[0])
69 diMuboost=diMuboost/float(ncycles)
70 xSecboost=xSecboost/float(ncycles)
71 print("POT = " ,nTot, " number of events:", f.Get("cbmsim").GetEntries(), ' diMuboost=', diMuboost, ' XsecBoost=', xSecboost)
72 return nTot,diMuboost,xSecboost

83def processFile(fin,noCharm=True):
84 f = ROOT.TFile.Open(os.environ['EOSSHIP']+path+fin)
85 nPot,diMuboost,xSecboost = PoT(f)
86 sTree = f.Get("cbmsim")
87 outFile = fin.replace(".root","_mu.root")
88 fmu = ROOT.TFile(outFile,"recreate")
89 newTree = sTree.CloneTree(0)
90 for n in range(sTree.GetEntries()):
91 sTree.GetEntry(n)
92 nMu = muonUpdateWeight(sTree,diMuboost,xSecboost,noCharm)
93 if nMu>0:
94 rc = newTree.Fill()
95 ff = f.FileHeader.Clone('Extracted Muon Background File')
96 txt = ff.GetTitle()
97 tmp = txt.split('=')[1]
98 newTxt = txt.replace(tmp.split('with')[0].replace(' ',''),str(nPot))
99 ff.SetTitle(newTxt)
100 fmu.cd()
101 ff.Write("FileHeader", ROOT.TObject.kSingleKey)
102 newTree.AutoSave()
103 fmu.Close()
104 return 0
