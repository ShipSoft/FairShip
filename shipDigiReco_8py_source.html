346 def findVetoHitOnTrack(self,track):
347 distMin = 99999.
348 hitID = -1
349 xx = track.getFittedState()
350 rep = ROOT.genfit.RKTrackRep(xx.getPDG())
351 state = ROOT.genfit.StateOnPlane(rep)
352 rep.setPosMom(state,xx.getPos(),xx.getMom())
353 for i,vetoHit in enumerate(self.digiSBT.det):
354 vetoHitPos = vetoHit.GetXYZ()
355 try:
356 rep.extrapolateToPoint(state,vetoHitPos,False)
357 except:
358 error = "shipDigiReco::findVetoHitOnTrack extrapolation did not worked"
359 ut.reportError(error)
360 if global_variables.debug:
361 print(error)
362 continue
363 dist = (rep.getPos(state) - vetoHitPos).Mag()
364 if dist < distMin:
365 distMin = dist
366 hitID = i
367 return ROOT.vetoHitOnTrack(hitID, distMin)

376 def fracMCsame(self, trackids):
377 track={}
378 nh=len(trackids)
379 for tid in trackids:
380 if tid in track:
381 track[tid] += 1
382 else:
383 track[tid] = 1
384 if track != {}:
385 tmax = max(track, key=track.get)
386 else:
387 track = {-999: 0}
388 tmax = -999
389 frac=0.0
390 if nh > 0:
391 frac = float(track[tmax]) / float(nh)
392 return frac, tmax
