347 def _compute_seed_state(self, atrack, meas, trackParams):
348 """Compute seed position and momentum for the track fitter.
349
350 When PatRec track parameters (k_y, b_y) are available, use them
351 to place the seed at the first hit's Z with the correct Y position
352 and momentum direction. Otherwise fall back to the default seed
353 at the decay vessel centre.
354 """
355 params = trackParams.get(atrack)
356 if params and params.get("k_y12") is not None:
357 # Use station 1-2 parameters to seed near the first measurement
358 k_y = params["k_y12"]
359 b_y = params["b_y12"]
360 # Seed Z at the first measurement's Z coordinate
361 z_seed = meas[0][2] # z is the 3rd element of the TVectorD
362 y_seed = k_y * z_seed + b_y
363 posM = ROOT.TVector3(0, y_seed, z_seed)
364 # Use slope as py/pz ratio; assume 3 GeV total momentum
365 p_total = 3.0 * u.GeV
366 pz = p_total / ROOT.TMath.Sqrt(1.0 + k_y * k_y)
367 py = k_y * pz
368 momM = ROOT.TVector3(0, py, pz)
369 logger.debug(
370 "seed from PatRec: z=%.1f y=%.1f k_y=%.4f p=(0, %.2f, %.2f)",
371 z_seed,
372 y_seed,
373 k_y,
374 py,
375 pz,
376 )
377 else:
378 posM = ROOT.TVector3(0, 0, 5812.0) # decay vessel centre
379 momM = ROOT.TVector3(0, 0, 3.0 * u.GeV)
380 return posM, momM
