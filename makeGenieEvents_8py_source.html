4"""Generate GENIE neutrino interaction events from flux histograms.
5
6This “launcher” wraps `genie_utils` (gmkspl/gevgen/gntpc helpers) and adds:
7- Argument parsing with sensible defaults and validation
8- Optional *nudet* mode (disables charm/tau decays via GXMLPATH)
9- Automatic neutrino/antineutrino scaling based on flux hist sums
10- Structured logging and robust error reporting
11
12Typical use
13-----------
14$ python $FAIRSHIP/macro/makeGenieEvents sim \
15 --seed 65539 \
16 --output ./work \
17 --filedir /eos/experiment/ship/data/Mbias/background-prod-2018 \
18 --target iron \
19 --nevents 1000 \
20 --particles 16 -16 \
21 --emin 0.5 --emax 350 \
22 --xsec-file gxspl-FNALsmall.xml \
23 --flux-file pythia8_Geant4_1.0_withCharm_nu.root \
24 --event-generator-list CC \
25 --nudet
26
27Notes
28-----
29- This tool *does not* modify your parent shell environment.
30- `--nudet` sets `GXMLPATH` for the child process only (you can override path
31 with `--gxmlpath`).
32
33"""

73def extract_nu_over_nubar(
74 neutrino_flux: Path, particles: Sequence[int]
75) -> dict[int, float]:
76 """Compute ν/ν̄ = sum(nu)/sum(nubar) from 1D flux histograms."""
77 f = ROOT.TFile(str(neutrino_flux), "READ")
78 if not f or f.IsZombie():
79 raise FileNotFoundError(f"Cannot open flux file: {neutrino_flux}")
80 try:
81 ratios: dict[int, float] = {}
82 for pdg in particles:
83 fam = abs(int(pdg))
84 h_nu = f.Get(get_1D_flux_name(fam))
85 h_nubar = f.Get(get_1D_flux_name(-fam))
86 if not h_nu or not h_nubar:
87 raise FileNotFoundError(
88 f"Missing hists {get_1D_flux_name(fam)} / "
89 f"{get_1D_flux_name(-fam)} in {neutrino_flux}"
90 )
91 s_nu = float(h_nu.GetSumOfWeights())
92 s_nubar = float(h_nubar.GetSumOfWeights())
93 if s_nubar <= 0.0:
94 raise FileNotFoundError(
95 f"ν̄ sum of weights is zero for family {fam} in {neutrino_flux}"
96 )
97 ratios[fam] = s_nu / s_nubar
98 log_output = (
99 f"ν/ν̄ ratio for |PDG|={fam}: "
100 f"{ratios[fam]:.4f} (nu={s_nu}, nubar={s_nubar})"
101 )
102 logging.info(log_output)
103 return ratios
104 finally:
105 f.Close()

149def make_events(
150 *,
151 run: int,
152 nevents: int,
153 particles: Sequence[int],
154 targetcode: str,
155 process: str | None,
156 emin: float,
157 emax: float,
158 neutrino_flux: Path,
159 splines: Path,
160 seed: int,
161 env_vars: Mapping[str, str | None] | None,
162 work_dir: Path,
163 nu_over_nubar: Mapping[int, float],
164) -> None:
165 """Generate events for PDGs, convert to GST, and attach 2D flux."""
166 _ensure_dir(work_dir)
167 pdg_db = ROOT.TDatabasePDG()
168
169 for pdg in particles:
170 fam = abs(pdg)
171 part = pdg_db.GetParticle(int(pdg))
172 pdg_name = part.GetName() if part else str(pdg)
173
174 # anti-neutrinos scaled by flux ratio
175 N = int(nevents) if pdg > 0 else int(nevents / nu_over_nubar[fam])
176
177 out_dir = work_dir / f"genie-{pdg_name}_{N}_events"
178 _ensure_dir(out_dir)
179
180 nudet_suffix = "_nudet" if env_vars and env_vars.get("GXMLPATH") else ""
181 filename = (
182 f"run_{run}_{pdg_name}_{N}_events_{targetcode}_{emin}_{emax}_GeV_"
183 f"{process or 'ALL'}{nudet_suffix}.ghep.root"
184 )
185 ghep_path = out_dir / filename
186 gst_path = out_dir / f"genie-{filename}"
187
188 logging.info(
189 f"Generating {N} events for PDG {pdg_name} (run={run}) -> {ghep_path}"
190 )
191
192 generate_genie_events(
193 nevents=N,
194 nupdg=int(pdg),
195 emin=emin,
196 emax=emax,
197 targetcode=targetcode,
198 inputflux=str(neutrino_flux),
199 spline=str(splines),
200 outputfile=str(ghep_path),
201 process=process,
202 seed=int(seed),
203 irun=int(run),
204 env_vars=env_vars,
205 )
206 make_ntuples(str(ghep_path), str(gst_path), env_vars=env_vars)
207 add_hists(str(neutrino_flux), str(gst_path), int(pdg))
208
209 logging.info(f"Done PDG {pdg_name}: {ghep_path} -> {gst_path}")

320def main(argv: Sequence[str] | None = None) -> int:
321 """Launch GENIE event generation or spline creation from CLI args."""
322 # CLI
323 parser = _build_parser()
324 args = parser.parse_args(argv)
325
326 # logging
327 level = logging.WARNING
328 if args.verbose == 1:
329 level = logging.INFO
330 elif args.verbose >= 2:
331 level = logging.DEBUG
332 logging.basicConfig(level=level, format="%(levelname)s %(name)s: %(message)s")
333
334 # ROOT env
335 shipRoot_conf.configure()
336
337 if args.cmd == "spline":
338 make_splines_cli(
339 target=args.target,
340 work_dir=args.work_dir,
341 nknots=args.nknots,
342 emax=args.emax,
343 )
344 return 0
345
346 # sim
347 env_vars = _build_env(nudet=bool(args.nudet), gxmlpath=args.gxmlpath)
348 targetcode = _target_code(args.target)
349
350 splines = (args.splinedir / args.xsec_file).resolve()
351 flux = (args.filedir / args.flux_file).resolve()
352 if not splines.is_file():
353 parser.error(f"Spline file not found: {splines}")
354 if not flux.is_file():
355 parser.error(f"Flux ROOT file not found: {flux}")
356
357 particles = _pdg_list(args.particles)
358 nu_over_nubar = extract_nu_over_nubar(flux, particles)
359
360 logging.info(
361 f"Seed: {args.seed} | "
362 f"Target: {args.target} ({targetcode}) | "
363 f"Process: {args.evtype or 'ALL'} | "
364 f"nudet={bool(args.nudet)}"
365 )
366 make_events(
367 run=int(args.run),
368 nevents=int(args.nevents),
369 particles=particles,
370 targetcode=targetcode,
371 process=args.evtype,
372 emin=float(args.emin),
373 emax=float(args.emax),
374 neutrino_flux=flux,
375 splines=splines,
376 seed=int(args.seed),
377 env_vars=env_vars,
378 work_dir=args.work_dir.resolve(),
379 nu_over_nubar=nu_over_nubar,
380 )
381 logging.info("Event generation completed successfully.")
382 return 0
