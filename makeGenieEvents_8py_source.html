1"""Generate GENIE neutrino interaction events from flux histograms.
2
3This “launcher” wraps `genie_utils` (gmkspl/gevgen/gntpc helpers) and adds:
4- Argument parsing with sensible defaults and validation
5- Optional *nudet* mode (disables charm/tau decays via GXMLPATH)
6- Automatic neutrino/antineutrino scaling based on flux hist sums
7- Structured logging and robust error reporting
8
9Typical use
10-----------
11$ python $FAIRSHIP/macro/makeGenieEvents sim \
12 --seed 65539 \
13 --output ./work \
14 --filedir /eos/experiment/ship/data/Mbias/background-prod-2018 \
15 --target iron \
16 --nevents 1000 \
17 --particles 16 -16 \
18 --emin 0.5 --emax 350 \
19 --xsec-file gxspl-FNALsmall.xml \
20 --flux-file pythia8_Geant4_1.0_withCharm_nu.root \
21 --event-generator-list CC \
22 --nudet
23
24Notes
25-----
26- This tool *does not* modify your parent shell environment.
27- `--nudet` sets `GXMLPATH` for the child process only (you can override path
28 with `--gxmlpath`).
29

70def extract_nu_over_nubar(
71 neutrino_flux: Path, particles: Sequence[int]
72) -> dict[int, float]:
73 """Compute ν/ν̄ = sum(nu)/sum(nubar) from 1D flux histograms."""
74 f = ROOT.TFile(str(neutrino_flux), "READ")
75 if not f or f.IsZombie():
76 raise FileNotFoundError(f"Cannot open flux file: {neutrino_flux}")
77 try:
78 ratios: dict[int, float] = {}
79 for pdg in particles:
80 fam = abs(int(pdg))
81 h_nu = f.Get(get_1D_flux_name(fam))
82 h_nubar = f.Get(get_1D_flux_name(-fam))
83 if not h_nu or not h_nubar:
84 raise FileNotFoundError(
85 f"Missing hists {get_1D_flux_name(fam)} / "
86 f"{get_1D_flux_name(-fam)} in {neutrino_flux}"
87 )
88 s_nu = float(h_nu.GetSumOfWeights())
89 s_nubar = float(h_nubar.GetSumOfWeights())
90 if s_nubar <= 0.0:
91 raise FileNotFoundError(
92 f"ν̄ sum of weights is zero for family {fam} in {neutrino_flux}"
93 )
94 ratios[fam] = s_nu / s_nubar
95 log_output = (
96 f"ν/ν̄ ratio for |PDG|={fam}: "
97 f"{ratios[fam]:.4f} (nu={s_nu}, nubar={s_nubar})"
98 )
99 logging.info(log_output)
100 return ratios
101 finally:
102 f.Close()

146def make_events(
147 *,
148 run: int,
149 nevents: int,
150 particles: Sequence[int],
151 targetcode: str,
152 process: str | None,
153 emin: float,
154 emax: float,
155 neutrino_flux: Path,
156 splines: Path,
157 seed: int,
158 env_vars: Mapping[str, str | None] | None,
159 work_dir: Path,
160 nu_over_nubar: Mapping[int, float],
161) -> None:
162 """Generate events for PDGs, convert to GST, and attach 2D flux."""
163 _ensure_dir(work_dir)
164 pdg_db = ROOT.TDatabasePDG()
165
166 for pdg in particles:
167 fam = abs(pdg)
168 part = pdg_db.GetParticle(int(pdg))
169 pdg_name = part.GetName() if part else str(pdg)
170
171 # anti-neutrinos scaled by flux ratio
172 N = int(nevents) if pdg > 0 else int(nevents / nu_over_nubar[fam])
173
174 out_dir = work_dir / f"genie-{pdg_name}_{N}_events"
175 _ensure_dir(out_dir)
176
177 nudet_suffix = "_nudet" if env_vars and env_vars.get("GXMLPATH") else ""
178 filename = (
179 f"run_{run}_{pdg_name}_{N}_events_{targetcode}_{emin}_{emax}_GeV_"
180 f"{process or 'ALL'}{nudet_suffix}.ghep.root"
181 )
182 ghep_path = out_dir / filename
183 gst_path = out_dir / f"genie-{filename}"
184
185 logging.info(
186 f"Generating {N} events for PDG {pdg_name} (run={run}) -> {ghep_path}"
187 )
188
189 generate_genie_events(
190 nevents=N,
191 nupdg=int(pdg),
192 emin=emin,
193 emax=emax,
194 targetcode=targetcode,
195 inputflux=str(neutrino_flux),
196 spline=str(splines),
197 outputfile=str(ghep_path),
198 process=process,
199 seed=int(seed),
200 irun=int(run),
201 env_vars=env_vars,
202 )
203 make_ntuples(str(ghep_path), str(gst_path), env_vars=env_vars)
204 add_hists(str(neutrino_flux), str(gst_path), int(pdg))
205
206 logging.info(f"Done PDG {pdg_name}: {ghep_path} -> {gst_path}")

317def main(argv: Sequence[str] | None = None) -> int:
318 """Launch GENIE event generation or spline creation from CLI args."""
319 # CLI
320 parser = _build_parser()
321 args = parser.parse_args(argv)
322
323 # logging
324 level = logging.WARNING
325 if args.verbose == 1:
326 level = logging.INFO
327 elif args.verbose >= 2:
328 level = logging.DEBUG
329 logging.basicConfig(level=level, format="%(levelname)s %(name)s: %(message)s")
330
331 # ROOT env
332 shipRoot_conf.configure()
333
334 if args.cmd == "spline":
335 make_splines_cli(
336 target=args.target,
337 work_dir=args.work_dir,
338 nknots=args.nknots,
339 emax=args.emax,
340 )
341 return 0
342
343 # sim
344 env_vars = _build_env(nudet=bool(args.nudet), gxmlpath=args.gxmlpath)
345 targetcode = _target_code(args.target)
346
347 splines = (args.splinedir / args.xsec_file).resolve()
348 flux = (args.filedir / args.flux_file).resolve()
349 if not splines.is_file():
350 parser.error(f"Spline file not found: {splines}")
351 if not flux.is_file():
352 parser.error(f"Flux ROOT file not found: {flux}")
353
354 particles = _pdg_list(args.particles)
355 nu_over_nubar = extract_nu_over_nubar(flux, particles)
356
357 logging.info(
358 f"Seed: {args.seed} | "
359 f"Target: {args.target} ({targetcode}) | "
360 f"Process: {args.evtype or 'ALL'} | "
361 f"nudet={bool(args.nudet)}"
362 )
363 make_events(
364 run=int(args.run),
365 nevents=int(args.nevents),
366 particles=particles,
367 targetcode=targetcode,
368 process=args.evtype,
369 emin=float(args.emin),
370 emax=float(args.emax),
371 neutrino_flux=flux,
372 splines=splines,
373 seed=int(args.seed),
374 env_vars=env_vars,
375 work_dir=args.work_dir.resolve(),
376 nu_over_nubar=nu_over_nubar,
377 )
378 logging.info("Event generation completed successfully.")
379 return 0
