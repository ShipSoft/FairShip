101Int_t UpstreamTagger::InitMedium(const char* name)
102{
103
104 static FairGeoLoader *geoLoad=FairGeoLoader::Instance();
105 static FairGeoInterface *geoFace=geoLoad->getGeoInterface();
106 static FairGeoMedia *media=geoFace->getMedia();
107 static FairGeoBuilder *geoBuild=geoLoad->getGeoBuilder();
108
109 FairGeoMedium *ShipMedium=media->getMedium(name);
110
111 if (!ShipMedium)
112 {
113 Fatal("InitMedium","Material %s not defined in media file.", name);
114 return -1111;
115 }
116 TGeoMedium* medium=gGeoManager->GetMedium(name);
117 if (medium!=NULL)
118 return ShipMedium->getMediumIndex();
119
120 return geoBuild->createMedium(ShipMedium);
121
122 return 0;
123}

127Bool_t UpstreamTagger::ProcessHits(FairVolume* vol)
128{
130 //Set parameters at entrance of volume. Reset ELoss.
131 if ( gMC->IsTrackEntering() ) {
132 fELoss = 0.;
133 fTime = gMC->TrackTime() * 1.0e09;
134 fLength = gMC->TrackLength();
135 gMC->TrackPosition(fPos);
136 gMC->TrackMomentum(fMom);
137 }
138
139 // Sum energy loss for all steps in the active volume
140 fELoss += gMC->Edep();
141
142 // Create vetoPoint at exit of active volume
143 if ( gMC->IsTrackExiting() ||
144 gMC->IsTrackStop() ||
145 gMC->IsTrackDisappeared() ) {
146 if (fELoss == 0. ) { return kFALSE; }
147
148 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
149
150 Int_t uniqueId;
151 gMC->CurrentVolID(uniqueId);
152 if (uniqueId>1000000) //Solid scintillator case
153 {
154 Int_t vcpy;
155 gMC->CurrentVolOffID(1, vcpy);
156 if (vcpy==5) uniqueId+=4; //Copy of half
157 }
158
159 TParticle* p = gMC->GetStack()->GetCurrentTrack();
160 Int_t pdgCode = p->GetPdgCode();
161 TLorentzVector Pos;
162 gMC->TrackPosition(Pos);
163 TLorentzVector Mom;
164 gMC->TrackMomentum(Mom);
165 Double_t xmean = (fPos.X()+Pos.X())/2. ;
166 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
167 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
168
169 AddHit(fTrackID, uniqueId, TVector3(xmean, ymean, zmean),
170 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
171 fELoss,pdgCode,TVector3(Pos.X(), Pos.Y(), Pos.Z()),
172 TVector3(Mom.Px(), Mom.Py(), Mom.Pz()) );
173
174 // Increment number of veto det points in TParticle
175 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
176 stack->AddPoint(kUpstreamTagger);
177 }
178
179 return kTRUE;
180}
