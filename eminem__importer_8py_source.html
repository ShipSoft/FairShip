4"""Convert pickled muon data to ROOT format (TTree or RNTuple).
5
6This script converts pickled muon data to ROOT format, supporting both TTree and RNTuple output formats.
7The input data should be a NumPy array with columns: px, py, pz, x, y, z, id, w.

9Usage:
10 python eminem_importer.py input_file.pkl [options]

12Example:
13 python eminem_importer.py muons_FullMC.pkl --output converted_data.root --format rntuple

86def load_pickled_data(input_file, verbose=False):
87 """
88 Load pickled data from file, handling both compressed and uncompressed files.
89
90 Args:
91 input_file (str): Path to the input pickled file
92 verbose (bool): Enable verbose output
93
94 Returns:
95 numpy.ndarray: Loaded data array
96
97 Raises:
98 FileNotFoundError: If input file doesn't exist
99 Exception: If file cannot be loaded or has wrong format
100 """
101 input_path = Path(input_file)
102
103 if not input_path.exists():
104 raise FileNotFoundError(f"Input file not found: {input_file}")
105
106 if verbose:
107 print(f"Loading data from: {input_file}")
108
109 try:
110 # Try gzip first (files may be gzipped regardless of extension)
111 try:
112 with gzip.open(input_file, "rb") as f:
113 data = pickle.loads(f.read())
114 except gzip.BadGzipFile:
115 with open(input_file, "rb") as f:
116 data = pickle.load(f)
117 except Exception as e:
118 raise RuntimeError(f"Failed to load pickled data: {e}") from e
119
120 if isinstance(data, dict):
121 required = ["px", "py", "pz", "x", "y", "z", "pdg_id", "weight"]
122
123 missing = [k for k in required if k not in data]
124 if missing:
125 raise ValueError(f"Missing keys in dict: {missing}")
126
127 # convert to column stack (ONLY here, once)
128 stacked = np.column_stack(
129 [
130 np.asarray(data["px"]),
131 np.asarray(data["py"]),
132 np.asarray(data["pz"]),
133 np.asarray(data["x"]),
134 np.asarray(data["y"]),
135 np.asarray(data["z"]),
136 np.asarray(data["pdg_id"]),
137 np.asarray(data["weight"]),
138 ]
139 )
140
141 if verbose:
142 print(f"Loaded data shape: {stacked.shape}")
143 print(f"Data type: {stacked.dtype}")
144
145 return stacked
146
147 if verbose:
148 print(f"Loaded data shape: {data.shape}")
149 print(f"Data type: {data.dtype}")
150
151 # Validate data format
152 if len(data.shape) != 2 or data.shape[1] != 8:
153 raise ValueError(f"Expected data with shape (N, 8), got {data.shape}")
154
155 return data

176def parse_custom_config_units(custom_config):
177 """
178 Parse custom configuration and convert string unit representations to shipunit objects.
179
180 Args:
181 custom_config (dict): Custom configuration dictionary
182
183 Returns:
184 dict: Configuration with parsed units
185 """
186 # Common unit mappings for shipunit
187 unit_map = {
188 "m": u.m,
189 "cm": u.cm,
190 "mm": u.mm,
191 "km": u.km,
192 "GeV": u.GeV,
193 "MeV": u.MeV,
194 "keV": u.keV,
195 "eV": u.eV,
196 "s": u.s,
197 "ms": u.ms,
198 "ns": u.ns,
199 "kg": u.kg,
200 "g": u.g,
201 # Add more mappings as needed
202 }
203
204 parsed_config = {}
205 for col_name, col_config in custom_config.items():
206 parsed_col_config = col_config.copy()
207
208 # Parse unit strings to shipunit objects
209 if "unit" in col_config and isinstance(col_config["unit"], str):
210 unit_str = col_config["unit"]
211 if unit_str in unit_map:
212 parsed_col_config["unit"] = unit_map[unit_str]
213 else:
214 print(f"Warning: Unknown unit '{unit_str}' for column '{col_name}', ignoring unit")
215 parsed_col_config.pop("unit", None)
216
217 if "offset_unit" in col_config and isinstance(col_config["offset_unit"], str):
218 offset_unit_str = col_config["offset_unit"]
219 if offset_unit_str in unit_map:
220 parsed_col_config["offset_unit"] = unit_map[offset_unit_str]
221 else:
222 print(f"Warning: Unknown offset unit '{offset_unit_str}' for column '{col_name}', ignoring offset unit")
223 parsed_col_config.pop("offset_unit", None)
224
225 parsed_config[col_name] = parsed_col_config
226
227 return parsed_config

333def apply_column_transformation(data_column, config):
334 """
335 Apply transformations to a data column based on its configuration.
336
337 Args:
338 data_column (numpy.ndarray): Raw data column
339 config (dict): Column configuration
340
341 Returns:
342 numpy.ndarray or shipunit quantity: Transformed data column
343 """
344 result = data_column.copy()
345
346 # Apply scale factor if specified
347 if "scale_factor" in config:
348 result = result * config["scale_factor"]
349
350 # Apply offset (before units if both are present)
351 if "offset" in config:
352 offset = config["offset"]
353 if config.get("offset_unit") is not None:
354 # If we have units, we need to handle the offset carefully
355 if config.get("unit") is not None:
356 # Both result and offset will have units
357 result = result * config["unit"] + offset * config["offset_unit"]
358 else:
359 # Only offset has units
360 result = result + offset * config["offset_unit"]
361 else:
362 # No units for offset
363 result = result + offset
364
365 # Apply units (if not already applied above)
366 if config.get("unit") is not None and "offset" not in config:
367 result = result * config["unit"]
368
369 if "type" in config:
370 result = result.astype(config["type"])
371
372 return result

375def create_root_dataframe(data, verbose=False, custom_config_path=None):
376 """
377 Create ROOT RDataFrame from numpy data using configuration-driven approach.
378
379 Args:
380 data (numpy.ndarray): Input data array with columns [px, py, pz, x, y, z, id, w]
381 verbose (bool): Enable verbose output
382 custom_config_path (str, optional): Path to custom configuration file
383
384 Returns:
385 ROOT.RDataFrame: ROOT dataframe with the converted data
386 """
387 if verbose:
388 print("Creating ROOT RDataFrame...")
389
390 # Get column configuration
391 column_config = get_column_config(custom_config_path)
392
393 # Create dataframe dictionary using configuration
394 df_dict = {}
395
396 for col_name, config in column_config.items():
397 # Extract raw data column
398 raw_data = data[:, config["index"]]
399
400 # Apply transformations
401 transformed_data = apply_column_transformation(raw_data, config)
402
403 # Store in dictionary
404 df_dict[col_name] = transformed_data
405
406 if verbose:
407 print(f" {col_name}: {config.get('description', 'No description')}")
408 if config.get("unit"):
409 print(f" Unit: {config['unit']}")
410 if config.get("offset"):
411 offset_unit = config.get("offset_unit", "")
412 print(f" Offset: {config['offset']} {offset_unit}")
413 if config.get("scale_factor"):
414 print(f" Scale factor: {config['scale_factor']}")
415
416 df = ROOT.RDF.FromNumpy(df_dict)
417
418 if verbose:
419 print(f"Created RDataFrame with {df.Count().GetValue()} entries")
420
421 return df

424def save_root_file(df, output_file, tree_name, output_format, verbose=False):
425 """
426 Save ROOT dataframe to file in specified format.
427
428 Args:
429 df (ROOT.RDataFrame): Input dataframe
430 output_file (str): Output file path
431 tree_name (str): Name of the tree/ntuple
432 output_format (str): Output format ('ttree' or 'rntuple')
433 verbose (bool): Enable verbose output
434 """
435 if verbose:
436 print(f"Saving to {output_file} in {output_format} format...")
437
438 # Set up snapshot options
439 opts = ROOT.RDF.RSnapshotOptions()
440
441 if output_format == "rntuple":
442 opts.fOutputFormat = ROOT.RDF.ESnapshotOutputFormat.kRNTuple
443 if verbose:
444 print("Using RNTuple format")
445 else:
446 # TTree is the default format
447 if verbose:
448 print("Using TTree format")
449
450 # Save the data
451 df.Snapshot(tree_name, output_file, "", opts)
452
453 if verbose:
454 print(f"Successfully saved data to {output_file}")

457def main():
458 """Convert pickled muon data to ROOT format (TTree or RNTuple)."""
