241class MyGeneratorAction(G4VUserPrimaryGeneratorAction):
242 "My Generator Action"
243
244 pos = G4ThreeVector(0 * cm, 0 * cm, -world_r / 2.0)

246 def GeneratePrimaries(self, anEvent):
247 global debug, nevTot, particleGun, fullTungsten
248 t_0 = time.time()
249 npart = 0
250 if particleGun:
251 # to evaluate interaction position
252 ztarget = G4ThreeVector(0 * cm, 0 * cm, -50.0 * m)
253 vertex = G4PrimaryVertex(ztarget, 0.0)
254 G4particle = G4PrimaryParticle(2212)
255 G4particle.Set4Momentum(0, 0, 400 * GeV, 400.0011 * GeV)
256 # store mother ID
257 w = 2212 + 0 * 100000
258 G4particle.SetWeight(w)
259 vertex.SetPrimary(G4particle)
260 npart += 1
261 else:
262 while npart == 0:
263 myPythia.GenerateEvent()
264 nevTot += 1
265 myTimer["pythia"] += time.time() - t_0
266 # pythia interaction happens at 0,0,0
267 # x = rnr.Uniform(-3.,3.)
268 # y = rnr.Uniform(-3.,3.)
269 # leave this smearing for later
270 # create new primaries and set them to the vertex
271 particles = myPythia.GetListOfParticles()
272 if fullTungsten:
273 z = rnr.Exp(10 * cm) - 50.0 * m # tungsten interaction length
274 else:
275 z = rnr.Exp(16 * cm) - 50.0 * m # Mo/H20/W average interaction length
276 ztarget = G4ThreeVector(0 * cm, 0 * cm, z)
277 vertex = G4PrimaryVertex(ztarget, 0.0)
278 v = ROOT.TLorentzVector()
279 for p in particles:
280 if p.GetStatusCode() != 1:
281 continue
282 pid = p.GetPdgCode()
283 mkey = p.GetMother(0) + 1
284 mother = myPythia.GetParticle(mkey)
285 if JpsiMainly and abs(pid) != 13:
286 continue
287 if tauOnly and abs(pid) != 16:
288 continue
289 if JpsiMainly and mother.GetPdgCode() != 443:
290 continue
291 if tauOnly:
292 mmkey = mother.GetMother(0) + 1
293 mmother = myPythia.GetParticle(mmkey)
294 if abs(mother.GetPdgCode()) != 431 and abs(mmother.GetPdgCode()) != 431:
295 myPythia.EventListing()
296 continue
297 qed = pid in qedlist # use cut only for photons, leptons/neutrinos, protons and neutrons
298 p.Momentum(v)
299 Ekin = v.E() - v.M()
300 if Ekin * GeV < ecut and (qed or allPart):
301 continue
302 G4particle = G4PrimaryParticle(pid)
303 G4particle.Set4Momentum(v.Px() * GeV, v.Py() * GeV, v.Pz() * GeV, v.E() * GeV)
304 # store mother ID
305 curPid = p.GetPdgCode() + 10000 # make it positive
306 moPid = mother.GetPdgCode() + 10000
307 w = curPid + moPid * 100000
308 G4particle.SetWeight(w)
309 vertex.SetPrimary(G4particle)
310 npart += 1
311 # if debug: myPythia.EventListing()
312 anEvent.AddPrimaryVertex(vertex)
313 if debug and not particleGun:
314 print("new event at ", ztarget.z / m)
315 myTimer["geant4_conv"] += time.time() - t_0
