7def readHists(h,fname,wanted=[]):
8 if fname[0:4] == "/eos":
9 eospath = gSystem.Getenv("EOSSHIP")+fname
10 f = TFile.Open(eospath)
11 else:
12 f = TFile(fname)
13 for akey in f.GetListOfKeys():
14 name = akey.GetName()
15 try: hname = int(name)
16 except: hname = name
17 if len(wanted)>0:
18 if not hname in wanted: continue
19 obj = akey.ReadObj()
20 cln = obj.Class().GetName()
21 if not cln.find('TCanv')<0:
22 h[hname] = obj.Clone()
23 if cln.find('TH')<0: continue
24 if hname in h:
25 rc = h[hname].Add(obj)
26 if not rc: print("Error when adding histogram ",hname)
27 else:
28 h[hname] = obj.Clone()
29 if h[hname].GetSumw2N()==0 : h[hname].Sumw2()
30 h[hname].SetDirectory(gROOT)
31 if cln == 'TH2D' or cln == 'TH2F':
32 for p in [ '_projx','_projy']:
33 if type(hname) == str: projname = hname+p
34 else: projname = str(hname)+p
35 if p.find('x')>-1: h[projname] = h[hname].ProjectionX()
36 else : h[projname] = h[hname].ProjectionY()
37 h[projname].SetName(name+p)
38 h[projname].SetDirectory(gROOT)
39 return
