12def readHists(h, fname, wanted=[]):
13 if fname[0:4] == "/eos":
14 eospath = gSystem.Getenv("EOSSHIP") + fname
15 f = TFile.Open(eospath)
16 else:
17 f = TFile(fname)
18 for akey in f.GetListOfKeys():
19 name = akey.GetName()
20 try:
21 hname = int(name)
22 except:
23 hname = name
24 if len(wanted) > 0:
25 if hname not in wanted:
26 continue
27 obj = akey.ReadObj()
28 cln = obj.Class().GetName()
29 if not cln.find("TCanv") < 0:
30 h[hname] = obj.Clone()
31 if cln.find("TH") < 0:
32 continue
33 if hname in h:
34 rc = h[hname].Add(obj)
35 if not rc:
36 print("Error when adding histogram ", hname)
37 else:
38 h[hname] = obj.Clone()
39 if h[hname].GetSumw2N() == 0:
40 h[hname].Sumw2()
41 h[hname].SetDirectory(gROOT)
42 if cln == "TH2D" or cln == "TH2F":
43 for p in ["_projx", "_projy"]:
44 if isinstance(hname, str):
45 projname = hname + p
46 else:
47 projname = str(hname) + p
48 if p.find("x") > -1:
49 h[projname] = h[hname].ProjectionX()
50 else:
51 h[projname] = h[hname].ProjectionY()
52 h[projname].SetName(name + p)
53 h[projname].SetDirectory(gROOT)
54 return
