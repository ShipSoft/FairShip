528{
531 gMC->TrackPosition(fPos);
532 gMC->TrackMomentum(fMom);
533 fVolumeID = -1;

539 mass * mass ) - mass;

546 AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
547 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,

549 }
