5"""Scan tracking benchmark over a grid of polar angles and multiplicities.
6
7Calls run_tracking_benchmark.py for each (theta, nTracks) point, collects
8metrics into a combined JSON, and produces summary plots with matplotlib.
9
10Example usage:
11 # 1D theta scan at single-track multiplicity
12 python macro/run_tracking_scan.py -n 200 --scan-mode theta
13
14 # 1D multiplicity scan at theta=0
15 python macro/run_tracking_scan.py -n 200 --scan-mode nTracks
16
17 # Full 2D grid
18 python macro/run_tracking_scan.py -n 500 --scan-mode grid
19
20 # Custom theta values, keep intermediate files
21 python macro/run_tracking_scan.py -n 200 --thetas 0,5,15,30 --keep-files
22"""

84# Build scan grid: each point is (thetaMin, thetaMax, nTracks, label_theta)
85# In theta mode: fixed theta per point (thetaMin == thetaMax)
86# In nTracks mode: uniform theta range, vary multiplicity

301 def _get_values(points: list, metric: str) -> tuple[list, list, list]:
302 """Extract x, y, yerr from sorted scan points, skipping None values."""
303 x, y, yerr = [], [], []
304 for p in points:
305 val = p[1][metric]["value"]
306 if val is None:
307 continue
308 x.append(p[0])
309 y.append(val)
310 yerr.append(p[1][metric].get("uncertainty") or 0)
311 return x, y, yerr
