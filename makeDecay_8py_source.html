15# Reference values for Molybdenum (historical defaults). Cross-section ratios
16# scale heavy-flavour ~ A and mbias ~ A^(0.71).
17heavyflavour_Ascale = 1
18mbias_Ascale = 0.71

20A_REF = 98.0
21CHICC_REF = 1.7e-3 # prob to produce primary ccbar pair/pot on Mo
22CHIBB_REF = 1.6e-7 # prob to produce primary bbbar pair/pot on Mo
23TARGET_A = {"W": 184.0, "Mo": 98.0}

57fname = args.input.replace(".root", "")

60A = args.A if args.A is not None else TARGET_A[args.target_composition]
61scale = (A / A_REF) ** (heavyflavour_Ascale - mbias_Ascale)
62
63if args.chicc is not None:
64 chicc = args.chicc

70chibb = args.chibb if args.chibb is not None else CHIBB_REF * scale
