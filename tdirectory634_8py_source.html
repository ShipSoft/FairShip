6def _TDirectory_getitem(self, key):
7 """Injection of TDirectory.__getitem__ that raises AttributeError on failure.
8
9 Method that is assigned to TDirectory.__getitem__. It relies on Get to
10 obtain the object from the TDirectory and adds on top:
11 - Raising an AttributeError if the object does not exist
12 - Caching the result of a successful get for future re-attempts.
13 Once cached, the same object is retrieved every time.
14 This pythonization is inherited by TDirectoryFile and TFile.
15
16 Example:
17 ```
18 myfile.mydir.mysubdir.myHist.Draw()
19 ```
20 """
21 if not hasattr(self, "_cached_items"):
22 self._cached_items = dict()
23
24 if key in self._cached_items:
25 return self._cached_items[key]
26
27 result = self.Get(key)
28 if not result:
29 raise KeyError(f"{repr(self)} object has no key '{key}'")
30
31 # Caching behavior seems to be more clear to the user; can always override said
32 # behavior (i.e. re-read from file) with an explicit Get() call
33 self._cached_items[key] = result
34 return result
