120TGeoVolume* SiliconTarget::CreateSiliconPlanes(const char* name,
121 Double_t width,
122 Double_t length,
123 Double_t spacing,
124 TGeoMedium* silicon,
125 Int_t layerId)
126{
127 Double_t strip_pitch = 75.5 * um;
128 Int_t nRows = 2;
129 Int_t nCols = 4;
130 Int_t nPlanes = 2;
131
132 TGeoBBox* SensorShape = new TGeoBBox("SensorShape", width / 2, length / 2, 0.3 * mm / 2);
133 TGeoVolume* SensorVolume = new TGeoVolume("SensorVolume", SensorShape, silicon);
134 SensorVolume->SetLineColor(kRed);
135 SensorVolume->SetTransparency(40);
136
137 auto* Strips = SensorVolume->Divide("SLICEX", 1, 1298, -width / 2, strip_pitch);
138 AddSensitiveVolume(Strips);
139
140 TGeoVolumeAssembly* trackingStation = new TGeoVolumeAssembly("TrackingStation");
141 // Each tracking station consists of X and Y planes
142 for (Int_t plane = 0; plane < nPlanes; plane++) {
143 TGeoVolumeAssembly* trackerPlane = new TGeoVolumeAssembly("TrackerPlane");
144 // Each plane consists of 8 modules
145 for (Int_t column = 0; column < nCols; column++) {
146 for (Int_t row = 0; row < nRows; row++) {
147 Int_t sensor_id = (layerId << 5) + (plane << 4) + (column << 2) + (row << 1);
148 // Add 1mm gap between sensors for realistic placement
149 trackerPlane->AddNode(SensorVolume,
150 sensor_id,
151 new TGeoTranslation((-1.5 * width - 2. * mm) + (column * width + column * 1 * mm),
152 (length / 2. + 0.5 * mm) - row * (length + 1. * mm),
153 0));
154 }
155 }
156 if (plane == 0) {
157 trackingStation->AddNode(trackerPlane, plane);
158 } else if (plane == 1) {
159 // Rotate the second plane by 90 degrees and translate such that the planes are evenly spaced.
160 trackingStation->AddNode(
161 trackerPlane,
162 plane,
163 new TGeoCombiTrans(TGeoTranslation(0, 0, spacing), TGeoRotation("y_rot", 0, 0, 90)));
164 }
165 }
166
167 return trackingStation;
168}
169void SiliconTarget::ConstructGeometry()
170{
171 InitMedium("tungstenalloySND");
172 TGeoMedium* tungsten = gGeoManager->GetMedium("tungstenalloySND");
173 InitMedium("air");
174 TGeoMedium* air = gGeoManager->GetMedium("air");
175 InitMedium("silicon");
176 TGeoMedium* Silicon = gGeoManager->GetMedium("silicon");
177
178 Double_t totalLength = fLayers * fTargetSpacing;
179
180 // --- Create an envelope volume for the detector (green, semi-transparent) ---
181 auto envBox = new TGeoBBox("SiliconTarget_env", fTargetWidth / 2., fTargetHeight / 2., totalLength / 2.);
182 auto envVol = new TGeoVolume("SiliconTarget", envBox, air);
183 envVol->SetLineColor(kGreen);
184 envVol->SetTransparency(50);
185
186 auto target = new TGeoBBox("Target", fTargetWidth / 2., fTargetHeight / 2., fTargetThickness / 2.);
187 auto targetVol = new TGeoVolume("TargetVol", target, tungsten);
188 targetVol->SetLineColor(kGray);
189 targetVol->SetTransparency(40);
190
191 for (Int_t i = 0; i < fLayers; i++) {
192 // Compute the center position (z) for the current W layer
193 Double_t zPos = -totalLength / 2 + i * fTargetSpacing;
194
195 // Place the tungsten layer
196 envVol->AddNode(targetVol, i, new TGeoTranslation(0, 0, zPos + fTargetThickness / 2.));
197
198 TGeoVolume* siliconPlanes = CreateSiliconPlanes("TrackerPlane",
199 fSensorWidth,
200 fSensorLength,
201 fTargetSpacing - fTargetThickness - 2. * fModuleOffset,
202 Silicon,
203 i);
204 envVol->AddNode(siliconPlanes, i, new TGeoTranslation(0, 0, zPos + fTargetThickness + fModuleOffset));
205 }
206
207 // Finally, add the envelope to the top volume with the global z offset fZCenter
208 gGeoManager->GetTopVolume()->AddNode(envVol, 1, new TGeoTranslation(0, 0, fZPosition));
209}

211Bool_t SiliconTarget::ProcessHits(FairVolume* vol)
212{
214 // Set parameters at entrance of volume. Reset ELoss.
215 if (gMC->IsTrackEntering()) {
216 fELoss = 0.;
217 fTime = gMC->TrackTime() * 1.0e09;
218 fLength = gMC->TrackLength();
219 gMC->TrackPosition(fPos);
220 gMC->TrackMomentum(fMom);
221 }
222
223 // Sum energy loss for all steps in the active volume
224 fELoss += gMC->Edep();
225
226 // Create SiliconTargetPoint at exit of active volume
227 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
228
229 if (fELoss == 0.) {
230 return kFALSE;
231 }
232
233 TParticle* p = gMC->GetStack()->GetCurrentTrack();
234 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
235 Int_t pdgCode = p->GetPdgCode();
236 TLorentzVector Pos;
237 gMC->TrackPosition(Pos);
238 TLorentzVector Mom;
239 gMC->TrackMomentum(Mom);
240
241 Double_t xmean = (fPos.X() + Pos.X()) / 2.;
242 Double_t ymean = (fPos.Y() + Pos.Y()) / 2.;
243 Double_t zmean = (fPos.Z() + Pos.Z()) / 2.;
244
245 AddHit(fTrackID,
246 fVolumeID,
247 TVector3(xmean, ymean, zmean),
248 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
249 fTime,
250 fLength,
251 fELoss,
252 pdgCode);
253
254 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
255 stack->AddPoint(kSiliconTarget);
256 }
257 return kTRUE;
258}
