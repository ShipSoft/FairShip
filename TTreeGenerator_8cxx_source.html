36Bool_t TTreeGenerator::Init(const char* fileName, int startEvent) {
37 if (startEvent < 0) {
38 LOG(error) << "TTreeGenerator: startEvent must be >= 0, got " << startEvent;
39 return kFALSE;
40 }
41
42 if (fInputFile) {
43 fInputFile->Close();
44 delete fInputFile;
45 fInputFile = nullptr;
46 }
47 fInputTree = nullptr;
48 fNEvents = 0;
49 fCurrentEvent = 0;
50
51 fInputFile = TFile::Open(fileName, "READ");
52 if (!fInputFile) {
53 LOG(error) << "TTreeGenerator: Cannot open file " << fileName;
54 delete fInputFile;
55 fInputFile = nullptr;
56 return kFALSE;
57 }
58
59 fInputTree = dynamic_cast<TTree*>(fInputFile->Get(fTreeName.Data()));
60 if (!fInputTree) {
61 LOG(error) << "TTreeGenerator: Cannot find tree " << fTreeName.Data()
62 << " in file " << fileName;
63 fInputFile->Close();
64 delete fInputFile;
65 fInputFile = nullptr;
66 fInputTree = nullptr;
67 return kFALSE;
68 }
69
70 fNEvents = fInputTree->GetEntries();
71 if (startEvent >= fNEvents) {
72 LOG(error) << "TTreeGenerator: startEvent " << startEvent
73 << " is out of range for " << fNEvents << " entries";
74 fInputFile->Close();
75 delete fInputFile;
76 fInputFile = nullptr;
77 fInputTree = nullptr;
78 fNEvents = 0;
79 return kFALSE;
80 }
81 fCurrentEvent = startEvent;
82 LOG(info) << "TTreeGenerator: Found " << fNEvents
83 << " events in tree, starting at " << startEvent;
84
85 if (!InitBranches()) {
86 fInputFile->Close();
87 delete fInputFile;
88 fInputFile = nullptr;
89 fInputTree = nullptr;
90 fNEvents = 0;
91 fCurrentEvent = 0;
92 return kFALSE;
93 }
94
95 return kTRUE;
96}

98Bool_t TTreeGenerator::ReadEvent(FairPrimaryGenerator* primGen) {
99 if (!fInputTree || fCurrentEvent >= fNEvents) {
100 return kFALSE;
101 }
102 if (fInputTree->GetEntry(fCurrentEvent) <= 0) {
103 LOG(error) << "TTreeGenerator: Failed to read entry " << fCurrentEvent;
104 return kFALSE;
105 }
106
107 // Position in cm, momentum in GeV/c
108 primGen->AddTrack(fPdgId, fPx, fPy, fPz, fX, fY, fZ, -1, true, 0., 0.,
109 fWeight);
110
111 fCurrentEvent++;
112
113 return kTRUE;
114}

116Bool_t TTreeGenerator::InitBranches() {
117 if (!fInputTree) return kFALSE;
118
119 bool ok = true;
120
121 ok &= (fInputTree->SetBranchAddress("px", &fPx) >= 0);
122 ok &= (fInputTree->SetBranchAddress("py", &fPy) >= 0);
123 ok &= (fInputTree->SetBranchAddress("pz", &fPz) >= 0);
124
125 ok &= (fInputTree->SetBranchAddress("x", &fX) >= 0);
126 ok &= (fInputTree->SetBranchAddress("y", &fY) >= 0);
127 ok &= (fInputTree->SetBranchAddress("z", &fZ) >= 0);
128
129 ok &= (fInputTree->SetBranchAddress("id", &fPdgId) >= 0);
130 ok &= (fInputTree->SetBranchAddress("w", &fWeight) >= 0);
131
132 if (!ok) {
133 LOG(error) << "TTreeGenerator: failed to bind one or more branches";
134 }
135
136 return ok ? kTRUE : kFALSE;
137}
