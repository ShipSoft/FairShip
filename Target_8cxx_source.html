99Int_t Target::InitMedium(const char* name)
100{
101 static FairGeoLoader* geoLoad = FairGeoLoader::Instance();
102 static FairGeoInterface* geoFace = geoLoad->getGeoInterface();
103 static FairGeoMedia* media = geoFace->getMedia();
104 static FairGeoBuilder* geoBuild = geoLoad->getGeoBuilder();
105
106 FairGeoMedium* ShipMedium = media->getMedium(name);
107
108 if (!ShipMedium) {
109 Fatal("InitMedium", "Material %s not defined in media file.", name);
110 return -1111;
111 }
112 TGeoMedium* medium = gGeoManager->GetMedium(name);
113 if (medium != NULL)
114 return ShipMedium->getMediumIndex();
115 return geoBuild->createMedium(ShipMedium);
116}

265void Target::ConstructGeometry()
266{
267 // cout << "Design = " << fDesign << endl;
268
269 InitMedium("air");
270 TGeoMedium* air = gGeoManager->GetMedium("air");
271
272 InitMedium("PlasticBase");
273 TGeoMedium* PBase = gGeoManager->GetMedium("PlasticBase");
274
275 InitMedium("NuclearEmulsion");
276 TGeoMedium* NEmu = gGeoManager->GetMedium("NuclearEmulsion");
277
278 TGeoMaterial* NEmuMat = NEmu->GetMaterial(); // I need the materials to build the mixture
279 TGeoMaterial* PBaseMat = PBase->GetMaterial();
280
281 Double_t rho_film = (NEmuMat->GetDensity() * 2 * EmulsionThickness + PBaseMat->GetDensity() * PlasticBaseThickness)
282 / (2 * EmulsionThickness + PlasticBaseThickness);
283 Double_t frac_emu =
284 NEmuMat->GetDensity() * 2 * EmulsionThickness
285 / (NEmuMat->GetDensity() * 2 * EmulsionThickness + PBaseMat->GetDensity() * PlasticBaseThickness);
286
287 if (fsingleemulsionfilm)
288 cout << "TARGET PRINTOUT: Single volume for emulsion film chosen: average density: " << rho_film
289 << " fraction in mass of emulsion " << frac_emu << endl;
290
291 TGeoMixture* emufilmmixture =
292 new TGeoMixture("EmulsionFilmMixture", 2.00); // two nuclear emulsions separated by the plastic base
293
294 emufilmmixture->AddElement(NEmuMat, frac_emu);
295 emufilmmixture->AddElement(PBaseMat, 1. - frac_emu);
296
297 TGeoMedium* Emufilm = new TGeoMedium("EmulsionFilm", 100, emufilmmixture);
298
299 InitMedium("lead");
300 TGeoMedium* lead = gGeoManager->GetMedium("lead");
301
302 InitMedium("tungsten");
303 TGeoMedium* tungsten = gGeoManager->GetMedium("tungsten");
304
305 InitMedium("Concrete");
306 TGeoMedium* Conc = gGeoManager->GetMedium("Concrete");
307
308 InitMedium("steel");
309 TGeoMedium* Steel = gGeoManager->GetMedium("steel");
310
311 Int_t NPlates = number_of_plates; // Number of doublets emulsion + Pb
312
313 // Definition of the target box containing emulsion bricks + target trackers (TT)
314 TGeoBBox* TargetBox = new TGeoBBox("TargetBox", XDimension / 2, YDimension / 2, ZDimension / 2);
315 TGeoVolume* volTarget = new TGeoVolume("volTarget", TargetBox, air);
316
317 //
318 // Volumes definition
319 //
320
321 TGeoBBox* Cell = new TGeoBBox("cell", BrickX / 2, BrickY / 2, CellWidth / 2);
322 TGeoVolume* volCell = new TGeoVolume("Cell", Cell, air);
323
324 // Brick
325 TGeoBBox* Brick = new TGeoBBox("brick", BrickX / 2, BrickY / 2, BrickZ / 2);
326 TGeoVolume* volBrick = new TGeoVolume("Brick", Brick, air);
327 volBrick->SetLineColor(kCyan);
328 volBrick->SetTransparency(1);
329 // need to separate the two cases, now with a ternary operator
330 auto* Absorber = new TGeoBBox("Absorber", EmulsionX / 2, EmulsionY / 2, LeadThickness / 2);
331 auto* volAbsorber = new TGeoVolume("volAbsorber", Absorber, tungsten);
332
333 volAbsorber->SetTransparency(1);
334 volAbsorber->SetLineColor(kGray);
335 for (Int_t n = 0; n < NPlates; n++) {
336 volBrick->AddNode(
337 volAbsorber,
338 n,
339 new TGeoTranslation(
340 0, 0, -BrickZ / 2 + BrickPackageZ / 2 + EmPlateWidth + LeadThickness / 2 + n * AllPlateWidth));
341 }
342 if (fsingleemulsionfilm) { // simplified configuration, unique sensitive layer for the whole emulsion plate
343 TGeoBBox* EmulsionFilm = new TGeoBBox("EmulsionFilm", EmulsionX / 2, EmulsionY / 2, EmPlateWidth / 2);
344 TGeoVolume* volEmulsionFilm = new TGeoVolume("Emulsion", EmulsionFilm, Emufilm); // TOP
345 volEmulsionFilm->SetLineColor(kBlue);
346
347 if (fPassive == 0) {
348 AddSensitiveVolume(volEmulsionFilm);
349 }
350
351 for (Int_t n = 0; n < NPlates + 1; n++) {
352 volBrick->AddNode(
353 volEmulsionFilm,
354 n,
355 new TGeoTranslation(0, 0, -BrickZ / 2 + BrickPackageZ / 2 + EmPlateWidth / 2 + n * AllPlateWidth));
356 }
357 } else { // more accurate configuration, two emulsion films divided by a plastic base
358 TGeoBBox* EmulsionFilm = new TGeoBBox("EmulsionFilm", EmulsionX / 2, EmulsionY / 2, EmulsionThickness / 2);
359 TGeoVolume* volEmulsionFilm = new TGeoVolume("Emulsion", EmulsionFilm, NEmu); // TOP
360 TGeoVolume* volEmulsionFilm2 = new TGeoVolume("Emulsion2", EmulsionFilm, NEmu); // BOTTOM
361 volEmulsionFilm->SetLineColor(kBlue);
362 volEmulsionFilm2->SetLineColor(kBlue);
363
364 if (fPassive == 0) {
365 AddSensitiveVolume(volEmulsionFilm);
366 AddSensitiveVolume(volEmulsionFilm2);
367 }
368 TGeoBBox* PlBase = new TGeoBBox("PlBase", EmulsionX / 2, EmulsionY / 2, PlasticBaseThickness / 2);
369 TGeoVolume* volPlBase = new TGeoVolume("PlasticBase", PlBase, PBase);
370 volPlBase->SetLineColor(kYellow - 4);
371 for (Int_t n = 0; n < NPlates + 1; n++) {
372 volBrick->AddNode(
373 volEmulsionFilm2,
374 n,
375 new TGeoTranslation(
376 0, 0, -BrickZ / 2 + BrickPackageZ / 2 + EmulsionThickness / 2 + n * AllPlateWidth)); // BOTTOM
377 volBrick->AddNode(volEmulsionFilm,
378 n,
379 new TGeoTranslation(0,
380 0,
381 -BrickZ / 2 + BrickPackageZ / 2 + 3 * EmulsionThickness / 2
382 + PlasticBaseThickness + n * AllPlateWidth)); // TOP
383 volBrick->AddNode(volPlBase,
384 n,
385 new TGeoTranslation(0,
386 0,
387 -BrickZ / 2 + BrickPackageZ / 2 + EmulsionThickness
388 + PlasticBaseThickness / 2
389 + n * AllPlateWidth)); // PLASTIC BASE
390 }
391 }
392
393 volBrick->SetVisibility(kTRUE);
394 TGeoVolume* top = gGeoManager->GetTopVolume();
395 TGeoVolumeAssembly* tTauNuDet = new TGeoVolumeAssembly("tTauNuDet");
396 top->AddNode(tTauNuDet, 1, new TGeoTranslation(0, 0, 0));
397
398 tTauNuDet->AddNode(volTarget, 1, new TGeoTranslation(0, 0, fCenterZ));
399 TGeoBBox* Row = new TGeoBBox("row", WallXDim / 2, BrickY / 2, WallZDim / 2);
400 TGeoVolume* volRow = new TGeoVolume("Row", Row, air);
401 volRow->SetLineColor(20);
402
403 Double_t d_cl_x = -WallXDim / 2;
404 for (int j = 0; j < fNCol; j++) {
405 volRow->AddNode(volBrick, j, new TGeoTranslation(d_cl_x + BrickX / 2, 0, 0));
406 d_cl_x += BrickX;
407 }
408 TGeoBBox* Wall = new TGeoBBox("wall", WallXDim / 2, WallYDim / 2, WallZDim / 2);
409 TGeoVolume* volWall = new TGeoVolume("Wall", Wall, air);
410 volWall->SetLineColor(kGreen);
411
412 Double_t d_cl_y = -WallYDim / 2;
413 for (int k = 0; k < fNRow; k++) {
414 volWall->AddNode(volRow, k, new TGeoTranslation(0, d_cl_y + BrickY / 2, 0));
415
416 // 2mm is the distance for the structure that holds the brick
417 d_cl_y += BrickY + Ydistance;
418 }
419 // Columns
420
421 Double_t d_cl_z = -ZDimension / 2 + TTrackerZ;
422 for (int l = 0; l < fNWall; l++) {
423 volTarget->AddNode(volWall, l, new TGeoTranslation(0, 0, d_cl_z + BrickZ / 2));
424
425 // 6 cm is the distance between 2 columns of consecutive Target for TT placement
426 d_cl_z += BrickZ + TTrackerZ;
427 }
428} // end construct geometry

430Bool_t Target::ProcessHits(FairVolume* vol)
431{
433 // Set parameters at entrance of volume. Reset ELoss.
434 if (gMC->IsTrackEntering()) {
435 fELoss = 0.;
436 fTime = gMC->TrackTime() * 1.0e09;
437 fLength = gMC->TrackLength();
438 gMC->TrackPosition(fPos);
439 gMC->TrackMomentum(fMom);
440 }
441 // Sum energy loss for all steps in the active volume
442 fELoss += gMC->Edep();
443
444 // Create muonPoint at exit of active volume
445 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
446 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
447 // Int_t fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
448 gMC->CurrentVolID(fVolumeID);
449 Int_t detID = fVolumeID;
450 // gGeoManager->PrintOverlaps();
451
452 // cout<< "detID = " << detID << endl;
453 Int_t MaxLevel = gGeoManager->GetLevel();
454 const Int_t MaxL = MaxLevel;
455 // cout << "MaxLevel = " << MaxL << endl;
456 // cout << gMC->CurrentVolPath()<< endl;
457
458 Int_t motherV[MaxL];
459 Bool_t EmTop = false;
460 Int_t NPlate = 0;
461 const char* name;
462
463 name = gMC->CurrentVolName();
464 // cout << name << endl;
465
466 if (strcmp(name, "Emulsion") == 0) {
467 NPlate = detID;
468 EmTop = 1;
469 }
470 if (strcmp(name, "Emulsion2") == 0) {
471 NPlate = detID;
472 EmTop = 0;
473 }
474
475 Int_t NWall = 0, NColumn = 0, NRow = 0;
476
477 for (Int_t i = 0; i < MaxL; i++) {
478 motherV[i] = gGeoManager->GetMother(i)->GetNumber();
479 const char* mumname = gMC->CurrentVolOffName(i);
480 if (motherV[0] == 1 && motherV[0] != detID) {
481 if (strcmp(mumname, "Brick") == 0)
482 NColumn = motherV[i];
483 if (strcmp(mumname, "Cell") == 0)
484 NRow = motherV[i];
485 if (strcmp(mumname, "Row") == 0)
486 NWall = motherV[i];
487 if ((strcmp(mumname, "Wall") == 0) && (motherV[i] == 2))
488 NWall += fNWall;
489 } else {
490
491 if (strcmp(mumname, "Cell") == 0)
492 NColumn = motherV[i];
493 if (strcmp(mumname, "Row") == 0)
494 NRow = motherV[i];
495 if (strcmp(mumname, "Wall") == 0)
496 NWall = motherV[i];
497 if ((strcmp(mumname, "volTarget") == 0) && (motherV[i] == 2))
498 NWall += fNWall;
499 }
500 // cout << i << " " << motherV[i] << " name = " << mumname << endl;
501 }
502
503 detID = (NWall + 1) * 1E7 + (NRow + 1) * 1E6 + (NColumn + 1) * 1E4 + 1E3 + (NPlate + 1) * 1E1 + EmTop * 1;
504
505 fVolumeID = detID;
506
507 if (fELoss == 0.) {
508 return kFALSE;
509 }
510 TParticle* p = gMC->GetStack()->GetCurrentTrack();
511 // Int_t MotherID =gMC->GetStack()->GetCurrentParentTrackNumber();
512 Int_t pdgCode = p->GetPdgCode();
513
514 TLorentzVector Pos;
515 gMC->TrackPosition(Pos);
516 Double_t xmean = (fPos.X() + Pos.X()) / 2.;
517 Double_t ymean = (fPos.Y() + Pos.Y()) / 2.;
518 Double_t zmean = (fPos.Z() + Pos.Z()) / 2.;
519
520 AddHit(fTrackID,
521 fVolumeID,
522 TVector3(xmean, ymean, zmean),
523 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
524 fTime,
525 fLength,
526 fELoss,
527 pdgCode);
528
529 // Increment number of muon det points in TParticle
530 ShipStack* stack = (ShipStack*)gMC->GetStack();
531 stack->AddPoint(ktauTarget);
532 }
533
534 return kTRUE;
535}

537void Target::DecodeBrickID(Int_t detID, Int_t& NWall, Int_t& NRow, Int_t& NColumn, Int_t& NPlate, Bool_t& EmTop)
538{
539 NWall = detID / 1E7;
540 NRow = (detID - NWall * 1E7) / 1E6;
541 NColumn = (detID - NWall * 1E7 - NRow * 1E6) / 1E4;
542 NPlate = (detID - NWall * 1E7 - NRow * 1E6 - NColumn * 1E4 - 1E3) / 1E1;
543 EmTop = (detID - NWall * 1E7 - NRow * 1E6 - NColumn * 1E4 - 1E3 - NPlate * 1E1) / 1E0;
544}
