486InitStatus ecalDrawer::Init()
487{
488 fEventN=0;
489
490
491 FairRootManager* io=FairRootManager::Instance();
492 if (!io)
493 {
494 Fatal("Init", "Can't find IOManager.");
495 return kFATAL;
496 }
497 fMCTracks = dynamic_cast<TClonesArray*>(io->GetObject("MCTrack"));
498 if (!fMCTracks)
499 {
500 Fatal("Init", "Can't find array of MC tracks");
501 return kFATAL;
502 }
503 fPoints = dynamic_cast<TClonesArray*>(io->GetObject("EcalPoint"));
504 if (!fPoints)
505 {
506 Fatal("Init", "Can't find array of Ecal Points");
507 return kFATAL;
508 }
509 fStr = dynamic_cast<ecalStructure*>(io->GetObject("EcalStructure"));
510 if (!fStr)
511 {
512 Fatal("Init", "Can't find calorimeter structure for drawing");
513 return kFATAL;
514 }
515 fClusters = dynamic_cast<TClonesArray*>(io->GetObject("EcalClusters"));
516 if (!fClusters)
517 {
518 Fatal("Init", "Can't find array of calorimeter clusters");
519 return kFATAL;
520 }
521 fInf=fStr->GetEcalInf();
522 fCX=fInf->GetXSize(); fCY=fInf->GetYSize();
523 fCX*=fCellSize*4+4; fCX+=1;
524 fCY*=fCellSize*4+4; fCY+=1;
525 fC=new TASImage(fCX, fCY);
526
527 list<ecalCell*> cells;
528 list<ecalCell*>::const_iterator p;
529 fStr->GetCells(cells);
530 for(p=cells.begin();p!=cells.end();++p)
531 fCells.push_back(new ecalDrawerItem((*p), 0.0, 0.0, 0.0));
532 fTxt->SetTextFont(43);
533 fTxt->SetTextSizePixels(8);
534 fTxt->SetTextAlign(21);
535 return kSUCCESS;
536}
