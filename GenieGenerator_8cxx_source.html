67Double_t GenieGenerator::MeanMaterialBudget(const Double_t *start, const Double_t *end, Double_t *mparam)
68{
69 //
70 // Calculate mean material budget and material properties between
71 // the points "start" and "end".
72 //
73 // "mparam" - parameters used for the energy and multiple scattering
74 // corrections:
75 //
76 // mparam[0] - mean density: sum(x_i*rho_i)/sum(x_i) [g/cm3]
77 // mparam[1] - equivalent rad length fraction: sum(x_i/X0_i) [adimensional]
78 // mparam[2] - mean A: sum(x_i*A_i)/sum(x_i) [adimensional]
79 // mparam[3] - mean Z: sum(x_i*Z_i)/sum(x_i) [adimensional]
80 // mparam[4] - length: sum(x_i) [cm]
81 // mparam[5] - Z/A mean: sum(x_i*Z_i/A_i)/sum(x_i) [adimensional]
82 // mparam[6] - number of boundary crosses
83 // mparam[7] - maximum density encountered (g/cm^3)
84 // mparam[8] - equivalent interaction length fraction: sum(x_i/I0_i) [adimensional]
85 // mparam[9] - maximum cross section encountered (mbarn)
86 //
87 // Origin: Marian Ivanov, Marian.Ivanov@cern.ch
88 //
89 // Corrections and improvements by
90 // Andrea Dainese, Andrea.Dainese@lnl.infn.it,
91 // Andrei Gheata, Andrei.Gheata@cern.ch
92 // Thomas Ruf, Thomas.Ruf@cern.ch
93 //
94
95 mparam[0]=0; mparam[1]=1; mparam[2]=0; mparam[3]=0; mparam[4]=0;
96 mparam[5]=0; mparam[6]=0; mparam[7]=0; mparam[8]=0; mparam[9]=0;
97 //
98 Double_t bparam[7]; // total parameters
99 Double_t lparam[7]; // local parameters
100 Double_t mbarn = 1E-3*1E-24*TMath::Na(); // cm^2 * Avogadro
101
102 for (Int_t i=0;i<7;i++) bparam[i]=0;
103
104 if (!gGeoManager) {
105 //AliFatalClass("No TGeo\n");
106 return 0.;
107 }
108 //
109 Double_t length;
110 Double_t dir[3];
111 length = TMath::Sqrt((end[0]-start[0])*(end[0]-start[0])+
112 (end[1]-start[1])*(end[1]-start[1])+
113 (end[2]-start[2])*(end[2]-start[2]));
114 mparam[4]=length;
115 if (length<TGeoShape::Tolerance()) return 0.0;
116 Double_t invlen = 1./length;
117 dir[0] = (end[0]-start[0])*invlen;
118 dir[1] = (end[1]-start[1])*invlen;
119 dir[2] = (end[2]-start[2])*invlen;
120
121 // Initialize start point and direction
122 TGeoNode *currentnode = 0;
123 TGeoNode *startnode = gGeoManager->InitTrack(start, dir);
124 if (!startnode) {
125 //AliErrorClass(Form("start point out of geometry: x %f, y %f, z %f",
126 // start[0],start[1],start[2]));
127 return 0.0;
128 }
129 TGeoMaterial *material = startnode->GetVolume()->GetMedium()->GetMaterial();
130 lparam[0] = material->GetDensity();
131 if (lparam[0] > mparam[7]) mparam[7]=lparam[0];
132 lparam[1] = material->GetRadLen();
133 lparam[2] = material->GetA();
134 lparam[3] = material->GetZ();
135 lparam[4] = length;
136 lparam[5] = lparam[3]/lparam[2];
137 lparam[6] = material->GetIntLen();
138 Double_t n = lparam[0]/lparam[2];
139 Double_t sigma = 1./(n*lparam[6])/mbarn;
140 if (sigma > mparam[9]) mparam[9]=sigma;
141 if (material->IsMixture()) {
142 TGeoMixture * mixture = (TGeoMixture*)material;
143 lparam[5] =0;
144 Double_t sum =0;
145 for (Int_t iel=0;iel<mixture->GetNelements();iel++){
146 sum += mixture->GetWmixt()[iel];
147 lparam[5]+= mixture->GetZmixt()[iel]*mixture->GetWmixt()[iel]/mixture->GetAmixt()[iel];
148 }
149 lparam[5]/=sum;
150 }
151
152 // Locate next boundary within length without computing safety.
153 // Propagate either with length (if no boundary found) or just cross boundary
154 gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
155 Double_t step = 0.0; // Step made
156 Double_t snext = gGeoManager->GetStep();
157 // If no boundary within proposed length, return current density
158 if (!gGeoManager->IsOnBoundary()) {
159 mparam[0] = lparam[0];
160 mparam[1] = lparam[4]/lparam[1];
161 mparam[2] = lparam[2];
162 mparam[3] = lparam[3];
163 mparam[4] = lparam[4];
164 mparam[8] = lparam[4]/lparam[6];
165 return lparam[0];
166 }
167 // Try to cross the boundary and see what is next
168 Int_t nzero = 0;
169 while (length>TGeoShape::Tolerance()) {
170 currentnode = gGeoManager->GetCurrentNode();
171 if (snext<2.*TGeoShape::Tolerance()) nzero++;
172 else nzero = 0;
173 if (nzero>3) {
174 // This means navigation has problems on one boundary
175 // Try to cross by making a small step
176 //AliErrorClass("Cannot cross boundary\n");
177 mparam[0] = bparam[0]/step;
178 mparam[1] = bparam[1];
179 mparam[2] = bparam[2]/step;
180 mparam[3] = bparam[3]/step;
181 mparam[5] = bparam[5]/step;
182 mparam[8] = bparam[6];
183 mparam[4] = step;
184 mparam[0] = 0.; // if crash of navigation take mean density 0
185 mparam[1] = 1000000; // and infinite rad length
186 return bparam[0]/step;
187 }
188 mparam[6]+=1.;
189 step += snext;
190 bparam[1] += snext/lparam[1];
191 bparam[2] += snext*lparam[2];
192 bparam[3] += snext*lparam[3];
193 bparam[5] += snext*lparam[5];
194 bparam[6] += snext/lparam[6];
195 bparam[0] += snext*lparam[0];
196
197 if (snext>=length) break;
198 if (!currentnode) break;
199 length -= snext;
200 material = currentnode->GetVolume()->GetMedium()->GetMaterial();
201 lparam[0] = material->GetDensity();
202 if (lparam[0] > mparam[7]) mparam[7]=lparam[0];
203 lparam[1] = material->GetRadLen();
204 lparam[2] = material->GetA();
205 lparam[3] = material->GetZ();
206 lparam[5] = lparam[3]/lparam[2];
207 lparam[6] = material->GetIntLen();
208 n = lparam[0]/lparam[2];
209 sigma = 1./(n*lparam[6])/mbarn;
210 if (sigma > mparam[9]) mparam[9]=sigma;
211 if (material->IsMixture()) {
212 TGeoMixture * mixture = (TGeoMixture*)material;
213 lparam[5]=0;
214 Double_t sum =0;
215 for (Int_t iel=0;iel<mixture->GetNelements();iel++){
216 sum+= mixture->GetWmixt()[iel];
217 lparam[5]+= mixture->GetZmixt()[iel]*mixture->GetWmixt()[iel]/mixture->GetAmixt()[iel];
218 }
219 lparam[5]/=sum;
220 }
221 gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
222 snext = gGeoManager->GetStep();
223 }
224 mparam[0] = bparam[0]/step;
225 mparam[1] = bparam[1];
226 mparam[2] = bparam[2]/step;
227 mparam[3] = bparam[3]/step;
228 mparam[5] = bparam[5]/step;
229 mparam[8] = bparam[6];
230 return bparam[0]/step;
231}

233std::vector<double> GenieGenerator::Rotate(Double_t x, Double_t y, Double_t z, Double_t px, Double_t py, Double_t pz)
234{
235 //rotate vector px,py,pz to point at x,y,z at origin.
236 Double_t theta=atan(sqrt(x*x+y*y)/z);
237 Double_t c=cos(theta);
238 Double_t s=sin(theta);
239 //rotate around y-axis
240 Double_t px1=c*px+s*pz;
241 Double_t pzr=-s*px+c*pz;
242 Double_t phi=atan2(y,x);
243 c=cos(phi);
244 s=sin(phi);
245 //rotate around z-axis
246 Double_t pxr=c*px1-s*py;
247 Double_t pyr=s*px1+c*py;
248 std::vector<double> pout;
249 pout.push_back(pxr);
250 pout.push_back(pyr);
251 pout.push_back(pzr);
252 //cout << "Info GenieGenerator: rotated" << pout[0] << " " << pout[1] << " " << pout[2] << " " << x << " " << y << " " << z <<endl;
253 return pout;
254}

491 pout[0] = cos(phi)*pt;
492 pout[1] = sin(phi)*pt;
493 pout[2] = pzv*pzv-pt*pt;
494 //printf("p= %f pt=%f px,py,pz**2=%f,%f,%f\n",pzv,pt,pout[0],pout[1],pout[2]);
495
496 if (pout[2]>=0.) {
497 pout[2]=TMath::Sqrt(pout[2]);
498 if (gRandom->Uniform(-1.,1.)<0.) pout[0]=-pout[0];
499 if (gRandom->Uniform(-1.,1.)<0.) pout[1]=-pout[1];
500 //cout << "Info GenieGenerator: neutrino pxyz " << pout[0] << ", " << pout[1] << ", " << pout[2] << endl;
501 // xyz at start and end
502 start[0]=(pout[0]/pout[2])*(start[2]-ztarget);
503 start[1]=(pout[1]/pout[2])*(start[2]-ztarget);
504 //cout << "Info GenieGenerator: neutrino xyz-start " << start[0] << "-" << start[1] << "-" << start[2] << endl;
505 txnu=pout[0]/pout[2];
506 tynu=pout[1]/pout[2];
507 end[0]=txnu*(end[2]-ztarget);
508 end[1]=tynu*(end[2]-ztarget);
509 //cout << "Info GenieGenerator: neutrino xyz-end " << end[0] << "-" << end[1] << "-" << end[2] << endl;
510 //get material density between these two points
511 bparam=MeanMaterialBudget(start, end, mparam);
512 //printf("param %e %e %e \n",bparam,mparam[6],mparam[7]);
513 }
514 }
515 //loop over trajectory between start and end to pick an interaction point
516 Double_t prob2int = -1.;
517 Double_t x;
518 Double_t y;
519 Double_t z;
520 Int_t count=0;
521 while (prob2int<gRandom->Uniform(0.,1.)) {
522 //place x,y,z uniform along path
523 z=gRandom->Uniform(start[2],end[2]);
524 x=txnu*(z-ztarget);
525 y=tynu*(z-ztarget);
526 if (mparam[6]<0.5){
527 //mparam is number of boundaries along path. mparam[6]=0.: uniform material budget along path, use present x,y,z
528 prob2int=2.;
529 }else{
530 //get local material at this point, to calculate probability that interaction is at this point.
531 TGeoNode *node = gGeoManager->FindNode(x,y,z);
532 TGeoMaterial *mat = 0;
533 if (node && !gGeoManager->IsOutside()) {
534 mat = node->GetVolume()->GetMaterial();
535 //cout << "Info GenieGenerator: mat " << count << ", " << mat->GetName() << ", " << mat->GetDensity() << endl;
536 //density relative to Prob largest density along this trajectory, i.e. use rho(Pt)
537 prob2int= mat->GetDensity()/mparam[7];
538 if (prob2int>1.) cout << "***WARNING*** GenieGenerator: prob2int > Maximum density????" << prob2int << " maxrho:" << mparam[7] << " material: " << mat->GetName() << endl;
539 count+=1;
540 }else{
541 prob2int=0.;
542 }
543 }
544 }
545 //cout << "Info GenieGenerator: prob2int " << prob2int << ", " << count << endl;
546
547 Double_t zrelative=z-ztarget;
548 Double_t tof = TMath::Sqrt(x * x + y * y + zrelative * zrelative) / 2.99792458e+10; // speed of light in cm/s
549 cpg->AddTrack(neu,pout[0],pout[1],pout[2],x,y,z,-1,false,TMath::Sqrt(pout[0]*pout[0]+pout[1]*pout[1]+pout[2]*pout[2]),tof,mparam[0]*mparam[4]);
550 if (not fNuOnly){
551 // second, outgoing lepton
552 std::vector<double> pp = Rotate(x,y,zrelative,pxl,pyl,pzl);
553 Int_t oLPdgCode = neu;
554 if (cc){oLPdgCode = copysign(TMath::Abs(neu)-1,neu);}
555 if (nuel){oLPdgCode = 11;}
556 cpg->AddTrack(oLPdgCode,pp[0],pp[1],pp[2],x,y,z,0,true,El,tof,mparam[0]*mparam[4]);
557// last, all others
558 for(int i=0; i<nf; i++)
559 {
560 pp = Rotate(x,y,zrelative,pxf[i],pyf[i],pzf[i]);
561 cpg->AddTrack(pdgf[i],pp[0],pp[1],pp[2],x,y,z,0,true,Ef[i],tof,mparam[0]*mparam[4]);
562 //cout << "f " << pdgf[i] << " pz "<< pzf[i] << endl;
563 }
564 //cout << "Info GenieGenerator Return from GenieGenerator" << endl;
565 }
566 return kTRUE;
567}
