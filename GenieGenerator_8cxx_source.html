39Bool_t GenieGenerator::Init(const char* fileName, const int startEvent) {
40 if (fGenOption != 0 && fGenOption != 3) {
41 LOG(fatal) << "Invalid GenieGen Option: " << fGenOption
42 << " Please check the option provided with --GenieOption "
43 << endl;
44 return kFALSE;
45 }
46 fNuOnly = false;
47 LOG(info) << "Opening input file " << fileName;
48 if (startEvent < 0) {
49 LOG(error) << "GenieGenerator: startEvent must be >= 0, got " << startEvent;
50 return kFALSE;
51 }
52 fInputFile = TFile::Open(fileName, "READ");
53 if (!fInputFile) {
54 LOG(error) << "GenieGenerator: error opening input file " << fileName;
55 delete fInputFile;
56 fInputFile = nullptr;
57 return kFALSE;
58 }
59 fTree = dynamic_cast<TTree*>(fInputFile->Get("gst"));
60 if (!fTree) {
61 LOG(error) << "GenieGenerator: cannot find tree gst in file " << fileName;
62 fInputFile->Close();
63 delete fInputFile;
64 fInputFile = nullptr;
65 return kFALSE;
66 }
67 fNevents = fTree->GetEntries();
68 if (startEvent >= fNevents) {
69 LOG(error) << "GenieGenerator: startEvent " << startEvent
70 << " is out of range for " << fNevents << " entries";
71 fInputFile->Close();
72 delete fInputFile;
73 fInputFile = nullptr;
74 fTree = nullptr;
75 return kFALSE;
76 }
77 fn = startEvent;
78 bool ok = true;
79 ok &= (fTree->SetBranchAddress("Ev", &Ev) >= 0);
80 ok &= (fTree->SetBranchAddress("pxv", &pxv) >= 0);
81 ok &= (fTree->SetBranchAddress("pyv", &pyv) >= 0);
82 ok &= (fTree->SetBranchAddress("pzv", &pzv) >= 0);
83 ok &= (fTree->SetBranchAddress("neu", &neu) >= 0);
84 ok &= (fTree->SetBranchAddress("cc", &cc) >= 0);
85 ok &= (fTree->SetBranchAddress("nuel", &nuel) >= 0);
86 ok &= (fTree->SetBranchAddress("vtxx", &vtxx) >= 0);
87 ok &= (fTree->SetBranchAddress("vtxy", &vtxy) >= 0);
88 ok &= (fTree->SetBranchAddress("vtxz", &vtxz) >= 0);
89 ok &= (fTree->SetBranchAddress("vtxt", &vtxt) >= 0);
90 ok &= (fTree->SetBranchAddress("El", &El) >= 0);
91 ok &= (fTree->SetBranchAddress("pxl", &pxl) >= 0);
92 ok &= (fTree->SetBranchAddress("pyl", &pyl) >= 0);
93 ok &= (fTree->SetBranchAddress("pzl", &pzl) >= 0);
94 ok &= (fTree->SetBranchAddress("Ef", &Ef) >= 0);
95 ok &= (fTree->SetBranchAddress("pxf", &pxf) >= 0);
96 ok &= (fTree->SetBranchAddress("pyf", &pyf) >= 0);
97 ok &= (fTree->SetBranchAddress("pzf", &pzf) >= 0);
98 ok &= (fTree->SetBranchAddress("nf", &nf) >= 0);
99 ok &= (fTree->SetBranchAddress("pdgf", &pdgf) >= 0);
100 if (!ok) {
101 LOG(error)
102 << "GenieGenerator: failed to bind one or more required branches";
103 fInputFile->Close();
104 delete fInputFile;
105 fInputFile = nullptr;
106 fTree = nullptr;
107 return kFALSE;
108 }
109 fFirst = kTRUE;
110 return kTRUE;
111}

113std::vector<double> GenieGenerator::Rotate(Double_t x, Double_t y, Double_t z,
114 Double_t px, Double_t py,
115 Double_t pz) {
116 // rotate vector px,py,pz to point at x,y,z at origin.
117 Double_t theta = atan(sqrt(x * x + y * y) / z);
118 Double_t c = cos(theta);
119 Double_t s = sin(theta);
120 // rotate around y-axis
121 Double_t px1 = c * px + s * pz;
122 Double_t pzr = -s * px + c * pz;
123 Double_t phi = atan2(y, x);
124 c = cos(phi);
125 s = sin(phi);
126 // rotate around z-axis
127 Double_t pxr = c * px1 - s * py;
128 Double_t pyr = s * px1 + c * py;
129 // cout << "Info GenieGenerator: rotated" << pxr << " " << pyr << " "
130 // << pzr << " " << x << " " << y << " " << z <<endl;
131 return {pxr, pyr, pzr};
132}

148// ----- Passing the event ---------------------------------------------
149Bool_t GenieGenerator::ReadEventGeometryDriver(FairPrimaryGenerator* cpg) {
150 // Use GENIE geometry driver.
151 // Get event from GENIE TTree. If we reach the end of the file, return
152 // false.
153 if (fTree->GetEntry(fn) == 0) return kFALSE;
154
155 if (fn % 100 == 0) {
156 cout << "Info GenieGenerator: neutrino event-nr " << fn << endl;
157 }
158
159 fn++;
160
161 // Add the neutrino to the MCTrack stack:
162 cpg->AddTrack(neu, // Neutrino PDG
163 pxv, pyv, pzv, // Neutrino momentum
164 vtxx * m, vtxy * m,
165 vtxz * m, // Event vertex [in cm!]
166 -1, // Parent
167 false); // Don't track this particle
168 IncrementCounter("generated_events");
169 if (cc) IncrementCounter("cc_events");
170 if (nuel) IncrementCounter("nue_elastic_events");
171 // Add the outgoing lepton and hadrons, if not in nu-only mode
172 if (!fNuOnly) {
173 // Add final state lepton to the MCTrack stack:
174 int outgoing_lepton_pdg = neu;
175 if (cc) outgoing_lepton_pdg = copysign(TMath::Abs(neu) - 1, neu);
176 if (nuel) outgoing_lepton_pdg = 11;
177
178 bool track_outgoing_lepton = (cc || nuel);
179 cpg->AddTrack(outgoing_lepton_pdg, pxl, pyl, pzl, vtxx * m, vtxy * m,
180 vtxz * m, 0, track_outgoing_lepton);
181 IncrementCounter("outgoing_leptons_stored");
182
183 // Add final state hadrons to the MCTrack stack
184 for (int i_hadron = 0; i_hadron < nf; i_hadron++) {
185 cpg->AddTrack(pdgf[i_hadron], pxf[i_hadron], pyf[i_hadron], pzf[i_hadron],
186 vtxx * m, vtxy * m, vtxz * m, 0, true);
187 IncrementCounter("outgoing_hadrons_stored");
188 }
189 }
190 return kTRUE;
191}

194Bool_t GenieGenerator::ReadEvent(FairPrimaryGenerator* cpg) {
195 if (fGenOption == 3) return this->ReadEventGeometryDriver(cpg);
196 // Read simple event format from GENIE. Vertex positions need to be
197 // generated here
198 // some start/end positions in z (emulsion to Tracker 1)
199 Double_t start[3] = {0., 0., startZ};
200 Double_t end[3] = {0., 0., endZ};
201 char ts[20];
202 // cout << "Enter GenieGenerator " << endl;
203 // pick histogram: 1100=100 momentum bins, 1200=25 momentum bins.
204 Int_t idbase = 1200;
205 if (fFirst) {
206 Double_t bparam = 0.;
207 Double_t mparam[10];
208 bparam = shipgen::MeanMaterialBudget(start, end, mparam);
209 cout << "Info GenieGenerator: MaterialBudget " << start[2] << " - "
210 << end[2] << endl;
211 cout << "Info GenieGenerator: MaterialBudget " << bparam << endl;
212 cout << "Info GenieGenerator: MaterialBudget 0 " << mparam[0] << endl;
213 cout << "Info GenieGenerator: MaterialBudget 1 " << mparam[1] << endl;
214 cout << "Info GenieGenerator: MaterialBudget 2 " << mparam[2] << endl;
215 cout << "Info GenieGenerator: MaterialBudget 3 " << mparam[3] << endl;
216 cout << "Info GenieGenerator: MaterialBudget 4 " << mparam[4] << endl;
217 cout << "Info GenieGenerator: MaterialBudget 5 " << mparam[5] << endl;
218 cout << "Info GenieGenerator: MaterialBudget 6 " << mparam[6] << endl;
219 cout << "Info GenieGenerator: MaterialBudget " << mparam[0] * mparam[4]
220 << endl;
221 // read the (log10(p),log10(pt)) hists to be able to draw a pt for every
222 // neutrino momentum loop over neutrino types
223 printf("Reading (log10(p),log10(pt)) Hists from file: %s\n",
224 fInputFile->GetName());
225 for (Int_t idnu = 12; idnu < 17; idnu += 2) {
226 for (Int_t idadd = -1; idadd < 2; idadd += 2) {
227 Int_t idhnu = static_cast<int>(idbase + idnu);
228 if (idadd < 0) idhnu += 1000;
229 sprintf(ts, "%d", idhnu);
230 // pickup corresponding (log10(p),log10(pt)) histogram
231 if (fInputFile->FindObjectAny(ts)) {
232 TH2* h2tmp = dynamic_cast<TH2*>(fInputFile->Get(ts));
233 printf("HISTID=%d, Title:%s\n", idhnu, h2tmp->GetTitle());
234 sprintf(ts, "px_%d", idhnu);
235 // make its x-projection, to later be able to convert log10(p) to its
236 // bin-number
237 pxhist[idhnu] = h2tmp->ProjectionX(ts, 1, -1);
238 Int_t nbinx = h2tmp->GetNbinsX();
239 // printf("idhnu=%d ts=%s nbinx=%d\n",idhnu,ts,nbinx);
240 // project all slices on the y-axis
241 for (Int_t k = 1; k < nbinx + 1; k += 1) {
242 sprintf(ts, "h%d%d", idhnu, k);
243 // printf("idnu %d idhnu %d bin%d ts=%s\n",idnu,idhnu,k,ts);
244 pyslice[idhnu][k] = h2tmp->ProjectionY(ts, k, k);
245 }
246 }
247 }
248 }
249 fFirst = kFALSE;
250 }
251
252 if (fn == fNevents) {
253 LOG(warning) << "End of input file. Rewind.";
254 }
255 fTree->GetEntry(fn % fNevents);
256 fn++;
257 if (fn % 100 == 0) {
258 LOG(info) << "Info GenieGenerator: neutrino event-nr " << fn << endl;
259 }
260
261 // Incoming neutrino, get a random px,py
262 Double_t mparam[10];
263 Double_t pout[3] = {0., 0., -1.};
264 Double_t txnu = 0;
265 Double_t tynu = 0;
266 // Does this neutrino fly through material? Otherwise draw another pt..
267 while (pout[2] < 0.) {
268 // get pt of this neutrino from 2D hists.
269 Int_t idhnu = TMath::Abs(neu) + idbase;
270 if (neu < 0) idhnu += 1000;
271 Int_t nbinmx = pxhist[idhnu]->GetNbinsX();
272 Double_t pl10 = log10(pzv);
273 Int_t nbx = pxhist[idhnu]->FindBin(pl10);
274 if (nbx < 1) nbx = 1;
275 if (nbx > nbinmx) nbx = nbinmx;
276 Double_t ptlog10 = pyslice[idhnu][nbx]->GetRandom();
277 // hist was filled with: log10(pt+0.01)
278 Double_t pt = pow(10., ptlog10) - 0.01;
279 // rotate pt in phi:
280 Double_t phi = gRandom->Uniform(0., 2 * TMath::Pi());
281 pout[0] = cos(phi) * pt;
282 pout[1] = sin(phi) * pt;
283 pout[2] = pzv * pzv - pt * pt;
284
285 if (pout[2] >= 0.) {
286 pout[2] = TMath::Sqrt(pout[2]);
287 if (gRandom->Uniform(-1., 1.) < 0.) pout[0] = -pout[0];
288 if (gRandom->Uniform(-1., 1.) < 0.) pout[1] = -pout[1];
289 // xyz at start and end
290 start[0] = (pout[0] / pout[2]) * (start[2] - ztarget);
291 start[1] = (pout[1] / pout[2]) * (start[2] - ztarget);
292 txnu = pout[0] / pout[2];
293 tynu = pout[1] / pout[2];
294 end[0] = txnu * (end[2] - ztarget);
295 end[1] = tynu * (end[2] - ztarget);
296 shipgen::MeanMaterialBudget(start, end, mparam);
297 }
298 }
299 // loop over trajectory between start and end to pick an interaction point
300 Double_t prob2int = -1.;
301 Double_t x = 0.;
302 Double_t y = 0.;
303 Double_t z = 0.;
304 while (prob2int < gRandom->Uniform(0., 1.)) {
305 IncrementCounter("interaction_sampling_trials");
306 // place x,y,z uniform along path
307 z = gRandom->Uniform(start[2], end[2]);
308 x = txnu * (z - ztarget);
309 y = tynu * (z - ztarget);
310 if (mparam[6] < 0.5) {
311 // mparam is number of boundaries along path. mparam[6]=0.: uniform
312 // material budget along path, use present x,y,z
313 prob2int = 2.;
314 } else {
315 // get local material at this point, to calculate probability that
316 // interaction is at this point.
317 TGeoNode* node = gGeoManager->FindNode(x, y, z);
318 TGeoMaterial* mat = nullptr;
319 if (node && !gGeoManager->IsOutside()) {
320 mat = node->GetVolume()->GetMaterial();
321 // relative to Prob largest density along this trajectory, i.e. use
322 // rho(Pt)
323 prob2int = mat->GetDensity() / mparam[7];
324 if (prob2int > 1.)
325 LOG(warning) << " GenieGenerator: prob2int > Maximum density????"
326 << prob2int << " maxrho:" << mparam[7]
327 << " material: " << mat->GetName() << endl;
328 } else {
329 prob2int = 0.;
330 }
331 }
332 }
333
334 Double_t zrelative = z - ztarget;
335 Double_t tof = TMath::Sqrt(x * x + y * y + zrelative * zrelative) /
336 2.99792458e+10; // speed of light in cm/s
337 cpg->AddTrack(
338 neu, pout[0], pout[1], pout[2], x, y, z, -1, false,
339 TMath::Sqrt(pout[0] * pout[0] + pout[1] * pout[1] + pout[2] * pout[2]),
340 tof, mparam[0] * mparam[4]);
341 IncrementCounter("generated_events");
342 if (cc) IncrementCounter("cc_events");
343 if (nuel) IncrementCounter("nue_elastic_events");
344 if (!fNuOnly) {
345 // second, outgoing lepton
346 std::vector<double> pp = Rotate(x, y, zrelative, pxl, pyl, pzl);
347 Int_t oLPdgCode = neu;
348 if (cc) {
349 oLPdgCode = copysign(TMath::Abs(neu) - 1, neu);
350 }
351 if (nuel) {
352 oLPdgCode = 11;
353 }
354 cpg->AddTrack(oLPdgCode, pp[0], pp[1], pp[2], x, y, z, 0, true, El, tof,
355 mparam[0] * mparam[4]);
356 IncrementCounter("outgoing_leptons_stored");
357 // last, all others
358 for (int i = 0; i < nf; i++) {
359 pp = Rotate(x, y, zrelative, pxf[i], pyf[i], pzf[i]);
360 cpg->AddTrack(pdgf[i], pp[0], pp[1], pp[2], x, y, z, 0, true, Ef[i], tof,
361 mparam[0] * mparam[4]);
362 IncrementCounter("outgoing_hadrons_stored");
363 // cout << "f " << pdgf[i] << " pz "<< pzf[i] << endl;
364 }
365 // cout << "Info GenieGenerator Return from GenieGenerator" << endl;
366 }
367 return kTRUE;
368}
