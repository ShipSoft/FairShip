25 def digitize(self):
26 """Digitize SND/MTC MC hits.
27
28 Example of fiberID: 123051820, where:
29 - 1: MTC unique ID
30 - 23: layer number
31 - 0: station type (0 for +5 degrees, 1 for -5 degrees, 2 for scint plane)
32 - 5: z-layer number (0-5)
33 - 1820: local fibre ID within the station
34 Example of SiPM global channel (what is seen in the output file): 123001123, where:
35 - 1: MTC unique ID
36 - 23: layer number
37 - 0: station type (0 for +5 degrees, 1 for -5 degrees)
38 - 0: mat number (only 0 by June 2025). In future, if multiple mats per station are used,
39 this number will be 0-N. Currently, this digit is allocated by SiPM channel ID!
40 - 1: SiPM number (automatically assigned based on fibre aggregation settings)
41 - 123: number of the SiPM channel (0-N). The channel number depends on the fibre aggregation setting.
42 """
43 hit_container = {}
44 mc_points = {}
45 norm = {}
46 for k, mc_point in enumerate(self.intree.MTCDetPoint):
47 det_id = mc_point.GetDetectorID()
48 station_type = mc_point.GetStationType() # 0 for +5 degrees, 1 for -5 degrees, 2 for scint plane, extraction: int(fDetectorID / 100000) % 10
49 energy_loss = mc_point.GetEnergyLoss()
50
51 if station_type == 0:
52 # +5 degrees fiber station uses U fibers
53 fibre_map = self.sipm_to_fibre_map_U
54 elif station_type == 1:
55 # -5 degrees fiber station uses V fibers
56 fibre_map = self.sipm_to_fibre_map_V
57 elif station_type == 2:
58 # Scint Plane. Preserve the same logic as for fibre stations,
59 # but use the det_id directly as the global channel.
60 global_channel = det_id
61 if global_channel not in hit_container:
62 hit_container[global_channel] = []
63 mc_points[global_channel] = {}
64 norm[global_channel] = 0
65
66 weight = 1
67 # Append (energy_loss, weight) instead of (mc_point, weight)
68 hit_container[global_channel].append([mc_point, weight])
69 d_e = energy_loss * weight
70 mc_points[global_channel][k] = d_e
71 norm[global_channel] += d_e
72 continue
73 else:
74 # Skip any other station_type values
75 continue
76 # For station_type 0 or 1, look up the local fibre ID
77 loc_fibre_id = det_id % 1_000_000
78 if loc_fibre_id not in fibre_map:
79 # If there is no entry for this fibre ID, skip
80 print(
81 f"MTC digitization: no mapping found for fibre ID {loc_fibre_id} in station type {station_type}. Skipping."
82 )
83 continue
84
85 for sipm_chan, chan_info in fibre_map[loc_fibre_id].items():
86 global_channel = (det_id // 1_000_000) * 1_000_000 + sipm_chan
87 if global_channel not in hit_container:
88 hit_container[global_channel] = []
89 mc_points[global_channel] = {}
90 norm[global_channel] = 0
91
92 weight = chan_info["weight"]
93 hit_container[global_channel].append([mc_point, weight])
94 d_e = energy_loss * weight
95 mc_points[global_channel][k] = d_e
96 norm[global_channel] += d_e
97
98 for det_id in hit_container:
99 all_points = ROOT.std.vector("MTCDetPoint*")()
100 all_weights = ROOT.std.vector("Float_t")()
101
102 for entry in hit_container[det_id]:
103 all_points.push_back(entry[0])
104 all_weights.push_back(entry[1])
105 det_hit = ROOT.MTCDetHit(det_id, all_points, all_weights)
106 self.det.push_back(det_hit)
