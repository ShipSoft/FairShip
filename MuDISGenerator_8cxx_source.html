37Bool_t MuDISGenerator::Init(const char* fileName, const int firstEvent)
38{
39 LOGF(info, "Opening input file %s", fileName);
40
41 iMuon = 0;
42 dPart = 0;
43 dPartSoft = 0;
44 fInputFile = TFile::Open(fileName);
45 if (!fInputFile) {
46 LOG(fatal) << "Error opening input file";
47 return kFALSE;
48 }
49 fTree = fInputFile->Get<TTree>("DIS");
50 fNevents = fTree->GetEntries();
51 fn = firstEvent;
52
53 fTree->SetBranchAddress("InMuon", &iMuon); // incoming muon
54 fTree->SetBranchAddress("DISParticles", &dPart);
55 fTree->SetBranchAddress("SoftParticles", &dPartSoft); // Soft interaction particles
56 LOG(info) << "MuDISGenerator: Initialization successful.";
57 return kTRUE;
58}

68Bool_t MuDISGenerator::ReadEvent(FairPrimaryGenerator* cpg)
69{
70 if (fn == fNevents) {
71 LOG(warning) << "End of input file. Rewind.";
72 }
73 fTree->GetEntry(fn % fNevents);
74 fn++;
75 if (fn % 10 == 0) {
76 LOG(info) << "Info MuDISGenerator: MuDIS event-nr " << fn;
77 }
78
79 int nf = dPart->GetEntries();
80 LOG(debug) << "*********************************************************";
81 LOG(debug) << "muon DIS Generator debug " << iMuon->GetEntries() << " " << iMuon->AddrAt(0) << " nf " << nf
82 << " fn=" << fn;
83
84 // some start/end positions in z (f.i. emulsion to Tracker 1)
85 Double_t start[3] = {0., 0., startZ};
86 Double_t end[3] = {0., 0., endZ};
87
88 // incoming muon array('d',[pid,px,py,pz,E,x,y,z,w,t])
89 TVectorD* mu = dynamic_cast<TVectorD*>(iMuon->AddrAt(0));
90 LOG(debug) << "muon DIS Generator in muon " << int(mu[0][0]);
91 Double_t x = mu[0][5] * 100.; // come in m -> cm
92 Double_t y = mu[0][6] * 100.; // come in m -> cm
93 Double_t z = mu[0][7] * 100.; // come in m -> cm
94 Double_t w = mu[0][8]; // weight of the original muon ( normalised to a spill)
95 Double_t cross_sec = mu[0][10]; // in mbarns
96 Double_t t_muon = mu[0][11]; // in ns
97 Double_t DIS_multiplicity = 1 / mu[0][12]; // 1/nDIS
98
99 // calculate start/end positions along this muon, and amount of material in between
100
101 Double_t txmu = mu[0][1] / mu[0][3];
102 Double_t tymu = mu[0][2] / mu[0][3];
103 start[0] = x - (z - start[2]) * txmu;
104 start[1] = y - (z - start[2]) * tymu;
105 end[0] = x - (z - end[2]) * txmu;
106 end[1] = y - (z - end[2]) * tymu;
107 LOG(debug) << "MuDIS: mu xyz position " << x << ", " << y << ", " << z;
108 LOG(debug) << "MuDIS: mu pxyz position " << mu[0][1] << ", " << mu[0][2] << ", " << mu[0][3];
109 LOG(debug) << "MuDIS: mu weight*DISmultiplicity " << w;
110 LOG(debug) << "MuDIS: mu DIS cross section " << cross_sec;
111 LOG(debug) << "MuDIS: start position " << start[0] << ", " << start[1] << ", " << start[2];
112 LOG(debug) << "MuDIS: end position " << end[0] << ", " << end[1] << ", " << end[2];
113
114 Double_t bparam;
115 Double_t mparam[10];
116 bparam = shipgen::MeanMaterialBudget(start, end, mparam);
117 // loop over trajectory between start and end to pick an interaction point
118 Double_t prob2int = 0.;
119 Double_t xmu = 0.;
120 Double_t ymu = 0.;
121 Double_t zmu = 0.;
122 Int_t count = 0;
123 LOG(debug) << "Info MuDISGenerator Start prob2int while loop, bparam= " << bparam << ", " << bparam * 1.e8;
124 LOG(debug) << "Info MuDISGenerator What was maximum density, mparam[7]= " << mparam[7] << ", " << mparam[7] * 1.e8;
125
126 while (prob2int < gRandom->Uniform(0., 1.)) {
127 zmu = gRandom->Uniform(start[2], end[2]);
128 xmu = x - (z - zmu) * txmu;
129 ymu = y - (z - zmu) * tymu;
130 // get local material at this point
131 TGeoNode* node = gGeoManager->FindNode(xmu, ymu, zmu);
132 TGeoMaterial* mat = 0;
133 if (node && !gGeoManager->IsOutside())
134 mat = node->GetVolume()->GetMaterial();
135 LOG(debug) << "Info MuDISGenerator: mat " << count << ", " << mat->GetName() << ", " << mat->GetDensity();
136 // density relative to Prob largest density along this trajectory, i.e. use rho(Pt)
137 prob2int = mat->GetDensity() / mparam[7];
138 if (prob2int > 1.) {
139 LOG(warning) << "MuDISGenerator: prob2int > Maximum density????";
140 LOG(warning) << "prob2int: " << prob2int;
141 LOG(warning) << "maxrho: " << mparam[7];
142 LOG(warning) << "material: " << mat->GetName();
143 }
144 count += 1;
145 }
146
147 LOG(debug) << "Info MuDISGenerator: prob2int " << prob2int << ", " << count;
148 LOG(debug) << "MuDIS: put position " << xmu << ", " << ymu << ", " << zmu;
149
150 Double_t total_mom =
151 TMath::Sqrt(TMath::Power(mu[0][1], 2) + TMath::Power(mu[0][2], 2) + TMath::Power(mu[0][3], 2)); // in GeV
152
153 Double_t distance =
154 TMath::Sqrt(TMath::Power(x - xmu, 2) + TMath::Power(y - ymu, 2) + TMath::Power(z - zmu, 2)); // in cm
155
156 Double_t v = c_light * total_mom / TMath::Sqrt(TMath::Power(total_mom, 2) + TMath::Power(muon_mass, 2));
157 Double_t t_rmu = distance / v;
158
159 // Adjust time based on the relative positions
160 if (zmu < z) {
161 t_rmu = -t_rmu;
162 }
163
164 Double_t t_DIS = (t_muon + t_rmu) / 1e9; // time taken in seconds to reach [xmu,ymu,zmu]
165
166 cpg->AddTrack(static_cast<int>(mu[0][0]), // incoming muon track ()
167 mu[0][1],
168 mu[0][2],
169 mu[0][3],
170 xmu,
171 ymu,
172 zmu,
173 -1,
174 false, // tracking disabled
175 mu[0][4],
176 t_DIS, // shift time of the incoming muon track wrt t_muon from the input file.
177 w * DIS_multiplicity); // muon weight associated with a spill* DISmultiplicity
178
179 // outgoing DIS particles, [did,dpx,dpy,dpz,E], put density along trajectory as weight, g/cm^2
180
181 w = mparam[0] * mparam[4]; // modify weight, by multiplying with average density * track length
182 int index = 0;
183 for (auto&& particle : *dPart) {
184 TVectorD* Part = dynamic_cast<TVectorD*>(particle);
185 LOG(debug) << "muon DIS Generator out part " << int((*Part)[0]);
186 LOG(debug) << "muon DIS Generator out part mom " << (*Part)[1] << " " << (*Part)[2] << " " << (*Part)[3] << " "
187 << (*Part)[4];
188 LOG(debug) << "muon DIS Generator out part pos " << xmu << " " << ymu << "" << zmu;
189 LOG(debug) << "muon DIS Generator out part w " << w;
190
191 if (index == 0) {
192 cpg->AddTrack(static_cast<int>((*Part)[0]),
193 (*Part)[1],
194 (*Part)[2],
195 (*Part)[3],
196 xmu,
197 ymu,
198 zmu,
199 0,
200 true,
201 (*Part)[4],
202 t_DIS,
203 cross_sec); // save DIS cross section in MCTrack[1]
204 } else {
205 cpg->AddTrack(static_cast<int>((*Part)[0]),
206 (*Part)[1],
207 (*Part)[2],
208 (*Part)[3],
209 xmu,
210 ymu,
211 zmu,
212 0,
213 true,
214 (*Part)[4],
215 t_DIS,
216 w);
217 }
218 index += 1;
219 }
220
221 // Soft interaction tracks
222 for (auto&& softParticle : *dPartSoft) {
223 TVectorD* SoftPart = dynamic_cast<TVectorD*>(softParticle);
224 if ((*SoftPart)[7] > zmu) {
225 continue;
226 } // Soft interactions after the DIS point are not saved
227 Double_t t_soft = (*SoftPart)[8] / 1e9; // Time in seconds
228 cpg->AddTrack(static_cast<int>((*SoftPart)[0]),
229 (*SoftPart)[1],
230 (*SoftPart)[2],
231 (*SoftPart)[3],
232 (*SoftPart)[5],
233 (*SoftPart)[6],
234 (*SoftPart)[7],
235 0,
236 true,
237 (*SoftPart)[4],
238 t_soft,
239 w);
240 }
241
242 return kTRUE;
243}
