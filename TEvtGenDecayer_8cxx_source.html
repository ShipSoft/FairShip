56//_____________________________________________________________________________
57void TEvtGenDecayer::Init()
58{
59 // Initialize both EvtGen and Pythia8 decayers
60
61 // Initialize fallback Pythia8 decayer
62 fPythia8Decayer = new TPythia8Decayer();
63 fPythia8Decayer->Init();
64
65 // Initialize EvtGen if decay files are specified
66 if (fDecayFile != "" && fParticleFile != "") {
67 // Set up EvtGen random engine with ROOT's random number generator seed
68 UInt_t seed = gRandom->GetSeed();
69
70 // Also seed Pythia8 with the same seed for consistency
71 TPythia8* pythia8 = TPythia8::Instance();
72 if (pythia8) {
73 pythia8->Pythia8()->readString(Form("Random:seed = %u", seed));
74 pythia8->Pythia8()->readString("Random:setSeed = on");
75 pythia8->Pythia8()->init(); // Re-initialize with new seed
76 LOG(debug) << "TEvtGenDecayer: Pythia8 RNG seeded with " << seed;
77 }
78
79 EvtRandomEngine* randomEngine = new EvtMTRandomEngine(seed);
80 EvtRandom::setRandomEngine(randomEngine);
81
82 LOG(info) << "TEvtGenDecayer: Random engine initialized with seed " << seed;
83
84 // Set up external generators with PHOTOS radiative corrections
85 std::string photonType("gamma");
86 bool useEvtGenRandom(true);
87 bool convertPythiaCodes(false);
88 std::string pythiaDir("");
89
90 EvtExternalGenList* externalGenList = new EvtExternalGenList(convertPythiaCodes, pythiaDir, photonType, useEvtGenRandom);
91
92 // Get the interface to the radiative correction engine (PHOTOS)
93 EvtAbsRadCorr* radCorrEngine = externalGenList->getPhotosModel();
94 if (radCorrEngine) {
95 LOG(info) << "TEvtGenDecayer: PHOTOS radiative corrections enabled";
96 } else {
97 LOG(warning) << "TEvtGenDecayer: PHOTOS not available, proceeding without radiative corrections";
98 }
99
100 // Get the interface to other external generators
101 std::list<EvtDecayBase*> extraModels = externalGenList->getListOfModels();
102
103 // Create EvtGen instance
104 fEvtGen = new EvtGen(fDecayFile.Data(), fParticleFile.Data(),
105 randomEngine, radCorrEngine, &extraModels, 1, false);
106
107 LOG(debug) << "TEvtGenDecayer: EvtGen created with " << EvtPDL::entries() << " particle entries";
108
109 LOG(info) << "TEvtGenDecayer: EvtGen initialized with decay file: " << fDecayFile.Data()
110 << ", particle file: " << fParticleFile.Data();
111 } else {
112 LOG(debug) << "TEvtGenDecayer: No EvtGen files specified, using Pythia8 only";
113 }
114
115 // Log summary at end of Init
116 LOG(info) << "TEvtGenDecayer initialized - EvtGen: " << (fEvtGen ? "YES" : "NO")
117 << ", Pythia8 fallback: " << (fPythia8Decayer ? "YES" : "NO")
118 << ", Particles configured for EvtGen: " << fEvtGenParticles.size();
119
120 for (size_t i = 0; i < fEvtGenParticles.size(); i++) {
121 LOG(debug) << " EvtGen particle: PDG " << fEvtGenParticles[i];
122 }
123}

125//_____________________________________________________________________________
126void TEvtGenDecayer::Decay(Int_t pdg, TLorentzVector* p)
127{
128 // Main decay method - route to EvtGen or Pythia8 based on particle type
129
130 // Debug output for decay calls (highest verbosity)
131 LOG(debug2) << "TEvtGenDecayer::Decay called for PDG " << pdg
132 << " (E=" << p->E() << ", p=" << p->P() << ")";
133
134
135 // Check if this particle should use EvtGen
136 Bool_t useEvtGen = UseEvtGenForParticle(pdg);
137 Bool_t evtGenAvailable = (fEvtGen != nullptr);
138
139 LOG(debug1) << "UseEvtGenForParticle(" << pdg << ") = " << useEvtGen
140 << ", EvtGen available = " << evtGenAvailable;
141
142 if (useEvtGen && evtGenAvailable) {
143 LOG(debug) << "Routing PDG " << pdg << " to EvtGen";
144 DecayWithEvtGen(pdg, p);
145 } else {
146 LOG(debug) << "Routing PDG " << pdg << " to Pythia8 fallback";
147 DecayWithPythia8(pdg, p);
148 }
149}

151//_____________________________________________________________________________
152Bool_t TEvtGenDecayer::UseEvtGenForParticle(Int_t pdg)
153{
154 // Check if this particle should be handled by EvtGen
155 LOG(debug2) << "UseEvtGenForParticle: Checking PDG " << pdg << " against EvtGen particle list";
156
157 for (Int_t evtGenPdg : fEvtGenParticles) {
158 if (pdg == evtGenPdg) {
159 LOG(debug2) << "UseEvtGenForParticle: Match found - PDG " << pdg << " should use EvtGen";
160 return kTRUE;
161 }
162 }
163
164 LOG(warning) << "UseEvtGenForParticle: PDG " << pdg << " not in EvtGen particle list, falling back to Pythia8";
165 return kFALSE;
166}

168//_____________________________________________________________________________
169void TEvtGenDecayer::DecayWithEvtGen(Int_t pdg, TLorentzVector* p)
170{
171 // Decay particle using EvtGen
172
173 if (!fEvtGen) {
174 // Fall back to Pythia8 if EvtGen not available
175 DecayWithPythia8(pdg, p);
176 return;
177 }
178
179 LOG(debug1) << "DecayWithEvtGen: Decaying PDG " << pdg << " with EvtGen";
180
181 try {
182 // Convert TLorentzVector to EvtGen format
183 EvtVector4R p4(p->E(), p->Px(), p->Py(), p->Pz());
184
185 // Get EvtGen particle ID from PDG code
186 EvtId evtId = EvtPDL::evtIdFromStdHep(pdg);
187
188 // Check if EvtId is valid before creating particle
189 if (evtId.getId() < 0 || static_cast<size_t>(evtId.getId()) >= EvtPDL::entries()) {
190 LOG(warning) << "TEvtGenDecayer: Invalid EvtId for PDG " << pdg
191 << " (EvtId: " << evtId.getId() << "), falling back to Pythia8";
192 DecayWithPythia8(pdg, p);
193 return;
194 }
195
196 // Create EvtGen particle
197 EvtParticle* parent = EvtParticleFactory::particleFactory(evtId, p4);
198
199 if (parent) {
200 // Generate decay
201 fEvtGen->generateDecay(parent);
202
203 // Convert EvtGen decay products to ROOT format
204 ConvertEvtGenDecay(parent);
205
206 LOG(debug1) << "TEvtGenDecayer: EvtGen decay products converted and stored";
207
208 // Clean up
209 parent->deleteTree();
210 } else {
211 LOG(warning) << "TEvtGenDecayer: Could not create EvtParticle for PDG " << pdg;
212 // Fall back to Pythia8
213 DecayWithPythia8(pdg, p);
214 }
215 } catch (const std::exception& e) {
216 LOG(error) << "TEvtGenDecayer: EvtGen exception for PDG " << pdg << ": " << e.what();
217 // Fall back to Pythia8 on any exception
218 DecayWithPythia8(pdg, p);
219 }
220}

222//_____________________________________________________________________________
223void TEvtGenDecayer::ConvertEvtGenDecay(EvtParticle* parent)
224{
225 // Convert EvtGen decay products to ROOT/GEANT4 format
226 // Clear previous decay products
227 fEvtGenProducts->Clear();
228 fHasEvtGenDecay = kTRUE;
229
230 int nDaughters = parent->getNDaug();
231 LOG(debug2) << "ConvertEvtGenDecay: Converting " << nDaughters << " daughter particles";
232
233 // Convert each daughter particle to TParticle format
234 for (int i = 0; i < nDaughters; i++) {
235 EvtParticle* daughter = parent->getDaug(i);
236 EvtVector4R p4 = daughter->getP4();
237 int pdgId = EvtPDL::getStdHep(daughter->getId());
238
239 // Create TParticle (ROOT format)
240 TParticle* particle = new ((*fEvtGenProducts)[i]) TParticle();
241 particle->SetPdgCode(pdgId);
242 particle->SetMomentum(p4.get(1), p4.get(2), p4.get(3), p4.get(0)); // px, py, pz, E
243 particle->SetProductionVertex(0, 0, 0, 0); // Produced at origin
244 particle->SetStatusCode(1); // Stable particle
245
246 LOG(debug2) << " Daughter " << i << ": PDG=" << pdgId
247 << ", E=" << p4.get(0) << ", px=" << p4.get(1)
248 << ", py=" << p4.get(2) << ", pz=" << p4.get(3);
249 }
250
251 LOG(debug1) << "ConvertEvtGenDecay: Stored " << fEvtGenProducts->GetEntriesFast() << " particles";
252}

268//_____________________________________________________________________________
269Int_t TEvtGenDecayer::ImportParticles(TClonesArray *particles)
270{
271 // Import particles - return EvtGen products if available, otherwise delegate to Pythia8
272 if (fHasEvtGenDecay && fEvtGenProducts) {
273 LOG(debug1) << "ImportParticles: Returning " << fEvtGenProducts->GetEntriesFast() << " EvtGen particles";
274
275 // Copy EvtGen particles to the provided array
276 particles->Clear();
277 int nParticles = fEvtGenProducts->GetEntriesFast();
278 for (int i = 0; i < nParticles; i++) {
279 TParticle* srcParticle = static_cast<TParticle*>(fEvtGenProducts->At(i));
280 new ((*particles)[i]) TParticle(*srcParticle);
281 }
282
283 // Reset the flag for next decay
284 fHasEvtGenDecay = kFALSE;
285 return nParticles;
286 } else {
287 // Fall back to Pythia8
288 LOG(debug1) << "ImportParticles: Delegating to Pythia8";
289 if (fPythia8Decayer) {
290 return fPythia8Decayer->ImportParticles(particles);
291 }
292 return 0;
293 }
294}
