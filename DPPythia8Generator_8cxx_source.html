52Bool_t DPPythia8Generator::Init() {
53 if (fUseRandom1) fRandomEngine = std::make_shared<PyTr1Rng>();
54 if (fUseRandom3) fRandomEngine = std::make_shared<PyTr3Rng>();
55 fPythia->setRndmEnginePtr(fRandomEngine);
56 // fPythiaHadDecay->setRndmEnginePtr(fRandomEngine);
57 fn = 0;
58
59 if (!fpbrem) {
60 if (debug) {
61 std::cout << "Beam Momentum " << fMom << std::endl;
62 }
63 fPythia->settings.mode("Beams:idA", fId);
64 fPythia->settings.mode("Beams:idB", 2212);
65 fPythia->settings.mode("Beams:frameType", 2);
66 fPythia->settings.parm("Beams:eA", fMom); // codespell:ignore parm
67 fPythia->settings.parm("Beams:eB", 0.); // codespell:ignore parm
68
69 if (fdy)
70 fPythia->settings.parm("PhaseSpace:mHatMin", // codespell:ignore parm
71 fDPminM);
72
73 } else {
74 if (!fpbremPDF) {
75 LOG(fatal) << "Failed in retrieving dark photon PDF for production by "
76 "proton bremstrahlung!";
77 return kFALSE;
78 }
79 }
80
81 TDatabasePDG* pdgBase = TDatabasePDG::Instance();
82 Double_t root_ctau = pdgBase->GetParticle(fDP)->Lifetime();
83
84 if (debug) {
85 std::cout << "Final particle parameters for PDGID " << fDP << ":"
86 << std::endl;
87 List(fDP);
88 }
89 if (debug) {
90 std::cout << "tau root PDG database " << root_ctau
91 << "[s] ctau root = " << root_ctau * 3e10 << "[cm]" << std::endl;
92 }
93 fctau = fPythia->particleData.tau0(fDP); //* 3.3333e-12
94 if (debug) {
95 std::cout << "ctau pythia " << fctau << "[mm]" << std::endl;
96 }
97 int initPass = fPythia->init();
98 if (debug) {
99 std::cout << "Pythia initialisation bool: " << initPass << std::endl;
100 }
101
102 if (!initPass) {
103 LOG(fatal) << "Pythia initialisation failed";
104 return kFALSE;
105 }
106
107 return kTRUE;
108}

118// ----- Passing the event ---------------------------------------------
119Bool_t DPPythia8Generator::ReadEvent(FairPrimaryGenerator* cpg) {
120 auto [dx, dy] = CalculateBeamOffset(fsmearBeam, fPaintBeam);
121 Double_t tp, tS, zp, xp, yp, zS, xS, yS, pz, px, py, e, w;
122 Double_t tm, zm, xm, ym, pmz, pmx, pmy, em;
123 Int_t im;
124 Int_t imout;
125 Double_t tN, zN, xN, yN, pzN, pxN, pyN, eN;
126
127 int iDP =
128 0; // index of the chosen DP, also ensures that at least 1 DP is produced
129 // Pythia indices of the particles stored on the FairShip MCTrack stack.
130 // In this implementation the exported chain is:
131 // [Pythia event[1] beam particle, mother meson, dark photon, daughters...]
132 // Pythia event[0] is the internal system entry and is not exported.
133 std::vector<int>
134 dec_chain; // pythia indices of the particles to be stored on the stack
135 std::vector<int> dpvec; // pythia indices of DP particles
136
137 do {
138 // clear containers at every retry to avoid mixing different events
139 dec_chain.clear();
140 dpvec.clear();
141
142 // bit for proton brem
143 if (fpbrem) {
144 fPythia->event.reset();
145 double dpmom = 0;
146 double thetain = 0;
147 fpbremPDF->GetRandom2(dpmom, thetain);
148 double dpm = fPythia->particleData.m0(fDP);
149 double dpe = sqrt(dpmom * dpmom + dpm * dpm);
150 double phiin = 2. * M_PI * gRandom->Rndm();
151
152 if (debug > 1) {
153 std::cout << " Adding DP gun with p " << dpmom << " m " << dpm << " e "
154 << dpe << " theta,phi " << thetain << "," << phiin
155 << std::endl
156 << std::flush;
157 }
158 fPythia->event.append(fDP, 1, 0, 0, dpmom * sin(thetain) * cos(phiin),
159 dpmom * sin(thetain) * sin(phiin),
160 dpmom * cos(thetain), dpe, dpm);
161 }
162
163 if (!fPythia->next()) LOG(fatal) << "fPythia->next() failed";
164
165 // fPythia->event.list();
166 for (int i = 0; i < fPythia->event.size(); i++) {
167 // find all DP
168 if (abs(fPythia->event[i].id()) == fDP) {
169 dpvec.push_back(i);
170 }
171 }
172 iDP = dpvec.size();
173 fnDPtot += iDP;
174 if (iDP == 0) {
175 fnRetries += 1;
176 } else {
177 // filter DPs by vessel acceptance
178 std::vector<int> accepted;
179 for (int idx : dpvec) {
180 if (IsInVesselAcceptance(fPythia->event[idx].px(),
181 fPythia->event[idx].py(),
182 fPythia->event[idx].pz())) {
183 accepted.push_back(idx);
184 }
185 }
186 if (accepted.empty()) {
187 iDP = 0;
188 dpvec.clear();
189 fnGeoRejects += 1;
190 continue;
191 }
192 int r = static_cast<int>(gRandom->Uniform(0, accepted.size()));
193 int i = accepted[r];
194 // production vertex
195 zp = fPythia->event[i].zProd();
196 xp = fPythia->event[i].xProd();
197 yp = fPythia->event[i].yProd();
198 tp = fPythia->event[i].tProd();
199 // momentum
200 pz = fPythia->event[i].pz();
201 px = fPythia->event[i].px();
202 py = fPythia->event[i].py();
203 e = fPythia->event[i].e();
204 // old decay vertex
205 if (debug > 1) {
206 int id1 = fPythia->event[i].daughter1();
207 int id2 = fPythia->event[i].daughter2();
208 std::cout << " Debug: decay product of A: ";
209 if (id1 > 0)
210 std::cout << fPythia->event[id1].id() << " ";
211 else
212 std::cout << "none ";
213 if (id2 > 0)
214 std::cout << fPythia->event[id2].id();
215 else
216 std::cout << "none";
217 std::cout << std::endl;
218 }
219
220 // new decay vertex
221 Double_t LS = gRandom->Uniform(fLmin / mm, fLmax / mm); // in mm
222 Double_t p = TMath::Sqrt(px * px + py * py + pz * pz);
223 Double_t lam = LS / p;
224 xS = xp + lam * px;
225 yS = yp + lam * py;
226 zS = zp + lam * pz;
227 Double_t gam = e / TMath::Sqrt(e * e - p * p);
228 Double_t beta = p / e;
229 tS = tp + LS / beta; // all in Pythia units
230 w = TMath::Exp(-LS / (beta * gam * fctau)) *
231 ((fLmax / mm - fLmin / mm) / (beta * gam * fctau));
232
233 // direct mother of the DP
234 im = (Int_t)fPythia->event[i].mother1();
235
236 if (!fpbrem && im > 0) {
237 // Meson-production chain: store event[1] (beam nucleon), mother, DP
238 if (fPythia->event.size() > 1) {
239 zN = fPythia->event[1].zProd();
240 xN = fPythia->event[1].xProd();
241 yN = fPythia->event[1].yProd();
242 pzN = fPythia->event[1].pz();
243 pxN = fPythia->event[1].px();
244 pyN = fPythia->event[1].py();
245 eN = fPythia->event[1].e();
246 tN = fPythia->event[1].tProd();
247
248 cpg->AddTrack((Int_t)fPythia->event[1].id(), pxN, pyN, pzN,
249 xN * mm + dx, yN * mm + dy, zN * mm, -1, false, eN,
250 tN * mm / c_light,
251 w); // event[0] is the root of the exported chain
252
253 dec_chain.push_back(1);
254
255 if (debug > 1)
256 std::cout << std::endl
257 << " insert nucleon ancestor id 0"
258 << " pdg=" << fPythia->event[1].id() << " pz = " << pzN
259 << " [GeV], z = " << zN << " [mm] t = " << tN << " [mm/c]"
260 << std::endl;
261 }
262
263 zm = fPythia->event[im].zProd();
264 xm = fPythia->event[im].xProd();
265 ym = fPythia->event[im].yProd();
266 pmz = fPythia->event[im].pz();
267 pmx = fPythia->event[im].px();
268 pmy = fPythia->event[im].py();
269 em = fPythia->event[im].e();
270 tm = fPythia->event[im].tProd();
271
272 // mother points to track 0 (event[1]) in the exported chain
273 imout = 0;
274
275 cpg->AddTrack(
276 (Int_t)fPythia->event[im].id(), pmx, pmy, pmz, xm * mm + dx,
277 ym * mm + dy, zm * mm, imout, false, em, tm * mm / c_light,
278 w); // convert pythia's (x,y,z[mm], t[mm/c]) to ([cm], [s])
279
280 // DP points to the direct mother in the exported chain, which is track
281 // 1
282 cpg->AddTrack(fDP, px, py, pz, xp * mm + dx, yp * mm + dy, zp * mm, 1,
283 false, e, tp * mm / c_light, w);
284
285 // bookkeep the indices of stored particles
286 dec_chain.push_back(im);
287 dec_chain.push_back(i);
288
289 if (debug > 1)
290 std::cout << std::endl
291 << " insert mother id " << im
292 << " pdg=" << fPythia->event[im].id() << " pmz = " << pmz
293 << " [GeV], zm = " << zm << " [mm] tm = " << tm
294 << " [mm/c]" << std::endl;
295 if (debug > 1)
296 std::cout << " ----> insert DP id " << i << " pdg=" << fDP
297 << " pz = " << pz << " [GeV] zp = " << zp
298 << " [mm] tp = " << tp << " [mm/c]" << std::endl;
299 } else {
300 // Proton bremsstrahlung or no valid mother: export only the DP as root
301 cpg->AddTrack(fDP, px, py, pz, xp * mm + dx, yp * mm + dy, zp * mm, -1,
302 false, e, tp * mm / c_light, w);
303 dec_chain.push_back(i);
304
305 if (debug > 1)
306 std::cout << " ----> insert DP (root) id " << i << " pdg=" << fDP
307 << " pz = " << pz << " [GeV] zp = " << zp
308 << " [mm] tp = " << tp << " [mm/c]" << std::endl;
309 }
310 iDP = i;
311 }
312 } while (iDP ==
313 0); // ----------- avoid rare empty events w/o any DP's produced
314
315 if (fShipEventNr % 100 == 0) {
316 LOGF(info, "ship event %i / pythia event-nr %i", fShipEventNr, fn);
317 }
318 fShipEventNr += 1;
319 // number of prefix entries (nucleon + mother + DP, or just DP for fpbrem)
320 size_t prefix_len = dec_chain.size();
321 // fill a container with pythia indices of the DP decay chain
322 if (debug > 1) std::cout << "Filling daughter particles" << std::endl;
323 // if (!hadDecay){
324 for (int k = 0; k < fPythia->event.size(); k++) {
325 // if daughter of DP, copy
326 if (debug > 1)
327 std::cout << k << " pdg =" << fPythia->event[k].id() << " mum "
328 << fPythia->event[k].mother1() << std::endl;
329 im = fPythia->event[k].mother1();
330 while (im > 0) {
331 if (im == iDP) {
332 break;
333 } // pick the decay products of only 1 chosen DP
334 // if ( abs(fPythia->event[im].id())==fDP && im == iDP ){break;}
335 else {
336 im = fPythia->event[im].mother1();
337 }
338 }
339 if (im < 1) {
340 if (debug > 1) std::cout << "reject" << std::endl;
341 continue;
342 }
343 if (debug > 1) std::cout << "accept" << std::endl;
344 dec_chain.push_back(k);
345 }
346
347 // go over daughters and store them on the stack
348 // dec_chain now contains prefix entries followed by daughters
349 for (std::vector<int>::iterator it = dec_chain.begin() + prefix_len;
350 it != dec_chain.end(); ++it) {
351 // pythia index of the particle to store
352 int k = *it;
353 // find mother position on the output stack: impy -> im
354 int impy = fPythia->event[k].mother1();
355 std::vector<int>::iterator itm =
356 std::find(dec_chain.begin(), dec_chain.end(), impy);
357 im = -1; // safety
358 if (itm != dec_chain.end()) {
359 im = itm - dec_chain.begin(); // convert iterator into sequence number
360 }
361
362 Bool_t wanttracking = false;
363 if (fPythia->event[k].isFinal()) {
364 wanttracking = true;
365 }
366 pz = fPythia->event[k].pz();
367 px = fPythia->event[k].px();
368 py = fPythia->event[k].py();
369 e = fPythia->event[k].e();
370
371 cpg->AddTrack((Int_t)fPythia->event[k].id(), px, py, pz, xS * mm, yS * mm,
372 zS * mm, im, wanttracking, e, tS * mm / c_light, w);
373 }
374 return kTRUE;
375}
