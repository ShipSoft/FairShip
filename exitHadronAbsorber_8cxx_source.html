68Bool_t exitHadronAbsorber::ProcessHits(FairVolume* vol) {
70 TString volName = gMC->CurrentVolName();
71
72 if (volName.Contains("exitHadronAbsorber")) {
73 if (gMC->IsTrackEntering()) {
74 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
75 fEventID = gMC->CurrentEvent();
76 TParticle* p = gMC->GetStack()->GetCurrentTrack();
77 fUniqueID = p->GetUniqueID();
78 Int_t pdgCode = p->GetPdgCode();
79 Int_t motherId = p->GetFirstMother();
80 gMC->TrackMomentum(fMom);
81 if (!fOnlyMuons || TMath::Abs(pdgCode) == 13) {
82 fTime = gMC->TrackTime() * 1.0e09;
83 fLength = gMC->TrackLength();
84 gMC->TrackPosition(fPos);
85 if (((fMom.E() - fMom.M()) > EMax) &&
86 (fDecayedParentIDs.count(motherId) == 0)) {
87 AddHit(fEventID, fTrackID, 111,
88 TVector3(fPos.X(), fPos.Y(), fPos.Z()),
89 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength, 0,
90 pdgCode, TVector3(p->Vx(), p->Vy(), p->Vz()),
91 TVector3(p->Px(), p->Py(), p->Pz()));
92 auto* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
93 stack->AddPoint(kVETO);
94 }
95 }
96 }
97
98 if ((!fCylindricalPlane) && fzPos > 1E8) {
99 gMC->StopTrack();
100 }
101 }
102
103 if (fNsplits > 0 && (!fSplitOnce)) {
104 Int_t currentTrackId = gMC->GetStack()->GetCurrentTrackNumber();
105
106 if (fCloneTracks.count(currentTrackId) > 0 ||
107 fContinuationTracks.count(currentTrackId) > 0) {
108 return kTRUE;
109 }
110
111 Int_t track_pid = gMC->TrackPid();
112 bool kaon_or_pion =
113 (TMath::Abs(track_pid) == 211 || TMath::Abs(track_pid) == 321);
114
115 if (kaon_or_pion) {
116 TParticle* part = gMC->GetStack()->GetCurrentTrack();
117 if (!part) return kTRUE;
118
119 Double_t delta_s = gMC->TrackStep();
120
121 if (delta_s > 0) {
122 TLorentzVector pos, mom;
123 gMC->TrackPosition(pos);
124 gMC->TrackMomentum(mom);
125
126 Double_t mass = gMC->TrackMass();
127 Double_t tau = gMC->ParticleLifeTime(track_pid); // in nanoseconds
128 Double_t c_tau = tau * 29.9792458; // in cm/ns
129
130 Double_t instantP = mom.P();
131
132 Double_t lambda_decay = (instantP / mass) * c_tau;
133 Double_t P_decay = 1.0 - TMath::Exp(-delta_s / lambda_decay);
134 if ((P_decay > 0.0) && ((mom.E() - mom.M()) > EMax)) {
135 double polX = 0, polY = 0, polZ = 0;
136 TVector3 polVector;
137 part->GetPolarisation(polVector);
138 polX = polVector.X();
139 polY = polVector.Y();
140 polZ = polVector.Z();
141 Int_t trueParentId = part->GetFirstMother();
142
143 Double_t decayBranchWeight = fCurrentSurvivalFactor * P_decay;
144 Double_t cloneWeight = decayBranchWeight / fNsplits;
145 for (int i = 0; i < fNsplits; ++i) {
146 TrackBuffer clone;
147 clone.pdg = track_pid;
148 clone.px = mom.Px();
149 clone.py = mom.Py();
150 clone.pz = mom.Pz();
151 clone.e = mom.E();
152 clone.x = pos.X();
153 clone.y = pos.Y();
154 clone.z = pos.Z();
155 clone.t = pos.T();
156 clone.polx = polX;
157 clone.poly = polY;
158 clone.polz = polZ;
159 clone.weight = cloneWeight;
160 clone.parentID = trueParentId;
161 fSecondaryBuffer.push_back(clone);
162 }
163 fCurrentSurvivalFactor *= (1.0 - P_decay);
164 }
165 }
166 }
167 }
168 return kTRUE;
169}

247void exitHadronAbsorber::PostTrack() {
248 Int_t currentTrackId = gMC->GetStack()->GetCurrentTrackNumber();
249
250 if (fCloneTracks.count(currentTrackId) > 0 ||
251 fContinuationTracks.count(currentTrackId) > 0) {
252 return;
253 }
254
255 Int_t track_pid = gMC->TrackPid();
256 bool kaon_or_pion =
257 (TMath::Abs(track_pid) == 211 || TMath::Abs(track_pid) == 321);
258
259 if (fNsplits > 0 && kaon_or_pion) {
260 bool isNaturalDecay = false;
261 bool isParticleDestroyed = gMC->IsTrackStop() || gMC->IsTrackDisappeared();
262 TArrayI processes;
263 gMC->StepProcesses(processes);
264 for (int i = 0; i < processes.GetSize(); i++) {
265 if (processes[i] == kPDecay) {
266 isNaturalDecay = true;
267 break;
268 }
269 }
270
271 TParticle* part = gMC->GetStack()->GetCurrentTrack();
272 if (!part) return;
273
274 TLorentzVector finalPos, finalMom;
275 gMC->TrackPosition(finalPos);
276 gMC->TrackMomentum(finalMom);
277
278 double polX = 0, polY = 0, polZ = 0;
279 TVector3 polVector;
280 part->GetPolarisation(polVector);
281 polX = polVector.X();
282 polY = polVector.Y();
283 polZ = polVector.Z();
284 Int_t trueParentId = part->GetFirstMother();
285
286 auto* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
287
288 // All remaining weight used if cloning happens at point where original
289 // particle decays
290 if (isNaturalDecay) {
291 Double_t finalEndpointWeight = fCurrentSurvivalFactor / fNsplits;
292 for (int i = 0; i < fNsplits; ++i) {
293 TrackBuffer clone;
294 clone.pdg = track_pid;
295 clone.px = finalMom.Px();
296 clone.py = finalMom.Py();
297 clone.pz = finalMom.Pz();
298 clone.e = finalMom.E();
299 clone.x = finalPos.X();
300 clone.y = finalPos.Y();
301 clone.z = finalPos.Z();
302 clone.t = finalPos.T();
303 clone.polx = polX;
304 clone.poly = polY;
305 clone.polz = polZ;
306 clone.weight = finalEndpointWeight;
307 clone.parentID = trueParentId;
308 fSecondaryBuffer.push_back(clone);
309 }
310 fCurrentSurvivalFactor = 0.0;
311 fDecayedParentIDs.insert(currentTrackId);
312 }
313
314 // tracks which do not decay are stopped with stoptrack and added back with
315 // a given weight
316 if (!isNaturalDecay && !isParticleDestroyed && stack) {
317 Int_t ntr;
318 stack->PushTrack(1, trueParentId, track_pid, finalMom.Px(), finalMom.Py(),
319 finalMom.Pz(), finalMom.E(), finalPos.X(), finalPos.Y(),
320 finalPos.Z(), finalPos.T(), polX, polY, polZ,
321 kPNoProcess, ntr, fCurrentSurvivalFactor, 999);
322 fContinuationTracks.insert(ntr);
323 }
324
325 gMC->StopTrack();
326 }

329void exitHadronAbsorber::PreTrack() {
330 bool stackbufferisnotempty = !fSecondaryBuffer.empty();
331 if (stackbufferisnotempty) {
332 auto* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
333 Int_t ntr;
334 for (const auto& trk : fSecondaryBuffer) {
335 stack->PushTrack(1, trk.parentID, trk.pdg, trk.px, trk.py, trk.pz, trk.e,
336 trk.x, trk.y, trk.z, trk.t, trk.polx, trk.poly, trk.polz,
337 kPNoProcess, ntr, trk.weight, 999);
338 fCloneTracks.insert(ntr);
339 }
340 // Clear the buffer so we don't duplicate them for the next track
341 fSecondaryBuffer.clear();
342 }
343
344 // Reset relative survival factor to 1.0
345 fCurrentSurvivalFactor = 1.0;
346
347 gMC->TrackMomentum(fMom);
348 if ((fMom.E() - fMom.M()) < EMax) {
349 gMC->StopTrack();
350 return;
351 }
352 TParticle* p = gMC->GetStack()->GetCurrentTrack();
353 Int_t currentID = gMC->GetStack()->GetCurrentTrackNumber();
354
355 if (fCloneTracks.find(currentID) != fCloneTracks.end()) {
356 // Force the decay time to 0
357 gMC->ForceDecayTime(0);
358 }
359
360 Int_t pdgCode = p->GetPdgCode();
361
362 // record statistics for neutrinos, electrons and photons
363 // add pi0 111 eta 221 eta' 331 omega 223
364 Int_t idabs = TMath::Abs(pdgCode);
365 if (idabs < 18 || idabs == 22 || idabs == 111 || idabs == 221 ||
366 idabs == 223 || idabs == 331 || idabs == 211 || idabs == 321 ||
367 idabs == 2212) {
368 Double_t wspill = p->GetWeight();
369 Int_t idhnu = idabs + 10000;
370 if (pdgCode < 0) {
371 idhnu += 10000;
372 }
373 Double_t l10ptot =
374 TMath::Min(TMath::Max(TMath::Log10(fMom.P()), -0.3), 1.69999);
375 Double_t l10pt =
376 TMath::Min(TMath::Max(TMath::Log10(fMom.Pt()), -2.), 0.4999);
377 TString key = fVetoName;
378 key += idhnu;
379 TH1D* h1 = dynamic_cast<TH1D*>(fout->Get(key));
380 if (h1) {
381 h1->Fill(fMom.P(), wspill);
382 }
383 key = fVetoName;
384 key += idhnu + 1000;
385 TH2D* h2 = dynamic_cast<TH2D*>(fout->Get(key));
386 if (h2) {
387 h2->Fill(l10ptot, l10pt, wspill);
388 }
389 key = fVetoName;
390 key += idhnu + 2000;
391 h2 = dynamic_cast<TH2D*>(fout->Get(key));
392 if (h2) {
393 h2->Fill(l10ptot, l10pt, wspill);
394 }
395 if (withNtuple) {
396 fNtuple->Fill(pdgCode, fMom.Px(), fMom.Py(), fMom.Pz(), fPos.X(),
397 fPos.Y(), fPos.Z());
398 }
399 if (fSkipNeutrinos && (idabs == 12 || idabs == 14 || idabs == 16)) {
400 gMC->StopTrack();
401 }
402 }

454void RegisterDaughtersRecursively(TGeoVolume* volume,
455 exitHadronAbsorber* detector) {
456 if (!volume) return;
457 Int_t nDaughters = volume->GetNdaughters();
458 for (Int_t i = 0; i < nDaughters; ++i) {
459 TGeoNode* daughterNode = volume->GetNode(i);
460 if (daughterNode) {
461 TGeoVolume* daughterVol = daughterNode->GetVolume();
462 if (daughterVol) {
463 // register daughter volume
464 detector->AddSensitiveVolume(daughterVol);
465 LOG(info) << "[exitHadronAbsorber] Registered subvolume: "
466 << daughterVol->GetName();
467 // call function recursively
468 RegisterDaughtersRecursively(daughterVol, detector);
469 }
470 }

474void exitHadronAbsorber::ConstructGeometry() {
475 static FairGeoLoader* geoLoad = FairGeoLoader::Instance();
476 static FairGeoInterface* geoFace = geoLoad->getGeoInterface();
477 static FairGeoMedia* media = geoFace->getMedia();
478 static FairGeoBuilder* geoBuild = geoLoad->getGeoBuilder();
479
480 FairGeoMedium* ShipMedium = media->getMedium("vacuums");
481 TGeoMedium* vac = gGeoManager->GetMedium("vacuums");

484 gGeoManager->GetTopVolume();
485 TGeoNavigator* nav = gGeoManager->GetCurrentNavigator();
486 Double_t zLoc;
487 if (fzPos > 1E8) {
488 // Add thin sensitive plane after hadron absorber
489 Float_t distance = 1.;

501 } else {
502 zLoc = fzPos;

504

506 TString shapename_prefix(myname); // Use a prefix for shapenames

549 }
550 if ((fNsplits > 0) && (!fSplitOnce)) {
551 TString parentVolumeName = "/target_vacuum_box_1";
552 nav->cd(parentVolumeName.Data());
553 TGeoVolume* vol = nav->GetCurrentNode()->GetVolume();
554 AddSensitiveVolume(vol);
555 // register each unique daughter volume
556 RegisterDaughtersRecursively(vol, this);
557 }
558}
