86Bool_t exitHadronAbsorber::ProcessHits(FairVolume* vol)
87{
89 if ( gMC->IsTrackEntering() ) {
90 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
91 fEventId = gMC->CurrentEvent();
92 TParticle* p = gMC->GetStack()->GetCurrentTrack();
93 fUniqueID = p->GetUniqueID();
94 Int_t pdgCode = p->GetPdgCode();
95 gMC->TrackMomentum(fMom);
96 if (!(fOnlyMuons && TMath::Abs(pdgCode)!=13)){
97 fTime = gMC->TrackTime() * 1.0e09;
98 fLength = gMC->TrackLength();
99 gMC->TrackPosition(fPos);
100 if ( (fMom.E()-fMom.M() )>EMax) {
101 AddHit(fTrackID, 111, TVector3(fPos.X(),fPos.Y(),fPos.Z()),
102 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
103 0,pdgCode,TVector3(p->Vx(), p->Vy(), p->Vz()),TVector3(p->Px(), p->Py(), p->Pz()), fEventId);
104 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
105 stack->AddPoint(kVETO);
106 }
107 }
108 }
109 if ((!fCylindricalPlane) && fzPos > 1E8) {
110 gMC->StopTrack();
111 }
112 return kTRUE;
113}

116{
117 FairDetector::Initialize();
118 TSeqCollection* fileList=gROOT->GetListOfFiles();
119 fout = dynamic_cast<TFile*>(fileList->At(0));
120 // book hists for Genie neutrino momentum distribution
121 // add also leptons, and photon
122 // add pi0 111 eta 221 eta' 331 omega 223 for DM production
123 TDatabasePDG* PDG = TDatabasePDG::Instance();
124 for(Int_t idnu=11; idnu<26; idnu+=1){
125 // nu or anti-nu
126 for (Int_t idadd=-1; idadd<3; idadd+=2){
127 Int_t idw=idnu;
128 if (idnu==18){idw=22;}
129 if (idnu==19){idw=111;}
130 if (idnu==20){idw=221;}
131 if (idnu==21){idw=223;}
132 if (idnu==22){idw=331;}
133 if (idnu==23){idw=211;}
134 if (idnu==24){idw=321;}
135 if (idnu==25){idw=2212;}
136 Int_t idhnu=10000+idw;
137 if (idadd==-1){
138 if (idnu>17){continue;}
139 idhnu+=10000;
140 idw=-idnu;
141 }
142 TString name=PDG->GetParticle(idw)->GetName();
143 TString title = name;title+=" momentum (GeV)";
144 TString key = fVetoName;key+=idhnu;
145 [[maybe_unused]] TH1D* Hidhnu = new TH1D(key,title,400,0.,400.);
146 title = name;title+=" log10-p vs log10-pt";
147 key = fVetoName;key+=idhnu+1000;

149 title = name;title+=" log10-p vs log10-pt";
150 key = fVetoName;key+=idhnu+2000;
151 [[maybe_unused]] TH2D* Hidhnu200 = new TH2D(key,title,25,-0.3,1.7,100,-2.,0.5);

166void exitHadronAbsorber::PreTrack(){
167 gMC->TrackMomentum(fMom);
168 if ( (fMom.E()-fMom.M() )<EMax){
169 gMC->StopTrack();
170 return;
171 }
172 TParticle* p = gMC->GetStack()->GetCurrentTrack();
173 Int_t pdgCode = p->GetPdgCode();
174// record statistics for neutrinos, electrons and photons
175// add pi0 111 eta 221 eta' 331 omega 223
176 Int_t idabs = TMath::Abs(pdgCode);
177 if (idabs<18 || idabs==22 || idabs==111 || idabs==221 || idabs==223 || idabs==331
178 || idabs==211 || idabs==321 || idabs==2212 ){
179 Double_t wspill = p->GetWeight();
180 Int_t idhnu=idabs+10000;
181 if (pdgCode<0){ idhnu+=10000;}
182 Double_t l10ptot = TMath::Min(TMath::Max(TMath::Log10(fMom.P()),-0.3),1.69999);
183 Double_t l10pt = TMath::Min(TMath::Max(TMath::Log10(fMom.Pt()),-2.),0.4999);
184 TString key = fVetoName; key+=idhnu;
185 TH1D* h1 = dynamic_cast<TH1D*>(fout->Get(key));
186 if (h1){h1->Fill(fMom.P(),wspill);}
187 key=fVetoName;key+=idhnu+1000;
188 TH2D* h2 = dynamic_cast<TH2D*>(fout->Get(key));
189 if (h2){h2->Fill(l10ptot,l10pt,wspill);}
190 key=fVetoName;key+=idhnu+2000;
191 h2 = dynamic_cast<TH2D*>(fout->Get(key));
192 if (h2){h2->Fill(l10ptot,l10pt,wspill);}
193 if(withNtuple){
194 fNtuple->Fill(pdgCode,fMom.Px(),fMom.Py(), fMom.Pz(),fPos.X(),fPos.Y(),fPos.Z());
195 }
196 if (fSkipNeutrinos && (idabs == 12 || idabs == 14 || idabs == 16)) {
197 gMC->StopTrack();
198 }
199 }

234void exitHadronAbsorber::ConstructGeometry()
235{
236 static FairGeoLoader *geoLoad=FairGeoLoader::Instance();
237 static FairGeoInterface *geoFace=geoLoad->getGeoInterface();
238 static FairGeoMedia *media=geoFace->getMedia();
239 static FairGeoBuilder *geoBuild=geoLoad->getGeoBuilder();
240
241 FairGeoMedium *ShipMedium=media->getMedium("vacuums");
242 TGeoMedium* vac=gGeoManager->GetMedium("vacuums");
243 if (vac==NULL)
244 geoBuild->createMedium(ShipMedium);
245 vac =gGeoManager->GetMedium("vacuums");
246 gGeoManager->GetTopVolume();
247 TGeoNavigator* nav = gGeoManager->GetCurrentNavigator();
248 Double_t zLoc;
249 if (fzPos>1E8){
250 //Add thin sensitive plane after hadron absorber
251 Float_t distance = 1.;
252 Double_t local[3]= {0,0,0};
253 if (!nav->cd("/MuonShieldArea_1/AbsorberVol_1")) {
254 nav->cd("/MuonShieldArea_1/MagnAbsorb_MagRetL_1");
255 distance = -1.;
256 }
257 TGeoBBox* tmp = dynamic_cast<TGeoBBox*>(nav->GetCurrentNode()->GetVolume()->GetShape());
258 local[2] = distance * tmp->GetDZ();
259 Double_t global[3] = {0,0,0};
260 nav->LocalToMaster(local,global);
261 zLoc = global[2] + distance * 1.*cm;
262 }else{zLoc = fzPos;} // use external input
263
264 TString myname(this->GetName());
265 TString shapename_prefix(myname); // Use a prefix for shapenames
266
267 Double_t xLocPlane = 0.0;
268 Double_t yLocPlane = 0.0;
269 Double_t zLocPlane = zLoc;
270
271 if (!fCylindricalPlane){
272 TGeoVolume *sensPlane = gGeoManager->MakeBox(shapename_prefix+fVetoName+"_box",vac,3.56*m-1.*mm,1.7*m-1.*mm,1.*mm);
273 std::cout << this->GetName() << ", ConstructGeometry(): Created Box with dimensions: 3.56*m-1.*mm,1.7*m-1.*mm,1.*mm" << std::endl;
274
275 if (!fUseCaveCoordinates) {
276 nav->cd("/MuonShieldArea_1/");
277 } else {
278 Double_t local[3] = {0, 0, zLoc};
279 Double_t global[3];
280 nav->cd("/target_vacuum_box_1/TargetArea_1");
281 nav->LocalToMaster(local, global);
282 nav->cd("/cave_1");
283 yLocPlane = global[1];
284 zLocPlane = global[2];
285 xLocPlane = global[0];
286 }
287 sensPlane->SetLineColor(kBlue - 10);
288 //nav->GetCurrentNode()->GetVolume()->AddNode(sensPlane, fzPos<250?2:1, new TGeoTranslation(0, 0, zLoc));
289 nav->GetCurrentNode()->GetVolume()->AddNode(sensPlane, 1, new TGeoTranslation(xLocPlane, yLocPlane, zLocPlane));
290 AddSensitiveVolume(sensPlane);
291 }
292 else { // add cylindrical sensPlane
293 // if (zLoc < 250*cm){//corresponds to target...
294 // Parameters: name, medium, inner radius, outer radius, half-length (Z), phi1, phi2
295 TGeoVolume *sensPlaneCyl = gGeoManager->MakeTube(shapename_prefix+fVetoName+"_tube",vac,12.51*cm,12.52*cm,zLoc/2.);
296 std::cout << this->GetName() << ", ConstructGeometry(): Created Tube with dimensions: 12.51*cm,12.52*cm,zLoc/2." << std::endl;
297 nav->cd("/target_vacuum_box_1/TargetArea_1/HeVolume_1");
298 nav->GetCurrentNode()->GetVolume()->AddNode(sensPlaneCyl, 1, new TGeoTranslation(0, 0, 0));
299 AddSensitiveVolume(sensPlaneCyl);
300 }
