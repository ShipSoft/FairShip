267Bool_t FixedTargetGenerator::ReadEvent(FairPrimaryGenerator* cpg)
268{
269 Double_t zinter=0;
270 Double_t ZoverA = 1.;
271 if (targetName.Data() !=""){
272// calculate primary proton interaction point:
273// loop over trajectory between start and end to pick an interaction point, copied from GenieGenerator and adapted to hadrons
274 Double_t prob2int = -1.;
275 Double_t rndm = 0.;
276 Double_t sigma;
277 Int_t count=0;
278 Double_t zinterStart = start[2];
279 if (Option == "charm" || Option == "beauty"){
280// simulate more downstream interaction points for interactions down in the cascade
281 if (!(nTree->GetBranch("k"))){ck=1;}
282 }
283 else {ck=1;}
284 while (ck>0.5){
285 while (prob2int<rndm) {
286 //place x,y,z uniform along path
287 zinter = gRandom->Uniform(zinterStart,end[2]);
288 Double_t point[3]={xOff,yOff,zinter};
289 bparam = fMaterialInvestigator->MeanMaterialBudget(start, point, mparam);
290 Double_t interLength = mparam[8];
291 TGeoNode *node = gGeoManager->FindNode(point[0],point[1],point[2]);
292 TGeoMaterial *mat = 0;
293 if (node && !gGeoManager->IsOutside()) {
294 mat = node->GetVolume()->GetMaterial();
295 Double_t n = mat->GetDensity()/mat->GetA();
296 ZoverA = mat->GetZ() / mat->GetA();
297 sigma = 1./(n*mat->GetIntLen())/mbarn;
298 prob2int = TMath::Exp(-interLength)*sigma/maxCrossSection;
299 }else{
300 prob2int=0.;
301 }
302 rndm = gRandom->Uniform(0.,1.);
303 count+=1;
304 }
305 zinterStart = zinter;
306 ck-=1;
307 }
308 zinter = zinter*cm;
309 }
310 Pythia8::Pythia* fPythia;
311 if (G4only){
312 cpg->AddTrack(2212,0.,0.,fMom,xOff/cm,yOff/cm,start[2],-1,kTRUE,-1.,0.,1.);
313 return kTRUE;
314 }else if (Option == "Primary"){
315 if (gRandom->Uniform(0.,1.) < ZoverA ){
316 fPythiaP->next();
317 if (withEvtGen){evtgenP->decay();}
318 fPythia = fPythiaP;
319 }else{
320 fPythiaN->next();
321 if (withEvtGen){evtgenN->decay();}
322 fPythia = fPythiaN;
323 }
324 }else{
325 if (nEntry==nEvents){
326 LOG(INFO) << "Rewind input file: " << nEntry;
327 nEntry=0;}
328 nTree->GetEvent(nEntry);
329 nEntry+=1;
330 // sanity check, count number of p.o.t. on input file.
331 Double_t pt=TMath::Sqrt( (n_mpx*n_mpx)+(n_mpy*n_mpy));
332 // every event appears twice, i.e.
333 if (pt<1.e-5 && n_mid==2212){
334 pot+=+0.5;
335 ntotprim+=1;}
336 Int_t idabs=int(TMath::Abs(n_id));
337 if (idabs==431){ nDsprim+=1;}
338 fPythiaP->event.reset();
339 Int_t nID1 = n_id;
340 fPythiaP->event.append(int(n_id),1,0,0,n_px,n_py,n_pz,n_E,n_M,0.,9.);
341 TMCProcess procID = kPTransportation;
342 if (n_mid==2212 && (n_mpx*n_mpx+n_mpy*n_mpy)<1E-5) {procID = kPPrimary;} // probably primary and not from cascade
343 cpg->AddTrack(int(n_mid),n_mpx,n_mpy,n_mpz, xOff/cm,yOff/cm,zinter/cm,-1,kFALSE,n_mE,0.,wspill,procID);
344// second charm hadron in the event
345 nTree->GetEvent(nEntry);
346 if (nID1 * n_id > 0){LOG(INFO) << "same sign charm: " << nEntry << ", " << nID1 << ", " << n_id;}
347 nEntry+=1;
348 fPythiaP->event.append(int(n_id),1,0,0,n_px,n_py,n_pz,n_E,n_M,0.,9.);
349 fPythiaP->next();
350 fPythia = fPythiaP;
351 }
352 if (withEvtGen){
353 fPythia->moreDecays();} // let the very short lived resonances decay via Pythia8
354 if(Debug && !G4only){fPythia->event.list();}
355 TMCProcess procID;
356 for(Int_t ii=1; ii<fPythia->event.size(); ii++){
357 Double_t e = fPythia->event[ii].e();
358 Double_t m = fPythia->event[ii].m();
359 Double_t pz = fPythia->event[ii].pz();
360 Int_t id = fPythia->event[ii].id();
361 Bool_t wanttracking=kTRUE;
362 if (e-m<EMax || !fPythia->event[ii].isFinal() || pz<0) {wanttracking=kFALSE;}
363 if (DrellYan || PhotonCollision || OnlyMuons){
364// don't track underlying event
365 if (fabs(id)!=13){wanttracking=kFALSE;}
366 }
367 Double_t z = fPythia->event[ii].zProd()+zinter;
368 Double_t x = fPythia->event[ii].xProd()+xOff;
369 Double_t y = fPythia->event[ii].yProd()+yOff;
370 Double_t tof = fPythia->event[ii].tProd() / (10*c_light) ; // to go from mm to s
371 Double_t px = fPythia->event[ii].px();
372 Double_t py = fPythia->event[ii].py();
373 Int_t im = fPythia->event[ii].mother1()-1;
374 procID = kPPrimary;
375 if (Option != "Primary"){
376 procID = kPDecay;
377 im+=1;
378 if (ii==1) {procID = kPHadronic;}
379 }else{
380 if (ii<3){im=-1;}
381 }
382 cpg->AddTrack(id,px,py,pz,x/cm,y/cm,z/cm,im,wanttracking,e,tof,wspill,procID);
383 if(withNtuple){
384 int idabs = TMath::Abs(id);
385 if (idabs<10) {continue;}
386 if (idabs<18 || idabs==22 || idabs==111 || idabs==221 || idabs==223 || idabs==331
387 || idabs==211 || idabs==113 || idabs==333 || idabs==321 || idabs==2212 ){
388 Int_t moID = fPythia->event[im+1].id();
389 fNtuple->Fill(0,id,px,py,pz,e,x/cm,y/cm,z/cm,moID);
390 }
391 }
392 }
393 return kTRUE;
394}
