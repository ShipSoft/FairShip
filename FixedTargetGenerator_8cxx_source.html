287 Bool_t FixedTargetGenerator::ReadEvent(FairPrimaryGenerator* cpg)
288 {
289  Double_t zinter=0;
290  Double_t ZoverA = 1.;
291  if (targetName.Data() !=""){
292 // calculate primary proton interaction point:
293 // loop over trajectory between start and end to pick an interaction point, copied from GenieGenerator and adapted to hadrons
294  Double_t prob2int = -1.;
295  Double_t rndm = 0.;
296  Double_t sigma;
297  Int_t count=0;
298  Double_t zinterStart = start[2];
299  if (Option == "charm" || Option == "beauty"){
300 // simulate more downstream interaction points for interactions down in the cascade
301  if (!(nTree->GetBranch("k"))){ck=1;}
302  }
303  else {ck=1;}
304  while (ck>0.5){
305  while (prob2int<rndm) {
306  //place x,y,z uniform along path
307  zinter = gRandom->Uniform(zinterStart,end[2]);
308  Double_t point[3]={xOff,yOff,zinter};
309  bparam = fMaterialInvestigator->MeanMaterialBudget(start, point, mparam);
310  Double_t interLength = mparam[8];
311  TGeoNode *node = gGeoManager->FindNode(point[0],point[1],point[2]);
312  TGeoMaterial *mat = 0;
313  if (node && !gGeoManager->IsOutside()) {
314  mat = node->GetVolume()->GetMaterial();
315  Double_t n = mat->GetDensity()/mat->GetA();
316  ZoverA = mat->GetZ() / mat->GetA();
317  sigma = 1./(n*mat->GetIntLen())/mbarn;
318  prob2int = TMath::Exp(-interLength)*sigma/maxCrossSection;
319  }else{
320  prob2int=0.;
321  }
322  rndm = gRandom->Uniform(0.,1.);
323  count+=1;
324  }
325  zinterStart = zinter;
326  ck-=1;
327  }
328  zinter = zinter*cm;
329  }
330  Pythia8::Pythia* fPythia;
331  if (G4only){
332  cpg->AddTrack(2212,0.,0.,fMom,xOff/cm,yOff/cm,start[2],-1,kTRUE,-1.,0.,1.);
333  return kTRUE;
334  }else if (Option == "Primary"){
335  if (gRandom->Uniform(0.,1.) < ZoverA ){
336  fPythiaP->next();
337  if (withEvtGen){evtgenP->decay();}
338  fPythia = fPythiaP;
339  }else{
340  fPythiaN->next();
341  if (withEvtGen){evtgenN->decay();}
342  fPythia = fPythiaN;
343  }
344  }else{
345  if (nEntry==nEvents){
346  LOG(INFO) << "Rewind input file: " << nEntry;
347  nEntry=0;}
348  nTree->GetEvent(nEntry);
349  nEntry+=1;
350  // sanity check, count number of p.o.t. on input file.
351  Double_t pt=TMath::Sqrt( (n_mpx*n_mpx)+(n_mpy*n_mpy));
352  // every event appears twice, i.e.
353  if (pt<1.e-5 && n_mid==2212){
354  pot+=+0.5;
355  ntotprim+=1;}
356  Int_t idabs=int(TMath::Abs(n_id));
357  if (idabs==431){ nDsprim+=1;}
358  fPythiaP->event.reset();
359  Int_t nID1 = n_id;

361  TMCProcess procID = kPTransportation;

363  cpg->AddTrack(int(n_mid),n_mpx,n_mpy,n_mpz, xOff/cm,yOff/cm,zinter/cm,-1,kFALSE,n_mE,0.,wspill,procID);
