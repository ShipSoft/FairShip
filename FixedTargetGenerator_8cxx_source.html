277Bool_t FixedTargetGenerator::ReadEvent(FairPrimaryGenerator* cpg)
278{
279 Double_t zinter=0;
280 Double_t ZoverA = 1.;
281 if (targetName.Data() !=""){
282// calculate primary proton interaction point:
283// loop over trajectory between start and end to pick an interaction point, copied from GenieGenerator and adapted to hadrons
284 Double_t prob2int = -1.;
285 Double_t rndm = 0.;
286 Double_t sigma;
287 Int_t count=0;
288 Double_t zinterStart = start[2];
289 if (Option == "charm" || Option == "beauty"){
290// simulate more downstream interaction points for interactions down in the cascade
291 if (!(nTree->GetBranch("k"))){ck=1;}
292 }
293 else {ck=1;}
294 while (ck>0.5){
295 while (prob2int<rndm) {
296 //place x,y,z uniform along path
297 zinter = gRandom->Uniform(zinterStart,end[2]);
298 Double_t point[3]={xOff,yOff,zinter};
299 bparam = fMaterialInvestigator->MeanMaterialBudget(start, point, mparam);
300 Double_t interLength = mparam[8];
301 TGeoNode *node = gGeoManager->FindNode(point[0],point[1],point[2]);
302 TGeoMaterial *mat = 0;
303 if (node && !gGeoManager->IsOutside()) {
304 mat = node->GetVolume()->GetMaterial();
305 Double_t n = mat->GetDensity()/mat->GetA();
306 ZoverA = mat->GetZ() / mat->GetA();
307 sigma = 1./(n*mat->GetIntLen())/mbarn;
308 prob2int = TMath::Exp(-interLength)*sigma/maxCrossSection;
309 }else{
310 prob2int=0.;
311 }
312 rndm = gRandom->Uniform(0.,1.);
313 count+=1;
314 }
315 zinterStart = zinter;
316 ck-=1;
317 }
318 zinter = zinter*cm;
319 }
320 Pythia8::Pythia* fPythia;
321 if (G4only){
322 cpg->AddTrack(2212,0.,0.,fMom,xOff/cm,yOff/cm,start[2],-1,kTRUE,-1.,0.,1.);
323 return kTRUE;
324 }else if (Option == "Primary"){
325 if (gRandom->Uniform(0.,1.) < ZoverA ){
326 fPythiaP->next();
327 if (withEvtGen){evtgenP->decay();}
328 fPythia = fPythiaP;
329 }else{
330 fPythiaN->next();
331 if (withEvtGen){evtgenN->decay();}
332 fPythia = fPythiaN;
333 }
334 }else{
335 if (nEntry==nEvents){
336 LOG(INFO) << "Rewind input file: " << nEntry;
337 nEntry=0;}
338 nTree->GetEvent(nEntry);
339 nEntry+=1;
340 // sanity check, count number of p.o.t. on input file.
341 Double_t pt=TMath::Sqrt( (n_mpx*n_mpx)+(n_mpy*n_mpy));
342 // every event appears twice, i.e.
343 if (pt<1.e-5 && n_mid==2212){
344 pot+=+0.5;
345 ntotprim+=1;}
346 Int_t idabs = static_cast<int>(TMath::Abs(n_id));
347 if (idabs==431){ nDsprim+=1;}
348 fPythiaP->event.reset();
349 Int_t nID1 = n_id;
350 fPythiaP->event.append(static_cast<int>(n_id), 1, 0, 0, n_px, n_py, n_pz, n_E, n_M, 0., 9.);
351 TMCProcess procID = kPTransportation;
352 if (n_mid==2212 && (n_mpx*n_mpx+n_mpy*n_mpy)<1E-5) {procID = kPPrimary;} // probably primary and not from cascade
353 cpg->AddTrack(static_cast<int>(n_mid),
354 n_mpx,
355 n_mpy,
356 n_mpz,
357 xOff / cm,
358 yOff / cm,
359 zinter / cm,
360 -1,
361 kFALSE,
362 n_mE,
363 0.,
364 wspill,
365 procID);
366 // second charm hadron in the event
367 nTree->GetEvent(nEntry);
368 if (nID1 * n_id > 0){LOG(INFO) << "same sign charm: " << nEntry << ", " << nID1 << ", " << n_id;}
369 nEntry+=1;
370 fPythiaP->event.append(static_cast<int>(n_id), 1, 0, 0, n_px, n_py, n_pz, n_E, n_M, 0., 9.);
371 fPythiaP->next();
372 fPythia = fPythiaP;
373 }
374 if (withEvtGen){
375 fPythia->moreDecays();} // let the very short lived resonances decay via Pythia8
376 if(Debug && !G4only){fPythia->event.list();}
377 TMCProcess procID;
378 for(Int_t ii=1; ii<fPythia->event.size(); ii++){
379 Double_t e = fPythia->event[ii].e();
380 Double_t m = fPythia->event[ii].m();
381 Double_t pz = fPythia->event[ii].pz();
382 Int_t id = fPythia->event[ii].id();
383 Bool_t wanttracking=kTRUE;
384 if (e-m<EMax || !fPythia->event[ii].isFinal() || pz<0) {wanttracking=kFALSE;}
385 if (DrellYan || PhotonCollision || OnlyMuons){
386// don't track underlying event
387 if (fabs(id)!=13){wanttracking=kFALSE;}
388 }
389 Double_t z = fPythia->event[ii].zProd()+zinter;
390 Double_t x = fPythia->event[ii].xProd()+xOff;
391 Double_t y = fPythia->event[ii].yProd()+yOff;
392 Double_t tof = fPythia->event[ii].tProd() / (10*c_light) ; // to go from mm to s
393 Double_t px = fPythia->event[ii].px();
394 Double_t py = fPythia->event[ii].py();
395 Int_t im = fPythia->event[ii].mother1()-1;
396 procID = kPPrimary;
397 if (Option != "Primary"){
398 procID = kPDecay;
399 im+=1;
400 if (ii==1) {procID = kPHadronic;}
401 }else{
402 if (ii<3){im=-1;}
403 }
404 cpg->AddTrack(id,px,py,pz,x/cm,y/cm,z/cm,im,wanttracking,e,tof,wspill,procID);
405 if(withNtuple){
406 int idabs = TMath::Abs(id);
407 if (idabs<10) {continue;}
408 if (idabs<18 || idabs==22 || idabs==111 || idabs==221 || idabs==223 || idabs==331
409 || idabs==211 || idabs==113 || idabs==333 || idabs==321 || idabs==2212 ){
410 Int_t moID = fPythia->event[im+1].id();
411 fNtuple->Fill(0,id,px,py,pz,e,x/cm,y/cm,z/cm,moID);
412 }
413 }
414 }
415 return kTRUE;
416}
