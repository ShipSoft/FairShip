225  def Width_H_L(self, H, L):
226  """
227  Returns the RPV neutralino decay width into neutral meson and lepton
228 
229  Inputs:
230  - H is a string (name of the meson)
231  - L is a string (name of the lepton)
232  """
233  if self.MNMN < (mass(H)+mass(L)):
234  return 0.
235 
236  phsp=math.sqrt(self.MNMN**4+mass(H)**4+mass(L)**4-\
237  2*self.MNMN**2*mass(H)**2-2*self.MNMN**2*mass(L)**2-\
238  2*mass(H)**2*mass(L)**2)
239  tmp_width=0
240  if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
241  tmp_width = phsp/(self.sfmasssfmass**4*128*u.pi*self.MNMN**3)*\
242  c.GST2['sneutrino']*c.decayConstant[H]**2*\
243  (self.MNMN**2+mass(L)**2-mass(H)**2)
244  else:
245  tmp_width = phsp/(self.sfmasssfmass**4*128*u.pi*self.MNMN**3)*\
246  c.GST2['slepton']*c.decayConstant[H]**2*\
247  (self.MNMN**2+mass(L)**2-mass(H)**2)
248 
249  if 'K*' in H or 'D*' in H or "phi" in H:
250  if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
251  tmp_width = phsp/(self.sfmasssfmass**4*2*u.pi*self.MNMN**3)*\
252  c.GST2['tneutrino']*c.decayConstant[H]**2*\
253  (2*(self.MNMN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MNMN**2+mass(L)**2))
254  else:
255  tmp_width = phsp/(self.sfmasssfmass**4*2*u.pi*self.MNMN**3)*\
256  c.GST2['tlepton']*c.decayConstant[H]**2*\
257  (2*(self.MNMN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MNMN**2+mass(L)**2))
258  width=self.U2U2[1]*tmp_width
259  # contributions both from decay and production couplings
260  if self.benchbench==1 and ('K_' in H or 'K*' in H):
261  width=(self.U2U2[0]+self.U2U2[1])*tmp_width
262  # contributions only from production coupling
263  if self.benchbench==2 and ('eta' in H or 'phi' in H):
264  width=self.U2U2[0]*tmp_width
265 
266  # Majorana case (charge conjugate channels)
267  width = 2.*width
268  return width
