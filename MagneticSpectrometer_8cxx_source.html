316Bool_t MagneticSpectrometer::ProcessHits(FairVolume* vol)
317{
319 //Set parameters at entrance of volume. Reset ELoss.
320 if ( gMC->IsTrackEntering() ) {
321 fELoss = 0.;
322 fTime = gMC->TrackTime() * 1.0e09;
323 fLength = gMC->TrackLength();
324 gMC->TrackPosition(fPos);
325 gMC->TrackMomentum(fMom);
326 }
327 // Sum energy loss for all steps in the active volume
328 fELoss += gMC->Edep();
329
330 // Create muonPoint at exit of active volume
331 if ( gMC->IsTrackExiting() ||
332 gMC->IsTrackStop() ||
333 gMC->IsTrackDisappeared() ) {
334 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
335 fVolumeID = vol->getMCid();
336 if (fELoss == 0. ) { return kFALSE; }
337 TParticle* p=gMC->GetStack()->GetCurrentTrack();
338 Int_t pdgCode = p->GetPdgCode();
339 Int_t detID=0;
340 gMC->CurrentVolID(detID);
341
342 // cout<< "detID = " << detID << endl;
343 Int_t MaxLevel = gGeoManager->GetLevel();
344 const Int_t MaxL = MaxLevel;
345 //cout << "MaxLevel = " << MaxL << endl;
346 //cout << gMC->CurrentVolPath()<< endl;
347 Int_t NRpc =0;
348 const char *name;
349 name = gMC->CurrentVolName();
350 //cout << name << endl;
351 Int_t motherID = gGeoManager->GetMother(0)->GetNumber();
352 const char *mumname = gMC->CurrentVolOffName(0);
353 //cout<<mumname<<" "<< motherID<<endl;
354 detID = motherID;
355 //cout<< "detID = " << detID << endl;
356 //cout<<endl;
357 fVolumeID = detID;
358
359 TLorentzVector Pos;
360 gMC->TrackPosition(Pos);
361 Double_t xmean = (fPos.X()+Pos.X())/2. ;
362 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
363 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
364
365 AddHit(fTrackID, fVolumeID, TVector3(xmean, ymean, zmean), TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,fELoss, pdgCode);
366
367 // Increment number of muon det points in TParticle
368 ShipStack* stack = (ShipStack*) gMC->GetStack();
369 stack->AddPoint(ktauRpc);
370 }
371
372 return kTRUE;
373}
