33def yBeam(Mproton = 0.938272081, pbeam = 400.):
34 Ebeam = ROOT.TMath.Sqrt(pbeam**2+Mproton**2)
35 betaCM = pbeam / (Ebeam + Mproton)
36 y_beam = 0.5*ROOT.TMath.Log( (1+betaCM)/(1-betaCM)) # https://arxiv.org/pdf/1604.02651.pdf
37 return y_beam
