43def yBeam(Mproton=0.938272081, pbeam=400.0):
44 Ebeam = ROOT.TMath.Sqrt(pbeam**2 + Mproton**2)
45 betaCM = pbeam / (Ebeam + Mproton)
46 y_beam = 0.5 * ROOT.TMath.Log((1 + betaCM) / (1 - betaCM)) # https://arxiv.org/pdf/1604.02651.pdf
47 return y_beam
