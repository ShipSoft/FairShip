40def inspect_file(inputfile, muonfile, print_table=False):
41 """Inspecting the produced file for successfully added muon veto points."""
42 input_file = r.TFile.Open(inputfile, "read")
43 input_tree = input_file.cbmsim
44
45 muon_file = r.TFile.Open(muonfile, "read")
46 muon_tree = muon_file.DIS
47
48 muons_found = False
49
50 table_data = []
51
52 for i, (muon_event, input_event) in enumerate(zip(muon_tree, input_tree)):
53 muon_hits = len(muon_event.muon_vetoPoints)
54 muoncount = 0
55 if muon_hits:
56 for hit in input_event.vetoPoint:
57 if hit.GetTrackID() == 0:
58 muons_found = True
59 muoncount += 1
60
61 muon_ubthits = len(muon_event.muon_UpstreamTaggerPoints)
62 muon_ubtcount = 0
63 if muon_ubthits:
64 for hit in input_event.UpstreamTaggerPoint:
65 if hit.GetTrackID() == 0:
66 muon_ubtcount += 1
67
68 table_data.append(
69 [
70 i,
71 muon_hits,
72 len(input_event.vetoPoint) - muoncount,
73 muoncount,
74 len(input_event.vetoPoint),
75 muon_ubthits,
76 len(input_event.UpstreamTaggerPoint) - muon_ubtcount,
77 muon_ubtcount,
78 len(input_event.UpstreamTaggerPoint),
79 ]
80 )
81
82 if print_table:
83 print(tabulate(table_data, headers=headers, tablefmt="grid"))
84
85 input_file.Close()
86 muon_file.Close()
87
