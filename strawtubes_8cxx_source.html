116Bool_t strawtubes::ProcessHits(FairVolume* vol)
117{
119 //Set parameters at entrance of volume. Reset ELoss.
120 if ( gMC->IsTrackEntering() ) {
121 fELoss = 0.;
122 fTime = gMC->TrackTime() * 1.0e09;
123 fLength = gMC->TrackLength();
124 gMC->TrackPosition(fPos);
125 gMC->TrackMomentum(fMom);
126 }
127 // Sum energy loss for all steps in the active volume
128 fELoss += gMC->Edep();
129
130 // Create strawtubesPoint at exit of active volume
131 if ( gMC->IsTrackExiting() ||
132 gMC->IsTrackStop() ||
133 gMC->IsTrackDisappeared() ) {
134 if (fELoss == 0. ) { return kFALSE; }
135 TParticle* p=gMC->GetStack()->GetCurrentTrack();
136 Int_t pdgCode = p->GetPdgCode();
137 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
138 Int_t straw_uniqueId;
139 gMC->CurrentVolID(straw_uniqueId);
140 if (fVolumeID == straw_uniqueId) {
141 //std::cout << pdgCode<< " same volume again ? "<< straw_uniqueId << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
142 return kTRUE; }
143 fVolumeID = straw_uniqueId;
144 // # d = |pq . u x v|/|u x v|
145 TVector3 bot,top;
146 StrawEndPoints(straw_uniqueId,bot,top);
147 TLorentzVector Pos;
148 gMC->TrackPosition(Pos);
149 Double_t xmean = (fPos.X()+Pos.X())/2. ;
150 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
151 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
152 TVector3 pq = TVector3(top.x()-xmean,top.y()-ymean,top.z()-zmean );
153 TVector3 u = TVector3(bot.x()-top.x(),bot.y()-top.y(),bot.z()-top.z() );
154 TVector3 v = TVector3(fPos.X()-Pos.X(),fPos.Y()-Pos.Y(),fPos.Z()-Pos.Z());
155 TVector3 uCrossv = u.Cross(v);
156 Double_t dist2Wire = fabs(pq.Dot(uCrossv))/(uCrossv.Mag()+1E-8);
157 Double_t deltaTrackLength = gMC->TrackLength() - fLength;
158 AddHit(fTrackID, straw_uniqueId, TVector3(xmean, ymean, zmean),
159 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, deltaTrackLength,
160 fELoss,pdgCode,dist2Wire);
161 if (dist2Wire>fInner_Straw_diameter/2){
162 std::cout << "addhit " << dist2Wire<< " straw id " << straw_uniqueId << " pdgcode " << pdgCode<< " dot prod " << pq.Dot(uCrossv)<< std::endl;
163 std::cout << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
164 std::cout << " entry:" << fPos.X()<< " " << fPos.Y()<< " " << fPos.Z() << std::endl;
165 std::cout << " exit:" << Pos.X()<< " " << Pos.Y()<< " " << Pos.Z() << std::endl;
166 std::cout << " mean:" << xmean<< " " << ymean << " " << zmean << std::endl;
167 std::cout << " bot:" << bot.x()<< " " << bot.y() << " " << bot.z() << std::endl;
168 std::cout << " top:" << top.x()<< " " << top.y() << " " << top.z() << std::endl;
169 pq.Print();
170 u.Print();
171 v.Print();
172 uCrossv.Print();
173 }
174 // Increment number of strawtubes det points in TParticle
175 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
176 stack->AddPoint(kStraw);
177 }
178 return kTRUE;
179}

298void strawtubes::ConstructGeometry()
299{
304 TGeoVolume *top = gGeoManager->GetTopVolume();
305 InitMedium("air");
306 TGeoMedium *air = gGeoManager->GetMedium("air");
307 InitMedium("ShipSens");
308 TGeoMedium *Se = gGeoManager->GetMedium("ShipSens");
309 InitMedium("aluminium");
310 TGeoMedium *Al = gGeoManager->GetMedium("aluminium");
311 InitMedium("mylar");
312 TGeoMedium *mylar = gGeoManager->GetMedium("mylar");
313 InitMedium("STTmix9010_2bar");
314 TGeoMedium *sttmix9010_2bar = gGeoManager->GetMedium("STTmix9010_2bar");
315 InitMedium("tungsten");
316 TGeoMedium *tungsten = gGeoManager->GetMedium("tungsten");
317 InitMedium(fFrame_material);
318 TGeoMedium *FrameMatPtr = gGeoManager->GetMedium(fFrame_material);
319 InitMedium(fMedium.c_str());
320 TGeoMedium* med = gGeoManager->GetMedium(fMedium.c_str());
321
322 gGeoManager->SetVisLevel(4);
323 gGeoManager->SetTopVisible();
324
325 //epsilon to avoid overlapping volumes
326 //Double_t eps=0.1;
327 Double_t eps=0.0001;
328 Double_t epsS=0.0001;
329 //width of frame
330 Double_t framewidth = 40.;
331 //width of view
332 Double_t viewwidth = fDeltaz_view - eps;
333 //width of layer
334 Double_t layerwidth = fOuter_Straw_diameter;
335
336 Double_t rmin, rmax, dx, dy, dz, z, density,a,w;
337 Double_t par[20];
338 Int_t nel,numed,isvol,ifield;
339 Double_t radl, absl, TStationz;
340
341 Double_t yDim = (fStraws_per_layer+1) * fStraw_pitch /2. ; // put everything inside vacbox
342 //arguments of box are half-lengths;
343 TGeoBBox* detbox1 = new TGeoBBox(
344 "detbox1", fStraw_length + fFrame_lateral_width, f_station_height + fFrame_lateral_width, fDeltaz_frame / 2.);
345 TGeoBBox* detbox2 = new TGeoBBox("detbox2", fStraw_length + eps, f_station_height + eps, fDeltaz_frame / 2. + eps);
346
347 TGeoCompositeShape* detcomp1 = new TGeoCompositeShape("detcomp1", "detbox1-detbox2");
348 // Volume: straw
349 rmin = fInner_Straw_diameter/2.;
350 rmax = fOuter_Straw_diameter/2.;
351 //third argument is halflength of tube
352 TGeoTube *straw_tube = new TGeoTube("straw",rmin,rmax,fStraw_length-4.*eps);
353 TGeoVolume *straw = new TGeoVolume("straw",straw_tube, mylar);
354 straw->SetLineColor(4);
355 straw->SetVisibility(kTRUE);
356 // Volume: gas
357 rmin = fWire_thickness/2.+epsS;
358 rmax = fInner_Straw_diameter/2.-epsS;
359 TGeoTube *gas_tube = new TGeoTube("gas",rmin,rmax,fStraw_length-6.*eps);
360 TGeoVolume *gas = new TGeoVolume("gas",gas_tube, sttmix9010_2bar);
361 gas->SetLineColor(5); //only the gas is sensitive
362 AddSensitiveVolume(gas);
363
364 // Volume: wire
365 rmin=0.;
366 rmax = fWire_thickness/2.;
367 TGeoTube *wire_tube = new TGeoTube("wire",rmin,rmax,fStraw_length-8.*eps);
368 TGeoVolume *wire = new TGeoVolume("wire",wire_tube, tungsten);
369 wire->SetLineColor(6);
370 Int_t statnb;
371
372 // Tracking stations
373 // statnb=station number; vnb=view number; lnb=layer number; snb=straw number
374
375 //New scalable endpoints of vacuum boxes which cover rotated view frames
376
377 Double_t x_prime =
378 (fVacBox_x + 0.6 * fFrame_lateral_width + 2 * eps) * TMath::Cos(fView_angle * TMath::Pi() / 180.0)
379 + (f_station_height + fFrame_lateral_width + 2 * eps) * TMath::Sin(fView_angle * TMath::Pi() / 180.0);
380 Double_t y_prime =
381 (fVacBox_x + 0.6 * fFrame_lateral_width + 2 * eps) * TMath::Sin(fView_angle * TMath::Pi() / 180.0)
382 + (f_station_height + fFrame_lateral_width + 2 * eps) * TMath::Cos(fView_angle * TMath::Pi() / 180.0);
383
384 TGeoBBox* vacbox = new TGeoBBox("vacbox", x_prime + eps, y_prime + eps, 2. * fDeltaz_view);
385
386 fFrame_material.ToLower();
387
388 for (statnb = 1; statnb < 5; statnb++) {
389 // Tracking station loop
390 TString nmstation = "Tr";
391 std::stringstream ss;
392 ss << statnb;
393 nmstation = nmstation + ss.str();
394 TGeoVolume* vac;
395 switch (statnb) {
396 case 1:
397 TStationz = fT1z;
398 vac = new TGeoVolume(nmstation, vacbox, med);
399 top->AddNode(vac, statnb, new TGeoTranslation(0, 0, TStationz));
400 break;
401 case 2:
402 TStationz = fT2z;
403 vac = new TGeoVolume(nmstation, vacbox, med);
404 top->AddNode(vac, statnb, new TGeoTranslation(0, 0, TStationz));
405 break;
406 case 3:
407 TStationz = fT3z;
408 vac = new TGeoVolume(nmstation, vacbox, med);
409 top->AddNode(vac, statnb, new TGeoTranslation(0, 0, TStationz));
410 break;
411 case 4:
412 TStationz = fT4z;
413 vac = new TGeoVolume(nmstation, vacbox, med);
414 top->AddNode(vac, statnb, new TGeoTranslation(0, 0, TStationz));
415 break;
416 default:
417 break;
418 }
419
420 for (Int_t vnb = 0; vnb < 4; vnb++) {
421 // View loop
422 TString nmview;
423 Double_t angle;
424 TGeoRotation r5;
425 TGeoTranslation t5;
426
427 switch (vnb) {
428 case 0:
429 angle = 0.;
430 nmview = nmstation + "_x1";
431 break;
432 case 1:
433 angle = fView_angle;
434 nmview = nmstation + "_u";
435 break;
436 case 2:
437 angle = -fView_angle;
438 nmview = nmstation + "_v";
439 break;
440 case 3:
441 angle = 0.;
442 nmview = nmstation + "_x2";
443 break;
444 default:
445 angle = 0.;
446 nmview = nmstation + "_x1";
447 }
448
449 TGeoVolume* viewframe;
450 if (fFrame_material.Contains("aluminium")) {
451 viewframe = new TGeoVolume(nmview, detcomp1, Al);
452 } else {
453 viewframe = new TGeoVolume(nmview, detcomp1, FrameMatPtr);
454 }
455
456 // z-translate the viewframe from station z pos
457 t5.SetTranslation(0, 0, (vnb - 3. / 2.) * fDeltaz_view);
458 // Rotate the frame box by angle degrees around the z axis (0 if it isn't a stereo view)
459 r5.SetAngles(angle, 0, 0);
460 TGeoCombiTrans c5(t5, r5);
461 TGeoHMatrix* h5 = new TGeoHMatrix(c5);
462
463 vac->AddNode(viewframe, statnb * 1e6 + vnb * 1e5, h5);
464 viewframe->SetLineColor(kRed);
465
466 for (Int_t lnb = 0; lnb < 2; lnb++) {
467
468 // Layer loop
469 TString nmlayer = nmview + "_layer_";
470 nmlayer += lnb;
471 TGeoBBox* layer = new TGeoBBox(
472 "layer box", fStraw_length + eps / 4, f_station_height + eps / 4, layerwidth / 2. + eps / 4);
473 TGeoVolume* layerbox = new TGeoVolume(nmlayer, layer, med);
474
475 // The layer box sits in the viewframe.
476 // Hence, z-translate the layer w.r.t. the view
477 TGeoTranslation t3;
478 t3.SetTranslation(0, 0, (vnb - 3. / 2.) * fDeltaz_view + (lnb - 1. / 2.) * fDeltaz_layer12);
479 TGeoCombiTrans d3(t3, r5);
480 TGeoHMatrix* j3 = new TGeoHMatrix(d3);
481 vac->AddNode(layerbox, statnb * 1e6 + vnb * 1e5 + lnb * 1e4, j3);
482
483 TGeoRotation r6s;
484 TGeoTranslation t6s;
485 for (Int_t snb = 1; snb < fStraws_per_layer; snb++) {
486 // Straw loop
487 t6s.SetTranslation(0, f_station_height - fStraw_pitch * snb + lnb * fOffset_layer12, 0);
488 r6s.SetAngles(90, 90, 0);
489 TGeoCombiTrans c6s(t6s, r6s);
490 TGeoHMatrix* h6s = new TGeoHMatrix(c6s);
491 layerbox->AddNode(straw, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 1e3 + snb, h6s);
492 layerbox->AddNode(gas, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 2e3 + snb, h6s);
493 layerbox->AddNode(wire, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 3e3 + snb, h6s);
494
495 // End of straw loop
496 }
497 // End of layer loop
498 }
499 // End of view loop
500 }
501
502 // End of station
503 }
504 std::cout << "tracking stations added" << std::endl;
505}
506// ----- Public method StrawDecode -------------------------------------------
507// ----- returns station, view, layer, straw number in a tuple -----------------------------------
508std::tuple<Int_t, Int_t, Int_t, Int_t> strawtubes::StrawDecode(Int_t detID)
509{
510 Int_t statnb, vnb, lnb, snb;
511 statnb = detID / 1e6;
512 vnb = (detID - statnb * 1e6) / 1e5;
513 lnb = (detID - statnb * 1e6 - vnb * 1e5) / 1e4;
514 snb = detID - statnb * 1e6 - vnb * 1e5 - lnb * 1e4 - 2e3;
515
516 if (statnb < 1 || statnb > 4 || vnb < 0 || vnb > 3 || lnb < 0 || lnb > 1 || snb < 1 || snb > 299) {
517 LOG(warning) << "Invalid strawtubes detID:";
518 LOG(warning) << detID << " -> station: " << statnb << ", view: " << vnb << ", layer: " << lnb
519 << ", straw: " << snb;
520 LOG(warning) << "strawtubes detID is 7-digit!";
521 return std::make_tuple(0, -1, -1, 0);
522 } else {
523 return std::make_tuple(statnb, vnb, lnb, snb);
524 }
525}

569 path += wire;

578 Double_t bot[3] = {0, 0, -S->GetDZ()};
579 Double_t Gtop[3], Gbot[3];
580 nav->LocalToMaster(top, Gtop); nav->LocalToMaster(bot, Gbot);
581 vtop.SetXYZ(Gtop[0], Gtop[1], Gtop[2]);
582 vbot.SetXYZ(Gbot[0], Gbot[1], Gbot[2]);

584void strawtubes::StrawEndPointsOriginal(Int_t detID, TVector3 &bot, TVector3 &top)
585// method to get end points by emulating the geometry
586{
587 Double_t sinangle, cosangle;
588 const auto [statnb, vnb, lnb, snb] = StrawDecode(detID);
589 switch (vnb) {
590 case 0:
591 sinangle = 0.;
592 cosangle = 1.;
593 break;
594 case 1:
595 sinangle = fsinphi;
596 cosangle = fcosphi;
597 break;
598 case 2:
599 sinangle = -fsinphi;
600 cosangle = fcosphi;
601 break;
602 case 3:
603 sinangle = 0.;
604 cosangle = 1.;
605 break;
606 default:
607 sinangle = 0.;
608 cosangle = 1.;
609 }
610
611 // from ConstructGeometry above
612 Double_t yDim = (fStraws_per_layer + 1) * fStraw_pitch / 2.;
613 Double_t ypos = f_station_height - fStraw_pitch * snb + lnb * fOffset_layer12;
614 Double_t xtop = -fStraw_length * cosangle - ypos * sinangle;
615 Double_t ytop = -fStraw_length * sinangle + ypos * cosangle;
616 Double_t xbot = fStraw_length * cosangle - ypos * sinangle;
617 Double_t ybot = fStraw_length * sinangle + ypos * cosangle;
618
619 Double_t TStationz;
620 switch (statnb) {
621 case 1:
622 TStationz = fT1z;
623 break;
624 case 2:
625 TStationz = fT2z;
626 break;
627 case 3:
628 TStationz = fT3z;
629 break;
630 case 4:
631 TStationz = fT4z;
632 break;
633 default:
634 TStationz = fT1z;
635 }
