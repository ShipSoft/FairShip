116Bool_t strawtubes::ProcessHits(FairVolume* vol)
117{
119 //Set parameters at entrance of volume. Reset ELoss.
120 if ( gMC->IsTrackEntering() ) {
121 fELoss = 0.;
122 fTime = gMC->TrackTime() * 1.0e09;
123 fLength = gMC->TrackLength();
124 gMC->TrackPosition(fPos);
125 gMC->TrackMomentum(fMom);
126 }
127 // Sum energy loss for all steps in the active volume
128 fELoss += gMC->Edep();
129
130 // Create strawtubesPoint at exit of active volume
131 if ( gMC->IsTrackExiting() ||
132 gMC->IsTrackStop() ||
133 gMC->IsTrackDisappeared() ) {
134 if (fELoss == 0. ) { return kFALSE; }
135 TParticle* p=gMC->GetStack()->GetCurrentTrack();
136 Int_t pdgCode = p->GetPdgCode();
137 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
138 Int_t straw_uniqueId;
139 gMC->CurrentVolID(straw_uniqueId);
140 if (fVolumeID == straw_uniqueId) {
141 //std::cout << pdgCode<< " same volume again ? "<< straw_uniqueId << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
142 return kTRUE; }
143 fVolumeID = straw_uniqueId;
144 // # d = |pq . u x v|/|u x v|
145 TVector3 bot,top;
146 StrawEndPoints(straw_uniqueId,bot,top);
147 TLorentzVector Pos;
148 gMC->TrackPosition(Pos);
149 Double_t xmean = (fPos.X()+Pos.X())/2. ;
150 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
151 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
152 TVector3 pq = TVector3(top.x()-xmean,top.y()-ymean,top.z()-zmean );
153 TVector3 u = TVector3(bot.x()-top.x(),bot.y()-top.y(),bot.z()-top.z() );
154 TVector3 v = TVector3(fPos.X()-Pos.X(),fPos.Y()-Pos.Y(),fPos.Z()-Pos.Z());
155 TVector3 uCrossv = u.Cross(v);
156 Double_t dist2Wire = fabs(pq.Dot(uCrossv))/(uCrossv.Mag()+1E-8);
157 Double_t deltaTrackLength = gMC->TrackLength() - fLength;
158 AddHit(fTrackID, straw_uniqueId, TVector3(xmean, ymean, zmean),
159 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, deltaTrackLength,
160 fELoss,pdgCode,dist2Wire);
161 if (dist2Wire > f_inner_straw_diameter / 2){
162 std::cout << "addhit " << dist2Wire<< " straw id " << straw_uniqueId << " pdgcode " << pdgCode<< " dot prod " << pq.Dot(uCrossv)<< std::endl;
163 std::cout << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
164 std::cout << " entry:" << fPos.X()<< " " << fPos.Y()<< " " << fPos.Z() << std::endl;
165 std::cout << " exit:" << Pos.X()<< " " << Pos.Y()<< " " << Pos.Z() << std::endl;
166 std::cout << " mean:" << xmean<< " " << ymean << " " << zmean << std::endl;
167 std::cout << " bot:" << bot.x()<< " " << bot.y() << " " << bot.z() << std::endl;
168 std::cout << " top:" << top.x()<< " " << top.y() << " " << top.z() << std::endl;
169 pq.Print();
170 u.Print();
171 v.Print();
172 uCrossv.Print();
173 }
174 // Increment number of strawtubes det points in TParticle
175 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
176 stack->AddPoint(kStraw);
177 }
178 return kTRUE;
179}

271void strawtubes::ConstructGeometry()
272{
277 TGeoVolume *top = gGeoManager->GetTopVolume();
278 InitMedium("air");
279 TGeoMedium *air = gGeoManager->GetMedium("air");
280 InitMedium("ShipSens");
281 TGeoMedium *Se = gGeoManager->GetMedium("ShipSens");
282 InitMedium("aluminium");
283 TGeoMedium *Al = gGeoManager->GetMedium("aluminium");
284 InitMedium("mylar");
285 TGeoMedium *mylar = gGeoManager->GetMedium("mylar");
286 InitMedium("STTmix8020_1bar");
287 TGeoMedium *sttmix8020_1bar = gGeoManager->GetMedium("STTmix8020_1bar");
288 InitMedium("tungsten");
289 TGeoMedium *tungsten = gGeoManager->GetMedium("tungsten");
290 InitMedium(f_frame_material);
291 TGeoMedium *FrameMatPtr = gGeoManager->GetMedium(f_frame_material);
292 InitMedium(fMedium.c_str());
293 TGeoMedium* med = gGeoManager->GetMedium(fMedium.c_str());
294
295 gGeoManager->SetVisLevel(4);
296 gGeoManager->SetTopVisible();
297
298 // Epsilon to avoid overlapping volumes
299 Double_t eps = 0.0001;
300 // Straw (half) length
301 Double_t straw_length = f_aperture_width + 2. * eps;
302 // Width of frame: standard HEA 500 I-beam width
303 Double_t frame_width = 49.;
304 // Offset due to floor space limitation
305 Double_t floor_offset = 14.;
306
307 Double_t rmin, rmax, T_station_z;
308
309 // Arguments of boxes are half-lengths
310 TGeoBBox* detbox1 = new TGeoBBox(
311 "detbox1", f_aperture_width + frame_width, f_aperture_height + frame_width - floor_offset / 2., f_station_length);
312 TGeoBBox* detbox2 = new TGeoBBox(
313 "detbox2",
314 straw_length + eps,
315 f_aperture_height + TMath::Tan(f_view_angle * TMath::Pi() / 180.0) * straw_length * 2 + f_offset_layer / TMath::Cos(f_view_angle * TMath::Pi() / 180.0) + eps,
316 f_station_length + eps);
317 TGeoTranslation* move_up = new TGeoTranslation("move_up", 0, floor_offset / 2., 0);
318 move_up->RegisterYourself();
319
320 // Composite shape to create frame
321 TGeoCompositeShape* detcomp1 = new TGeoCompositeShape("detcomp1", "(detbox1:move_up)-detbox2");
322
323 // Volume: straw
324 rmin = f_inner_straw_diameter / 2.;
325 rmax = f_outer_straw_diameter / 2.;
326 // Third argument is half-length of tube
327 TGeoTube *straw_tube = new TGeoTube("straw", rmin, rmax, straw_length);
328 TGeoVolume *straw = new TGeoVolume("straw", straw_tube, mylar);
329 straw->SetLineColor(4);
330 straw->SetVisibility(kTRUE);
331
332 // Volume: gas
333 rmin = f_wire_thickness / 2. + eps;
334 rmax = f_inner_straw_diameter / 2. - eps;
335 TGeoTube *gas_tube = new TGeoTube("gas", rmin, rmax, straw_length - 2. * eps);
336 TGeoVolume *gas = new TGeoVolume("gas", gas_tube, sttmix8020_1bar);
337 gas->SetLineColor(5);
338 // Only the gas is sensitive
339 AddSensitiveVolume(gas);
340
341 // Volume: wire
342 rmin = 0.;
343 rmax = f_wire_thickness / 2.;
344 TGeoTube *wire_tube = new TGeoTube("wire", rmin, rmax, straw_length - 4. * eps);
345 TGeoVolume *wire = new TGeoVolume("wire", wire_tube, tungsten);
346 wire->SetLineColor(6);
347
348 // Tracking stations
349 // statnb = station number; vnb = view number; lnb = layer number; snb = straw number
350
351 // Station box to contain all components
352 TGeoBBox* statbox = new TGeoBBox("statbox", f_station_width, f_station_height - floor_offset / 2., f_station_length);
353
354 f_frame_material.ToLower();
355
356 for (Int_t statnb = 1; statnb < 5; statnb++) {
357 // Tracking station loop
358 TString nmstation = "Tr";
359 std::stringstream ss;
360 ss << statnb;
361 nmstation = nmstation + ss.str();
362 switch (statnb) {
363 case 1:
364 T_station_z = f_T1_z;
365 break;
366 case 2:
367 T_station_z = f_T2_z;
368 break;
369 case 3:
370 T_station_z = f_T3_z;
371 break;
372 case 4:
373 T_station_z = f_T4_z;
374 break;
375 default:
376 T_station_z = f_T1_z;
377 }
378
379 TGeoVolume* vol = new TGeoVolume(nmstation, statbox, med);
380 // z-translate the station to its (absolute) position
381 top->AddNode(vol, statnb, new TGeoTranslation(0, floor_offset / 2., T_station_z));
382
383 TGeoVolume* statframe = new TGeoVolume(nmstation + "_frame", detcomp1, FrameMatPtr);
384 vol->AddNode(statframe, statnb * 1e6, new TGeoTranslation(0, -floor_offset / 2., 0));
385 statframe->SetLineColor(kRed);
386
387 for (Int_t vnb = 0; vnb < 4; vnb++) {
388 // View loop
389 TString nmview;
390 Double_t angle;
391 Double_t stereo_growth;
392 Double_t stereo_pitch;
393 Double_t offset_layer;
394 Int_t straws_per_layer;
395
396 switch (vnb) {
397 case 0:
398 angle = 0.;
399 nmview = nmstation + "_y1";
400 break;
401 case 1:
402 angle = f_view_angle;
403 nmview = nmstation + "_u";
404 break;
405 case 2:
406 angle = -f_view_angle;
407 nmview = nmstation + "_v";
408 break;
409 case 3:
410 angle = 0.;
411 nmview = nmstation + "_y2";
412 break;
413 default:
414 angle = 0.;
415 nmview = nmstation + "_y1";
416 }
417
418 // Adjustments in the stereo views
419 // stereo_growth: extension of stereo views beyond aperture
420 // stereo_pitch: straw pitch in stereo views
421 // offset_layer: layer offset in stereo views
422 // straws_per_layer: number of straws in one layer with stereo extension
423 // If angle == 0., all numbers return the case of non-stereo views.
424 stereo_growth = TMath::Tan(TMath::Abs(angle) * TMath::Pi() / 180.0) * straw_length;
425 stereo_pitch = f_straw_pitch / TMath::Cos(TMath::Abs(angle) * TMath::Pi() / 180.0);
426 offset_layer = f_offset_layer / TMath::Cos(TMath::Abs(angle) * TMath::Pi() / 180.0);
427 straws_per_layer = std::ceil(2 * (f_aperture_height + stereo_growth) / stereo_pitch);
428
429 for (Int_t lnb = 0; lnb < 2; lnb++) {
430 // Layer loop
431 TString nmlayer = nmview + "_layer_";
432 nmlayer += lnb;
433 TGeoBBox* layer = new TGeoBBox(
434 "layer box", straw_length + eps / 4, f_aperture_height + stereo_growth * 2 + offset_layer + eps / 4, f_outer_straw_diameter / 2. + eps / 4);
435 TGeoVolume* layerbox = new TGeoVolume(nmlayer, layer, med);
436
437 // The layer box sits in the viewframe.
438 // Hence, z-translate the layer w.r.t. the view
439 vol->AddNode(layerbox, statnb * 1e6 + vnb * 1e5 + lnb * 1e4, new TGeoTranslation(0, -floor_offset / 2., (vnb - 3. / 2.) * f_delta_z_view + (lnb - 1. / 2.) * f_delta_z_layer));
440
441 TGeoRotation r6s;
442 TGeoTranslation t6s;
443 for (Int_t snb = 1; snb <= straws_per_layer; snb++) {
444 // Straw loop
445 // y-translate the straw to its position
446 t6s.SetTranslation(0, f_aperture_height + stereo_growth - (snb - 1. / 2.) * stereo_pitch + lnb * offset_layer, 0);
447 // Rotate the straw with stereo angle
448 r6s.SetAngles(90 + angle, 90, 0);
449 TGeoCombiTrans c6s(t6s, r6s);
450 TGeoHMatrix* h6s = new TGeoHMatrix(c6s);
451 layerbox->AddNode(straw, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 1e3 + snb, h6s);
452 layerbox->AddNode(gas, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 2e3 + snb, h6s);
453 layerbox->AddNode(wire, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 3e3 + snb, h6s);
454 // End of straw loop
455 }
456
457 if (lnb == 1) {
458 // Add one more straw at the bottom of the second layer to cover aperture entirely
459 t6s.SetTranslation(0, f_aperture_height + stereo_growth - (straws_per_layer - 1. / 2.) * stereo_pitch - lnb * offset_layer, 0);
460 r6s.SetAngles(90 + angle, 90, 0);
461 TGeoCombiTrans c6s(t6s, r6s);
462 TGeoHMatrix* h6s = new TGeoHMatrix(c6s);
463 layerbox->AddNode(straw, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 1e3 + straws_per_layer + 1, h6s);
464 layerbox->AddNode(gas, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 2e3 + straws_per_layer + 1, h6s);
465 layerbox->AddNode(wire, statnb * 1e6 + vnb * 1e5 + lnb * 1e4 + 3e3 + straws_per_layer + 1, h6s);
466 }
467 // End of layer loop
468 }
469 // End of view loop
470 }
471 // End of tracking station loop
472 }
473}
474// ----- Public method StrawDecode -------------------------------------------
475// ----- returns station, view, layer, straw number in a tuple -----------------------------------
476std::tuple<Int_t, Int_t, Int_t, Int_t> strawtubes::StrawDecode(Int_t detID)
477{
478 Int_t statnb, vnb, lnb, snb;
479 statnb = detID / 1e6;
480 vnb = (detID - statnb * 1e6) / 1e5;
481 lnb = (detID - statnb * 1e6 - vnb * 1e5) / 1e4;
482 snb = detID - statnb * 1e6 - vnb * 1e5 - lnb * 1e4 - 2e3;
483
484 if (statnb < 1 || statnb > 4 || vnb < 0 || vnb > 3 || lnb < 0 || lnb > 1 || snb < 1 || snb > 317) {
485 LOG(warning) << "Invalid strawtubes detID:";
486 LOG(warning) << detID << " -> station: " << statnb << ", view: " << vnb << ", layer: " << lnb
487 << ", straw: " << snb;
488 LOG(warning) << "strawtubes detID is 7-digit!";
489 return std::make_tuple(0, -1, -1, 0);
490 } else {
491 return std::make_tuple(statnb, vnb, lnb, snb);
492 }
493}
