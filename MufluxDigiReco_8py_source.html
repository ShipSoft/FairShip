22 def StripX(x):

24  STRIP_XWIDTH = 0.8625 # internal STRIP V, WIDTH, in cm

26  EXT_STRIP_XWIDTH_R = 0.86 # measured Right external STRIP V, WIDTH,in cm (beam along z, out from the V plane)
27  V_STRIP_OFF = 0.200
28  NR_VER_STRIPS = 184
29  total_width = (NR_VER_STRIPS - 2) * STRIP_XWIDTH + EXT_STRIP_XWIDTH_L + EXT_STRIP_XWIDTH_R + (NR_VER_STRIPS - 1) * V_STRIP_OFF
30  x_start = (total_width - EXT_STRIP_XWIDTH_R + EXT_STRIP_XWIDTH_L) / 2
31  # calculating strip as an integer
32  strip_x = (x_start - EXT_STRIP_XWIDTH_L + 1.5 * STRIP_XWIDTH + V_STRIP_OFF - x)//(STRIP_XWIDTH + V_STRIP_OFF)
33  if not (0 < strip_x <= NR_VER_STRIPS):
34  print("WARNING: X strip outside range!")
