181// Test vector operations
182template<typename T>
183bool test_vector_operations(const char* class_name)
184{
185 std::cout << "Testing " << class_name << "... ";
186
187 // Test 1: Can we create a vector?
188 if (!test_copyable<T>()) {
189 std::cout << "FAIL: Not copyable (private copy constructor?)" << std::endl;
190 return false;
191 }
192
193 try {
194 // Test 2: Create default objects
195 std::vector<T> vec;
196 vec.reserve(10);
197
198 // Test 3: Emplace back (uses default constructor)
199 vec.emplace_back();
200
201 // Test 4: Push back (uses copy constructor)
202 T obj;
203 vec.push_back(obj);
204
205 // Test 5: Access elements with const reference
206 const T& const_ref = vec[0];
207 test_const_access(const_ref);
208
209 // Test 6: Vector resize (uses copy constructor)
210 vec.resize(5);
211
212 std::cout << "PASS" << std::endl;
213 return true;
214 } catch (const std::exception& e) {
215 std::cout << "FAIL: " << e.what() << std::endl;
216 return false;
217 }
218}
