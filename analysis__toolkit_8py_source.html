39 def define_candidate_time(self, candidate):
40 """Calculate time associated with the candidate decay vertex using strawtubes MCPoint info."""
41 t0 = self.tree.ShipEventHeader.GetEventTime()
42 candidate_pos = ROOT.TVector3()
43 candidate.GetVertex(candidate_pos)
44
45 d1, d2 = candidate.GetDaughter(0), candidate.GetDaughter(1)
46 d1_mc, d2_mc = self.tree.fitTrack2MC[d1], self.tree.fitTrack2MC[d2]
47
48 time_vtx_from_strawhits = []
49
50 for hit in self.tree.strawtubesPoint:
51 if not (
52 int(str(hit.GetDetectorID())[:1]) == 1
53 or int(str(hit.GetDetectorID())[:1]) == 2
54 ):
55 continue
56
57 if not (hit.GetTrackID() == d1_mc or hit.GetTrackID() == d2_mc):
58 continue
59
60 t_straw = hit.GetTime()
61
62 dist = np.sqrt(
63 (candidate_pos.X() - hit.GetX()) ** 2
64 + (candidate_pos.Y() - hit.GetY()) ** 2
65 + (candidate_pos.Z() - hit.GetZ()) ** 2
66 ) # distance to the vertex in cm
67
68 d_mom = self.tree.MCTrack[hit.GetTrackID()].GetP() / u.GeV
69 mass = self.tree.MCTrack[hit.GetTrackID()].GetMass()
70 v = u.c_light * d_mom / np.sqrt(d_mom**2 + (mass) ** 2)
71
72 t_vertex = t_straw - (dist / v)
73
74 time_vtx_from_strawhits.append(t_vertex)
75
76 t_vtx = np.average(time_vtx_from_strawhits) + t0
77
78 return t_vtx # units in ns

80 def impact_parameter(self, candidate):
81 """Calculate the impact parameter of the candidate relative to (0,0,target.z0)."""
82 candidate_pos = ROOT.TVector3()
83 candidate.GetVertex(candidate_pos)
84
85 candidate_mom = ROOT.TLorentzVector()
86 candidate.Momentum(candidate_mom)
87 target_point = ROOT.TVector3(0, 0, self.ship_geo.target.z0)
88
89 projection_factor = 0
90 if hasattr(candidate_mom, "P"):
91 P = candidate_mom.P()
92 else:
93 P = candidate_mom.Mag()
94 for i in range(3):
95 projection_factor += (
96 candidate_mom(i) / P * (target_point(i) - candidate_pos(i))
97 )
98
99 dist = 0
100 for i in range(3):
101 dist += (
102 target_point(i)
103 - candidate_pos(i)
104 - projection_factor * candidate_mom(i) / P
105 ) ** 2
106 dist = ROOT.TMath.Sqrt(dist)
107
108 return dist # in cm

110 def dist_to_innerwall(self, candidate):
111 """Calculate the minimum distance(in XY plane) of the candidate decay vertex to the inner wall of the decay vessel. If outside the decay volume, or if distance > 100cm,Return 0."""
112 candidate_pos = ROOT.TVector3()
113 candidate.GetVertex(candidate_pos)
114 position = (candidate_pos.X(), candidate_pos.Y(), candidate_pos.Z())
115
116 nsteps = 8
117 dalpha = 2 * ROOT.TMath.Pi() / nsteps
118 min_distance = float("inf")
119
120 node = self.geometry_manager.FindNode(*position)
121 if not node:
122 return 0 # is outside the decay volume
123
124 # Loop over directions in the XY plane
125 for n in range(nsteps):
126 alpha = n * dalpha
127 direction = (
128 ROOT.TMath.Sin(alpha),
129 ROOT.TMath.Cos(alpha),
130 0.0,
131 ) # Direction vector in XY plane
132 self.geometry_manager.InitTrack(*position, *direction)
133 if not self.geometry_manager.FindNextBoundary():
134 continue
135 # Get the distance to the boundary and update the minimum distance
136 distance = self.geometry_manager.GetStep()
137 min_distance = min(min_distance, distance)
138
139 return min_distance if min_distance < 100 * u.m else 0
