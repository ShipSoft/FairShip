38 def define_candidate_time(self, candidate):
39 """Calculate time associated with the candidate decay vertex using strawtubes MCPoint info."""
40 t0 = self.tree.ShipEventHeader.GetEventTime()
41 candidate_pos = ROOT.TVector3()
42 candidate.GetVertex(candidate_pos)
43
44 d1, d2 = candidate.GetDaughter(0), candidate.GetDaughter(1)
45 d1_mc, d2_mc = self.tree.fitTrack2MC[d1], self.tree.fitTrack2MC[d2]
46
47 time_vtx_from_strawhits = []
48
49 for hit in self.tree.strawtubesPoint:
50 if not (
51 int(str(hit.GetDetectorID())[:1]) == 1
52 or int(str(hit.GetDetectorID())[:1]) == 2
53 ):
54 continue
55
56 if not (hit.GetTrackID() == d1_mc or hit.GetTrackID() == d2_mc):
57 continue
58
59 t_straw = hit.GetTime()
60
61 dist = np.sqrt(
62 (candidate_pos.X() - hit.GetX()) ** 2
63 + (candidate_pos.Y() - hit.GetY()) ** 2
64 + (candidate_pos.Z() - hit.GetZ()) ** 2
65 ) # distance to the vertex in cm
66
67 d_mom = self.tree.MCTrack[hit.GetTrackID()].GetP() / u.GeV
68 mass = self.tree.MCTrack[hit.GetTrackID()].GetMass()
69 v = u.c_light * d_mom / np.sqrt(d_mom**2 + (mass) ** 2)
70
71 t_vertex = t_straw - (dist / v)
72
73 time_vtx_from_strawhits.append(t_vertex)
74
75 t_vtx = np.average(time_vtx_from_strawhits) + t0
76
77 return t_vtx # units in ns

79 def impact_parameter(self, candidate):
80 """Calculate the impact parameter of the candidate relative to (0,0,target.z0)."""
81 candidate_pos = ROOT.TVector3()
82 candidate.GetVertex(candidate_pos)
83
84 candidate_mom = ROOT.TLorentzVector()
85 candidate.Momentum(candidate_mom)
86 target_point = ROOT.TVector3(0, 0, self.ship_geo.target.z0)
87
88 projection_factor = 0
89 if hasattr(candidate_mom, "P"):
90 P = candidate_mom.P()
91 else:
92 P = candidate_mom.Mag()
93 for i in range(3):
94 projection_factor += (
95 candidate_mom(i) / P * (target_point(i) - candidate_pos(i))
96 )
97
98 dist = 0
99 for i in range(3):
100 dist += (
101 target_point(i)
102 - candidate_pos(i)
103 - projection_factor * candidate_mom(i) / P
104 ) ** 2
105 dist = ROOT.TMath.Sqrt(dist)
106
107 return dist # in cm

109 def dist_to_innerwall(self, candidate):
110 """Calculate the minimum distance(in XY plane) of the candidate decay vertex to the inner wall of the decay vessel. If outside the decay volume, or if distance > 100cm,Return 0."""
111 candidate_pos = ROOT.TVector3()
112 candidate.GetVertex(candidate_pos)
113 position = (candidate_pos.X(), candidate_pos.Y(), candidate_pos.Z())
114
115 nsteps = 8
116 dalpha = 2 * ROOT.TMath.Pi() / nsteps
117 min_distance = float("inf")
118
119 node = self.geometry_manager.FindNode(*position)
120 if not node:
121 return 0 # is outside the decay volume
122
123 # Loop over directions in the XY plane
124 for n in range(nsteps):
125 alpha = n * dalpha
126 direction = (
127 ROOT.TMath.Sin(alpha),
128 ROOT.TMath.Cos(alpha),
129 0.0,
130 ) # Direction vector in XY plane
131 self.geometry_manager.InitTrack(*position, *direction)
132 if not self.geometry_manager.FindNextBoundary():
133 continue
134 # Get the distance to the boundary and update the minimum distance
135 distance = self.geometry_manager.GetStep()
136 min_distance = min(min_distance, distance)
137
138 return min_distance if min_distance < 100 * u.m else 0
