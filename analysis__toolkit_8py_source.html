37 def define_candidate_time(self, candidate):
38 """Calculate time associated with the candidate decay vertex using strawtubes MCPoint info."""
39 t0 = self.tree.ShipEventHeader.GetEventTime()
40 candidate_pos = ROOT.TVector3()
41 candidate.GetVertex(candidate_pos)
42
43 d1, d2 = candidate.GetDaughter(0), candidate.GetDaughter(1)
44 d1_mc, d2_mc = self.tree.fitTrack2MC[d1], self.tree.fitTrack2MC[d2]
45
46 time_vtx_from_strawhits = []
47
48 for hit in self.tree.strawtubesPoint:
49 if not (
50 int(str(hit.GetDetectorID())[:1]) == 1
51 or int(str(hit.GetDetectorID())[:1]) == 2
52 ):
53 continue
54
55 if not (hit.GetTrackID() == d1_mc or hit.GetTrackID() == d2_mc):
56 continue
57
58 t_straw = hit.GetTime()
59
60 dist = np.sqrt(
61 (candidate_pos.X() - hit.GetX()) ** 2
62 + (candidate_pos.Y() - hit.GetY()) ** 2
63 + (candidate_pos.Z() - hit.GetZ()) ** 2
64 ) # distance to the vertex in cm
65
66 d_mom = self.tree.MCTrack[hit.GetTrackID()].GetP() / u.GeV
67 mass = self.tree.MCTrack[hit.GetTrackID()].GetMass()
68 v = u.c_light * d_mom / np.sqrt(d_mom**2 + (mass) ** 2)
69
70 t_vertex = t_straw - (dist / v)
71
72 time_vtx_from_strawhits.append(t_vertex)
73
74 t_vtx = np.average(time_vtx_from_strawhits) + t0
75
76 return t_vtx # units in ns

78 def impact_parameter(self, candidate):
79 """Calculate the impact parameter of the candidate relative to (0,0,target.z0)."""
80 candidate_pos = ROOT.TVector3()
81 candidate.GetVertex(candidate_pos)
82
83 candidate_mom = ROOT.TLorentzVector()
84 candidate.Momentum(candidate_mom)
85 target_point = ROOT.TVector3(0, 0, self.ship_geo.target.z0)
86
87 projection_factor = 0
88 if hasattr(candidate_mom, "P"):
89 P = candidate_mom.P()
90 else:
91 P = candidate_mom.Mag()
92 for i in range(3):
93 projection_factor += (
94 candidate_mom(i) / P * (target_point(i) - candidate_pos(i))
95 )
96
97 dist = 0
98 for i in range(3):
99 dist += (
100 target_point(i)
101 - candidate_pos(i)
102 - projection_factor * candidate_mom(i) / P
103 ) ** 2
104 dist = ROOT.TMath.Sqrt(dist)
105
106 return dist # in cm

108 def dist_to_innerwall(self, candidate):
109 """Calculate the minimum distance(in XY plane) of the candidate decay vertex to the inner wall of the decay vessel. If outside the decay volume, or if distance > 100cm,Return 0."""
110 candidate_pos = ROOT.TVector3()
111 candidate.GetVertex(candidate_pos)
112 position = (candidate_pos.X(), candidate_pos.Y(), candidate_pos.Z())
113
114 nsteps = 8
115 dalpha = 2 * ROOT.TMath.Pi() / nsteps
116 min_distance = float("inf")
117
118 node = self.geometry_manager.FindNode(*position)
119 if not node:
120 return 0 # is outside the decay volume
121
122 # Loop over directions in the XY plane
123 for n in range(nsteps):
124 alpha = n * dalpha
125 direction = (
126 ROOT.TMath.Sin(alpha),
127 ROOT.TMath.Cos(alpha),
128 0.0,
129 ) # Direction vector in XY plane
130 self.geometry_manager.InitTrack(*position, *direction)
131 if not self.geometry_manager.FindNextBoundary():
132 continue
133 # Get the distance to the boundary and update the minimum distance
134 distance = self.geometry_manager.GetStep()
135 min_distance = min(min_distance, distance)
136
137 return min_distance if min_distance < 100 * u.m else 0
