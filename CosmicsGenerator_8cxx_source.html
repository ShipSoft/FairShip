16double Co3Rng::fSpectrumL(double theta, double minE, Bool_t generateP = 1){
17 // 2 Options: a) generateP, b) calcInt
18 // see doi: 10.1016/j.nuclphysbps.2005.07.056. for flux details
19 // ad a) returns a random P between minE and 100GeV taken from the
20 // zenith angle dependent momentum distribution
21 // Here, the inverse of the function is computed and a random
22 // number between 0 and 1 mapped to the interval [minE, 100[ GeV
23 // ad b) return the momentum-integrated flux for a given zenith angle
24 // from minE to 100 GeV. Result in cm-2s-1
25
26 theta = 180 * theta / TMath::Pi(); // theta in degrees
27 double a = -0.8816 / 10000 / (1 / theta - 0.1117 / 1000 * theta) - 0.1096 - 0.01966 * TMath::Exp(-0.02040 * theta);
28 double b =
29 0.4169 / 100 / (1 / theta - 0.9891 / 10000 * theta) + 4.0395 - 4.3118 * TMath::Exp(0.9235 / 1000 * theta);
30 double btilde = b + 1.0 / TMath::Ln10();
31 double gamma = sqrt(-TMath::Ln10() * a);
32 double offset = 0.5 * btilde / a;
33 double norm = TMath::Erf(gamma * (TMath::Log(100) + offset)) - TMath::Erf(gamma * (offset + TMath::Log(minE)));
34
35 if (generateP) {
36 double r3 = rng->Uniform();
37 return exp(TMath::ErfInverse(r3 * norm + TMath::Erf(gamma * (offset + TMath::Log(minE)))) / gamma - offset);
38 }
39 else{
40 double c = -0.3516/1000 * theta*theta + 0.8861/100 * theta - 2.5985 -0.8745/100000*TMath::Exp(0.1457*theta);
41 double scale = 0.5*TMath::Sqrt(TMath::Pi())/gamma * TMath::Power(10,(-0.25/a*btilde*btilde + c));
42 return scale * norm;
43 }
44}

46Bool_t CosmicsGenerator::DetectorBox(){
47 // check, if a given staring setup x,y,z,px,py,pz will lead to a
48 // close enough to the detector
49
50 if ((TMath::Abs(x-(y+yBox)*px/py) < xBox && TMath::Abs(z-z0-(y+yBox)*pz/py) < zBox) ||
51 (TMath::Abs(x-(y-yBox)*px/py) < xBox && TMath::Abs(z-z0-(y-yBox)*pz/py) < zBox) ||
52 (TMath::Abs(y-(x+xBox)*py/px) < yBox && TMath::Abs(z-z0-(x+xBox)*pz/px) < zBox) ||
53 (TMath::Abs(y-(x-xBox)*py/px) < yBox && TMath::Abs(z-z0-(x-xBox)*pz/px) < zBox)) {
54 return true;
55 }
56 return false;
57}

59void CosmicsGenerator::GenerateDynamics(){
60 // generate starting conditions for CM until the DetectorBox is hit
61 do{
62 weighttest += weight; nTest++; //book keeping
63 //momentum components
64 double phi = fRandomEngine->Uniform(0,2*TMath::Pi());
65 theta = fRandomEngine->fTheta->GetRandom();
66 px = TMath::Sin(phi)*TMath::Sin(theta);
67 pz = TMath::Cos(phi)*TMath::Sin(theta);
68 py = -TMath::Cos(theta);
69 //staring location
70 x = fRandomEngine->Uniform(-xdist/2,xdist/2);
71 z = fRandomEngine->Uniform(z0 - zdist/2, z0 + zdist/2);
72 }while(!DetectorBox());
73 nInside++;
74}

76Bool_t CosmicsGenerator::Init(Bool_t largeMom){
77 //general
78 fRandomEngine = new Co3Rng();
79 TDatabasePDG* pdgBase = TDatabasePDG::Instance();
80 mass = pdgBase->GetParticle(13)->Mass(); // muons!
81 cout<<"----------------------------------------------------------------------"<<endl;
82 cout<<"configuration for the CMBG as defined in $FAIRSHIP/python/CMBG_conf.py: "<<endl;
83 cout<<"x_dist: "<<xdist<<endl;
84 cout<<"z_dist: "<<zdist<<endl;
85 cout<<"x_box: "<<xBox<<endl;
86 cout<<"y_box: "<<yBox<<endl;
87 cout<<"z_box: "<<zBox<<endl;
88 cout<<"n_EVENTS:"<<n_EVENTS<<endl;
89 cout<<"minE: "<<minE<<endl<<endl;
90 if (xdist*zdist*n_EVENTS == 0){cout<<"check the configuration for unphysical behavior."<<endl<<"We stop the execution."<<endl<<endl; return kFALSE;}
91
92 high = largeMom;
93 if (high) cout<<"Simulation for high momentum"<<endl;
94 else cout<<"Simulation for low momentum"<<endl;
95
96 // calculating weights for this run
97 // weight_flux: expected #muons per spill/ #simulated events per spill: FluxIntegral*xdist*zdist/EVENTS;
98 // the respective integrals are calculated from the fluxes
99 // weight_DetectorBox: only consider CM hitting the DetectorBox
100 // this is gained from a MC test of 10xEVENTS events
101 double weight_flux, weight_DetectorBox;
102 FluxIntegral = 0;
103 if (!high) { // momentum range 1 GeV - 100 GeV
104 if (minE > 100) {cout<<"choose minE < 100 !"<<endl; return kFALSE;}
105 double dt = TMath::Pi()/2/100;
106 for (double t= dt/2; t< TMath::Pi()/2; t += dt){
107 FluxIntegral += fRandomEngine->fSpectrumL(t, minE, 0);
108 }
109 FluxIntegral = 2*TMath::Pi()/3*FluxIntegral*dt*10000;
110 cout<< "LowE CM flux with P < minE = "<<minE<<" : "<<FluxIntegral<< "m-2s-1"<<endl;
111 }
112 else { // momentum range 100 GeV - 1000 GeV
113 FluxIntegral = 2*TMath::Pi()/3*fRandomEngine->fSpectrumH->Integral(100,1000);
114 cout << "High E CM flux: " << FluxIntegral << "m-2s-1" << endl;
115 }
116 weight_flux = FluxIntegral*xdist*zdist/n_EVENTS/10000;
117 nInside = 0; nTest = 0; weighttest = 0; // book keeping
118 y = 1900; //all muons start 19m over beam axis
119 for (; nInside < 10*n_EVENTS;){
120 GenerateDynamics();
121 }
122 weight_DetectorBox = 0.10 * nTest/n_EVENTS;
123 weight = weight_flux / weight_DetectorBox;
124 cout<<"weight_DetectorBox: "<< weight_DetectorBox<<", weight: "<< weight<<endl;
125 cout<<"----------------------------------------------------------------------"<<endl<<endl;
126 nInside = 0; nTest = 0; weighttest = 0; // book keeping
127
128 return kTRUE;
129}

131Bool_t CosmicsGenerator::ReadEvent(FairPrimaryGenerator* cpg){
132 // muon or anti-muon
133 PID = 26*(fRandomEngine->Uniform(0,1) < 1.0/2.278) - 13;
134 // starting conditions
135 GenerateDynamics();
136 //momentum in the two regions, < or > 100 GeV
137 if (!high) P = fRandomEngine->fSpectrumL(theta, minE);
138 else P = fRandomEngine->fSpectrumH->GetRandom();
139 px = px*P;
140 py = py*P;
141 pz = pz*P;
142 // transfer to Geant4
143 cpg->AddTrack(PID,px,py,pz,x,y,z,-1,true,TMath::Sqrt(P*P+mass*mass),0,weight); // -1 = Mother ID, true = tracking, SQRT(x) = Energy, 0 = t
144 if (!nInside%10000){cout<<nInside/10000<<"10k events have been simulated"<<endl;}
145 return kTRUE;
146}
