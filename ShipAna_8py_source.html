127def VertexError(t1,t2,PosDir,CovMat,scalFac):
128# with improved Vx x,y resolution
129 a,u = PosDir[t1]['position'],PosDir[t1]['direction']
130 c,v = PosDir[t2]['position'],PosDir[t2]['direction']
131 Vsq = v.Dot(v)
132 Usq = u.Dot(u)
133 UV = u.Dot(v)
134 ca = c-a
135 denom = Usq*Vsq-UV**2
136 tmp2 = Vsq*u-UV*v
137 Va = ca.Dot(tmp2)/denom
138 tmp2 = UV*u-Usq*v
139 Vb = ca.Dot(tmp2)/denom
140 X = (a+c+Va*u+Vb*v) * 0.5
141 l1 = a - X + u*Va # l2 = c - X + v*Vb
142 dist = 2. * ROOT.TMath.Sqrt( l1.Dot(l1) )
143 T = ROOT.TMatrixD(3,12)
144 for i in range(3):
145 for k in range(4):
146 for j in range(3):
147 KD = 0
148 if i==j: KD = 1
149 if k==0 or k==2:
150 # cova and covc
151 temp = ( u[j]*Vsq - v[j]*UV )*u[i] + (u[j]*UV-v[j]*Usq)*v[i]
152 sign = -1
153 if k==2 : sign = +1
154 T[i][3*k+j] = 0.5*( KD + sign*temp/denom )
155 elif k==1:
156 # covu
157 aNAZ = denom*( ca[j]*Vsq-v.Dot(ca)*v[j] )
158 aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( u[j]*Vsq-v[j]*UV )
159 bNAZ = denom*( ca[j]*UV+(u.Dot(ca)*v[j]) - 2*ca.Dot(v)*u[j] )
160 bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( u[j]*Vsq-v[j]*UV )
161 T[i][3*k+j] = 0.5*( Va*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
162 elif k==3:
163 # covv
164 aNAZ = denom*( 2*ca.Dot(u)*v[j] - ca.Dot(v)*u[j] - ca[j]*UV )
165 aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( v[j]*Usq-u[j]*UV )
166 bNAZ = denom*( ca.Dot(u)*u[j]-ca[j]*Usq )
167 bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( v[j]*Usq-u[j]*UV )
168 T[i][3*k+j] = 0.5*(Vb*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
169 transT = ROOT.TMatrixD(12,3)
170 transT.Transpose(T)
171 CovTracks = ROOT.TMatrixD(12,12)
172 tlist = [t1,t2]
173 for k in range(2):
174 for i in range(6):
175 for j in range(6):
176 xfac = 1.
177 if i>2: xfac = scalFac[tlist[k]]
178 if j>2: xfac = xfac * scalFac[tlist[k]]
179 CovTracks[i+k*6][j+k*6] = CovMat[tlist[k]][i][j] * xfac
180 # if i==5 or j==5 : CovMat[tlist[k]][i][j] = 0 # ignore error on z-direction
181 tmp = ROOT.TMatrixD(3,12)
182 tmp.Mult(T,CovTracks)
183 covX = ROOT.TMatrixD(3,3)
184 covX.Mult(tmp,transT)
185 return X,covX,dist

215def ImpactParameter(point,tPos,tMom):
216 t = 0
217 if hasattr(tMom,'P'): P = tMom.P()
218 else: P = tMom.Mag()
219 for i in range(3): t += tMom(i)/P*(point(i)-tPos(i))

222 dist = ROOT.TMath.Sqrt(dist)
223 return dist

274def myVertex(t1,t2,PosDir):
275 # closest distance between two tracks
276 # d = |pq . u x v|/|u x v|
277 a = ROOT.TVector3(PosDir[t1][0](0) ,PosDir[t1][0](1), PosDir[t1][0](2))
278 u = ROOT.TVector3(PosDir[t1][1](0),PosDir[t1][1](1),PosDir[t1][1](2))
279 c = ROOT.TVector3(PosDir[t2][0](0) ,PosDir[t2][0](1), PosDir[t2][0](2))
280 v = ROOT.TVector3(PosDir[t2][1](0),PosDir[t2][1](1),PosDir[t2][1](2))
281 pq = a-c
282 uCrossv = u.Cross(v)
283 dist = pq.Dot(uCrossv)/(uCrossv.Mag()+1E-8)
284 # u.a - u.c + s*|u|**2 - u.v*t = 0
285 # v.a - v.c + s*v.u - t*|v|**2 = 0
286 E = u.Dot(a) - u.Dot(c)
287 F = v.Dot(a) - v.Dot(c)
288 A,B = u.Mag2(), -u.Dot(v)
289 C,D = u.Dot(v), -v.Mag2()
290 t = -(C*E-A*F)/(B*C-A*D)
291 X = c.x()+v.x()*t
292 Y = c.y()+v.y()*t
293 Z = c.z()+v.z()*t
294 return X,Y,Z,abs(dist)
295def RedoVertexing(t1,t2):
296 PosDir = {}
297 for tr in [t1,t2]:
298 xx = sTree.FitTracks[tr].getFittedState()
299 PosDir[tr] = [xx.getPos(),xx.getDir()]
300 xv,yv,zv,doca = myVertex(t1,t2,PosDir)
301# as we have learned, need iterative procedure
302 dz = 99999.
303 reps,states,newPosDir = {},{},{}
304 parallelToZ = ROOT.TVector3(0., 0., 1.)
305 rc = True
306 step = 0
307 while dz > 0.1:
308 zBefore = zv
309 newPos = ROOT.TVector3(xv,yv,zv)
310 # make a new rep for track 1,2
311 for tr in [t1,t2]:
312 xx = sTree.FitTracks[tr].getFittedState()
313 reps[tr] = ROOT.genfit.RKTrackRep(xx.getPDG())
314 states[tr] = ROOT.genfit.StateOnPlane(reps[tr])
315 reps[tr].setPosMom(states[tr],xx.getPos(),xx.getMom())
316 try:
317 reps[tr].extrapolateToPoint(states[tr], newPos, False)
318 except:
319 print('SHiPAna: extrapolation did not worked')
320 rc = False
321 break
322 newPosDir[tr] = [reps[tr].getPos(states[tr]),reps[tr].getDir(states[tr])]
323 if not rc: break
324 xv,yv,zv,doca = myVertex(t1,t2,newPosDir)
325 dz = abs(zBefore-zv)
326 step+=1
327 if step > 10:
328 print('abort iteration, too many steps, pos=',xv,yv,zv,' doca=',doca,'z before and dz',zBefore,dz)
329 rc = False
330 break
331 if not rc: return xv,yv,zv,doca,-1 # extrapolation failed, makes no sense to continue

334 mom = reps[tr].getMom(states[tr])

378def ecalCluster2MC(aClus):
379 # return MC track most contributing, and its fraction of energy
380 trackid = ctypes.c_int()
381 energy_dep = ctypes.c_double()
382 mcLink = {}
383 for i in range( aClus.Size() ):
384 mccell = ecalStructure.GetHitCell(aClus.CellNum(i)) # Get i'th cell of the cluster.
385 for n in range( mccell.TrackEnergySize()):
386 mccell.GetTrackEnergySlow(n, trackid, energy_dep)
387 if not abs(trackid.value)<sTree.MCTrack.GetEntries(): tid = -1
388 else: tid = trackid.value
389 if tid not in mcLink: mcLink[tid]=0
390 mcLink[tid]+=energy_dep.value
391# find trackid most contributing
392 eMax,mMax = 0,-1
393 for m in mcLink:
394 if mcLink[m]>eMax:
395 eMax = mcLink[m]
396 mMax = m
397 return mMax,eMax/aClus.Energy()
