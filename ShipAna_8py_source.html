142 def VertexError(t1,t2,PosDir,CovMat,scalFac):
143 # with improved Vx x,y resolution
144  a,u = PosDir[t1]['position'],PosDir[t1]['direction']
145  c,v = PosDir[t2]['position'],PosDir[t2]['direction']
146  Vsq = v.Dot(v)
147  Usq = u.Dot(u)
148  UV = u.Dot(v)
149  ca = c-a
150  denom = Usq*Vsq-UV**2
151  tmp2 = Vsq*u-UV*v
152  Va = ca.Dot(tmp2)/denom
153  tmp2 = UV*u-Usq*v
154  Vb = ca.Dot(tmp2)/denom
155  X = (a+c+Va*u+Vb*v) * 0.5
156  l1 = a - X + u*Va # l2 = c - X + v*Vb
157  dist = 2. * ROOT.TMath.Sqrt( l1.Dot(l1) )
158  T = ROOT.TMatrixD(3,12)
159  for i in range(3):
160  for k in range(4):
161  for j in range(3):
162  KD = 0
163  if i==j: KD = 1
164  if k==0 or k==2:
165  # cova and covc
166  temp = ( u[j]*Vsq - v[j]*UV )*u[i] + (u[j]*UV-v[j]*Usq)*v[i]
167  sign = -1
168  if k==2 : sign = +1
169  T[i][3*k+j] = 0.5*( KD + sign*temp/denom )
170  elif k==1:
171  # covu
172  aNAZ = denom*( ca[j]*Vsq-v.Dot(ca)*v[j] )
173  aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( u[j]*Vsq-v[j]*UV )
174  bNAZ = denom*( ca[j]*UV+(u.Dot(ca)*v[j]) - 2*ca.Dot(v)*u[j] )
175  bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( u[j]*Vsq-v[j]*UV )
176  T[i][3*k+j] = 0.5*( Va*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
177  elif k==3:
178  # covv
179  aNAZ = denom*( 2*ca.Dot(u)*v[j] - ca.Dot(v)*u[j] - ca[j]*UV )
180  aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( v[j]*Usq-u[j]*UV )
181  bNAZ = denom*( ca.Dot(u)*u[j]-ca[j]*Usq )
182  bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( v[j]*Usq-u[j]*UV )
183  T[i][3*k+j] = 0.5*(Vb*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
184  transT = ROOT.TMatrixD(12,3)
185  transT.Transpose(T)
186  CovTracks = ROOT.TMatrixD(12,12)
187  tlist = [t1,t2]
188  for k in range(2):
189  for i in range(6):
190  for j in range(6):
191  xfac = 1.
192  if i>2: xfac = scalFac[tlist[k]]
193  if j>2: xfac = xfac * scalFac[tlist[k]]
194  CovTracks[i+k*6][j+k*6] = CovMat[tlist[k]][i][j] * xfac
195  # if i==5 or j==5 : CovMat[tlist[k]][i][j] = 0 # ignore error on z-direction
196  tmp = ROOT.TMatrixD(3,12)
197  tmp.Mult(T,CovTracks)
198  covX = ROOT.TMatrixD(3,3)
199  covX.Mult(tmp,transT)
200  return X,covX,dist

293 def myVertex(t1,t2,PosDir):
294  # closest distance between two tracks
295  # d = |pq . u x v|/|u x v|
296  a = ROOT.TVector3(PosDir[t1][0](0) ,PosDir[t1][0](1), PosDir[t1][0](2))
297  u = ROOT.TVector3(PosDir[t1][1](0),PosDir[t1][1](1),PosDir[t1][1](2))
298  c = ROOT.TVector3(PosDir[t2][0](0) ,PosDir[t2][0](1), PosDir[t2][0](2))
299  v = ROOT.TVector3(PosDir[t2][1](0),PosDir[t2][1](1),PosDir[t2][1](2))
300  pq = a-c
301  uCrossv = u.Cross(v)
302  dist = pq.Dot(uCrossv)/(uCrossv.Mag()+1E-8)
303  # u.a - u.c + s*|u|**2 - u.v*t = 0
304  # v.a - v.c + s*v.u - t*|v|**2 = 0
305  E = u.Dot(a) - u.Dot(c)
306  F = v.Dot(a) - v.Dot(c)
307  A,B = u.Mag2(), -u.Dot(v)
308  C,D = u.Dot(v), -v.Mag2()
309  t = -(C*E-A*F)/(B*C-A*D)
310  X = c.x()+v.x()*t
311  Y = c.y()+v.y()*t
312  Z = c.z()+v.z()*t
313  return X,Y,Z,abs(dist)
