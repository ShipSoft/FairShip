128def fillp1(hist):
129 # scan filled bins in hist, and fill intermediate bins with linear interpolation
130 nb = hist.GetNbinsX()
131 i1 = hist.FindBin(pbeaml, 0., 0.)
132 y1 = hist.GetBinContent(i1)
133 p1 = hist.GetBinCenter(i1)
134 for i in range(i1 + 1, nb + 1):
135 if hist.GetBinContent(i) > 0.:
136 y2 = hist.GetBinContent(i)
137 p2 = hist.GetBinCenter(i)
138 tg = (y2 - y1) / (p2 - p1)
139 for ib in range(i1 + 1, i):
140 px = hist.GetBinCenter(ib)
141 yx = (px - p1) * tg + y1
142 hist.Fill(px, yx)
143 i1 = i
144 y1 = y2
145 p1 = p2
