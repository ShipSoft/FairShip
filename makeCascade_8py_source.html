142def fillp1(hist):
143# scan filled bins in hist, and fill intermediate bins with lineair interpolation
144 nb=hist.GetNbinsX()
145 i1=hist.FindBin(pbeaml,0.,0.)
146 y1=hist.GetBinContent(i1)
147 p1=hist.GetBinCenter(i1)
148 for i in range(i1+1,nb+1):
149 if hist.GetBinContent(i)>0.:
150 y2=hist.GetBinContent(i)
151 p2=hist.GetBinCenter(i)
152 tg=(y2-y1)/(p2-p1)
153 for ib in range(i1+1,i):
154 px=hist.GetBinCenter(ib)
155 yx=(px-p1)*tg+y1
156 hist.Fill(px,yx)
157 i1=i
158 y1=y2
159 p1=p2
