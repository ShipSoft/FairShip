209void ecalReco::ReconstructXY(ecalCell* max, Float_t& x, Float_t& y)
210{
211 // Now use just center of gravity
212 Float_t e=max->GetEnergy();
213 list<ecalCell*> cls;
214 list<ecalCell*>::const_iterator p;
215
216 x=max->GetCenterX()*max->GetEnergy();
217 y=max->GetCenterY()*max->GetEnergy();
218 max->GetNeighborsList(cls);
219 for(p=cls.begin();p!=cls.end();++p)
220 {
221 x+=(*p)->GetCenterX()*(*p)->GetEnergy();
222 y+=(*p)->GetCenterY()*(*p)->GetEnergy();
223 e+=(*p)->GetEnergy();

225 x/=e; y/=e;
