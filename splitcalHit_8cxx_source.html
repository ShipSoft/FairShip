136std::string splitcalHit::GetDetectorElementName(int& id){
137
138 std::string encodedID = GetPaddedString(id);
139 //std::cout << "-- encodedID = " << encodedID <<std::endl;
140 int isPrec, nL, nMx, nMy, nS;
141 Decoder(encodedID, isPrec, nL, nMx, nMy, nS);
142
143 std::string name;
144 if (isPrec==1) {
145 name = "ECALdet_gas_";
146 SetIsX(true);
147 SetIsY(true);
148 } else if (nL%2==0) {
149 name = "stripGivingY_";
150 SetIsX(false);
151 SetIsY(true);
152 } else {
153 name = "stripGivingX_";
154 SetIsX(true);
155 SetIsY(false);
156 }
157 name = name + std::to_string(id);
158 // std::cout << "--GetDetectorElementName - name = " << name <<std::endl;
159
160 return name;
161
162}
