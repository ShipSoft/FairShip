101// ----- Public method SetCovP -----------------------------------------
102void ShipParticle::SetCovP(const Double_t* covElements)
103{
104 // Set 4x4 symmetric momentum covariance matrix from 10 unique elements
105 // Order: (0,0), (0,1), (0,2), (0,3), (1,1), (1,2), (1,3), (2,2), (2,3), (3,3)
106 fCovP[0] = covElements[0]; // (0,0)
107 fCovP[1] = covElements[1]; // (0,1)
108 fCovP[2] = covElements[2]; // (0,2)
109 fCovP[3] = covElements[3]; // (0,3)
110 fCovP[4] = covElements[4]; // (1,1)
111 fCovP[5] = covElements[5]; // (1,2)
112 fCovP[6] = covElements[6]; // (1,3)
113 fCovP[7] = covElements[7]; // (2,2)
114 fCovP[8] = covElements[8]; // (2,3)
115 fCovP[9] = covElements[9]; // (3,3)
116}

119// ----- Public method SetCovV -----------------------------------------
120void ShipParticle::SetCovV(const Double_t* covElements)
121{
122 // Set 3x3 symmetric vertex covariance matrix from 6 unique elements
123 // Order: (0,0), (0,1), (0,2), (1,1), (1,2), (2,2)
124 fCovV[0] = covElements[0]; // (0,0)
125 fCovV[1] = covElements[1]; // (0,1)
126 fCovV[2] = covElements[2]; // (0,2)
127 fCovV[3] = covElements[3]; // (1,1)
128 fCovV[4] = covElements[4]; // (1,2)
129 fCovV[5] = covElements[5]; // (2,2)
130}

147// ----- Public method GetMass -----------------------------------------
148Double_t ShipParticle::GetMass() const
149{
150 return TMath::Sqrt(fE * fE - fPx * fPx - fPy * fPy - fPz * fPz);
151}

164}
