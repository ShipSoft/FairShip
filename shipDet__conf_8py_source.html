13def getParameter(x, ship_geo, latestShipGeo):
14 lv = x.split(".")
15 last = lv[len(lv) - 1]
16 top = ""
17 for l in range(len(lv) - 1):
18 top += lv[l]
19 if l < len(lv) - 2:
20 top += "."
21 a = getattr(ship_geo, top)
22 if hasattr(a, last):
23 return getattr(a, last)
24 # not in the list of recorded parameters. probably added after
25 # creation of file. Check newest geometry_config:
26 a = getattr(latestShipGeo, top)
27 return getattr(a, last)

30def posHcal(z, hfile, HcalOption):
31 HcalZSize = 0
32 sz = hfile + "z" + str(z) + ".geo"
33 floc = os.environ["FAIRSHIP"] + "/geometry"
34 f_hcal = floc + "/" + hfile
35 f_hcalz = floc + "/" + sz
36 f = open(f_hcal)
37 rewrite = True
38 if sz in os.listdir(floc):
39 test = os.popen("diff " + f_hcal + " " + f_hcalz).read()
40 if str.count(test, "---") == 1 and not test.find("Position") < 0:
41 rewrite = False # only different is z position
42 if rewrite:
43 fn = open(f_hcalz, "w")
44 for l in f.readlines():
45 if rewrite:
46 if not l.find("ZPos") < 0:
47 l = "ZPos=" + str(z) + " #Position of Hcal center [cm]\n"
48 fn.write(l)
49 if not l.find("HcalZSize") < 0:
50 HcalZSize = float(l[len("HcalZSize") + 1 :].split("#")[0])
51 f.close()
52 if rewrite:
53 fn.close()
54 if HcalOption == 2:
55 hcal = ROOT.hcal("Hcal", ROOT.kFALSE, sz)
56 else:
57 hcal = ROOT.hcal("Hcal", ROOT.kTRUE, sz)
58 return hcal, HcalZSize
