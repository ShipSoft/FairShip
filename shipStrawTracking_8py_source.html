507 def extrapolateToPlane(fT,z):
508  # etrapolate to a plane perpendicular to beam direction (z)
509  rc,pos,mom = False,None,None
510  parallelToZ = ROOT.TVector3(0., 0., 1.)
511  fst = fT.getFitStatus()
512  if fst.isFitConverged():
513  # test for fit status for each point
514  if fT.getPoint(0).getFitterInfo() and fT.getPoint(1).getFitterInfo():
515  fstate0,fstate1 = fT.getFittedState(0),fT.getFittedState(1)
516  fPos0,fPos1 = fstate0.getPos(),fstate1.getPos()
517  if abs(z-fPos0.z()) < abs(z-fPos1.z()): fstate = fstate0
518  else: fstate = fstate1
519  zs = z
520  NewPosition = ROOT.TVector3(0., 0., zs)
521  rep = ROOT.genfit.RKTrackRep(13*cmp(fstate.getPDG(),0) )
522  state = ROOT.genfit.StateOnPlane(rep)
523  pos,mom = fstate.getPos(),fstate.getMom()
524  rep.setPosMom(state,pos,mom)
525  try:
526  rep.extrapolateToPlane(state, NewPosition, parallelToZ )
527  pos,mom = state.getPos(),state.getMom()
528  rc = True
529  except:
530  print('error with extrapolation')
531  pass
532  if not rc:
533  # use linear extrapolation
534  px,py,pz = mom.X(),mom.Y(),mom.Z()
535  lam = (z-pos.Z())/pz
536  pos = ROOT.TVector3( pos.X()+lam*px, pos.Y()+lam*py, z )
537  return rc,pos,mom

543 def decodeDetectorID(detID):
544  """
545  Decodes detector ID.
546  Parameters
547  ----------
548  detID : int or array-like
549  Detector ID values.
550  Returns
551  -------
552  statnb : int or array-like
553  Station numbers.
554  vnb : int or array-like
555  View numbers.
556  pnb : int or array-like
557  Plane numbers.
558  lnb : int or array-like
559  Layer numbers.
560  snb : int or array-like
561  Straw tube numbers.
562  """
563 
564  statnb = detID // 10000000
565  vnb = (detID - statnb * 10000000) // 1000000
566  pnb = (detID - statnb * 10000000 - vnb * 1000000) // 100000
567  lnb = (detID - statnb * 10000000 - vnb * 1000000 - pnb * 100000) // 10000
568  snb = detID - statnb * 10000000 - vnb * 1000000 - pnb * 100000 - lnb * 10000 - 2000
569 
570  return statnb, vnb, pnb, lnb, snb

574 def fracMCsame(trackids):
575  """
576  Estimates max fraction of true hit labels for a recognized track.
577  trackids : array_like
578  hit indexes of a recognized track
579  Retunrs
580  -------
581  frac : float
582  Max fraction of true hit labels.
583  tmax : int
584  True hit label with max fraction in a recognized track.
585  """
586 
587  track={}
588  nh=len(trackids)
589 
590  for tid in trackids:
591  if tid in track:
592  track[tid] += 1
593  else:
594  track[tid] = 1
595 
596  #now get track with largest number of hits
597  if track != {}:
598  tmax = max(track, key=track.get)
599  else:
600  track = {-999: 0}
601  tmax = -999
602 
603  frac=0.0
604  if nh > 0:
605  frac = float(track[tmax]) / float(nh)
606 
607  return frac, tmax
