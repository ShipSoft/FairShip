180 def sqrt_lambda(self,a,b,c):
181 """
182 Useful function for decay kinematics. Returns 0 for kinematically forbidden region
183 """
184 l = a**2 + b**2 + c**2 - 2*a*b - 2*b*c - 2*a*c
185 if l<0:
186 return 0
187 else:
188 return math.sqrt(l)

190 def QCD_correction(self):
191 """
192 Returns 3-loops QCD correction to HNL decay width into quarks
193 """
194 alpha_s = ROOT.TGraph( os.path.expandvars('$FAIRSHIP/python/alpha_s.dat') )
195 a_s = alpha_s.Eval(self.MN)
196 qcd_corr = a_s / math.pi
197 qcd_corr += 5.2 * (a_s / math.pi)**2.
198 qcd_corr += 26.4 * (a_s / math.pi)**3.
199 return qcd_corr

201 def Width_3nu(self):
202 """
203 Returns the HNL decay width into three neutrinos
204 """
205 width = (c.GF**2.)*(self.MN**5.)*sum(self.U2)/(192.*(u.pi**3.))
206 width = 2.*width # Majorana case (charge conjugate channels)
207 return width

209 def Width_nu_f_fbar(self, alpha, beta):
210 """
211 Returns the HNL tree level decay width into the fermion-antifermion pair and a neutrino (decay through Z boson or Z and W)
212
213 Inputs:
214 - alpha is a flavour of neutrino (1,2 or 3), determine the mixing angle U_alpha
215 - beta is a flavour of the fermions: (1, 2 or 3) for charge leptons or (4, 5, 6, 7, 8, 9) for quarks
216 """
217 if (alpha not in [1,2,3]) or (beta not in [1,2,3,4,5,6,7,8,9]):
218 print('Width_nu_f_fbar ERROR: unknown channel alpha =',alpha,' beta =',beta)
219 quit()
220 l = [None,'e-','mu-','tau-','u','d','s','c','b','t']
221 x = mass(l[beta]) / self.MN
222 if x > 0.5: # the decay is kinematically forbidden
223 return 0
224 width = (c.GF**2.)*(self.MN**5.)*self.U2[alpha-1]/(192.*(math.pi**3.))
225 L = 0.
226 if x>0.01:
227 logContent = (1. - 3.*x**2. - (1.-x**2.)*math.sqrt(1. - 4.*x**2.) ) / ( (x**2.)*(1 + math.sqrt(1. - 4.*x**2.)) )
228 else:
229 logContent = x**4 + 4.*x**6 + 14.*x**8
230 if logContent > 0:
231 L = math.log( logContent )
232 if beta < 4: # lepton decay
233 NZ = 1
234 if alpha == beta: # interference case
235 C1 = 0.25*(1. + 4.*c.s2thetaw + 8.*c.s2thetaw**2.)
236 C2 = 0.5*c.s2thetaw*(2.*c.s2thetaw +1.)
237 else: # no interference
238 C1 = 0.25*(1. - 4.*c.s2thetaw + 8.*c.s2thetaw**2.)
239 C2 = 0.5*c.s2thetaw*(2.*c.s2thetaw -1.)
240 else: # decay into quarks
241 NZ = 3
242 if (beta in [4,7,9]): # up quarks
243 C1 = 0.25*(1. - 8./3.*c.s2thetaw + 32./9.*c.s2thetaw**2.)
244 C2 = 1./3.*c.s2thetaw*(4./3.*c.s2thetaw - 1.)
245 if (beta in [5,6,8]): # down quarks
246 C1 = 0.25*(1. - 4./3.*c.s2thetaw + 8./9.*c.s2thetaw**2.)
247 C2 = 1./6.*c.s2thetaw*(2./3.*c.s2thetaw - 1.)
248 width = width*NZ*( C1*( (1.-14.*x**2. -2.*x**4. -12.*x**6.)*math.sqrt(1-4.*x**2) +12.*x**4. *(-1.+x**4.)*L )
249 + 4.*C2*( x**2. *(2.+10.*x**2. -12.*x**4.) * math.sqrt(1.-4.*x**2) + 6.*x**4. *(1.-2.*x**2+2.*x**4)*L ) )
250 width = 2.*width # Majorana case (charge conjugate channels)
251 return width

253 def Integrand(self,xx,xi):
254 """
255 Function to integrate needed for numerical integration using ROOT. Needed for 3-body decays through W boson.
256 First argument is a list of 1 number, argument. Second is the list of 3 numbers xi = mi/MN
257 """
258 x = xx[0]
259 xl2 = xi[0]**2
260 xu2 = xi[1]**2
261 xd2 = xi[2]**2
262 if x==0: # Workaround for division by zero
263 return 0
264 res = 1./x
265 res *= (x - xl2 - xd2)
266 res *= (1. + xu2 - x)
267 res *= self.sqrt_lambda(x,xl2,xd2)
268 res *= self.sqrt_lambda(1.,x,xu2)
269 return res

271 def I(self,x1,x2,x3):
272 """
273 Numerical integral needed for 3-body decays through W boson.
274 xi = mi/MN
275 """
276 theFunction = self.Integrand
277 func = ROOT.TF1('func',theFunction,0,1,3) # Setting function
278 func.SetParameters(x1,x2,x3)
279 xmin = (x1 + x3)**2
280 xmax = (1. - x2)**2
281 wf1 = ROOT.Math.WrappedTF1(func) # Setting simple Gauss integration method
282 ig = ROOT.Math.GaussIntegrator()
283 ig.SetFunction(wf1)
284 ig.SetRelTolerance(0.001)
285 res = 12. * ig.Integral(xmin,xmax)
286 return res

288 def Width_l1_l2_nu2(self, alpha, beta):
289 """
290 Returns the HNL decay width into two different flavour leptons and a neutrino (decay through W boson)
291
292 Inputs:
293 - alpha is a flavour of the first lepton (1,2 or 3), determine the mixing angle U_alpha
294 - beta is a flavour of the second lepton and neutrino (1, 2 or 3)
295 """
296 if (alpha not in [1,2,3]) or (beta not in [1,2,3]):
297 print('Width_l1_l2_nu2 ERROR: unknown channel alpha =',alpha,' beta =',beta)
298 quit()
299 if alpha==beta: # The interference case is handled by Width_nu_f_fbar function, workaround for a total width calculation
300 return 0
301 l = [None,'e-','mu-','tau-']
302 x1 = mass(l[alpha])/self.MN
303 x2 = mass(l[beta])/self.MN
304 if x1+x2>1: # The decay is kinematically forbidden
305 return 0
306 width = (c.GF**2.)*(self.MN**5.)*self.U2[alpha-1]/(192.*(math.pi**3.))
307 width = width*self.I(x1,x2,0)
308 width = 2.*width # Majorana case (charge conjugate channels)
309 return width

311 def Width_l_u_d(self, alpha, beta, gamma):
312 """
313 Returns the HNL tree level decay width into a charged lepton, up quark and down quark (decay through W boson)
314
315 Inputs:
316 - alpha is a flavour of the charged lepton (1,2 or 3), determine the mixing angle U_alpha
317 - beta is the up quark generation (1, 2, 3 for u, c, t)
318 - gamma is the down quark generation (1, 2, 3 for d, s, b)
319 """
320 if (alpha not in [1,2,3]) or (beta not in [1,2,3]) or (gamma not in [1,2,3]):
321 print('Width_l_u_d ERROR: unknown channel alpha =',alpha,' beta =',beta,' gamma =',gamma)
322 quit()
323 l = [None,'e-','mu-','tau-']
324 u = [None,'u','c','t']
325 d = [None,'d','s','b']
326 xl = mass(l[alpha])/self.MN
327 xu = mass(u[beta])/self.MN
328 xd = mass(d[gamma])/self.MN
329 if xl+xu+xd>1: # The decay is kinematically forbidden
330 return 0
331 width = (c.GF**2.)*(self.MN**5.)*self.U2[alpha-1]/(192.*(math.pi**3.))
332 width *= 3*self.CKMelemSq[(beta, gamma)]
333 width *= self.I(xl,xu,xd)
334 width = 2.*width # Majorana case (charge conjugate channels)
335 return width

337 def Width_H0_nu(self, H, alpha):
338 """
339 Returns the HNL decay width into neutral meson and neutrino
340
341 Inputs:
342 - H is a string (name of the meson)
343 - alpha is a flavour of the nu (1, 2 or 3), determine the mixing angle U_alpha
344 """
345 if (H not in ['pi0','eta','rho0','omega','eta1','phi','eta_c']) or (alpha not in [1,2,3]):
346 print('Width_H0_nu ERROR: unknown channel H0 =',H,' alpha =',alpha)
347 quit()
348 x = mass(H)/self.MN
349 if x > 1: # The decay is kinematically forbidden
350 return 0.
351 width = (c.GF**2.)*(c.decayConstant[H]**2.)*(self.MN**3.)*self.U2[alpha-1]/(32.*math.pi)
352 if H in ['pi0','eta','eta1','eta_c']: # pseudoscalar mesons
353 width *= (1 - x**2.)**2.
354 if H in ['rho0','omega','phi']: # vector mesons
355 width *= (1. + 2*x**2.)*(1. - x**2.)**2.
356 if H=='rho0':
357 width *= (1. - 2.*c.s2thetaw)**2.
358 if H=='omega':
359 width *= (16.*c.s2thetaw**2.)/9.
360 if H=='phi':
361 width *= (1. - 4./3.*c.s2thetaw)**2.
362 width = 2.*width # Majorana case (charge conjugate channels)
363 return width

365 def Width_H_l(self, H, alpha):
366 """
367 Returns the HNL decay width into charged meson and lepton
368
369 Inputs:
370 - H is a string (name of the positively charged meson)
371 - alpha is the lepton flavour (1, 2 or 3), determine the mixing angle U_alpha
372 """
373 if (H not in ['pi+','rho+','D_s+','D*_s+']) or (alpha not in [1,2,3]):
374 print('Width_H_l ERROR: unknown channel H =',H,' alpha =',alpha)
375 quit()
376 l = [None,'e-','mu-','tau-']
377 xl = mass(l[alpha])/self.MN
378 xh = mass(H)/self.MN
379 if xl+xh > 1: # The decay is kinematically forbidden
380 return 0.
381 width = (c.GF**2.)*(c.decayConstant[H]**2.)*self.CKMelemSq[H]*(self.MN**3.)*self.U2[alpha-1]/(16.*math.pi)
382 width *= self.sqrt_lambda(1., xl**2, xh**2)
383 if H in ['pi+','D_s+']: # pseudoscalar mesons
384 width *= ( (1. - xl**2.)**2. - xh**2. *(1. + xl**2.) )
385 if H in ['rho+','D*_s+']: # vector mesons
386 width *= ( (1. - xl**2.)**2. + xh**2.*(1. + xl**2.) - 2.*xh**4. )
387 width = 2.*width # Majorana case (charge conjugate channels)
388 return width

390 def Width_charged_leptons(self):
391 """
392 Returns the total HNL leptonic decay width with charged leptons in final state
393 """
394 width = 0.
395 for l1 in [1,2,3]:
396 for l2 in [1,2,3]:
397 width += self.Width_nu_f_fbar(l1,l2)
398 width += self.Width_l1_l2_nu2(l1,l2)
399 return width

401 def Width_neutral_mesons(self):
402 """
403 Returns the total HNL decay width into a neutral meson and a neutrino
404 """
405 mesons = ['pi0','eta','rho0','omega','eta1','phi','eta_c']
406 width = 0.
407 for m in mesons:
408 for l in [1,2,3]:
409 width += self.Width_H0_nu(m,l)
410 return width

412 def Width_charged_mesons(self):
413 """
414 Returns the total HNL decay width into a charged meson and a charged lepton
415 """
416 mesons = ['pi+','rho+','D_s+','D*_s+']
417 width = 0.
418 for m in mesons:
419 for l in [1,2,3]:
420 width += self.Width_H_l(m,l)
421 return width

423 def Width_quarks_neutrino(self):
424 """
425 Returns the total HNL decay width with quarks and neutrino in final state. Uses 3-loops alpha_s correction
426 """
427 if self.MN<1.: # decay into quarks is not applicable at low HNL mass
428 return 0
429 width = 0.
430 for l in [1,2,3]:
431 for q in [4,5,6,7,8,9]:
432 width += self.Width_nu_f_fbar(l,q)
433 width *= (1. + self.QCD_corr)
434 return width
435
436 def Width_quarks_lepton(self):
437 """
438 Returns the total HNL decay width with quarks and charged lepton in final state. Uses 3-loops alpha_s correction
439 """
440 if self.MN<1.: # decay into quarks is not applicable at low HNL mass
441 return 0
442 width = 0.
443 for l in [1,2,3]:
444 for u in [1,2,3]:
445 for d in [1,2,3]:
446 width += self.Width_l_u_d(l,u,d)
447 width *= (1. + self.QCD_corr)
448 return width

450 def NDecayWidth(self):
451 """
452 Returns the total HNL decay width
453 """
454 leptonWidth = self.Width_3nu() + self.Width_charged_leptons()
455 mesonWidth = self.Width_neutral_mesons() + self.Width_charged_mesons()
456 quarkWidth = self.Width_quarks_neutrino() + self.Width_quarks_lepton()
457 totalWidth = leptonWidth + max([mesonWidth, quarkWidth])
458 return totalWidth

460 def findBranchingRatio(self, decayString):
461 """
462 Returns the branching ratio of the selected HNL decay channel
463
464 Inputs:
465 - decayString is a string describing the decay, in the form 'N -> stuff1 ... stuffN'
466 """
467 br = 0.
468 totalWidth = self.NDecayWidth()
469
470 if (decayString not in self.decays) and (decayString not in ['N -> hadrons','N -> charged hadrons']):
471 print('findBranchingRatio ERROR: unknown decay %s'%decayString)
472 quit()
473
474 if decayString == 'N -> nu nu nu' or decayString == 'N -> 3nu': br = self.Width_3nu() / totalWidth # inclusive
475 if decayString == 'N -> e- e+ nu_e': br = self.Width_nu_f_fbar(1,1) / totalWidth
476 if decayString == 'N -> e- e+ nu_mu': br = self.Width_nu_f_fbar(2,1) / totalWidth
477 if decayString == 'N -> e- e+ nu_tau': br = self.Width_nu_f_fbar(3,1) / totalWidth
478 if decayString == 'N -> e- mu+ nu_mu': br = self.Width_l1_l2_nu2(1,2) / totalWidth
479 if decayString == 'N -> mu- e+ nu_e': br = self.Width_l1_l2_nu2(2,1) / totalWidth
480 if decayString == 'N -> pi0 nu_e': br = self.Width_H0_nu('pi0',1) / totalWidth
481 if decayString == 'N -> pi0 nu_mu': br = self.Width_H0_nu('pi0',2) / totalWidth
482 if decayString == 'N -> pi0 nu_tau': br = self.Width_H0_nu('pi0',3) / totalWidth
483 if decayString == 'N -> pi+ e-': br = self.Width_H_l('pi+',1) / totalWidth
484 if decayString == 'N -> mu- mu+ nu_e': br = self.Width_nu_f_fbar(1,2) / totalWidth
485 if decayString == 'N -> mu- mu+ nu_mu': br = self.Width_nu_f_fbar(2,2) / totalWidth
486 if decayString == 'N -> mu- mu+ nu_tau': br = self.Width_nu_f_fbar(3,2) / totalWidth
487 if decayString == 'N -> pi+ mu-': br = self.Width_H_l('pi+',2) / totalWidth
488 if decayString == 'N -> eta nu_e': br = self.Width_H0_nu('eta',1) / totalWidth
489 if decayString == 'N -> eta nu_mu': br = self.Width_H0_nu('eta',2) / totalWidth
490 if decayString == 'N -> eta nu_tau': br = self.Width_H0_nu('eta',3) / totalWidth
491 if decayString == 'N -> rho0 nu_e': br = self.Width_H0_nu('rho0',1) / totalWidth
492 if decayString == 'N -> rho0 nu_mu': br = self.Width_H0_nu('rho0',2) / totalWidth
493 if decayString == 'N -> rho0 nu_tau': br = self.Width_H0_nu('rho0',3) / totalWidth
494 if decayString == 'N -> rho+ e-': br = self.Width_H_l('rho+',1) / totalWidth
495 if decayString == 'N -> omega nu_e': br = self.Width_H0_nu('omega',1) / totalWidth
496 if decayString == 'N -> omega nu_mu': br = self.Width_H0_nu('omega',2) / totalWidth
497 if decayString == 'N -> omega nu_tau': br = self.Width_H0_nu('omega',3) / totalWidth
498 if decayString == 'N -> rho+ mu-': br = self.Width_H_l('rho+',2) / totalWidth
499 if decayString == 'N -> eta1 nu_e': br = self.Width_H0_nu('eta1',1) / totalWidth
500 if decayString == 'N -> eta1 nu_mu': br = self.Width_H0_nu('eta1',2) / totalWidth
501 if decayString == 'N -> eta1 nu_tau': br = self.Width_H0_nu('eta1',3) / totalWidth
502 if decayString == 'N -> phi nu_e': br = self.Width_H0_nu('phi',1) / totalWidth
503 if decayString == 'N -> phi nu_mu': br = self.Width_H0_nu('phi',2) / totalWidth
504 if decayString == 'N -> phi nu_tau': br = self.Width_H0_nu('phi',3) / totalWidth
505 if decayString == 'N -> e- tau+ nu_tau': br = self.Width_l1_l2_nu2(1,3) / totalWidth
506 if decayString == 'N -> tau- e+ nu_e': br = self.Width_l1_l2_nu2(3,1) / totalWidth
507 if decayString == 'N -> mu- tau+ nu_tau': br = self.Width_l1_l2_nu2(2,3) / totalWidth
508 if decayString == 'N -> tau- mu+ nu_mu': br = self.Width_l1_l2_nu2(3,2) / totalWidth
509 if decayString == 'N -> D_s+ e-': br = self.Width_H_l('D_s+',1) / totalWidth
510 if decayString == 'N -> D_s+ mu-': br = self.Width_H_l('D_s+',2) / totalWidth
511 if decayString == 'N -> D*_s+ e-': br = self.Width_H_l('D*_s+',1) / totalWidth
512 if decayString == 'N -> D*_s+ mu-': br = self.Width_H_l('D*_s+',2) / totalWidth
513 if decayString == 'N -> eta_c nu_e': br = self.Width_H0_nu('eta_c',1) / totalWidth

515 if decayString == 'N -> eta_c nu_tau': br = self.Width_H0_nu('eta_c',3) / totalWidth

519 quarkWidth = self.Width_quarks_neutrino() + self.Width_quarks_lepton()
520 br = max([mesonWidth, quarkWidth]) / totalWidth
521 if decayString == 'N -> charged hadrons':
522 br = max([self.Width_charged_mesons(), self.Width_quarks_lepton()]) / totalWidth
