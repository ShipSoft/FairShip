94TGeoVolume* veto::GeoTrapezoid(TString xname,
95 Double_t z_thick,
96 Double_t x_thick_start,
97 Double_t x_thick_end,
98 Double_t y_thick_start,
99 Double_t y_thick_end,
100 Int_t color,
101 TGeoMedium* material,
102 Bool_t sens = kFALSE)
103{
104 Double_t dz = z_thick / 2;
105 Double_t dx1 = x_thick_start / 2 - 1.E-6;
106 Double_t dx2 = x_thick_end / 2 - 1.E-6;
107 Double_t dy1 = y_thick_start / 2 - 1.E-6;
108 Double_t dy2 = y_thick_end / 2 - 1.E-6;
109 TGeoArb8* T1 = new TGeoArb8("T1" + xname, dz + 1.E-6);
110 T1->SetVertex(0, -dx1, -dy1);
111 T1->SetVertex(1, -dx1, dy1);
112 T1->SetVertex(2, dx1, dy1);
113 T1->SetVertex(3, dx1, -dy1);
114 T1->SetVertex(4, -dx2, -dy2);
115 T1->SetVertex(5, -dx2, dy2);
116 T1->SetVertex(6, dx2, dy2);
117 T1->SetVertex(7, dx2, -dy2);
118
119 TGeoVolume* T = new TGeoVolume(xname, T1, material);
120 T->SetLineColor(color);
121 if (sens) {
122 AddSensitiveVolume(T);
123 } // and make the volumes sensitive..
124 return T;
125}

127TGeoVolume* veto::GeoTrapezoidHollow(TString xname,
128 Double_t wallthick,
129 Double_t z_thick,
130 Double_t x_thick_start,
131 Double_t x_thick_end,
132 Double_t y_thick_start,
133 Double_t y_thick_end,
134 Int_t color,
135 TGeoMedium* material,
136 Bool_t sens = kFALSE)
137{
138
139 Double_t dx_start = x_thick_start / 2;
140 Double_t dy_start = y_thick_start / 2;
141 Double_t dx_end = x_thick_end / 2;
142 Double_t dy_end = y_thick_end / 2;
143 Double_t dz = z_thick / 2;
144
145 TString nm = xname.ReplaceAll("-", ""); // otherwise it will try to subtract "-" in TGeoComposteShape
146 Double_t dx1 = dx_start + wallthick;
147 Double_t dx2 = dx_end + wallthick;
148 Double_t dy1 = dy_start + wallthick;
149 Double_t dy2 = dy_end + wallthick;
150
151 TGeoArb8* T2 = new TGeoArb8("T2" + nm, dz);
152 T2->SetVertex(0, -dx1, -dy1);
153 T2->SetVertex(1, -dx1, dy1);
154 T2->SetVertex(2, dx1, dy1);
155 T2->SetVertex(3, dx1, -dy1);
156 T2->SetVertex(4, -dx2, -dy2);
157 T2->SetVertex(5, -dx2, dy2);
158 T2->SetVertex(6, dx2, dy2);
159 T2->SetVertex(7, dx2, -dy2);
160
161 Double_t tdx1 = dx_start;
162 Double_t tdx2 = dx_end;
163 Double_t tdy1 = dy_start;
164 Double_t tdy2 = dy_end;
165 TGeoArb8* T1 = new TGeoArb8("T1" + nm, dz + 1.E-6);
166 T1->SetVertex(0, -tdx1, -tdy1);
167 T1->SetVertex(1, -tdx1, tdy1);
168 T1->SetVertex(2, tdx1, tdy1);
169 T1->SetVertex(3, tdx1, -tdy1);
170 T1->SetVertex(4, -tdx2, -tdy2);
171 T1->SetVertex(5, -tdx2, tdy2);
172 T1->SetVertex(6, tdx2, tdy2);
173 T1->SetVertex(7, tdx2, -tdy2);
174
175 TGeoCompositeShape* T321 = new TGeoCompositeShape("T3" + nm, "T2" + nm + "-T1" + nm);
176 TGeoVolume* T = new TGeoVolume(xname, T321, material);

185double veto::wx(double z)
186{ // calculate x thickness at z
187
188 double wx1 = VetoStartInnerX;
189 double wx2 = VetoEndInnerX;
190 double z1 = 0 * m;
191 double z2 = 50 * m;
192
193 return (wx1 + (z - z1) * (wx2 - wx1) / (z2 - z1));
194}

198
199 double wy1 = VetoStartInnerY;
200 double wy2 = VetoEndInnerY;
201 double z1 = 0 * m;
202 double z2 = 50 * m;
203
204 return (wy1 + (z - z1) * (wy2 - wy1) / (z2 - z1));
205}

340TGeoVolumeAssembly* veto::GeoCornerRib(TString xname,
341 double ribThick,
342 double lt1,
343 double lt2,
344 double dz,
345 double slopeX,
346 double slopeY,
347 Int_t color,
348 TGeoMedium* material,
349 Bool_t sens = kFALSE)
350{
351
352 Double_t wz = dz * 2;
353 double d = ribThick * sqrt(2) / 2;
354 double dx = slopeX * wz;
355 double dy = slopeY * wz;
356
357 TGeoArb8* T1 = new TGeoArb8(dz);
358 T1->SetVertex(0, lt1, lt1 - d);
359 T1->SetVertex(1, 0, -d);
360 T1->SetVertex(2, 0, 0);
361 T1->SetVertex(3, lt1, lt1);
362
363 T1->SetVertex(4, dx + lt2, dy + lt2 - d);
364 T1->SetVertex(5, dx, dy - d);
365 T1->SetVertex(6, dx, dy);
366 T1->SetVertex(7, dx + lt2, dy + lt2);
367
368 TGeoArb8* T2 = new TGeoArb8(dz);
369 T2->SetVertex(0, lt1 - d, lt1);
370 T2->SetVertex(1, lt1, lt1);
371 T2->SetVertex(2, 0, 0);
372 T2->SetVertex(3, -d, 0);
373
374 T2->SetVertex(4, dx + lt2 - d, dy + lt2);
375 T2->SetVertex(5, dx + lt2, dy + lt2);
376 T2->SetVertex(6, dx, dy);
377 T2->SetVertex(7, dx - d, dy);
378
379 TGeoVolume* T1v = new TGeoVolume("part", T1, material);
380 TGeoVolume* T2v = new TGeoVolume("part", T2, material);
381 T1v->SetLineColor(color);
382 T2v->SetLineColor(color);
383 if (sens) {
384 AddSensitiveVolume(T1v);
385 }
386 if (sens) {
387 AddSensitiveVolume(T2v);
388 }
389
390 TGeoVolumeAssembly* T = new TGeoVolumeAssembly(xname);
391 T->AddNode(T1v, 1, new TGeoTranslation(0, 0, 0));
392 T->AddNode(T2v, 2, new TGeoTranslation(0, 0, 0));
393 return T;
394}

396int veto::makeId(double z, double x, double y)
397{
398 double Z = z / 10;
399 double r = sqrt(x * x + y * y);
400 double phi = 999;
401 if (y >= 0)
402 phi = acos(x / r);
403 else
404 phi = -acos(x / r) + 2 * TMath::Pi();
405
406 phi = phi * 180 / TMath::Pi();
407 return (int)Z * 1000000 + (int)r * 1000 + (int)phi;
408}

411{
412
413 int id = 999999;
414 int ShapeType = -1;
415
416 if (ShapeTypeName == "LiScX")

437void veto::AddBlock(TGeoVolumeAssembly* tInnerWall,
438 TGeoVolumeAssembly* tDecayVacuum,
439 TGeoVolumeAssembly* tOuterWall,
440 TGeoVolumeAssembly* tLongitRib,
441 TGeoVolumeAssembly* tVerticalRib,
442 TGeoVolumeAssembly* ttLiSc,
443 int& liScCounter,
444 int blockNr,
445 int nx,
446 int ny,
447 double z1,
448 double z2,
449 double Zshift,
450 double cell_thickness_z,
451 double wallThick,
452 double liscThick1,
453 double liscThick2,
454 double ribThick)
455{
456
457 TString blockName = "block";
458 blockName += blockNr;
459
460 int ribColor = 15;
461
462 double wz = (z2 - z1);
463 double slX = (wx(z2) - wx(z1)) / 2 / wz;
464 double slY = (wy(z2) - wy(z1)) / 2 / wz;
465
466 double dZ = (cell_thickness_z - ribThick) / 2; // half space between ribs
467
468 double tX = 0;
469 double tY = 0;
470 double tZ = 0;
471 TString name("");
472
473 double idX = 0;
474 double idY = 0;
475 double idZ = 0;
476
478 TString nameInnerWall = (TString)tInnerWall->GetName() + "_" + blockName;
479 TGeoVolume* TIW =
480 GeoTrapezoidHollow(nameInnerWall, wallThick, wz, wx(z1), wx(z2), wy(z1), wy(z2), ribColor, supportMedIn);
481 tInnerWall->AddNode(TIW, 0, new TGeoTranslation(0, 0, Zshift));
482
484 TString nameDecayVacuum = (TString)tDecayVacuum->GetName() + "_" + blockName;
485 TGeoVolume* TDV = GeoTrapezoid(nameDecayVacuum, wz, wx(z1), wx(z2), wy(z1), wy(z2), 1, decayVolumeMed);
486 TDV->SetVisibility(kFALSE);
487 tDecayVacuum->AddNode(TDV, 0, new TGeoTranslation(0, 0, Zshift));
488
490 TString nameOuterWall = (TString)tOuterWall->GetName() + "_" + blockName;
491 TGeoVolume* TOW = GeoTrapezoidHollow(nameOuterWall,
492 wallThick,
493 wz,
494 wx(z1) + 2 * (wallThick + liscThick1),
495 wx(z2) + 2 * (wallThick + liscThick2),
496 wy(z1) + 2 * (wallThick + liscThick1),
497 wy(z2) + 2 * (wallThick + liscThick2),
498 ribColor,
499 supportMedIn);
500 tOuterWall->AddNode(TOW, 0, new TGeoTranslation(0, 0, Zshift));
501
503
504 std::vector<TGeoVolume*> vLongitRibX(nx);
505 std::vector<TGeoVolume*> vLongitRibY(ny);
506
507 double z1_x_thickness_1 = wx(z1 + ribThick) + 2 * (wallThick + liscThick1);
508 double z1_x_thickness_2 = wx(z1 + cell_thickness_z) + 2 * (wallThick + liscThick1);
509
510 double z1_x1_Step = (z1_x_thickness_1 - nx * ribThick) / (nx + 1);
511 double z1_x2_Step = (z1_x_thickness_2 - nx * ribThick) / (nx + 1);
512
513 double z1_y_thickness_1 = wy(z1 + ribThick) + 2 * (wallThick + liscThick1);
514 double z1_y_thickness_2 = wy(z1 + cell_thickness_z) + 2 * (wallThick + liscThick1);
515
516 double z1_y1_Step = (z1_y_thickness_1 - ny * ribThick) / (ny + 1);
517 double z1_y2_Step = (z1_y_thickness_2 - ny * ribThick) / (ny + 1);
518
519 double z1_x1, z1_x2, z1_y1, z1_y2;
520
521 z1_x1 = z1_x_thickness_1 / 2;
522 z1_x2 = z1_x_thickness_2 / 2;
523
524 z1_y1 = z1_y_thickness_1 / 2 - liscThick1;
525 z1_y2 = z1_y_thickness_2 / 2 - liscThick2;
526
527 for (int i = 0; i < nx; i++) {
528
529 z1_x1 = z1_x1 - z1_x1_Step;
530 z1_x2 = z1_x2 - z1_x2_Step;
531
532 name = "";
533 name.Form("vLongitRibX_%s_phi%d", blockName.Data(), makeId(0, z1_x1, z1_y1));
534 vLongitRibX.at(i) = GeoSideObj(
535 name, dZ, ribThick, liscThick1, ribThick, liscThick2, z1_x2 - z1_x1, z1_y2 - z1_y1, ribColor, supportMedIn);
536
537 z1_x1 = z1_x1 - ribThick;
538 z1_x2 = z1_x2 - ribThick;
539 }
540
541 z1_y1 = z1_y_thickness_1 / 2;
542 z1_y2 = z1_y_thickness_2 / 2;
543
544 z1_x1 = z1_x_thickness_1 / 2 - liscThick1;
545 z1_x2 = z1_x_thickness_2 / 2 - liscThick2;
546
547 for (int i = 0; i < ny; i++) {
548
549 z1_y1 = z1_y1 - z1_y1_Step;
550 z1_y2 = z1_y2 - z1_y2_Step;
551
552 name = "";
553 name.Form("vLongitRibY_%s_phi%d", blockName.Data(), makeId(0, z1_x1, z1_y1));
554 vLongitRibY.at(i) = GeoSideObj(
555 name, dZ, liscThick1, ribThick, liscThick2, ribThick, z1_x2 - z1_x1, z1_y2 - z1_y1, ribColor, supportMedIn);
556 z1_y1 = z1_y1 - ribThick;
557 z1_y2 = z1_y2 - ribThick;
558 }
559
561
562 name = "CornerRib_L_" + blockName + "_id";
563 TGeoVolumeAssembly* CornerRib_L =
564 GeoCornerRib(name, ribThick, liscThick1, liscThick2, dZ, slX, slY, ribColor, supportMedIn);
565 name = "CornerRib_R_" + blockName + "_id";
566 TGeoVolumeAssembly* CornerRib_R =
567 GeoCornerRib(name, ribThick, liscThick1, liscThick2, dZ, slY, slX, ribColor, supportMedIn);
568
569 for (double zi = z1; zi < z2; zi += cell_thickness_z) {
570
571 int Zlayer = (int)zi / cell_thickness_z + 1;
572
574 TString nameVR("");
575 nameVR.Form("VetoVerticalRib_z%d", (int)zi);
576 TGeoVolume* TVR = GeoTrapezoidHollow(nameVR,
577 liscThick1,
578 ribThick,
579 wx(zi) + 2 * wallThick,
580 wx(zi + ribThick) + 2 * wallThick,
581 wy(zi) + 2 * wallThick,
582 wy(zi + ribThick) + 2 * wallThick,
583 ribColor,
584 supportMedIn);
585 tZ = Zshift - wz / 2 + zi - z1 + ribThick / 2;
586
587 tVerticalRib->AddNode(TVR, 0, new TGeoTranslation(0, 0, tZ));
588
589 if (z2 - zi < cell_thickness_z)
590 continue;
591
592 tX = wx(zi + ribThick) / 2 + wallThick;
593 tY = wy(zi + ribThick) / 2 + wallThick;
594 tZ = tZ + cell_thickness_z / 2;
595
597 tLongitRib->AddNode(
598 CornerRib_L, makeId(zi, tX, tY), new TGeoCombiTrans(tX, tY, tZ, new TGeoRotation("r", 0, 0, 0)));
599 tLongitRib->AddNode(
600 CornerRib_L, makeId(zi, -tX, -tY), new TGeoCombiTrans(-tX, -tY, tZ, new TGeoRotation("r", 0, 0, 180)));
601 tLongitRib->AddNode(
602 CornerRib_R, makeId(zi, -tX, tY), new TGeoCombiTrans(-tX, tY, tZ, new TGeoRotation("r", 0, 0, 90)));
603 tLongitRib->AddNode(
604 CornerRib_R, makeId(zi, tX, -tY), new TGeoCombiTrans(tX, -tY, tZ, new TGeoRotation("r", 0, 0, 270)));
605
606 double zi_x_thickness_1 = wx(zi + ribThick) + 2 * (wallThick + liscThick1);
607 double zi_x_thickness_2 = wx(zi + cell_thickness_z) + 2 * (wallThick + liscThick1);
608
609 double zi_x1_Step = (zi_x_thickness_1 - nx * ribThick) / (nx + 1);
610 double zi_x2_Step = (zi_x_thickness_2 - nx * ribThick) / (nx + 1);
611
612 double zi_y_thickness_1 = wy(zi + ribThick) + 2 * (wallThick + liscThick1);
613 double zi_y_thickness_2 = wy(zi + cell_thickness_z) + 2 * (wallThick + liscThick1);
614
615 double zi_y1_Step = (zi_y_thickness_1 - ny * ribThick) / (ny + 1);
616 double zi_y2_Step = (zi_y_thickness_2 - ny * ribThick) / (ny + 1);
617
618 double zi_x1, zi_x2, zi_y1, zi_y2;
619
620 zi_x1 = zi_x_thickness_1 / 2 - liscThick1;
621 zi_x2 = zi_x_thickness_2 / 2 - liscThick1;
622
623 zi_y1 = zi_y_thickness_1 / 2;
624 zi_y2 = zi_y_thickness_2 / 2;

951Bool_t veto::ProcessHits(FairVolume* vol)
952{
954 // Set parameters at entrance of volume. Reset ELoss.
955 if (gMC->IsTrackEntering()) {
956 fELoss = 0.;
957 fTime = gMC->TrackTime() * 1.0e09;
958 fLength = gMC->TrackLength();
959 gMC->TrackPosition(fPos);
960 gMC->TrackMomentum(fMom);
961 }
962 // Sum energy loss for all steps in the active volume
963 fELoss += gMC->Edep();
964
965 // Create vetoPoint at exit of active volume
966 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
967 if (fELoss == 0.) {
968 return kFALSE;
969 }
970
971 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
972
973 Int_t veto_uniqueId;
974 gMC->CurrentVolID(veto_uniqueId);
975 TParticle* p = gMC->GetStack()->GetCurrentTrack();
976 Int_t pdgCode = p->GetPdgCode();
977 TLorentzVector pos;
978 gMC->TrackPosition(pos);
979 TLorentzVector Mom;
980 gMC->TrackMomentum(Mom);
981 Double_t xmean = (fPos.X() + pos.X()) / 2.;
982 Double_t ymean = (fPos.Y() + pos.Y()) / 2.;
983 Double_t zmean = (fPos.Z() + pos.Z()) / 2.;
984 AddHit(fTrackID,
985 veto_uniqueId,
986 TVector3(xmean, ymean, zmean),
987 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
988 fTime,
989 fLength,
990 fELoss,
991 pdgCode,
992 TVector3(pos.X(), pos.Y(), pos.Z()),
993 TVector3(Mom.Px(), Mom.Py(), Mom.Pz()));
994
995 // Increment number of veto det points in TParticle
996 ShipStack* stack = (ShipStack*)gMC->GetStack();
997 stack->AddPoint(kVETO);
998 }
999
1000 return kTRUE;
