53def brMesonToGammaDP(mass,epsilon,mumPdg,doprint=False):
54 mMeson = PDG.GetParticle(mumPdg).Mass()
55 if (doprint==True): print("Mass of mother %d meson is %3.3f"%(mumPdg,mMeson))
56 if (mass<mMeson): br = 2*epsilon**2*pow((1-mass**2/mMeson**2),3)*mesonBRtoPhoton(mumPdg,doprint)
57 else: br = 0
58 if (doprint==True): print("Branching ratio of %d meson to DP is %.8g"%(mumPdg,br))

61def brMesonToMesonDP(mass,epsilon,mumPdg,dauPdg,doprint=False):
62 mMeson = PDG.GetParticle(mumPdg).Mass()
63 mDaughterMeson = PDG.GetParticle(dauPdg).Mass()
64 if (doprint==True): print("Mass of mother %d meson is %3.3f"%(mumPdg,mMeson))
65 if (doprint==True): print("Mass of daughter %d meson is %3.3f"%(dauPdg,mDaughterMeson))
66 if (mass<(mMeson-mDaughterMeson)):
67 fac1 = pow(mMeson**2.-mDaughterMeson**2.,-3.)
68 fac2 = pow((mass**2.-(mMeson+mDaughterMeson)**2.)*(mass**2.-(mMeson-mDaughterMeson)**2.),1.5)
69 massfactor = fac1*fac2
70 br = (epsilon**2.)*mesonBRtoPhoton(mumPdg,doprint)*massfactor
71 else: br = 0
72 if (doprint==True): print("Branching ratio of %d meson to DP is %.8g"%(mumPdg,br))
73 return br
