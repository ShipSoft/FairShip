90// ----- Passing the event -------------------------------------------
91Bool_t EvtCalcGenerator::ReadEvent(FairPrimaryGenerator* cpg)
92{
93 if (fn == fNevents) {
94 LOG(WARNING) << "End of input file. Rewind.";
95 fn = 0;
96 }
97
98 fTree->GetEntry(fn);
99 fn++;
100 if (fn % 100 == 0)
101 LOGF(info, "Info EvtCalcGenerator: event nr %d", fn);
102
103 Ndau = GetNdaughters(fTree);
104 // Vertex coordinates in the SHiP reference frame, expressed in [cm]
105 Double_t space_unit_conv = 100.; // m to cm
106 Double_t coord_shift = (zDecayVolume - ztarget) / space_unit_conv; // units m
107 Double_t vx_transf = GetVx(fTree) * space_unit_conv; // units cm
108 Double_t vy_transf = GetVy(fTree) * space_unit_conv; // units cm
109 Double_t vz_transf = (GetVz(fTree) - coord_shift) * space_unit_conv; // units cm
110
111 Double_t c = 2.99792458e+10; // speed of light [cm/s]
112 Double_t tof = TMath::Sqrt(vx_transf * vx_transf + vy_transf * vy_transf + vz_transf * vz_transf) / c;
113 Double_t decay_prob = GetDecayProb(fTree);
114 Double_t pdg_dau = 0;
115
116 // Mother LLP
117 Bool_t wanttracking = false;
118 Double_t pdg_llp = 999; // Geantino, placeholder
119
120 cpg->AddTrack(
121 pdg_llp, GetMotherPx(fTree), GetMotherPy(fTree), GetMotherPz(fTree),
122 vx_transf, vy_transf, vz_transf, -1., wanttracking,
123 GetMotherE(fTree), tof, decay_prob);
124
125 wanttracking = true;
126
127 // Secondaries
128 for (int iPart = 0; iPart < Ndau; ++iPart) {
129 pdg_dau = GetDauPDG(fTree, iPart);
130 if ( pdg_dau != -999) {
131 cpg->AddTrack(pdg_dau,
132 GetDauPx(fTree, iPart),
133 GetDauPy(fTree, iPart),
134 GetDauPz(fTree, iPart),
135 vx_transf,
136 vy_transf,
137 vz_transf,
138 0.,
139 wanttracking,
140 GetDauE(fTree, iPart),
141 tof,
142 decay_prob);
143 }
144 }
145
146 return kTRUE;
147}
