23// ----- Default constructor -------------------------------------------
24Bool_t EvtCalcGenerator::Init(const char* fileName, const int startEvent) {
25 if (startEvent < 0) {
26 LOG(error) << "EvtCalcGenerator: startEvent must be >= 0, got "
27 << startEvent;
28 return kFALSE;
29 }
30 fInputFile = std::unique_ptr<TFile>(TFile::Open(fileName, "read"));
31 LOGF(info, "Info EvtCalcGenerator: Opening input file %s", fileName);
32 if (!fInputFile) {
33 LOG(error) << "EvtCalcGenerator: error opening input file " << fileName;
34 fInputFile.reset();
35 return kFALSE;
36 }
37
38 fTree =
39 std::unique_ptr<TTree>(dynamic_cast<TTree*>(fInputFile->Get("LLP_tree")));
40 if (!fTree) {
41 LOG(error) << "EvtCalcGenerator: cannot find tree LLP_tree in file "
42 << fileName;
43 fInputFile.reset();
44 return kFALSE;
45 }
46 fNevents = fTree->GetEntries();
47 if (startEvent >= fNevents) {
48 LOG(error) << "EvtCalcGenerator: startEvent " << startEvent
49 << " is out of range for " << fNevents << " entries";
50 fTree.reset();
51 fInputFile.reset();
52 return kFALSE;
53 }
54 fn = startEvent;
55
56 auto* branches = fTree->GetListOfBranches();
57 if (!branches) {
58 LOG(error) << "EvtCalcGenerator: failed to access tree branches";
59 fTree.reset();
60 fInputFile.reset();
61 return kFALSE;
62 }
63 nBranches = branches->GetEntries();
64 if (nBranches <= 0) {
65 LOG(error) << "EvtCalcGenerator: tree LLP_tree has no branches";
66 fTree.reset();
67 fInputFile.reset();
68 return kFALSE;
69 }
70 branchVars.resize(nBranches);
71
72 for (int i = 0; i < nBranches; ++i) {
73 auto* branch = dynamic_cast<TBranch*>(branches->At(i));
74 if (!branch) {
75 LOG(error) << "EvtCalcGenerator: encountered an invalid branch entry";
76 fTree.reset();
77 fInputFile.reset();
78 return kFALSE;
79 }
80 if (fTree->FindBranch(branch->GetName())) {
81 if (fTree->SetBranchAddress(branch->GetName(), &branchVars[i]) < 0) {
82 LOG(error) << "EvtCalcGenerator: failed to bind branch "
83 << branch->GetName();
84 fTree.reset();
85 fInputFile.reset();
86 return kFALSE;
87 }
88 }
89 }
90
91 return kTRUE;
92}

176// ----- Passing the event -------------------------------------------
177Bool_t EvtCalcGenerator::ReadEvent(FairPrimaryGenerator* cpg) {
178 if (fn == fNevents) {
179 LOG(warning) << "End of input file. Rewind.";
180 fn = 0;
181 }
182
183 fTree->GetEntry(fn);
184 fn++;
185 if (fn % 100 == 0) LOGF(info, "Info EvtCalcGenerator: event nr %d", fn);
186
187 Ndau = GetNdaughters(fTree);
188 // Vertex coordinates in the SHiP reference frame, expressed in [cm]
189 Double_t space_unit_conv = 100.; // m to cm
190 Double_t coord_shift = (zDecayVolume - ztarget) / space_unit_conv; // units m
191 Double_t vx_transf = GetVx(fTree) * space_unit_conv; // units cm
192 Double_t vy_transf = GetVy(fTree) * space_unit_conv; // units cm
193 Double_t vz_transf =
194 (GetVz(fTree) - coord_shift) * space_unit_conv; // units cm
195
196 Double_t c = 2.99792458e+10; // speed of light [cm/s]
197 Double_t tof = TMath::Sqrt(vx_transf * vx_transf + vy_transf * vy_transf +
198 vz_transf * vz_transf) /
199 c;
200 Double_t decay_prob = GetDecayProb(fTree);
201 Double_t pdg_dau = 0;
202
203 // Mother LLP
204 Bool_t wanttracking = false;
205 Double_t pdg_llp = 999; // Geantino, placeholder
206
207 cpg->AddTrack(pdg_llp, GetMotherPx(fTree), GetMotherPy(fTree),
208 GetMotherPz(fTree), vx_transf, vy_transf, vz_transf, -1.,
209 wanttracking, GetMotherE(fTree), tof, decay_prob);
210
211 wanttracking = true;
212
213 // Secondaries
214 for (int iPart = 0; iPart < Ndau; ++iPart) {
215 pdg_dau = GetDauPDG(fTree, iPart);
216 if (pdg_dau != -999) {
217 cpg->AddTrack(pdg_dau, GetDauPx(fTree, iPart), GetDauPy(fTree, iPart),
218 GetDauPz(fTree, iPart), vx_transf, vy_transf, vz_transf, 0.,
219 wanttracking, GetDauE(fTree, iPart), tof, decay_prob);
220 }
221 }
222
223 return kTRUE;
224}
