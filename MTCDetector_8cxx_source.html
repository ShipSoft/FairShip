123TGeoVolume* MTCDetector::CreateSegmentedLayer(const char* name,
124 Double_t width,
125 Double_t height,
126 Double_t thickness,
127 Double_t cellSizeX,
128 Double_t cellSizeY,
129 TGeoMedium* material,
130 Int_t color,
131 Double_t transparency,
132 Int_t LayerId)
133{
134 auto mother = new TGeoBBox(Form("%s_mother", name), width / 2, height / 2, thickness / 2);
135 auto motherVol = new TGeoVolume(Form("%s_mother", name), mother, material);
136 motherVol->SetLineColor(color);
137 motherVol->SetTransparency(transparency);
138
139 auto cell = new TGeoBBox(Form("%s_cell", name), cellSizeX / 2, cellSizeY / 2, thickness / 2);
140 auto cellVol = new TGeoVolume(Form("%s_cell", name), cell, material);
141 cellVol->SetLineColor(color);
142 cellVol->SetTransparency(transparency);
143 AddSensitiveVolume(cellVol);
144 Int_t nX = Int_t(width / cellSizeX);
145 Int_t nY = Int_t(height / cellSizeY);
146
147 for (Int_t i = 0; i < nX; i++) {
148 for (Int_t j = 0; j < nY; j++) {
149 Double_t x = -width / 2 + cellSizeX * (i + 0.5);
150 Double_t y = -height / 2 + cellSizeY * (j + 0.5);
151 motherVol->AddNode(cellVol, 30000000 + LayerId * 100000 + i * nY + j, new TGeoTranslation(x, y, 0));
152 }
153 }
154 return motherVol;
155}

157TGeoVolume* MTCDetector::CreateSciFiModule(const char* name,
158 Double_t width,
159 Double_t height,
160 Double_t thickness,
161 Int_t LayerId)
162{
163 // Define sublayer thicknesses (in cm)
164 // These values mimic the GEANT4 setup:
165 Double_t lowerIronThick = 0.3; // 3 mm
166 Double_t fiberMatThick = 0.135; // 1.35 mm (each fiber mat)
167 Double_t airGap = 0.1; // 1 mm
168 Double_t upperIronThick = 0.3; // 3 mm
169 Double_t zLowerIronInt = -3.5 / 10;
170 Double_t zFiberMat1 = -1.325 / 10;
171 Double_t zAirGap = -0.15 / 10;
172 Double_t zFiberMat2 = 1.025 / 10;
173 Double_t zUpperIronInt = 3.2 / 10;
174 // Total module thickness = 0.3 + 0.135 + 0.1 + 0.135 + 0.3 ≈ 1.0 cm
175
176 // Create the mother volume for the SciFi module
177 auto modMother = new TGeoBBox(Form("%s_mother", name), width / 2, height / 2, thickness / 2);
178 auto modMotherVol = new TGeoVolume(Form("%s_mother", name), modMother, gGeoManager->GetMedium("SciFiMat"));
179 // AddSensitiveVolume(modMotherVol);
180 modMotherVol->SetLineColor(kGreen + 2);
181 modMotherVol->SetTransparency(40);
182
183 // --- Lower Internal Iron ---
184 auto lowerIronBox = new TGeoBBox(Form("%s_lowerIron", name), width / 2, height / 2, lowerIronThick / 2);
185 auto lowerIronVol = new TGeoVolume(Form("%s_lowerIron", name), lowerIronBox, gGeoManager->GetMedium("iron"));
186 lowerIronVol->SetLineColor(kGray + 1);
187 lowerIronVol->SetTransparency(20);
188 // AddSensitiveVolume(lowerIronVol);
189 modMotherVol->AddNode(lowerIronVol, 1, new TGeoTranslation(0, 0, zLowerIronInt));
190
191 // --- Fiber Mat U (Lower SciFi Mat) ---
192 auto fiberMatBoxU = new TGeoBBox(Form("%s_fiberMat_U", name), width / 2, height / 2, fiberMatThick / 2);
193 auto fiberMatVolU = new TGeoVolume(Form("%s_fiberMat_U", name), fiberMatBoxU, gGeoManager->GetMedium("SciFiMat"));
194 // AddSensitiveVolume(fiberMatVolU);
195 modMotherVol->AddNode(fiberMatVolU, 1, new TGeoTranslation(0, 0, zFiberMat1));
196
197 // --- Fiber Mat V (Upper SciFi Mat) ---
198 auto fiberMatBoxV = new TGeoBBox(Form("%s_fiberMat_V", name), width / 2, height / 2, fiberMatThick / 2);
199 auto fiberMatVolV = new TGeoVolume(Form("%s_fiberMat_V", name), fiberMatBoxV, gGeoManager->GetMedium("SciFiMat"));
200 // AddSensitiveVolume(fiberMatVolV);
201 modMotherVol->AddNode(fiberMatVolV, 2, new TGeoTranslation(0, 0, zFiberMat2));
202
203 // --- Upper Internal Iron ---
204 auto upperIronBox = new TGeoBBox(Form("%s_upperIron", name), width / 2, height / 2, upperIronThick / 2);
205 auto upperIronVol = new TGeoVolume(Form("%s_upperIron", name), upperIronBox, gGeoManager->GetMedium("iron"));
206 upperIronVol->SetLineColor(kGray + 1);
207 upperIronVol->SetTransparency(20);
208 modMotherVol->AddNode(upperIronVol, 1, new TGeoTranslation(0, 0, zUpperIronInt));
209
210 // -----------------------------
211 // Now, build the fiber arrays inside each fiber mat.
212 // Create a daughter "mother" volume in each fiber mat to hold the fibers.
213 Int_t fiber_layer_number = 1;
214 auto sciFiLayerMotherUBox = new TGeoBBox(
215 Form("%s_SciFiLayerMother_U", name), width / 2, height / 2, fiberMatThick / 2 / fiber_layer_number);
216 auto sciFiLayerMotherUVol =
217 new TGeoVolume(Form("%s_SciFiLayerMother_U", name), sciFiLayerMotherUBox, gGeoManager->GetMedium("SciFiMat"));
218 AddSensitiveVolume(sciFiLayerMotherUVol);
219 for (int j = 0; j < fiber_layer_number; j++) {
220 Double_t zCenter = -fiberMatThick / 2 + (j + 0.5) * fiberMatThick / fiber_layer_number;
221 fiberMatVolU->AddNode(
222 sciFiLayerMotherUVol, 10000000 + LayerId * 100000 + j, new TGeoTranslation(0, 0, zCenter));
223 }
224
225 auto sciFiLayerMotherVBox = new TGeoBBox(
226 Form("%s_SciFiLayerMother_V", name), width / 2, height / 2, fiberMatThick / 2 / fiber_layer_number);
227 auto sciFiLayerMotherVVol =
228 new TGeoVolume(Form("%s_SciFiLayerMother_V", name), sciFiLayerMotherVBox, gGeoManager->GetMedium("SciFiMat"));
229 AddSensitiveVolume(sciFiLayerMotherVVol);
230 for (int j = 0; j < fiber_layer_number; j++) {
231 Double_t zCenter = -fiberMatThick / 2 + (j + 0.5) * fiberMatThick / fiber_layer_number;
232 fiberMatVolV->AddNode(
233 sciFiLayerMotherVVol, 20000000 + LayerId * 100000 + j, new TGeoTranslation(0, 0, zCenter));
234 }
235 return modMotherVol;
236}

238void MTCDetector::ConstructGeometry()
239{
240 // Initialize media (using FairROOT’s interface)
241 InitMedium("SciFiMat");
242 TGeoMedium* SciFiMat = gGeoManager->GetMedium("SciFiMat");
243 TGeoMedium* air = gGeoManager->GetMedium("air");
244 TGeoMedium* ironMed = gGeoManager->GetMedium("iron");
245 // For the scintillator, you may use the same medium as SciFiMat or another if defined.
246 TGeoMedium* scintMed = gGeoManager->GetMedium("SciFiMat");
247 gGeoManager->SetVisLevel(4);
248 gGeoManager->SetTopVisible();
249
250 // Define the module spacing based on three sublayers:
251 // fIronThick (outer iron), fSciFiThick (SciFi module/fiber module), fScintThick (scintillator)
252 Double_t moduleSpacing = fIronThick + fSciFiThick + fScintThick;
253 Double_t totalLength = fLayers * moduleSpacing;
254
255 // --- Create an envelope volume for the detector (green, semi-transparent) ---
256 auto envBox = new TGeoBBox("MTC_env", fWidth / 2, fHeight / 2, totalLength / 2);
257 auto envVol = new TGeoVolume("MTC", envBox, air);
258 envVol->SetLineColor(kGreen);
259 envVol->SetTransparency(50);
260
261 // --- Outer Iron Layer (gray) ---
262 auto ironBox = new TGeoBBox("MTC_iron", fWidth / 2, fHeight / 2, fIronThick / 2);
263 auto ironVol = new TGeoVolume("MTC_iron", ironBox, ironMed);
264 ironVol->SetLineColor(kGray + 1);
265 ironVol->SetTransparency(20);
266 // Enable the field in the iron volume
267 if (fFieldY != 0)
268 ironVol->SetField(new TGeoUniformMagField(0, fFieldY, 0));
269
270 // --- Assemble the layers into the envelope ---
271 for (Int_t i = 0; i < fLayers; i++) {
272 // Compute the center position (z) for the current module
273 Double_t zPos = -totalLength / 2 + i * moduleSpacing;
274
275 // Place the Outer Iron layer (shifted down by half the SciFi+scint thickness)
276 envVol->AddNode(ironVol, i, new TGeoTranslation(0, 0, zPos + fIronThick / 2));
277 // Create a SciFi module with the current detector id 'i'
278 TGeoVolume* sciFiModuleVol = CreateSciFiModule("MTC_sciFi", fWidth, fHeight, fSciFiThick, i);
279 envVol->AddNode(sciFiModuleVol, i, new TGeoTranslation(0, 0, zPos + fIronThick + fSciFiThick / 2));
280 TGeoVolume* scintVol =
281 CreateSegmentedLayer("MTC_scint", fWidth, fHeight, fScintThick, 1.0, 1.0, scintMed, kAzure + 7, 30, i);
282 // Place the Scintillator layer (shifted up by half the iron thickness)
283 envVol->AddNode(scintVol, i, new TGeoTranslation(0, 0, zPos + fIronThick + fSciFiThick + fScintThick / 2));
284 }
285
286 // Finally, add the envelope to the top volume with the global z offset fZCenter
287 gGeoManager->GetTopVolume()->AddNode(envVol, 1, new TGeoTranslation(0, 0, fZCenter));
288}

295Bool_t MTCDetector::ProcessHits(FairVolume* vol)
296{
298 // Set parameters at entrance of volume. Reset ELoss.
299 if (gMC->IsTrackEntering()) {
300 fELoss = 0.;
301 fTime = gMC->TrackTime() * 1.0e09;
302 fLength = gMC->TrackLength();
303 gMC->TrackPosition(fPos);
304 gMC->TrackMomentum(fMom);
305 }
306 // Sum energy loss for all steps in the active volume
307 fELoss += gMC->Edep();
308
309 // Create vetoPoint when exiting active volume
310 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
311
312 if (fELoss == 0.) {
313 return kFALSE;
314 } // if you do not want hits with zero eloss
315
316 TParticle* p = gMC->GetStack()->GetCurrentTrack();
317 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
318 Int_t pdgCode = p->GetPdgCode();
319 Int_t detID;
320 gMC->CurrentVolID(detID);
321 TLorentzVector Pos;
322 gMC->TrackPosition(Pos);
323 TLorentzVector Mom;
324 gMC->TrackMomentum(Mom);
325 Double_t x, y, z;
326 if (detID / 10000000 == 3) {
327 x = (fPos.X() + Pos.X()) / 2.;
328 y = (fPos.Y() + Pos.Y()) / 2.;
329 z = (fPos.Z() + Pos.Z()) / 2.;
330 } else {
331 x = fPos.X();
332 y = fPos.Y();
333 z = (fPos.Z() + Pos.Z()) / 2.;
334 }
335
336 AddHit(fTrackID,
337 detID,
338 TVector3(x, y, z),
339 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), // entrance momentum
340 fTime,
341 fLength,
342 fELoss,
343 pdgCode);
344 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
345 stack->AddPoint(kMTC);
346 }
347 return kTRUE;
348}
