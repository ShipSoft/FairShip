47Double_t ycross(Double_t a, Double_t R, Double_t x)
48{
49 /*
50 * ycross:
51 * Compute the positive y-coordinate where the vertical line x intersects
52 * the circle of radius R centered at (a, 0). If the line does not intersect
53 * (i.e., (x-a)^2 > R^2), returns -1 as a flag.
54 */
55 Double_t y = -1;
56 Double_t A = R * R - (x - a) * (x - a);
57 if (!(A < 0)) {
58 y = TMath::Sqrt(A);
59 }
60 return y;
61}
62
63Double_t integralSqrt(Double_t ynorm)
64{
65 /*
66 * integralSqrt:
67 * Compute the analytic integral ∫₀^{ynorm} sqrt(1 - t^2) dt
68 * = ½ [ ynorm * sqrt(1 - ynorm^2) + arcsin(ynorm) ].
69 * This is used for normalizing the circular segment area.
70 */
71 Double_t y = 1. / 2. * (ynorm * TMath::Sqrt(1 - ynorm * ynorm) + TMath::ASin(ynorm));
72 return y;
73}
74
75Double_t fraction(Double_t R, Double_t x, Double_t y)
76{
77 /*
78 * fraction:
79 * Compute the fraction of the circle's total area that lies on one side
80 * of a vertical cut at horizontal distance x from the circle center,
81 * up to the intersection height y = sqrt(R^2 - x^2).
82 * Formula:
83 * F = 2 R^2 ∫₀^{y/R} sqrt(1 - t^2) dt - 2 x y
84 * result = F / (π R^2)
85 */
86 Double_t F = 2 * R * R * (integralSqrt(y / R));
87 F -= (2 * x * y);
88 Double_t result = F / (R * R * TMath::Pi());
89 return result;
90}
91
92Double_t area(Double_t a, Double_t R, Double_t xL, Double_t xR)
93{
94 /*
95 * area:
96 * Compute the fraction of the full circle (radius R, center at (a,0))
97 * that lies between the vertical boundaries x = xL and x = xR.
98 * Special cases:
99 * - If [xL, xR] fully covers the circle, returns 1.
100 * - If neither boundary intersects the circle, returns -1 (no overlap).
101 * Otherwise, uses ycross() to find intersection heights, fraction()
102 * to get segment areas, and combines them to yield the net fraction.
103 */
104 Double_t fracL = -1;
105 Double_t fracR = -1;
106 if (xL <= a - R && xR >= a + R) {
107 return 1;
108 }
109
110 Double_t leftC = ycross(a, R, xL);
111 Double_t rightC = ycross(a, R, xR);
112 if (leftC < 0 && rightC < 0) {
113 return -1;
114 }
115
116 if (!(rightC < 0)) {
117 fracR = fraction(R, abs(xR - a), rightC);
118 }
119 if (!(leftC < 0)) {
120 fracL = fraction(R, abs(xL - a), leftC);
121 }
122
123 Double_t theAnswer = 0;
124 if (!(leftC < 0)) {
125 if (xL < a) {
126 theAnswer += 1 - fracL;
127 } else {
128 theAnswer += fracL;
129 }
130 if (!(rightC < 0)) {
131 theAnswer -= 1;
132 }
133 }
134 if (!(rightC < 0)) {
135 if (xR > a) {
136 theAnswer += 1 - fracR;
137 } else {
138 theAnswer += fracR;
139 }
140 }
141 return theAnswer;
142}

257void MTCDetector::CreateSciFiModule(const char* name,
258 TGeoVolumeAssembly* modMotherVol,
259 Double_t width,
260 Double_t height,
261 Double_t thickness,
262 Int_t LayerId)
263{
264 // Define sublayer thicknesses (in cm)
265 // These values mimic the GEANT4 setup:
266 Double_t lowerIronThick = 0.3; // 3 mm
267 fiberMatThick = 0.135; // 1.35 mm (each fiber mat)
268 Double_t airGap = 0.1; // 1 mm
269 Double_t upperIronThick = 0.3; // 3 mm
270 Double_t zLowerIronInt = -3.5 / 10;
271 Double_t zFiberMat1 = -1.325 / 10;
272 Double_t zAirGap = -0.15 / 10;
273 Double_t zFiberMat2 = 1.025 / 10;
274 Double_t zUpperIronInt = 3.2 / 10;
275 // Total module thickness = 0.3 + 0.135 + 0.1 + 0.135 + 0.3 ≈ 1.0 cm
276
277 // --- Lower Internal Iron ---
278 TGeoBBox* lowerIronBox = new TGeoBBox(Form("%s_lowerIron", name), width / 2, height / 2, lowerIronThick / 2);
279 TGeoVolume* lowerIronVol = new TGeoVolume(Form("%s_lowerIron", name), lowerIronBox, gGeoManager->GetMedium("iron"));
280 lowerIronVol->SetLineColor(kGray + 1);
281 lowerIronVol->SetTransparency(20);
282 modMotherVol->AddNode(lowerIronVol, 0, new TGeoTranslation(0, 0, zLowerIronInt));
283
284 // --- Lower Epoxy matrix (replaces SciFiMat layer) ---
285 TGeoVolumeAssembly* ScifiVolU = new TGeoVolumeAssembly(Form("%s_scifi_U", name));
286 modMotherVol->AddNode(ScifiVolU, 0, new TGeoTranslation(0, 0, 0));
287 TGeoBBox* epoxyMatBoxU = new TGeoBBox(Form("%s_epoxyMat", name), width / 2, height / 2, fiberMatThick / 2);
288 TGeoVolume* ScifiMatVolU = new TGeoVolume(Form("%s_epoxyMat", name), epoxyMatBoxU, gGeoManager->GetMedium("Epoxy"));
289 ScifiMatVolU->SetLineColor(kYellow - 2);
290 ScifiMatVolU->SetTransparency(30);
291 ScifiMatVolU->SetVisibility(kFALSE);
292 ScifiMatVolU->SetVisDaughters(kFALSE);
293 ScifiVolU->AddNode(ScifiMatVolU, 0, new TGeoTranslation(0, 0, zFiberMat1));
294
295 // --- Upper Epoxy matrix (replaces SciFiMat layer) ---
296 TGeoVolumeAssembly* ScifiVolV = new TGeoVolumeAssembly(Form("%s_scifi_V", name));
297 modMotherVol->AddNode(ScifiVolV, 0, new TGeoTranslation(0, 0, 0));
298 TGeoBBox* epoxyMatBoxV = new TGeoBBox(Form("%s_epoxyMat", name), width / 2, height / 2, fiberMatThick / 2);
299 TGeoVolume* ScifiMatVolV = new TGeoVolume(Form("%s_epoxyMat", name), epoxyMatBoxV, gGeoManager->GetMedium("Epoxy"));
300 ScifiMatVolV->SetLineColor(kYellow - 2);
301 ScifiMatVolV->SetTransparency(30);
302 ScifiMatVolV->SetVisibility(kFALSE);
303 ScifiMatVolV->SetVisDaughters(kFALSE);
304 ScifiVolV->AddNode(ScifiMatVolV, 0, new TGeoTranslation(0, 0, zFiberMat2));
305
306 // --- Upper Internal Iron ---
307 TGeoBBox* upperIronBox = new TGeoBBox(Form("%s_upperIron", name), width / 2, height / 2, upperIronThick / 2);
308 TGeoVolume* upperIronVol = new TGeoVolume(Form("%s_upperIron", name), upperIronBox, gGeoManager->GetMedium("iron"));
309 upperIronVol->SetLineColor(kGray + 1);
310 upperIronVol->SetTransparency(20);
311 modMotherVol->AddNode(upperIronVol, 0, new TGeoTranslation(0, 0, zUpperIronInt));
312
313 // -----------------------------
314 // Now build the fibers inside each Epoxy block:
315
316 // Common fiber parameters (cm)
317 fSciFiActiveY = height;
318 Int_t numFiberLayers = 6;
319 Double_t layerThick = fiberMatThick / numFiberLayers;
320 Double_t fFiberRadius = 0.01125;
321 fFiberLength = fSciFiActiveY / cos(fSciFiBendingAngle * TMath::DegToRad())
322 - 2 * fFiberRadius * sin(fSciFiBendingAngle * TMath::DegToRad());
323 fFiberPitch = 0.025;
324 Int_t fNumFibers = static_cast<Int_t>(fSciFiActiveX / fFiberPitch);
325
326 // --- Define the SciFi fiber volume ---
327 TGeoTube* fiberTube = new TGeoTube("FiberTube", 0, fFiberRadius, fFiberLength / 2);
328 TGeoVolume* fiberVol = new TGeoVolume("FiberVol", fiberTube, gGeoManager->GetMedium("SciFiMat"));
329 AddSensitiveVolume(fiberVol);
330 fiberVol->SetLineColor(kMagenta);
331 fiberVol->SetTransparency(15);
332 fiberVol->SetVisibility(kFALSE);
333
334 // --- Rotations for U/V fibers ---
335 TGeoRotation* rotU = new TGeoRotation();
336 rotU->RotateY(fSciFiBendingAngle);
337 rotU->RotateX(90.);
338 TGeoRotation* rotV = new TGeoRotation();
339 rotV->RotateY(-fSciFiBendingAngle);
340 rotV->RotateX(90.);
341
342 // --- Place U-fibers inside lower Epoxy ---
343 for (int layer = 0; layer < numFiberLayers; ++layer) {
344 Double_t z0 = -fiberMatThick / 2 + (layer + 0.5) * (layerThick);
345 for (int j = 0; j < fNumFibers; ++j) {
346 Double_t x0 = -fSciFiActiveX / 2 + (j + 0.5) * fFiberPitch;
347 if (layer % 2 == 1) {
348 if (j == fNumFibers - 1) {
349 continue; // Skip the last layer for odd layers
350 }
351 x0 += fFiberPitch / 2;
352 }
353 TGeoCombiTrans* ct = new TGeoCombiTrans("", x0, 0, z0, rotU);
354 Int_t copyNo = 100000000 + 1000000 + 0 * 100000 + layer * 10000 + j;
355 ScifiMatVolU->AddNode(fiberVol, copyNo, ct);
356 }
357 }
358
359 // --- Place V-fibers inside upper Epoxy ---
360 for (int layer = 0; layer < numFiberLayers; ++layer) {
361 Double_t z0 = -fiberMatThick / 2 + (layer + 0.5) * (layerThick);
362 for (int j = 0; j < fNumFibers; ++j) {
363 Double_t x0 = -fSciFiActiveX / 2 + (j + 0.5) * fFiberPitch;
364 if (layer % 2 == 1) {
365 if (j == fNumFibers - 1) {
366 continue; // Skip the last layer for odd layers
367 }
368 x0 += fFiberPitch / 2;
369 }
370 TGeoCombiTrans* ct = new TGeoCombiTrans("", x0, 0, z0, rotV);
371 Int_t copyNo = 100000000 + 1000000 + 1 * 100000 + layer * 10000 + j;
372 ScifiMatVolV->AddNode(fiberVol, copyNo, ct);
373 }
374 }
375}

377void MTCDetector::ConstructGeometry()
378{
379 // Initialize media (using FairROOT’s interface)
380 InitMedium("SciFiMat");
381 TGeoMedium* SciFiMat = gGeoManager->GetMedium("SciFiMat");
382 InitMedium("Epoxy");
383 TGeoMedium* Epoxy = gGeoManager->GetMedium("Epoxy");
384 InitMedium("air");
385 TGeoMedium* air = gGeoManager->GetMedium("air");
386 TGeoMedium* ironMed = gGeoManager->GetMedium("iron");
387 // For the scintillator, you may use the same medium as SciFiMat or another if defined.
388 TGeoMedium* scintMed = gGeoManager->GetMedium("SciFiMat");
389 InitMedium("silicon");
390
391 // Define the module spacing based on three sublayers:
392 // fIronThick (outer iron), fSciFiThick (SciFi module/fiber module), fScintThick (scintillator)
393 Double_t moduleSpacing = fIronThick + fSciFiThick + fScintThick;
394 Double_t totalLength = fLayers * moduleSpacing;
395
396 // --- Create an envelope volume for the detector (green, semi-transparent) ---
397 auto envBox = new TGeoBBox("MTC_env", fWidth / 2, fHeight / 2, totalLength / 2);
398 auto envVol = new TGeoVolume("MTC", envBox, air);
399 envVol->SetLineColor(kGreen);
400 envVol->SetTransparency(50);
401
402 // --- Outer Iron Layer (gray) ---
403 auto ironBox = new TGeoBBox("MTC_iron", fWidth / 2, fHeight / 2, fIronThick / 2);
404 auto ironVol = new TGeoVolume("MTC_iron", ironBox, ironMed);
405 ironVol->SetLineColor(kGray + 1);
406 ironVol->SetTransparency(20);
407 // Enable the field in the iron volume
408 if (fFieldY != 0)
409 ironVol->SetField(new TGeoUniformMagField(0, fFieldY, 0));
410
411 // --- Assemble the layers into the envelope ---
412 TGeoVolumeAssembly* sensitiveModule = new TGeoVolumeAssembly("MTC_layer");
413 // Define a layer for the SciFi module
414 CreateSciFiModule("MTC", sensitiveModule, fWidth, fHeight, fSciFiThick, 1);
415 CreateScintModule("MTC",
416 sensitiveModule,
417 fSciFiThick / 2 + fScintThick / 2,
418 fWidth,
419 fHeight,
420 fScintThick,
421 1.0,
422 1.0,
423 scintMed,
424 kAzure + 7,
425 30,
426 1);
427
428 for (Int_t i = 0; i < fLayers; i++) {
429 // Compute the center position (z) for the current module
430 Double_t zPos = -totalLength / 2 + i * moduleSpacing;
431
432 // Place the Outer Iron layer (shifted down by half the SciFi+scint thickness)
433 envVol->AddNode(ironVol, i, new TGeoTranslation(0, 0, zPos + fIronThick / 2));
434 // Place the sensitive module (SciFi + Scintillator) at the correct z position
435 envVol->AddNode(sensitiveModule, i, new TGeoTranslation(0, 0, zPos + fIronThick + fSciFiThick / 2));
436 }
437
438 // Finally, add the envelope to the top volume with the global z offset fZCenter
439 gGeoManager->GetTopVolume()->AddNode(envVol, 1, new TGeoTranslation(0, 0, fZCenter));
440}

447Bool_t MTCDetector::ProcessHits(FairVolume* vol)
448{
450 // Set parameters at entrance of volume. Reset ELoss.
451 if (gMC->IsTrackEntering()) {
452 fELoss = 0.;
453 fTime = gMC->TrackTime() * 1.0e09;
454 fLength = gMC->TrackLength();
455 gMC->TrackPosition(fPos);
456 gMC->TrackMomentum(fMom);
457 TGeoNavigator* nav = gGeoManager->GetCurrentNavigator();
458 Int_t vol_local_id = nav->GetCurrentNode()->GetNumber() % 1000000; // Local ID within the mat or scint.
459 Int_t layer_id = nav->GetMother(3)->GetNumber(); // Get layer ID.
460 fVolumeID = 100000000 + layer_id * 1000000 + vol_local_id; // 1e8 + layer_id * 1e6 + fibre_local_id;
461 }
462 // Sum energy loss for all steps in the active volume
463 fELoss += gMC->Edep();
464
465 // Create vetoPoint when exiting active volume
466 if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared()) {
467
468 if (fELoss == 0.) {
469 return kFALSE;
470 } // if you do not want hits with zero eloss
471
472 TParticle* p = gMC->GetStack()->GetCurrentTrack();
473 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
474 Int_t pdgCode = p->GetPdgCode();
475 TLorentzVector Pos;
476 gMC->TrackPosition(Pos);
477 TLorentzVector Mom;
478 gMC->TrackMomentum(Mom);
479 Double_t x, y, z;
480 if (fVolumeID / 100000 == 3) {
481 x = (fPos.X() + Pos.X()) / 2.;
482 y = (fPos.Y() + Pos.Y()) / 2.;
483 z = (fPos.Z() + Pos.Z()) / 2.;
484 } else {
485 x = fPos.X();
486 y = fPos.Y();
487 z = (fPos.Z() + Pos.Z()) / 2.;
488 }
489
490 auto hit = AddHit(fTrackID,
491 fVolumeID,
492 TVector3(x, y, z),
493 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), // entrance momentum
494 fTime,
495 fLength,
496 fELoss,
497 pdgCode);
498 // hit->Print();
499 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
500 stack->AddPoint(kMTC);
501 }
502 return kTRUE;
503}

505void MTCDetector::SiPMOverlap()
506{
507 if (gGeoManager->FindVolumeFast("SiPMmapVolU") || gGeoManager->FindVolumeFast("SiPMmapVolV")) {
508 return;
509 }
510 Double_t fLengthScifiMat = fSciFiActiveY;
511 Double_t fWidthChannel = 0.025 * 2;
512 Int_t fNSiPMChan = 128;
513 Int_t fNSiPMs = 7;
514 Int_t fNMats = 1;
515 Double_t fEdge = 0.413 - 0.025 / 2;
516 Double_t initial_shift = fFiberLength * sin(fSciFiBendingAngle * TMath::DegToRad()) / 2;
517 Double_t fCharr = 64 * fWidthChannel;
518 Double_t firstChannelX = -fSciFiActiveX / 2;
519
520 // Contains all plane SiPMs, defined for horizontal fiber plane
521 // To obtain SiPM map for vertical fiber plane rotate by 90 degrees around Z
522 TGeoVolumeAssembly* SiPMmapVolU = new TGeoVolumeAssembly("SiPMmapVolU");
523 TGeoVolumeAssembly* SiPMmapVolV = new TGeoVolumeAssembly("SiPMmapVolV");
524
525 TGeoBBox* ChannelVol_box = new TGeoBBox("ChannelVol", fWidthChannel / 2, fLengthScifiMat / 2, fiberMatThick / 2);
526 TGeoVolume* ChannelVol = new TGeoVolume("ChannelVol", ChannelVol_box, gGeoManager->GetMedium("silicon"));
527 /*
528 Example of fiberID: 123051820, where:
529 - 1: MTC unique ID
530 - 23: layer number
531 - 0: station type (0 for +5 degrees, 1 for -5 degrees, 2 for scint plane)
532 - 5: z-layer number (0-5)
533 - 1820: local fibre ID within the station
534 Example of SiPM global channel (what is seen in the output file): 123004123, where:
535 - 1: MTC unique ID
536 - 23: layer number
537 - 0: station type (0 for +5 degrees, 1 for -5 degrees)
538 - 0: mat number (only 0 by June 2025)
539 - 4: SiPM number (0-N, where N is the number of SiPMs in the station)
540 - 123: number of the SiPM channel (0-127, 128 channels per SiPM)
541 */
542
543 int N = fNMats == 1 ? 1 : 0;
544 Double_t pos = fEdge + firstChannelX + initial_shift;
545 for (int imat = 0; imat < fNMats; imat++) {
546 for (int isipms = 0; isipms < fNSiPMs; isipms++) {
547 // pos+= fEdge;
548 for (int ichannel = 0; ichannel < fNSiPMChan; ichannel++) {
549 // +5 degrees
550 SiPMmapVolU->AddNode(
551 ChannelVol, 100000 * 0 + imat * 10000 + isipms * 1000 + ichannel, new TGeoTranslation(pos, 0, 0));
552 // -5 degrees
553 SiPMmapVolV->AddNode(
554 ChannelVol, 100000 * 1 + imat * 10000 + isipms * 1000 + ichannel, new TGeoTranslation(-pos, 0, 0));
555 pos += fWidthChannel;
556 }
557 }
558 }

561void MTCDetector::GetPosition(Int_t fDetectorID, TVector3& A, TVector3& B)
562{
563 /*
564 Example of fiberID: 123051820, where:
565 - 1: MTC unique ID
566 - 23: layer number
567 - 0: station type (0 for +5 degrees, 1 for -5 degrees, 2 for scint plane)
568 - 5: z-layer number (0-5)
569 - 1820: local fibre ID within the station
570 Example of SiPM global channel (what is seen in the output file): 123004123, where:
571 - 1: MTC unique ID
572 - 23: layer number
573 - 0: station type (0 for +5 degrees, 1 for -5 degrees)
574 - 0: mat number (only 0 by June 2025)
575 - 4: SiPM number (0-N, where N is the number of SiPMs in the station)
576 - 123: number of the SiPM channel (0-127, 128 channels per SiPM)
577 */
578
579 Int_t station_number = int(fDetectorID / 1e6) % 100;
580 Int_t plane_type = int(fDetectorID / 1e5) % 10; // 0 for horizontal, 1 for vertical

668void MTCDetector::SiPMmapping()
669{
670 // check if containers are already filled
671 if (!fibresSiPM_U.empty() || !fibresSiPM_V.empty() || !SiPMPos_U.empty() || !SiPMPos_V.empty()) {
672 LOG(WARN) << "SiPM mapping already done, skipping.";
673 return;
674 }
675 Float_t fibresRadius = -1;
676 Float_t dSiPM = -1;
677 TGeoNode* vol;
678 TGeoNode* fibre;
679 SiPMOverlap();
680 // Loop over both U and V planes
681 std::vector<std::pair<const char*, const char*>> sipm_planes = {{"SiPMmapVolU", "MTC_scifi_U"},
682 {"SiPMmapVolV", "MTC_scifi_V"}};
683 for (const auto& [sipm_vol, scifi_vol] : sipm_planes) {
684 auto sipm = gGeoManager->FindVolumeFast(sipm_vol);
685 if (!sipm)
686 continue;
687 TObjArray* Nodes = sipm->GetNodes();
688 auto plane = gGeoManager->FindVolumeFast(scifi_vol);
689 if (!plane)
690 continue;
691 for (int imat = 0; imat < plane->GetNodes()->GetEntries(); imat++) {
692 auto mat = static_cast<TGeoNode*>(plane->GetNodes()->At(imat));
693 Float_t t1 = mat->GetMatrix()->GetTranslation()[0];
694 auto vmat = mat->GetVolume();
695 for (int ifibre = 0; ifibre < vmat->GetNodes()->GetEntriesFast(); ifibre++) {
696 fibre = static_cast<TGeoNode*>(vmat->GetNodes()->At(ifibre));
697 if (fibresRadius < 0) {
698 auto tmp = fibre->GetVolume()->GetShape();
699 auto S = dynamic_cast<TGeoBBox*>(tmp);
700 fibresRadius = S->GetDX();
701 }
702 Int_t fID =
703 fibre->GetNumber() % 1000000 + imat * 1e4; // local fibre number, global fibre number = SO+fID
704 TVector3 Atop, Bbot;
705 GetPosition(fibre->GetNumber(), Atop, Bbot);
706 Float_t a = Atop[0];
707
708 // check for overlap with any of the SiPM channels in the same mat
709 for (Int_t nChan = 0; nChan < Nodes->GetEntriesFast(); nChan++) { // 7 SiPMs total times 128 channels
710 vol = static_cast<TGeoNode*>(Nodes->At(nChan));
711 Int_t N = vol->GetNumber();
712 Float_t xcentre = vol->GetMatrix()->GetTranslation()[0];
713 if (dSiPM < 0) {
714 TGeoBBox* B = dynamic_cast<TGeoBBox*>(vol->GetVolume()->GetShape());
715 dSiPM = B->GetDX();
716 }
717 if (TMath::Abs(xcentre - a) > 4 * fibresRadius) {
718 continue;
719 } // no need to check further
720 Float_t W = area(a, fibresRadius, xcentre - dSiPM, xcentre + dSiPM);
721 if (W < 0) {
722 continue;
723 }
724 std::array<float, 2> Wa;
725 Wa[0] = W;
726 Wa[1] = a;
727 if (sipm_vol == std::string("SiPMmapVolU")) {
728 fibresSiPM_U[N][fID] = Wa;
729 } else {
730 fibresSiPM_V[N][fID] = Wa;
731 }
732 }
733 }
734 }
735 // calculate also local SiPM positions based on fibre positions and their fraction
736 // probably an overkill, maximum difference between weighted average and central position < 6 micron.
737 if (sipm_vol == std::string("SiPMmapVolU")) {
738 for (auto [N, it] : fibresSiPM_U) {
739 Float_t m = 0;
740 Float_t w = 0;
741 for (auto [fibre, Wa] : it) {
742 m += Wa[0] * Wa[1];
743 w += Wa[0];
744 }
745 SiPMPos_U[N] = m / w;
746 }
747 // make inverse mapping, which fibre is associated to which SiPMs
748 for (auto [N, it] : fibresSiPM_U) {
749 for (auto [nfibre, Wa] : it) {
750 siPMFibres_U[nfibre][N] = Wa;
751 }
752 }
753 } else if (sipm_vol == std::string("SiPMmapVolV")) {
754 for (auto [N, it] : fibresSiPM_V) {
755 Float_t m = 0;
756 Float_t w = 0;
757 for (auto [fibre, Wa] : it) {
758 m += Wa[0] * Wa[1];
759 w += Wa[0];
760 }
761 SiPMPos_V[N] = m / w;
762 }
763 // make inverse mapping, which fibre is associated to which SiPMs
764 for (auto [N, it] : fibresSiPM_V) {
765 for (auto [nfibre, Wa] : it) {
766 siPMFibres_V[nfibre][N] = Wa;
767 }
768 }
769 }
770 }
771}
