128Bool_t TimeDet::ProcessHits(FairVolume* vol) {
130 // Set parameters at entrance of volume. Reset ELoss.
131 if (gMC->IsTrackEntering()) {
132 fELoss = 0.;
133 fTime = gMC->TrackTime() * 1.0e09;
134 fLength = gMC->TrackLength();
135 gMC->TrackPosition(fPos);
136 gMC->TrackMomentum(fMom);
137 }
138
139 // Sum energy loss for all steps in the active volume
140 fELoss += gMC->Edep();
141
142 // Create vetoPoint at exit of active volume
143 if (gMC->IsTrackExiting() || gMC->IsTrackStop() ||
144 gMC->IsTrackDisappeared()) {
145 if (fELoss == 0.) {
146 return kFALSE;
147 }
148
149 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
150 fEventID = gMC->CurrentEvent();
151 Int_t uniqueId;
152 gMC->CurrentVolID(uniqueId);
153 if (uniqueId > 1000000) // Solid scintillator case
154 {
155 Int_t vcpy;
156 gMC->CurrentVolOffID(1, vcpy);
157 if (vcpy == 5) uniqueId += 4; // Copy of half
158 }
159
160 TParticle* p = gMC->GetStack()->GetCurrentTrack();
161 Int_t pdgCode = p->GetPdgCode();
162 TLorentzVector Pos;
163 gMC->TrackPosition(Pos);
164 TLorentzVector Mom;
165 gMC->TrackMomentum(Mom);
166 Double_t xmean = (fPos.X() + Pos.X()) / 2.;
167 Double_t ymean = (fPos.Y() + Pos.Y()) / 2.;
168 Double_t zmean = (fPos.Z() + Pos.Z()) / 2.;
169
170 // cout << uniqueId << " :(" << xmean << ", " << ymean << ", " << zmean <<
171 // "): " << gMC->CurrentVolName() << endl;
172
173 AddHit(fEventID, fTrackID, uniqueId, TVector3(xmean, ymean, zmean),
174 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength, fELoss,
175 pdgCode, TVector3(Pos.X(), Pos.Y(), Pos.Z()),
176 TVector3(Mom.Px(), Mom.Py(), Mom.Pz()));
177
178 // Increment number of veto det points in TParticle
179 ShipStack* stack = dynamic_cast<ShipStack*>(gMC->GetStack());
180 stack->AddPoint(kTimeDet);
181 }
182
183 return kTRUE;
184}

202void TimeDet::UpdatePointTrackIndices(const std::map<Int_t, Int_t>& indexMap) {
203 for (auto& point : *fTimeDetPoints) {
204 Int_t oldTrackID = point.GetTrackID();
205 auto iter = indexMap.find(oldTrackID);
206 if (iter != indexMap.end()) {

215void TimeDet::ConstructGeometry() {
216 TGeoVolume* top = gGeoManager->GetTopVolume();
217
218 ShipGeo::InitMedium("polyvinyltoluene");
219 TGeoMedium* Scint = gGeoManager->GetMedium("polyvinyltoluene");
220
222
223 fDetector = new TGeoVolumeAssembly("Timing Detector");
224
225 TGeoVolume* plate =
226 gGeoManager->MakeBox("TimeDet", Scint, fxBar / 2, fyBar / 2, fzBar / 2);
227 plate->SetLineColor(kBlue);
228 AddSensitiveVolume(plate);
229
230 for (int ib = 0; ib < fNBars; ib++) {
231 int irow = 0, icol = 0;
232 GetBarRowCol(ib, irow, icol);
233
234 double xbar = 0, ybar = 0, zbar = 0;
235 xbar = GetXCol(icol);
236 ybar = GetYRow(irow);
237 zbar = GetZBar(irow, icol);
238
239 fDetector->AddNode(plate, ib, new TGeoTranslation(xbar, ybar, zbar));
240
241 // printf("%3i %3i %2i %8.3f %8.3f %8.3f\n",ib, irow,icol,
242 // xbar,ybar,zbar);
243 }
244
245 top->AddNode(fDetector, 1, new TGeoTranslation(0, 0, fzPos));
246
248
249 return;
250}

267double TimeDet::GetXCol(int ic) const {
268 ic += 1;
269 double x = fxBar * ic - fxOv * (ic - 1) - fxBar / 2;
270 return x - fxSize / 2 + fxCenter;
271}

273double TimeDet::GetYRow(int ir) const {
274 ir += 1;
275 double y = fyBar * ir - fyOv * (ir - 1) - fyBar / 2;
276 return y - fySize / 2 + fyCenter;
277}

279double TimeDet::GetZBar(int ir, int ic) const {
280 double z = (ir % 2) * fdzBarRow + (ic % 2) * fdzBarCol;
