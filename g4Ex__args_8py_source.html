161 myTimer['pythia']+=time.time()-t_0
162# pythia interaction happens at 0,0,0
163 #x = rnr.Uniform(-3.,3.)
164 #y = rnr.Uniform(-3.,3.)
165 # leave this smearing for later
166 pos = G4ThreeVector(0*cm, 0*cm, -50*m)
167 vertex = G4PrimaryVertex(pos,0.)
168 # create new primaries and set them to the vertex
169 particles = myPythia.GetListOfParticles()
170 for p in particles:
171 if p.GetStatusCode()!=1 : continue
172 pid = p.GetPdgCode()
173 if tauOnly and abs(pid) != 16: continue
174 if pid in notWanted : continue
175 G4particle = G4PrimaryParticle( pid )
176 v = TLorentzVector()
177 p.Momentum(v)
178 if v.E()*GeV < ecut : continue
179 G4particle.Set4Momentum( v.Px()*GeV,v.Py()*GeV,v.Pz()*GeV,v.E()*GeV )
180 vertex.SetPrimary( G4particle )
181# store mother ID
182 mkey = p.GetMother(0)+1
183 mother = myPythia.GetParticle(mkey)
184 curPid = p.GetPdgCode() + 10000 # make it positive
185 moPid = mother.GetPdgCode() + 10000
186 w = curPid + moPid * 100000
187 G4particle.SetWeight(w)
188 npart += 1
189 if tauOnly and debug: myPythia.EventListing()
190 anEvent.AddPrimaryVertex( vertex )
191 myTimer['geant4_conv']+=time.time()-t_0
