38strawtubesHit::strawtubesHit(strawtubesPoint* p, Double_t t0)
39 : ShipHit()
40{
41 TVector3 start = TVector3();
42 TVector3 stop = TVector3();
43 fDetectorID = p->GetDetectorID();
44 strawtubes* module =
45 dynamic_cast<strawtubes*>(FairRunSim::Instance()->GetListOfModules()->FindObject("strawtubes"));
46 Double_t v_drift = module->StrawVdrift();
47 Double_t sigma_spatial = module->StrawSigmaSpatial();
48 strawtubes::StrawEndPoints(fDetectorID, start, stop);
49 Double_t t_drift = fabs(gRandom->Gaus(p->dist2Wire(), sigma_spatial)) / v_drift;
50 fdigi = t0 + p->GetTime() + t_drift + (stop[0] - p->GetX()) / speedOfLight;
51 flag = true;
52}
