383 n = -1

385 fPos = ROOT.TVector3()
386 fMom = ROOT.TVector3()

391 DTrack.SetTitle(fT.__repr__())
392 fT.GetStartVertex(fPos)
393 hitlist = {}
394 hitlist[fPos.Z()] = [fPos.X(), fPos.Y()]
395 # look for HNL
396 if abs(fT.GetPdgCode()) == options.HiddenParticleID:
397 for da in sTree.MCTrack:
398 if da.GetMotherId() == n:
399 break
400 # end vertex of HNL
401 da.GetStartVertex(fPos)
402 hitlist[fPos.Z()] = [fPos.X(), fPos.Y()]
403 # loop over all sensitive volumes to find hits
404 for P in [
405 "vetoPoint",
406 "muonPoint",
407 "strawtubesPoint",
408 "ShipRpcPoint",
409 "TargetPoint",
410 "MTCDetPoint",
411 "SiliconTargetPoint",
412 "TimeDetPoint",
413 ]:
414 if not sTree.GetBranch(P):
415 continue
416 c = eval("sTree." + P)
417 for p in c:
418 if p.GetTrackID() == n:
419 if hasattr(p, "LastPoint"):
420 lp = p.LastPoint()
421 if lp.x() == lp.y() and lp.x() == lp.z() and lp.x() == 0:
422 # must be old data, don't expect hit at 0,0,0
423 hitlist[p.GetZ()] = [p.GetX(), p.GetY()]
424 else:
425 hitlist[lp.z()] = [lp.x(), lp.y()]
426 hitlist[2.0 * p.GetZ() - lp.z()] = [
427 2.0 * p.GetX() - lp.x(),
428 2.0 * p.GetY() - lp.y(),
429 ]
430 else:
431 hitlist[p.GetZ()] = [p.GetX(), p.GetY()]
432 if len(hitlist) == 1:
433 if fT.GetMotherId() < 0:
434 continue
435 if (
436 abs(sTree.MCTrack[fT.GetMotherId()].GetPdgCode())
437 == options.HiddenParticleID
438 ):
439 # still would like to draw track stub
440 # check for end vertex
441 evVx = False
442 for da in sTree.MCTrack:
443 if da.GetMotherId() == n:
444 evVx = True
445 break
446 if evVx:
447 hitlist[da.GetStartZ()] = [da.GetStartX(), da.GetStartY()]
448 else:
449 zEx = 10 * u.m
450 fT.GetMomentum(fMom)
451 lam = (zEx + fPos.Z()) / fMom.Z()
452 hitlist[zEx + fPos.Z()] = [
453 fPos.X() + lam * fMom.X(),
454 fPos.Y() + lam * fMom.Y(),
455 ]

477 def DrawFittedTracks(self, option=""):
478 n, ntot = -1, 0
479 for fT in sTree.FitTracks:
480 n += 1
481 fst = fT.getFitStatus()
482 if not fst.isFitConverged():
483 continue

485 continue

487 DTrack.SetPickable(ROOT.kTRUE)
488 DTrack.SetTitle(fT.__repr__())
489 fstate = fT.getFittedState(0)
490 fPos = fstate.getPos()
491 fMom = fstate.getMom()
492 pos = fPos
493 mom = fMom
494 pid = fstate.getPDG()
495 zs = self.z_start
496 for i in range(self.niter):
497 rc, newpos, newmom = TrackExtrapolateTool.extrapolateToPlane(fT, zs)

499 DTrack.SetNextPoint(newpos.X(), newpos.Y(), newpos.Z())
500 else:
501 print("error with extrapolation: z=", zs)
502 # use linear extrapolation
503 px, py, pz = mom.X(), mom.Y(), mom.Z()
504 lam = (zs - pos.Z()) / pz
505 DTrack.SetNextPoint(pos.X() + lam * px, pos.Y() + lam * py, zs)

507 DTrack.SetName("FitTrack_" + str(n))
