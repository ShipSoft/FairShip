402 def DrawMCTracks(self, option=""):
403 n = -1
404 ntot = 0
405 fPos = ROOT.TVector3()
406 fMom = ROOT.TVector3()
407 for fT in sTree.MCTrack:
408 n += 1
409 DTrack = ROOT.TEveLine()
410 DTrack.SetPickable(ROOT.kTRUE)
411 DTrack.SetTitle(fT.__repr__())
412 fT.GetStartVertex(fPos)
413 hitlist = {}
414 hitlist[fPos.Z()] = [fPos.X(), fPos.Y()]
415 # look for HNL
416 if abs(fT.GetPdgCode()) == options.HiddenParticleID:
417 for da in sTree.MCTrack:
418 if da.GetMotherId() == n:
419 break
420 # end vertex of HNL
421 da.GetStartVertex(fPos)
422 hitlist[fPos.Z()] = [fPos.X(), fPos.Y()]
423 # loop over all sensitive volumes to find hits
424 for P in [
425 "vetoPoint",
426 "strawtubesPoint",
427 "ShipRpcPoint",
428 "TargetPoint",
429 "MTCDetPoint",
430 "SiliconTargetPoint",
431 "TimeDetPoint",
432 ]:
433 if not sTree.GetBranch(P):
434 continue
435 c = eval("sTree." + P)
436 for p in c:
437 if p.GetTrackID() == n:
438 if hasattr(p, "LastPoint"):
439 lp = p.LastPoint()
440 if lp.x() == lp.y() and lp.x() == lp.z() and lp.x() == 0:
441 # must be old data, don't expect hit at 0,0,0
442 hitlist[p.GetZ()] = [p.GetX(), p.GetY()]
443 else:
444 hitlist[lp.z()] = [lp.x(), lp.y()]
445 hitlist[2.0 * p.GetZ() - lp.z()] = [
446 2.0 * p.GetX() - lp.x(),
447 2.0 * p.GetY() - lp.y(),
448 ]
449 else:
450 hitlist[p.GetZ()] = [p.GetX(), p.GetY()]
451 if len(hitlist) == 1:
452 if fT.GetMotherId() < 0:
453 continue
454 if (
455 abs(sTree.MCTrack[fT.GetMotherId()].GetPdgCode())
456 == options.HiddenParticleID
457 ):
458 # still would like to draw track stub
459 # check for end vertex
460 evVx = False
461 for da in sTree.MCTrack:
462 if da.GetMotherId() == n:
463 evVx = True
464 break
465 if evVx:
466 hitlist[da.GetStartZ()] = [da.GetStartX(), da.GetStartY()]
467 else:
468 zEx = 10 * u.m
469 fT.GetMomentum(fMom)
470 lam = (zEx + fPos.Z()) / fMom.Z()
471 hitlist[zEx + fPos.Z()] = [
472 fPos.X() + lam * fMom.X(),
473 fPos.Y() + lam * fMom.Y(),
474 ]
475 # sort in z
476 lz = list(hitlist.keys())
477 if len(lz) > 1:
478 lz.sort()
479 for z in lz:
480 DTrack.SetNextPoint(hitlist[z][0], hitlist[z][1], z)
481 p = pdg.GetParticle(fT.GetPdgCode())
482 if p:
483 pName = p.GetName()
484 else:
485 pName = str(fT.GetPdgCode())
486 DTrack.SetName("MCTrack_" + str(n) + "_" + pName)
487 c = ROOT.kYellow
488 if abs(fT.GetPdgCode()) == options.HiddenParticleID:
489 c = ROOT.kMagenta
490 DTrack.SetMainColor(c)
491 DTrack.SetLineWidth(3)
492 self.comp.AddElement(DTrack)
493 ntot += 1

496 def DrawFittedTracks(self, option=""):
497 n, ntot = -1, 0
498 for fT in sTree.FitTracks:
499 n += 1
500 fst = fT.getFitStatus()
501 if not fst.isFitConverged():
502 continue
503 if fst.getNdf() < 20:
504 continue
505 DTrack = ROOT.TEveLine()
506 DTrack.SetPickable(ROOT.kTRUE)
507 DTrack.SetTitle(fT.__repr__())
508 fstate = fT.getFittedState(0)
509 fPos = fstate.getPos()
510 fMom = fstate.getMom()
511 pos = fPos
512 mom = fMom
513 pid = fstate.getPDG()
514 zs = self.z_start
515 for i in range(self.niter):
516 rc, newpos, newmom = TrackExtrapolateTool.extrapolateToPlane(fT, zs)
517 if rc:
518 DTrack.SetNextPoint(newpos.X(), newpos.Y(), newpos.Z())
519 else:
520 print("error with extrapolation: z=", zs)
521 # use linear extrapolation
522 px, py, pz = mom.X(), mom.Y(), mom.Z()
523 lam = (zs - pos.Z()) / pz
524 DTrack.SetNextPoint(pos.X() + lam * px, pos.Y() + lam * py, zs)
525 zs += self.dz
526 DTrack.SetName("FitTrack_" + str(n))
527 c = ROOT.kWhite
528 if abs(pid) in self.trackColors:
529 c = self.trackColors[abs(pid)]
530 DTrack.SetMainColor(c)
531 DTrack.SetLineWidth(3)

534 print("draw ", ntot, " fitted tracks")
