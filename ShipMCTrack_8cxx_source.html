121Double_t ShipMCTrack::GetEnergy() const
122{
123 if (fM<0){
124// older data, mass not made persistent
125 Double_t mass = GetMass();
126 return TMath::Sqrt(mass*mass + fPx*fPx + fPy*fPy + fPz*fPz );
127 }else{
128 return TMath::Sqrt(fM*fM + fPx*fPx + fPy*fPy + fPz*fPz );
129 }
130}

157{
158 Double_t e = GetEnergy();
159 Double_t y = 0.5 * TMath::Log( (e+fPz) / (e-fPz) );
