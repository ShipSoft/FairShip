143
144Double_t ShipMCTrack::GetEnergy() const
145{
146 if (fM<0){
147// older data, mass not made persistent
148 Double_t mass = GetMass();
149 return TMath::Sqrt(mass*mass + fPx*fPx + fPy*fPy + fPz*fPz );
150 }else{
151 return TMath::Sqrt(fM*fM + fPx*fPx + fPy*fPy + fPz*fPz );
152 }
153}
154// ----- Public method GetMass -----------------------------------------
155Double_t ShipMCTrack::GetMass() const
156{
157 if (fM<0){
158// older data, mass not made persistent
159 if ( TDatabasePDG::Instance() ) {
160 TParticlePDG* particle = TDatabasePDG::Instance()->GetParticle(fPdgCode);
161 if ( particle ) { return particle->Mass(); }
162 else { return 0.; }
163 }
164 }
165 return fM;
166}

178// ----- Public method GetRapidity -------------------------------------
179Double_t ShipMCTrack::GetRapidity() const
180{
181 Double_t e = GetEnergy();
182 Double_t y = 0.5 * TMath::Log( (e+fPz) / (e-fPz) );
183 return y;
184}
