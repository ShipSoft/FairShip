26def wilson_interval(k: int, n: int) -> float:
27 """Wilson score interval half-width for a binomial proportion.
28
29 Parameters
30 ----------
31 k : int
32 Number of successes.
33 n : int
34 Number of trials.
35
36 Returns
37 -------
38 float
39 Half-width of the 68% Wilson score interval (~1 sigma).
40 """
41 if n == 0:
42 return 0.0
43 z = 1.0 # 1-sigma
44 p = k / n
45 denom = 1 + z**2 / n
46 spread = z * math.sqrt(p * (1 - p) / n + z**2 / (4 * n**2)) / denom
47 return spread

50class TrackingBenchmark:
51 """Compute tracking benchmark metrics from simulation and reconstruction files.
52
53 Parameters
54 ----------
55 sim_file : str
56 Path to MC simulation ROOT file (contains cbmsim tree).
57 reco_file : str
58 Path to reconstruction ROOT file (contains ship_reco_sim tree).
59 geo_file : str
60 Path to geometry ROOT file.
61 purity_cut : float
62 Minimum hit purity fraction for a reco track to be considered matched.
63 min_hits : int
64 Minimum number of straw hits for reconstructibility.
65 min_stations : int
66 Minimum number of tracking stations crossed for reconstructibility.
67 """

69 def __init__(
70 self,
71 sim_file: str,
72 reco_file: str,

75 min_hits: int = 25,

79 self.min_hits = min_hits
80 self.min_stations = min_stations

87
88 self.f_geo = ROOT.TFile.Open(geo_file, "read")

95 def _is_reconstructible(self, mc_track_id: int) -> bool:
96 """Check if an MC particle meets reconstructibility criteria.
97
98 A particle is reconstructible if it is a charged primary with
99 hits in >= min_stations tracking stations and >= min_hits total
100 straw hits. This matches the cuts in shipDigiReco.findTracks().
101 """
102 mc_track = self.sim_tree.MCTrack[mc_track_id]
103
104 # Must be primary (no mother)
105 if mc_track.GetMotherId() >= 0:
106 return False
107
108 # Must be charged
109 pdg_code = mc_track.GetPdgCode()
110 particle = self.PDG.GetParticle(pdg_code)
111 if particle is None or particle.Charge() == 0:
112 return False
113
114 # Count hits per station
115 stations: set[int] = set()
116 n_hits = 0
117 for hit in self.sim_tree.strawtubesPoint:
118 if hit.GetTrackID() != mc_track_id:
119 continue
120 n_hits += 1
121 det_id = hit.GetDetectorID()
122 station = int(det_id // 1_000_000)
123 stations.add(station)
124
125 return n_hits >= self.min_hits and len(stations) >= self.min_stations
126
127 def _get_ptruth_first(self, mc_track_id: int) -> tuple[float, float, float, float]:
128 """Get MC truth momentum at the first straw hit.
129
130 Follows the pattern from macro/ShipAna.py:getPtruthFirst().
131 """
132 for hit in self.sim_tree.strawtubesPoint:
133 if hit.GetTrackID() == mc_track_id:
134 px, py, pz = hit.GetPx(), hit.GetPy(), hit.GetPz()
135 p = math.sqrt(px**2 + py**2 + pz**2)
136 return p, px, py, pz
137 return -1.0, -1.0, -1.0, -1.0
138
139 def _get_truth_pos_first(self, mc_track_id: int) -> tuple[float, float, float]:
140 """Get MC truth position at the first straw hit."""
141 for hit in self.sim_tree.strawtubesPoint:
142 if hit.GetTrackID() == mc_track_id:
143 return hit.GetX(), hit.GetY(), hit.GetZ()
144 return 0.0, 0.0, 0.0
145
146 def _get_truth_slopes(self, mc_track_id: int) -> tuple[float, float]:
147 """Get MC truth track slopes tx=px/pz, ty=py/pz at first straw hit."""
148 for hit in self.sim_tree.strawtubesPoint:
149 if hit.GetTrackID() == mc_track_id:
150 px, py, pz = hit.GetPx(), hit.GetPy(), hit.GetPz()
151 if abs(pz) > 1e-10:
152 return px / pz, py / pz
153 return 0.0, 0.0
154 return 0.0, 0.0
155
156 def _fracMCsame(self, reco_track_idx: int) -> tuple[float, int]:
157 """Get the hit purity and dominant MC track ID for a reco track.
158
159 Uses the Tracklets branch to access hit indices, then checks
160 which MC track contributed most hits.
161 """
162 tracklet = self.reco_tree.Tracklets[reco_track_idx]
163 hit_indices = tracklet.getList()
164
165 track_counts: dict[int, int] = {}
166 n_hits = 0
167 for idx in hit_indices:
168 mc_id = self.sim_tree.strawtubesPoint[idx].GetTrackID()
169 track_counts[mc_id] = track_counts.get(mc_id, 0) + 1
170 n_hits += 1
171
172 if not track_counts:
173 return 0.0, -999
174
175 tmax = max(track_counts, key=track_counts.__getitem__)
176 frac = track_counts[tmax] / n_hits if n_hits > 0 else 0.0
177 return frac, tmax

179 def compute_metrics(self) -> dict[str, Any]:
180 """Run the full benchmark analysis over all events.
181
182 Returns
183 -------
184 dict
185 Dictionary of metrics compatible with compare_metrics.py format.
186 """
187 n_events = self.sim_tree.GetEntries()
188 n_reco_events = self.reco_tree.GetEntries()
189 if n_events != n_reco_events:
190 print(f"WARNING: sim has {n_events} events, reco has {n_reco_events}")
191 n_events = min(n_events, n_reco_events)
192
193 # Book histograms
194 h_dp_over_p = ROOT.TH1D(
195 "h_dp_over_p",
196 "Momentum resolution;(p_{reco} #minus p_{truth}) / p_{truth};Entries",
197 100,
198 -0.5,
199 0.5,
200 )
201 h_dp_vs_p = ROOT.TH2D(
202 "h_dp_vs_p",
203 "Momentum resolution vs p_{truth};p_{truth} [GeV/c];(p_{reco} #minus p_{truth}) / p_{truth}",
204 50,
205 0,
206 120,
207 100,
208 -0.5,
209 0.5,
210 )
211 h_dx = ROOT.TH1D(
212 "h_dx",
213 "x resolution at first hit;x_{reco} #minus x_{truth} [cm];Entries",
214 100,
215 -5.0,
216 5.0,
217 )
218 h_dy = ROOT.TH1D(
219 "h_dy",
220 "y resolution at first hit;y_{reco} #minus y_{truth} [cm];Entries",
221 100,
222 -5.0,
223 5.0,
224 )
225 h_dtx = ROOT.TH1D(
226 "h_dtx",
227 "Slope t_{x} resolution;t_{x,reco} #minus t_{x,truth} (t_{x} = p_{x}/p_{z});Entries",
228 100,
229 -0.01,
230 0.01,
231 )
232 h_dty = ROOT.TH1D(
233 "h_dty",
234 "Slope t_{y} resolution;t_{y,reco} #minus t_{y,truth} (t_{y} = p_{y}/p_{z});Entries",
235 100,
236 -0.01,
237 0.01,
238 )
239 h_hit_residual = ROOT.TH1D(
240 "h_hit_residual",
241 "Unbiased hit residual;drift distance residual [cm];Entries",
242 100,
243 -0.2,
244 0.2,
245 )
246 h_hit_pull = ROOT.TH1D(
247 "h_hit_pull",
248 "Unbiased hit pull;(residual) / #sigma;Entries",
249 100,
250 -5.0,
251 5.0,
252 )
253 h_chi2ndf = ROOT.TH1D("h_chi2ndf", "#chi^{2}/ndf;#chi^{2}/ndf;Entries", 100, 0, 20)
254 h_p_truth = ROOT.TH1D("h_p_truth", "p_{truth} (reconstructible);p [GeV/c];Entries", 50, 0, 120)
255 h_p_matched = ROOT.TH1D("h_p_matched", "p_{truth} (matched);p [GeV/c];Entries", 50, 0, 120)
256
257 # Counters
258 n_reconstructible = 0
259 n_matched_mc = 0 # reconstructible MC particles with >= 1 matched reco track
260 n_total_reco = 0
261 n_matched_reco = 0 # reco tracks passing purity cut
262 n_clone_reco = 0 # extra matches beyond the first for same MC particle
263
264 for i_event in range(n_events):
265 self.sim_tree.GetEvent(i_event)
266 self.reco_tree.GetEvent(i_event)
267
268 # Find reconstructible MC particles
269 reconstructible_ids: set[int] = set()
270 n_mc_tracks = len(self.sim_tree.MCTrack)
271 for mc_id in range(n_mc_tracks):
272 if self._is_reconstructible(mc_id):
273 reconstructible_ids.add(mc_id)
274 p_truth, _, _, _ = self._get_ptruth_first(mc_id)
275 if p_truth > 0:
276 h_p_truth.Fill(p_truth)
277
278 n_reconstructible += len(reconstructible_ids)
279
280 # Match reco tracks to MC
281 n_reco = self.reco_tree.FitTracks.size()
282 n_total_reco += n_reco
283
284 # Track which MC particles have been matched in this event
285 matched_mc_this_event: set[int] = set()
286
287 for i_reco in range(n_reco):
288 track = self.reco_tree.FitTracks[i_reco]
289 fit_status = track.getFitStatus()
290 if not fit_status.isFitConverged():
291 continue
292
293 ndf = fit_status.getNdf()
294 if ndf <= 0:
295 continue
296 chi2 = fit_status.getChi2() / ndf
297 h_chi2ndf.Fill(chi2)
298
299 # Use fitTrack2MC for the MC link (already computed by fracMCsame)
300 mc_id = self.reco_tree.fitTrack2MC[i_reco]
301
302 # Recompute purity to apply our cut
303 frac, _dominant_id = self._fracMCsame(i_reco)
304
305 if frac < self.purity_cut:
306 # Ghost track
307 continue
308
309 n_matched_reco += 1
310
311 if mc_id in reconstructible_ids:
312 if mc_id not in matched_mc_this_event:
313 matched_mc_this_event.add(mc_id)
314 else:
315 n_clone_reco += 1
316
317 # Resolution histograms (use first match only for resolution)
318 p_truth, _, _, _ = self._get_ptruth_first(mc_id)
319 x_t, y_t, _ = self._get_truth_pos_first(mc_id)
320 tx_t, ty_t = self._get_truth_slopes(mc_id)
321
322 if p_truth > 0:
323 try:
324 fitted_state = track.getFittedState()
325 p_reco = fitted_state.getMomMag()
326 mom = fitted_state.getMom()
327 pos = fitted_state.getPos()
328
329 dp_over_p = (p_reco - p_truth) / p_truth
330 h_dp_over_p.Fill(dp_over_p)
331 h_dp_vs_p.Fill(p_truth, dp_over_p)
332
333 h_dx.Fill(pos.X() - x_t)
334 h_dy.Fill(pos.Y() - y_t)
335
336 pz_reco = mom.Z()
337 if abs(pz_reco) > 1e-10:
338 tx_reco = mom.X() / pz_reco
339 ty_reco = mom.Y() / pz_reco
340 h_dtx.Fill(tx_reco - tx_t)
341 h_dty.Fill(ty_reco - ty_t)
342
343 h_p_matched.Fill(p_truth)
344 except Exception:
345 pass
346
347 # Per-hit unbiased residuals and pulls
348 n_meas = track.getNumPointsWithMeasurement()
349 for i_hit in range(n_meas):
350 try:
351 tp = track.getPointWithMeasurement(i_hit)
352 fitter_info = tp.getFitterInfo()
353 if fitter_info is None:
354 continue
355 res = fitter_info.getResidual()
356 # Index 0: 1D residual (straw drift-distance measurement)
357 res_val = res.getState()(0)
358 res_cov = res.getCov()(0, 0)
359 h_hit_residual.Fill(res_val)
360 if res_cov > 0:
361 h_hit_pull.Fill(res_val / math.sqrt(res_cov))
362 except (ROOT.genfit.Exception, AttributeError, IndexError) as e:
363 logger.debug("Skipping hit %d: %s", i_hit, e)
364 continue
365
366 n_matched_mc += len(matched_mc_this_event)
367
368 # Compute metrics
369 n_ghost_reco = n_total_reco - n_matched_reco
370
371 efficiency = n_matched_mc / n_reconstructible if n_reconstructible > 0 else 0.0
372 efficiency_unc = wilson_interval(n_matched_mc, n_reconstructible)
373
374 clone_rate = n_clone_reco / n_matched_reco if n_matched_reco > 0 else 0.0
375 clone_rate_unc = wilson_interval(n_clone_reco, n_matched_reco)
376
377 ghost_rate = n_ghost_reco / n_total_reco if n_total_reco > 0 else 0.0
378 ghost_rate_unc = wilson_interval(n_ghost_reco, n_total_reco)
379
380 # Fit dp/p with Gaussian
381 dp_p_sigma = h_dp_over_p.GetRMS()
382 dp_p_sigma_unc = h_dp_over_p.GetRMSError()
383 if h_dp_over_p.GetEntries() > 20:
384 fit_result = h_dp_over_p.Fit("gaus", "QS")
385 if fit_result and int(fit_result) == 0:
386 dp_p_sigma = fit_result.Parameter(2)
387 dp_p_sigma_unc = fit_result.ParError(2)
388
389 # Fit hit pull with Gaussian
390 hit_pull_underflow = int(h_hit_pull.GetBinContent(0))
391 hit_pull_overflow = int(h_hit_pull.GetBinContent(h_hit_pull.GetNbinsX() + 1))
392 if h_hit_pull.GetEntries() > 50:
393 hit_pull_mean = h_hit_pull.GetMean()
394 hit_pull_mean_unc = h_hit_pull.GetMeanError()
395 hit_pull_sigma = h_hit_pull.GetRMS()
396 hit_pull_sigma_unc = h_hit_pull.GetRMSError()
397 pull_fit = h_hit_pull.Fit("gaus", "QS")
398 if pull_fit and int(pull_fit) == 0:
399 hit_pull_mean = pull_fit.Parameter(1)
400 hit_pull_mean_unc = pull_fit.ParError(1)
401 hit_pull_sigma = pull_fit.Parameter(2)
402 hit_pull_sigma_unc = pull_fit.ParError(2)
403 else:
404 hit_pull_mean = math.nan
405 hit_pull_mean_unc = math.nan
406 hit_pull_sigma = math.nan
407 hit_pull_sigma_unc = math.nan
408
409 self.metrics = {
410 "tracking_benchmark": {
411 "n_events": {"value": int(n_events), "compare": "exact"},
412 "n_reconstructible": {"value": int(n_reconstructible), "compare": "exact"},
413 "n_total_reco": {"value": int(n_total_reco), "compare": "exact"},
414 "efficiency": {

416 "uncertainty": round(efficiency_unc, 6),
417 "compare": "statistical",
418 },
419 "clone_rate": {

421 "uncertainty": round(clone_rate_unc, 6),
422 "compare": "statistical",
423 },
424 "ghost_rate": {
425 "value": round(ghost_rate, 6),
426 "uncertainty": round(ghost_rate_unc, 6),
427 "compare": "statistical",
428 },
429 "dp_over_p_sigma": {
430 "value": round(dp_p_sigma, 6),
431 "uncertainty": round(dp_p_sigma_unc, 6),
432 "compare": "statistical",
433 },
434 "dx_rms": {
435 "value": round(h_dx.GetRMS(), 6),
436 "uncertainty": round(h_dx.GetRMSError(), 6),
437 "compare": "statistical",
438 },
439 "dy_rms": {
440 "value": round(h_dy.GetRMS(), 6),
441 "uncertainty": round(h_dy.GetRMSError(), 6),
442 "compare": "statistical",
443 },
444 "dtx_rms": {
445 "value": round(h_dtx.GetRMS(), 6),
446 "uncertainty": round(h_dtx.GetRMSError(), 6),
447 "compare": "statistical",
448 },
