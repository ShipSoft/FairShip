135def addVMCFields(shipGeo, controlFile = '', verbose = False, withVirtualMC = True):
136 '''
137 Define VMC B fields, e.g. global field, field maps, local or local+global fields
138 '''
139 print('Calling addVMCFields')
140
141 fieldMaker = ROOT.ShipFieldMaker(verbose)
142
143 # Read the input control file. If this is empty then the only fields that are
144 # defined (so far) are those within the C++ geometry classes
145 if controlFile != '':
146 fieldMaker.readInputFile(controlFile)
147
148 # Set the main spectrometer field map as a global field
149 if hasattr(shipGeo, 'Bfield'):
150 fieldsList = []
151 fieldMaker.defineFieldMap('MainSpecMap', shipGeo.Bfield.fieldMap,
152 ROOT.TVector3(0.0, 0.0, shipGeo.Bfield.z))
153 fieldsList.append('MainSpecMap')
154
155 if not shipGeo.hadronAbsorber.WithConstField:
156 fieldMaker.defineFieldMap('HadronAbsorberMap','files/FieldHadronStopper_raised_20190411.root', ROOT.TVector3(0.0,0.0,shipGeo.hadronAbsorber.z))
157 fieldsList.append('HadronAbsorberMap')
158
159 if not shipGeo.muShield.WithConstField:
160 field_center, _ = ShieldUtils.find_shield_center(shipGeo)
161 fieldMaker.defineFieldMap('muonShieldField', 'files/MuonShieldField.root',
162 ROOT.TVector3(0.0, 0.0, field_center), ROOT.TVector3(0.0, 0.0, 0.0), True)
163 fieldsList.append('muonShieldField')
164 # Combine the fields to obtain the global field
165 if len(fieldsList) > 1:
166 fieldMaker.defineComposite('TotalField', *fieldsList) #fieldsList MUST have length <=4
167 fieldMaker.defineGlobalField('TotalField')
168 else:
169 fieldMaker.defineGlobalField('MainSpecMap')
170 if withVirtualMC:
171 # Force the VMC to update/reset the fields defined by the fieldMaker object.
172 # Get the ROOT/Geant4 geometry manager
173 geom = ROOT.TG4GeometryManager.Instance()
174 # Let the geometry know about the fieldMaker object
175 geom.SetUserPostDetConstruction(fieldMaker)
176 # Update the fields via the overridden ShipFieldMaker::Construct() function
177 geom.ConstructSDandField()
178
179 # Return the fieldMaker object, otherwise it will "go out of scope" and its
180 # content will be deleted
181 return fieldMaker
