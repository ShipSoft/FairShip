20static const std::map<std::string, ModelSpec> kVertexModels = {
21 {"gaussian",
22 {5, "Gaus(X), Gaus(Y), Point(Z)",
23 [](ParticleGunParticle& p, const std::vector<Double32_t>& pars) {
24 p.X = (pars[1] > 0) ? gRandom->Gaus(pars[0], pars[1]) : pars[0];
25 p.Y = (pars[3] > 0) ? gRandom->Gaus(pars[2], pars[3]) : pars[2];
26 p.Z = pars[4];
27 }}},
28 {"exponential",
29 {5, "Exp(X), Exp(Y), Point(Z)",
30 [](ParticleGunParticle& p, const std::vector<Double32_t>& pars) {

32 ? gRandom->Exp(pars[1]) + pars[0]
33 : -gRandom->Exp(pars[1]) +
34 pars[0]; // Make this symmetric about a specified point
35 p.Y = (gRandom->Uniform() > 0.5) ? gRandom->Exp(pars[3]) + pars[2]
36 : -gRandom->Exp(pars[3]) + pars[2];
37 p.Z = pars[4];
38 }}},
39 {"uniform",
40 {5, "Uniform(X), Uniform(Y), Point(Z)",
41 [](ParticleGunParticle& p, const std::vector<Double32_t>& pars) {
42 p.X = (pars[1] > 0) ? gRandom->Uniform(pars[0] - pars[1] / 2.,
43 pars[0] + pars[1] / 2.)

45 p.Y = (pars[3] > 0) ? gRandom->Uniform(pars[2] - pars[3] / 2.,

76void ParticleGunGenerator::LoadHistoFromFile(
77 const std::string& inFile, const std::string& inHisto,
78 std::vector<std::string> varNames) {
79 TFile* loadFile = TFile::Open(inFile.c_str(), "READ");
80 if (!loadFile || loadFile->IsZombie()) {
81 delete loadFile;
82 LOG(fatal) << "ParticleGunGenerator: Cannot open file " << inFile;
83 throw std::runtime_error("ParticleGunGenerator: Cannot open file " +
84 inFile);
85 }
86
87 auto* raw = dynamic_cast<TH1*>(loadFile->Get(inHisto.c_str()));
88 if (!raw) {
89 loadFile->Close();
90 delete loadFile;
91 LOG(fatal) << "ParticleGunGenerator: Histogram " << inHisto
92 << " not found in file " << inFile;
93 throw std::runtime_error("ParticleGunGenerator: Histogram " + inHisto +
94 " not found in file " + inFile);
95 }
96 raw = static_cast<TH1*>(raw->Clone());
97 raw->SetDirectory(nullptr); // detach from ROOT's directory ownership
98 loadFile->Close();
99 delete loadFile;
100
101 const int dims = raw->GetDimension();
102 if (static_cast<int>(varNames.size()) != dims) {
103 LOG(fatal) << "ParticleGunGenerator: " << varNames.size()
104 << " variables given for a " << dims << "-D histogram";
105 throw std::runtime_error(
106 "ParticleGunGenerator: " + std::to_string(varNames.size()) +
107 " variables given for a " + std::to_string(dims) + "-D histogram");
108 } else {
109 ParticleGunParticle probe;
110 for (const auto& name : varNames) {
111 static_cast<void>(GetVar(probe, name));
112 }
113 }
114
115 // Check for variable name conflicts with already-loaded histograms
116 for (const auto& name : varNames) {
117 for (const auto& entry : fHistoEntries) {
118 for (const auto& existing : entry.varNames) {
119 if (existing == name) {
120 LOG(fatal)
121 << "ParticleGunGenerator: variable \"" << name
122 << "\" is already controlled by a previously loaded histogram";
123 throw std::runtime_error(
124 "ParticleGunGenerator: variable \"" + name +
125 "\" is already controlled by a previously loaded histogram");
126 }
127 }
128 }
129 }
130
131 fHistoEntries.push_back(
132 {std::shared_ptr<TH1>(raw), dims, std::move(varNames)});
133
134 LOG(info) << "ParticleGunGenerator: loaded " << dims << "-D histogram \""
135 << inHisto << "\" from " << inFile << " (" << fHistoEntries.size()
136 << " total)";
137}

139Bool_t ParticleGunGenerator::Init() {
140 const auto& vertexSpec = kVertexModels.at(fVertexModel);
141
142 if (static_cast<int>(fVertexPars.size()) != vertexSpec.expectedPars) {
143 throw std::runtime_error(
144 "ParticleGunGenerator: vertex model " + fVertexModel + " requires " +
145 std::to_string(vertexSpec.expectedPars) + " parameters, got " +
146 std::to_string(fVertexPars.size()));
147 }
148
149 // Check for particle type
150 TDatabasePDG* pdgBase = TDatabasePDG::Instance();
151 TParticlePDG* particle = pdgBase->GetParticle(m_pdgid);
152
153 if (!particle) {
154 LOG(fatal) << "ParticleGunGenerator: PDG " << m_pdgid << " not defined";
155 return kFALSE;
156 }
157 LOG(info) << this->ClassName() << ": particle with PDG =" << m_pdgid
158 << " Found";
159 fPDGMass = particle->Mass();
160
161 if (fPhiMax - fPhiMin > 360) {
162 LOG(fatal) << "ParticleGunGenerator:Init(): phi range is too wide: "
163 << fPhiMin << "<phi<" << fPhiMax;
164 return kFALSE;
165 }
166 if (fEkinRangeIsSet) {
167 if (fPRangeIsSet) {
168 LOG(fatal) << "ParticleGunGenerator:Init(): Cannot set P and Ekin ranges "
169 "simultaneously";
170 return kFALSE;
171 } else {
172 fPMin = TMath::Sqrt(fEkinMin * (fEkinMin + 2 * fPDGMass));
173 fPMax = TMath::Sqrt(fEkinMax * (fEkinMax + 2 * fPDGMass));
174 fPRangeIsSet = kTRUE;
175 fEkinRangeIsSet = kFALSE;
176 }
177 }
178 if (fPRangeIsSet && fPtRangeIsSet) {
179 LOG(fatal) << "ParticleGunGenerator:Init(): Cannot set P and Pt ranges "
180 "simultaneously";
181 return kFALSE;
182 }
183 if (fPRangeIsSet && fYRangeIsSet) {
184 LOG(fatal) << "ParticleGunGenerator:Init(): Cannot set P and Y ranges "
185 "simultaneously";
186 return kFALSE;
187 }
188 if ((fThetaRangeIsSet && fYRangeIsSet) ||
189 (fThetaRangeIsSet && fEtaRangeIsSet) ||
190 (fYRangeIsSet && fEtaRangeIsSet)) {
191 LOG(fatal) << "ParticleGunGenerator:Init(): Cannot set Y, Theta or Eta "
192 "ranges simultaneously";
193 return kFALSE;
194 }
195 return kTRUE;
196}

198void ParticleGunGenerator::SetBothCharges(bool flag, Double32_t fraction) {
199 if (flag && (!std::isfinite(fraction) || fraction < 0. || fraction > 1.)) {
200 LOG(fatal)
201 << "ParticleGunGenerator: charge fraction must be finite and in [0, 1]";
202 throw std::runtime_error(
203 "ParticleGunGenerator: charge fraction must be finite and in [0, 1]");
204 }
205 m_bothCharges = flag;
206 m_chargeFraction = flag ? fraction : 1.;
207 if (m_bothCharges) {
208 // Check for particle type
209 TDatabasePDG* pdgBase = TDatabasePDG::Instance();
210 TParticlePDG* particle = pdgBase->GetParticle(-m_pdgid);
211 if (!particle) {
212 LOG(warning) << "ParticleGunGenerator: PDG " << -m_pdgid
213 << " not defined. Only generating " << m_pdgid;
214 m_bothCharges = false;
215 m_chargeFraction = 1.;
216 } else
217 LOG(info)
218 << "ParticleGunGenerator: Setting to generate both charges, with "
219 "fraction of "
220 << m_pdgid << " set to " << m_chargeFraction;
221 } else
222 LOG(info) << "ParticleGunGenerator: Setting to only generate one charge "
223 "for PDG id"
224 << m_pdgid;
225}

235Bool_t ParticleGunGenerator::ReadEvent(FairPrimaryGenerator* primGen) {
236 int pdg_type;
237 for (Int_t k = 0; k < GetMultiplicity(); k++) {
238 pdg_type = GenPDGType();

240 OverrideFromHistogram(p);

246ParticleGunParticle ParticleGunGenerator::GenerateKinematics() {
247 ParticleGunParticle p;
248 // --- Vertex smearing ---
249 GenVertexModel(p);
250 if (fMomentumModel > 0)
251 GenMomentumModel(p);
252 else {
253 Double32_t pabs = 0, pt = 0, theta = 0, eta, y, mt, pz = 0;
254 // Generate particles
255 const Double32_t phi =
256 gRandom->Uniform(fPhiMin, fPhiMax) * TMath::DegToRad();
257
258 if (fPRangeIsSet) {
259 pabs = gRandom->Uniform(fPMin, fPMax);
260 } else if (fPtRangeIsSet) {
261 pt = gRandom->Uniform(fPtMin, fPtMax);
262 }
263
264 if (fThetaRangeIsSet) {
265 if (fCosThetaIsSet)
266 theta = acos(gRandom->Uniform(cos(fThetaMax * TMath::DegToRad()),
267 cos(fThetaMin * TMath::DegToRad())));
268 else {
269 theta = gRandom->Uniform(fThetaMin, fThetaMax) * TMath::DegToRad();
270 }
271 } else if (fEtaRangeIsSet) {
272 eta = gRandom->Uniform(fEtaMin, fEtaMax);
273 theta = 2 * TMath::ATan(TMath::Exp(-eta));
274 } else if (fYRangeIsSet) {
275 y = gRandom->Uniform(fYMin, fYMax);
276 mt = TMath::Sqrt(GetPDGMass() * GetPDGMass() + pt * pt);
277 pz = mt * TMath::SinH(y);
278 }
279
280 if (fThetaRangeIsSet || fEtaRangeIsSet) {
281 if (fPRangeIsSet) {
282 pz = pabs * TMath::Cos(theta);
283 pt = pabs * TMath::Sin(theta);
284 } else if (fPtRangeIsSet) {
285 pz = pt / TMath::Tan(theta);
286 }
287 } else if (fPRangeIsSet) {
288 pz = pabs;
289 }
290 p.Px = pt * TMath::Cos(phi);
291 p.Py = pt * TMath::Sin(phi);
292 p.Pz = pz;
293 }
294 return p;
295}

297void ParticleGunGenerator::OverrideFromHistogram(ParticleGunParticle& p) {
298 for (const auto& entry : fHistoEntries) {
299 switch (entry.dims) {
300 case 1:
301 SetVar(p, entry.varNames[0],
302 static_cast<TH1*>(entry.hist.get())->GetRandom());
303 break;
304 case 2:
305 static_cast<TH2*>(entry.hist.get())
306 ->GetRandom2(GetVar(p, entry.varNames[0]),
307 GetVar(p, entry.varNames[1]));
308 break;
309 case 3:
310 static_cast<TH3*>(entry.hist.get())
311 ->GetRandom3(GetVar(p, entry.varNames[0]),
312 GetVar(p, entry.varNames[1]),
313 GetVar(p, entry.varNames[2]));
314 break;
315 default:
316 LOG(error) << "ParticleGunGenerator: unsupported histogram dimension "
317 << entry.dims;
318 }
319 }
