146def makeFinalNtuples(norm=5.0e13, opt=""):
147 cuts = {"": "", "_onlyMuons": "abs(id)==13", "_onlyNeutrinos": "abs(id)==12||abs(id)==14||abs(id)==16"}
148 first = True
149 tuples = ""
150 fn = 1
151 for ecut in stats:
152 for i in range(len(stats[ecut])):
153 h[fn] = ROOT.TFile(files[ecut][i])
154 t = h[fn].FindObjectAny("pythia8-Geant4")
155 fn += 1
156 if first:
157 first = False
158 for leaf in t.GetListOfLeaves():
159 tuples += leaf.GetName() + ":"
160 tuples += "w:ecut"
161 fxx = fnew.replace(".root", opt + ".root")
162 if opt != "":
163 fxx = fxx.replace("_total", "")
164 h["N"] = ROOT.TFile(fxx, "RECREATE")
165 print("new file created", fxx)
166 if afterHadronAbsorber:
167 h["ntuple"] = ROOT.TNtuple("pythia8-Geant4", "flux after 3m hadron absorber", tuples)
168 else:
169 h["ntuple"] = ROOT.TNtuple(
170 "pythia8-Geant4", "flux after 3m hadron absorber, position&momentum before", tuples
171 )
172 gROOT.cd()
173 t.SetEventList(0)
174 t.Draw(">>temp", cuts[opt])
175 temp = gROOT.FindObjectAny("temp")
176 t.SetEventList(temp)
177 nev = temp.GetN()
178 for iev in range(nev):
179 t.GetEntry(temp.GetEntry(iev))
180 leaves = t.GetListOfLeaves()
181 vlist = array("f")
182 for x in range(leaves.GetEntries()):
183 vlist.append(leaves.At(x).GetValue())
184 # get kinetic energy "id:px:py:pz:x:y:z:ox:oy:oz:pythiaid:parentid"
185 # since 2016 "id:px:py:pz:x:y:z:opx:opy:opz:ox:oy:oz:pythiaid:parentid"
186 Psq = vlist[1] ** 2 + vlist[2] ** 2 + vlist[3] ** 2
187 if abs(vlist[0]) == 13:
188 Ekin = ROOT.TMath.Sqrt(Mmu2 + Psq) - Mmu
189 else:
190 Ekin = ROOT.TMath.Sqrt(Psq)
191 if Yandex:
192 if Ekin < ecut:
193 continue
194 if Ekin > 5.0:
195 weight = norm / (ntot[5.0] + ntot[0.5])
196 else:
197 weight = norm / (ntot[0.5])
198 if Yandex2 or Yandex3:
199 if Ekin < ecut:
200 continue
201 weight = norm / (ntot[10.0])
202 if JPsi:
203 weight = norm / (ntot[10.0])
204 if Tau:
205 weight = norm / (ntot[0.0])
206 if MoTarget:
207 if Ekin < ecut:
208 continue
209 if Ekin > 50.0:
210 weight = norm / (ntot[0.5] + ntot[10.0] + ntot[20.0] + ntot[50.0])
211 elif Ekin > 20.0:
212 weight = norm / (ntot[0.5] + ntot[10.0] + ntot[20.0])
213 elif Ekin > 10.0:
214 weight = norm / (ntot[0.5] + ntot[10.0])
215 else:
216 weight = norm / (ntot[0.5])
217 vlist.append(weight)
218 vlist.append(float(ecut))
219 h["ntuple"].Fill(vlist)
220 h["N"].cd()
