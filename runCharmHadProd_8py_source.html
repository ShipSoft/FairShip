164def compactifyCascade(run):

170 fName = path + "run" + str(i) + "/Cascade-run" + str(i) + "-parp16-MSTP82-1-MSEL" + msel + ".root"
171 f = open(path + "run" + str(i) + "/log" + str(i))
172 success = False
173 for line in f.readlines():
174 if not line.find("Macro finished successfully") < 0:
175 success = True
176 if not success:
177 print("job not finished properly", fName)
178 continue
179 cmd += fName + " "
180 f.close()
181 Ntot += NperJob
182 if cmd.find("root") < 0:
183 print("no file found, exit")
184 else:
185 stat = str(int(Ntot / 1e6)) + "Mpot"
186 outFile = (
187 "Cascade-run"
188 + str(run)
189 + "-"
190 + str(run + ncpus - 1)
191 + "-parp16-MSTP82-1-MSEL"
192 + msel
193 + "-"
194 + stat
195 + ".root"
196 )
197 rc = os.system("hadd -O " + outFile + " " + cmd)
