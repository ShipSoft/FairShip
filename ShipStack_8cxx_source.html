142TParticle* ShipStack::PopNextTrack(Int_t& iTrack)
143{
144
145 // If end of stack: Return empty pointer
146 if (fStack.empty()) {
147 iTrack = -1;
148 return NULL;
149 }
150
151 // If not, get next particle from stack
152 TParticle* thisParticle = fStack.top();
153 fStack.pop();
154
155 if ( !thisParticle) {
156 iTrack = 0;
157 return NULL;
158 }
159
160 fCurrentTrack = thisParticle->GetStatusCode();
161 iTrack = fCurrentTrack;
162

224// ----- Public method FillTrackArray ----------------------------------
225void ShipStack::FillTrackArray()
226{
227
228 LOG(DEBUG) << "ShipStack: Filling MCTrack array...";
229
230 // --> Reset index map and number of output tracks
231 fIndexMap.clear();
232 fNTracks = 0;
233
234 // --> Check tracks for selection criteria
235 SelectTracks();
236
237 // --> Loop over fParticles array and copy selected tracks
238 for (Int_t iPart=0; iPart<fNParticles; iPart++) {
239
240 fStoreIter = fStoreMap.find(iPart);
241 if (fStoreIter == fStoreMap.end() ) {
242 LOGF(fatal, "ShipStack: Particle %i not found in storage map! ", iPart);
243 }
244 Bool_t store = (*fStoreIter).second;
245
246 if (store) {
247 ShipMCTrack* track =
248 new( (*fTracks)[fNTracks]) ShipMCTrack(GetParticle(iPart));
249 fIndexMap[iPart] = fNTracks;
250 // --> Set the number of points in the detectors for this track
251 for (Int_t iDet=kVETO; iDet<kEndOfList; iDet++) {
252 pair<Int_t, Int_t> a(iPart, iDet);
253 track->SetNPoints(iDet, fPointsMap[a]);
254 }
255 fNTracks++;
256 } else { fIndexMap[iPart] = -2; }
257
258 }
259
260 // --> Map index for primary mothers
261 fIndexMap[-1] = -1;
262
263 // --> Screen output
264 //Print(1);

271// ----- Public method UpdateTrackIndex --------------------------------
272void ShipStack::UpdateTrackIndex(TRefArray* detList)
273{
274
275 LOG(DEBUG) << "ShipStack: Updating track indizes...";
276 Int_t nColl = 0;
277
278 // First update mother ID in MCTracks
279 for (Int_t i=0; i<fNTracks; i++) {
280 ShipMCTrack* track = (ShipMCTrack*)fTracks->At(i);
281 Int_t iMotherOld = track->GetMotherId();
282 fIndexIter = fIndexMap.find(iMotherOld);
283 if (fIndexIter == fIndexMap.end()) {
284 LOGF(fatal, "ShipStack: Particle index %i not found in dex map! ", iMotherOld);
285 }
286 track->SetMotherId( (*fIndexIter).second );
287 }
288
289
290 if(fDetList==0) {
291 // Now iterate through all active detectors
292 fDetIter = detList->MakeIterator();
293 fDetIter->Reset();
294 } else {
295 fDetIter->Reset();
296 }
297
298 FairDetector* det = NULL;
299 while( (det = (FairDetector*)fDetIter->Next() ) ) {
300
301
302 // --> Get hit collections from detector
303 Int_t iColl = 0;
304 TClonesArray* hitArray;
305 while ( (hitArray = det->GetCollection(iColl++)) ) {
306 nColl++;
307 Int_t nPoints = hitArray->GetEntriesFast();
308
309 // --> Update track index for all MCPoints in the collection
310 for (Int_t iPoint=0; iPoint<nPoints; iPoint++) {
311 FairMCPoint* point = (FairMCPoint*)hitArray->At(iPoint);
312 Int_t iTrack = point->GetTrackID();
313
314 fIndexIter = fIndexMap.find(iTrack);
315 if (fIndexIter == fIndexMap.end()) {
316 LOGF(fatal, "ShipStack: Particle index %i not found in index map! ", iTrack);
317 }
318 point->SetTrackID((*fIndexIter).second);
319 point->SetLink(FairLink("MCTrack", (*fIndexIter).second));
320 }
321
322 } // Collections of this detector
323 } // List of active detectors
324 LOGF(debug, "...stack and %i collections updated.", nColl);
325}

423// ----- Private method SelectTracks -----------------------------------
424void ShipStack::SelectTracks()
425{
426
427 // --> Clear storage map
428 fStoreMap.clear();
429
430 // --> Check particles in the fParticle array
431 for (Int_t i=0; i<fNParticles; i++) {
432
433 TParticle* thisPart = GetParticle(i);
434 Bool_t store = kTRUE;
435
436 // --> Get track parameters
437 Int_t iMother = thisPart->GetMother(0);
438 TLorentzVector p;
439 thisPart->Momentum(p);
440 Double_t energy = p.E();
441 Double_t mass = p.M();
442// Double_t mass = thisPart->GetMass();
443 Double_t eKin = energy - mass;
444
445 // --> Calculate number of points
446 Int_t nPoints = 0;
447 for (Int_t iDet=kVETO; iDet<kEndOfList; iDet++) {
448 pair<Int_t, Int_t> a(i, iDet);
449 if ( fPointsMap.find(a) != fPointsMap.end() ) {
450 nPoints += fPointsMap[a];
451 }
452 }
453
454 // --> Check for cuts (store primaries in any case)
455 if (iMother < 0) { store = kTRUE; }
456 else {
457 if (!fStoreSecondaries) { store = kFALSE; }
458 if (nPoints < fMinPoints) { store = kFALSE; }
459 if (eKin < fEnergyCut) { store = kFALSE; }
460 if ((nPoints >= fMinPoints && fMinPoints > 0) || (eKin >= fEnergyCut && fEnergyCut > 0)) {
461 store=kTRUE;
462 }
463 }
464 // --> Set storage flag
465 fStoreMap[i] = store;
466// special case for Ship generators, want to keep all original particles with their mother daughter relationship
467// independent if tracked or not. apply a dirty trick and use second mother to identify original generator particles.
468// doesn't work, always true: Int_t iMother2 = GetParticle(i)->GetMother(1); maybe should set Mother2 to -1 in the generator
469// if (iMother == iMother2) {fStoreMap[i] = kTRUE;}
470 }
471 // --> If flag is set, flag recursively mothers of selected tracks
472 if (fStoreMothers) {
473 for (Int_t i=0; i<fNParticles; i++) {
474 if (fStoreMap[i]) {
475 Int_t iMother = GetParticle(i)->GetMother(0);
476 {
477 while(iMother >= 0)
478 {
479 fStoreMap[iMother] = kTRUE;
480 iMother = GetParticle(iMother)->GetMother(0);
481 }
482 }
483 }
484 }
485 }
