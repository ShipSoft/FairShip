144Int_t Hpt::InitMedium(const char* name)
145{
146 static FairGeoLoader *geoLoad=FairGeoLoader::Instance();
147 static FairGeoInterface *geoFace=geoLoad->getGeoInterface();
148 static FairGeoMedia *media=geoFace->getMedia();
149 static FairGeoBuilder *geoBuild=geoLoad->getGeoBuilder();
150
151 FairGeoMedium *ShipMedium=media->getMedium(name);
152
153 if (!ShipMedium)
154 {
155 Fatal("InitMedium","Material %s not defined in media file.", name);
156 return -1111;
157 }
158 TGeoMedium* medium=gGeoManager->GetMedium(name);
159 if (medium!=NULL)
160 return ShipMedium->getMediumIndex();
161 return geoBuild->createMedium(ShipMedium);
162}

224 HPT_support_volume->SetLineColor(kGray - 2);
225 HPT_support_volume->SetVisibility(1);

231 HPT_honeycomb_volume->SetVisibility(1);

233 //SciFi planes

242 //SciFi mats
243 TGeoBBox* HPT_scifimat_hor_box = new TGeoBBox("HPT_scifimat_hor_box", scifimat_hor / 2, scifimat_width / 2, scifimat_z / 2);
244 TGeoVolume* HPT_scifimat_hor_volume = new TGeoVolume("HPT_scifimat_hor", HPT_scifimat_hor_box, SciFiMat);
245 HPT_scifimat_hor_volume->SetLineColor(kCyan);
246
247 TGeoBBox* HPT_scifimat_vert_box = new TGeoBBox("HPT_scifimat_vert_box", scifimat_width / 2, scifimat_vert / 2, scifimat_z / 2);
248 TGeoVolume* HPT_scifimat_vert_volume = new TGeoVolume("HPT_scifimat_vert", HPT_scifimat_vert_box, SciFiMat);
249 HPT_scifimat_vert_volume->SetLineColor(kGreen);
250
251 AddSensitiveVolume(HPT_scifimat_hor_volume);
252 AddSensitiveVolume(HPT_scifimat_vert_volume);

254 // Creating physical volumes and multiply
255 for (int i = 0; i < n_hor_planes; i++){
256 HPT_scifi_plane_hor_volume->AddNode(HPT_scifimat_hor_volume, i+1, new TGeoTranslation(0, (-(n_hor_planes-1)/2.0 + i)*scifimat_width, 0));
257 }
258 for (int i = 0; i < n_vert_planes; i++){
259 HPT_scifi_plane_vert_volume->AddNode(HPT_scifimat_vert_volume, 100+i+1 , new TGeoTranslation((-(n_vert_planes-1)/2.0 + i)*scifimat_width, 0, 0));
260 }
261
262 volDT->AddNode(HPT_support_volume, 0, new TGeoTranslation(0, 0, - DimZ / 2 + support_z / 2));
263 volDT->AddNode(HPT_scifi_plane_hor_volume, 0, new TGeoTranslation(0, 0, - DimZ / 2 + support_z + scifimat_z / 2));
264 volDT->AddNode(HPT_scifi_plane_vert_volume, 0, new TGeoTranslation(0, 0, - DimZ / 2 + support_z + scifimat_z + scifimat_z / 2));

268

270 for(int i=0;i<fnHPT;i++){
271 for (int j = 0; j < fntarget; j++){
272 volMagRegion->AddNode(volDT,(i+1)*1000+j*fnHPT,new TGeoTranslation(0,0, first_DT_position + i*(fDistance+DimZ) + j*(DZTarget+ fnHPT * DimZ + (fnHPT-1)*fDistance)));
273 }

278Bool_t Hpt::ProcessHits(FairVolume* vol)
279{
281 //Set parameters at entrance of volume. Reset ELoss.
282 if ( gMC->IsTrackEntering() ) {
283 fELoss = 0.;
284 fTime = gMC->TrackTime() * 1.0e09;
285 fLength = gMC->TrackLength();
286 gMC->TrackPosition(fPos);
287 gMC->TrackMomentum(fMom);
288 }
289 // Sum energy loss for all steps in the active volume
290 fELoss += gMC->Edep();
291
292 // Create muonPoint at exit of active volume
293 if ( gMC->IsTrackExiting() ||
294 gMC->IsTrackStop() ||
295 gMC->IsTrackDisappeared() ){
296 if (fELoss == 0. ) { return kFALSE; }
297 TParticle* p=gMC->GetStack()->GetCurrentTrack();
298 Int_t pdgCode = p->GetPdgCode();
299 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
300
301 gMC->CurrentVolID(fVolumeID);
302 Int_t detID = fVolumeID;
303 Int_t TTstationID;
304 gMC->CurrentVolOffID(2, TTstationID);
305 fVolumeID = TTstationID + detID;
306
307 TLorentzVector Pos;
308 gMC->TrackPosition(Pos);
309 Double_t xmean = (fPos.X()+Pos.X())/2. ;
310 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
311 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
312
313 AddHit(fTrackID, fVolumeID, TVector3(xmean, ymean, zmean),
314 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
315 fTime, fLength,fELoss, pdgCode);
316
317 // Increment number of muon det points in TParticle
318 ShipStack* stack = (ShipStack*) gMC->GetStack();
319 stack->AddPoint(ktauRpc);
320 }
321
322 return kTRUE;
323}
